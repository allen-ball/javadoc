006package ball.util.ant.taskdefs;
007
008import java.io.File;
009import java.util.Set;
010import java.util.TreeSet;
011import lombok.Getter;
012import lombok.NoArgsConstructor;
013import lombok.Setter;
014import lombok.ToString;
015import lombok.experimental.Accessors;
016import org.apache.tools.ant.BuildException;
017import org.apache.tools.ant.Task;
018import org.apache.tools.ant.taskdefs.optional.depend.ClassFile;
019import org.apache.tools.ant.taskdefs.optional.depend.ClassFileUtils;
020import org.apache.tools.ant.taskdefs.optional.depend.DirectoryIterator;
021import org.apache.tools.ant.types.Path;
022import org.apache.tools.ant.util.ClasspathUtils;
023
024import static java.lang.reflect.Modifier.isAbstract;
025import static java.util.Comparator.comparing;
026
027/**
028 * {@link.uri http://ant.apache.org/ Ant} {@link Task} to bootstrap
029 * {@link javax.annotation.processing.Processor}s.  Creates and invokes
030 * {@link Processor}s found on the class path.
031 *
032 * {@ant.task}
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 * @version $Revision: 4275 $
036 */
037@NoArgsConstructor @ToString
038public class BootstrapProcessorTask extends Task
039                                    implements AnnotatedAntTask,
040                                               ClasspathDelegateAntTask {
041    private static final String _JAVA = ".java";
042
043    @Getter @Setter @Accessors(chain = true, fluent = true)
044    private ClasspathUtils.Delegate delegate = null;
045    @Getter @Setter
046    private File basedir = null;
047    @Getter
048    private Path srcPath = null;
049    @Getter @Setter
050    private File destdir = null;
051
052    public void setSrcdir(Path srcdir) {
053        if (srcPath == null) {
054            srcPath = srcdir;
055        } else {
056            srcPath.append(srcdir);
057        }
058    }
059
060    public Path createSrc() {
061        if (srcPath == null) {
062            srcPath = new Path(getProject());
063        }
064
065        return srcPath.createPath();
066    }
067
068    @Override
069    public void init() throws BuildException {
070        super.init();
071        ClasspathDelegateAntTask.super.init();
072    }
073
074    @Override
075    public void execute() throws BuildException {
076        super.execute();
077        AnnotatedAntTask.super.execute();
078
079        if (getBasedir() == null) {
080            setBasedir(getProject().resolveFile("."));
081        }
082
083        if (getDestdir() == null) {
084            setDestdir(getBasedir());
085        }
086
087        try {
088            for (Class<?> type : getClassSet()) {
089                if (Processor.class.isAssignableFrom(type)) {
090                    if (! isAbstract(type.getModifiers())) {
091                        type.asSubclass(Processor.class).newInstance()
092                            .process(getClassSet(), getDestdir());
093                    }
094                }
095            }
096        } catch (BuildException exception) {
097            throw exception;
098        } catch (Throwable throwable) {
099            throwable.printStackTrace();
100            throw new BuildException(throwable);
101        }
102    }
103
104    protected Set<Class<?>> getClassSet() throws BuildException {
105        TreeSet<Class<?>> set = new TreeSet<>(comparing(t -> t.getName()));
106
107        try {
108            DirectoryIterator iterator =
109                new DirectoryIterator(getBasedir(), true);
110            ClassFile file = null;
111
112            while ((file = iterator.getNextClassFile()) != null) {
113                set.add(getClassForName(file.getFullClassName()));
114            }
115        } catch (BuildException exception) {
116            throw exception;
117        } catch (RuntimeException exception) {
118            throw exception;
119        } catch (Exception exception) {
120            throw new BuildException(exception);
121        }
122
123        return set;
124    }
125
126    protected File getJavaFile(Class<?> type) {
127        File file = null;
128
129        if (srcPath != null && type != null) {
130            while (type.getDeclaringClass() != null) {
131                type = type.getDeclaringClass();
132            }
133
134            while (type.getEnclosingClass() != null) {
135                type = type.getEnclosingClass();
136            }
137
138            String child =
139                ClassFileUtils.convertDotName(type.getCanonicalName()) + _JAVA;
140
141            for (String parent : srcPath.list()) {
142                file = new File(parent, child);
143
144                if (file.isFile()) {
145                    break;
146                } else {
147                    file = null;
148                }
149            }
150        }
151
152        return file;
153    }
154
155    /**
156     * Bootstrap processor interface.
157     */
158    public interface Processor {
159
160        /**
161         * Bootstrap method called by this {@link org.apache.tools.ant.Task}.
162         *
163         * @param       set     The {@link Set} of {@link Class}es to
164         *                      examine.
165         * @param       destdir The root of the hierarchy to record any
166         *                      output artifacts.
167         *
168         * @throws      Exception
169         *                      If the implementation throws an
170         *                      {@link Exception}.
171         */
172        public void process(Set<Class<?>> set, File destdir) throws Exception;
173    }
174}
