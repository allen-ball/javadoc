064/**
065 * Abstract base class for javadoc {@link org.apache.maven.plugin.Mojo}s.
066 *
067 * {@bean.info}
068 *
069 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
070 * @version $Revision: 8409 $
071 */
072@NoArgsConstructor(access = PROTECTED) @Getter @ToString @Slf4j
073public abstract class AbstractJavadocMojo extends AbstractMojo {
074    @Parameter(defaultValue = "${project.remoteArtifactRepositories}", readonly = true, required = true)
075    private List<ArtifactRepository> remoteRepositories = Collections.emptyList();
076
077    @Parameter(required = false)
078    private Link[] links = new Link[] { };
079
080    @Parameter(required = false)
081    private Offlinelink[] offlinelinks = new Offlinelink[] { };
082
083    @Parameter(defaultValue = "false", property = "javadoc.skip")
084    private boolean skip = false;
085
086    @Inject private MavenSession session = null;
087    @Inject private ArtifactHandlerManager manager = null;
088    @Inject private RepositorySystem system = null;
089    @Inject private ArtifactResolver resolver = null;
090
091    /**
092     * Method to produce a {@link Stream} of
093     * {@link MavenProject#getDependencies()} and
094     * {@link MavenProject#getDependencyManagement()}
095     * {@link Dependency Dependencies}.
096     *
097     * @param   project         The {@link MavenProject}.
098     *
099     * @return  The {@link Dependency} {@link Stream}.
100     */
101    protected Stream<Dependency> getDependencyManagementStream(MavenProject project) {
102        Stream<Dependency> stream =
103            Stream.of(project.getDependencies(),
104                      project.getDependencyManagement().getDependencies())
105            .filter(Objects::nonNull)
106            .flatMap(Collection::stream);
107
108        return stream;
109    }
110
111    /**
112     * Method to get the {@link Set} of {@link Link} {@link URL}s.
113     *
114     * @param   project         The {@link MavenProject}.
115     * @param   includeDependencyManagement
116     *                          Whether or not to include dependency
117     *                          management in the analysis.
118     *
119     * @return  The {@link Set} of {@link Link} {@link URL}s.
120     */
121    protected Set<URL> getLinkSet(MavenProject project, boolean includeDependencyManagement) {
122        Set<URL> set = new LinkedHashSet<>();
123
124        for (Link link : links) {
125            project.getArtifacts().stream()
126                .filter(link::include)
127                .map(link::getUrl)
128                .forEach(set::add);
129        }
130
131        if (includeDependencyManagement) {
132            for (Link link : links) {
133                getDependencyManagementStream(project)
134                    .filter(t -> isNotBlank(t.getVersion()))
135                    .map(JavadocArtifact::new)
136                    .filter(link::include)
137                    .map(link::getUrl)
138                    .forEach(set::add);
139            }
140        }
141
142        return set;
143    }
144
145    /**
146     * Method to get the {@link Map} of {@link Offlinelink}
147     * {@link Artifact}s to {@link URL}s.
148     *
149     * @param   project         The {@link MavenProject}.
150     * @param   includeDependencyManagement
151     *                          Whether or not to include dependency
152     *                          management in the analysis.
153     *
154     * @return  The {@link Map} of {@link Offlinelink} {@link Artifact}s to
155     *          {@link URL}s.
156     */
157    protected Map<Artifact,URL> getResolvedOfflinelinkMap(MavenProject project, boolean includeDependencyManagement) {
158        TreeMap<Artifact,URL> map = new TreeMap<>(Comparator.comparing(ArtifactUtils::versionlessKey));
159
160        for (Offlinelink offlinelink : offlinelinks) {
161            project.getArtifacts().stream()
162                .filter(t -> Objects.equals(t.getType(), "jar"))
163                .filter(t -> Objects.equals(t.getClassifier(), "javadoc"))
164                .filter(offlinelink::include)
165                .forEach(t -> map.putIfAbsent(t, offlinelink.getUrl()));
166        }
167
168        Set<Artifact> artifacts =
169            project.getArtifacts().stream()
170            .filter(t -> Objects.equals(t.getType(), "jar"))
171            .filter(t -> isBlank(t.getClassifier()))
172            .map(JavadocArtifact::new)
173            .collect(toCollection(() -> new TreeSet<>(map.comparator())));
174
175        if (includeDependencyManagement) {
176            getDependencyManagementStream(project)
177                .filter(t -> Objects.equals(t.getType(), "jar"))
178                .filter(t -> isBlank(t.getClassifier()))
179                .filter(t -> isNotBlank(t.getVersion()))
180                .map(JavadocArtifact::new)
181                .forEach(artifacts::add);
182        }
183
184        artifacts.removeAll(map.keySet());
185
186        ProjectBuildingRequest request = getProjectBuildingRequest();
187
188        for (Offlinelink offlinelink : offlinelinks) {
189            Set<Artifact> set =
190                artifacts.stream()
191                .filter(offlinelink::include)
192                .collect(toCollection(LinkedHashSet::new));
193
194            for (Artifact artifact : set) {
195                URL url = map.get(artifact);
196
197                if (url == null) {
198                    log.info("Resolving {}...", artifact);
199
200                    try {
201                        map.putIfAbsent(resolver.resolveArtifact(request, artifact).getArtifact(),
202                                        offlinelink.getUrl(artifact));
203                    } catch (Exception exception) {
204                        log.warn("{}: {}", artifact, exception.getMessage());
205                        log.debug("{}", exception);
206                    }
207                } else {
208                    if (! Objects.equals(url, offlinelink.getUrl(artifact))) {
209                        log.warn("{} matches {} but was previously resolved with {}",
210                                 artifact, offlinelink, url);
211                    }
212                }
213            }
214        }
215
216        return map;
217    }
218
219    private ProjectBuildingRequest getProjectBuildingRequest() {
220        List<ArtifactRepository> repoList = getRemoteRepositories();
221        Settings settings = session.getSettings();
222
223        system.injectMirror(repoList, settings.getMirrors());
224        system.injectProxy(repoList, settings.getProxies());
225        system.injectAuthentication(repoList, settings.getServers());
226
227        ProjectBuildingRequest request = new DefaultProjectBuildingRequest(session.getProjectBuildingRequest());
228
229        request.setRemoteRepositories(repoList);
230
231        return request;
232    }
233
234    @Override
235    public void execute() throws MojoExecutionException, MojoFailureException {
236    }
237
238    private class JavadocArtifact extends DefaultArtifact {
239        public JavadocArtifact(Artifact artifact) {
240            this(artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion());
241        }
242
243        public JavadocArtifact(Dependency dependency) {
244            this(dependency.getGroupId(), dependency.getArtifactId(), dependency.getVersion());
245        }
246
247        public JavadocArtifact(String gav) { this(gav.split("[:]")); }
248
249        public JavadocArtifact(String... gav) {
250            super(gav.length > 0 ? gav[0] : EMPTY,
251                  gav.length > 1 ? gav[1] : EMPTY,
252                  gav.length > 2 ? gav[gav.length - 1] : EMPTY,
253                  EMPTY, "jar", "javadoc", manager.getArtifactHandler("jar"));
254        }
255    }
