023import java.net.MalformedURLException;
024import java.net.URL;
025import java.util.Arrays;
026import lombok.Data;
027import lombok.Getter;
028import org.apache.maven.artifact.Artifact;
029import org.apache.maven.shared.artifact.filter.StrictPatternIncludesArtifactFilter;
030
031/**
032 * {@code <link/>} parameter.
033 *
034 * {@bean.info}
035 *
036 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
037 * @version $Revision: 8343 $
038 */
039@Data
040public class Link {
041    private String artifact = null;
042    private URL url = null;
043    @Getter(lazy = true)
044    private final StrictPatternIncludesArtifactFilter filter =
045        new StrictPatternIncludesArtifactFilter(Arrays.asList(artifact.split("[,\\p{Space}]+")));
046
047    /**
048     * See {@link StrictPatternIncludesArtifactFilter#include(Artifact)}.
049     */
050    public boolean include(Artifact artifact) {
051        return getFilter().include(artifact);
052    }
053
054    /**
055     * Method to return a {@link URL} substituting <code>{g}</code>,
056     * <code>{a}</code>, and <code>{v}</code> with
057     * {@link Artifact#getGroupId()}, {@link Artifact#getArtifactId()}, and
058     * {@link Artifact#getVersion()}, respectively.
059     *
060     * @param   artifact        The {@link Artifact}.
061     *
062     * @return  The {@link URL} after substitution.
063     */
064    public URL getUrl(Artifact artifact) {
065        URL url = getUrl();
066
067        if (url != null && artifact != null) {
068            try {
069                url =
070                    new URL(url.toString()
071                            .replaceAll("(?i)[{]g[}]", artifact.getGroupId())
072                            .replaceAll("(?i)[{]a[}]", artifact.getArtifactId())
073                            .replaceAll("(?i)[{]v[}]", artifact.getVersion()));
074            } catch (MalformedURLException exception) {
075                throw new IllegalArgumentException(exception);
076            }
077        }
078
079        return url;
080    }
081}
