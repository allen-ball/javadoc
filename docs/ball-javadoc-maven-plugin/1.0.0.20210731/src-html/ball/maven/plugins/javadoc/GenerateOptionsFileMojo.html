076/**
077 * {@link org.apache.maven.plugin.Mojo} to generate javadoc options file for
078 * {@code maven-javadoc-plugin}.
079 *
080 * {@maven.plugin.fields}
081 *
082 * {@bean.info}
083 *
084 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
085 * @version $Revision: 8343 $
086 */
087@Mojo(name = "generate-options-file",
088      requiresDependencyResolution = RUNTIME,
089      defaultPhase = GENERATE_SOURCES, requiresProject = true)
090@NoArgsConstructor @ToString @Slf4j
091public class GenerateOptionsFileMojo extends AbstractJavadocMojo {
092    private static final Pattern JAR_ENTRY_PATTERN = Pattern.compile("^(package|element)[^-]*-list$");
093
094    @Parameter(property = "outputDirectory", defaultValue = "${project.build.directory}/javadoc-options")
095    private File outputDirectory = null;
096
097    @Parameter(required = false)
098    private Link[] links = new Link[] { };
099
100    @Parameter(required = false)
101    private Offlinelink[] offlinelinks = new Offlinelink[] { };
102
103    @Inject private MavenProject project = null;
104    @Inject private MavenSession session = null;
105    @Inject private ArtifactHandlerManager manager = null;
106    @Inject private RepositorySystem system = null;
107    @Inject private ArtifactResolver resolver = null;
108
109    @Override
110    public void execute() throws MojoExecutionException, MojoFailureException {
111        super.execute();
112
113        try {
114            if (! isSkip()) {
115                Set<URL> set = getLinkSet();
116                Map<URL,List<Artifact>> map =
117                    getResolvedOfflinelinkMap().entrySet().stream()
118                    .collect(groupingBy(Map.Entry::getValue,
119                                        mapping(Map.Entry::getKey, toList())));
120
121                generateOutput(set, map);
122            } else {
123                log.info("Skipping javadoc options file generation.");
124            }
125        } catch (Throwable throwable) {
126            log.error("{}", throwable.getMessage(), throwable);
127
128            if (throwable instanceof MojoExecutionException) {
129                throw (MojoExecutionException) throwable;
130            } else if (throwable instanceof MojoFailureException) {
131                throw (MojoFailureException) throwable;
132            } else {
133                throw new MojoExecutionException(throwable.getMessage(), throwable);
134            }
135        }
136    }
137
138    private Set<URL> getLinkSet() {
139        Set<URL> set = new LinkedHashSet<>();
140
141        for (Link link : links) {
142            List<URL> list =
143                project.getArtifacts().stream()
144                .filter(link::include)
145                .map(link::getUrl)
146                .collect(toList());
147
148            set.addAll(list);
149        }
150
151        return set;
152    }
153
154    private Map<Artifact,URL> getResolvedOfflinelinkMap() {
155        Map<Artifact,URL> map = new TreeMap<>(Comparator.comparing(ArtifactUtils::versionlessKey));
156        Set<Artifact> javadocs = getJavadocJarDependencyManagementSet();
157        ProjectBuildingRequest request = getProjectBuildingRequest();
158
159        for (Offlinelink offlinelink : offlinelinks) {
160            Set<Artifact> set =
161                javadocs.stream()
162                .filter(offlinelink::include)
163                .map(JavadocArtifact::new)
164                .collect(toCollection(LinkedHashSet::new));
165
166            for (Artifact artifact : set) {
167                URL url = map.get(artifact);
168
169                if (url == null) {
170                    log.info("Resolving {}...", artifact);
171
172                    try {
173                        map.put(resolver.resolveArtifact(request, artifact).getArtifact(), offlinelink.getUrl(artifact));
174                    } catch (Exception exception) {
175                        log.warn("{}: {}", artifact, exception.getMessage());
176                        log.debug("{}", exception);
177                    }
178                } else {
179                    if (! Objects.equals(url, offlinelink.getUrl(artifact))) {
180                        log.warn("{} matches {} but was previously resolved with {}",
181                                 artifact, offlinelink, url);
182                    }
183                }
184            }
185        }
186
187        return map;
188    }
189
190    private Set<Artifact> getJavadocJarDependencyManagementSet() {
191        Set<Artifact> set =
192            Stream.of(project.getDependencies(),
193                      project.getDependencyManagement().getDependencies())
194            .filter(Objects::nonNull)
195            .flatMap(Collection::stream)
196            .filter(t -> isBlank(t.getClassifier()))
197            .filter(t -> Objects.equals(t.getType(), "jar"))
198            .filter(t -> (! isBlank(t.getVersion())))
199            .map(t -> new JavadocArtifact(t.getGroupId(), t.getArtifactId(), t.getVersion()))
200            .collect(toCollection(LinkedHashSet::new));
201
202        return set;
203    }
204
205    private ProjectBuildingRequest getProjectBuildingRequest() {
206        List<ArtifactRepository> repoList = getRemoteRepositories();
207        Settings settings = session.getSettings();
208
209        system.injectMirror(repoList, settings.getMirrors());
210        system.injectProxy(repoList, settings.getProxies());
211        system.injectAuthentication(repoList, settings.getServers());
212
213        ProjectBuildingRequest request = new DefaultProjectBuildingRequest(session.getProjectBuildingRequest());
214
215        request.setRemoteRepositories(repoList);
216
217        return request;
218    }
219
220    private void generateOutput(Set<URL> set, Map<URL,List<Artifact>> map) throws IOException {
221        Path parent = outputDirectory.toPath();
222
223        Files.createDirectories(parent);
224
225        Path options = parent.resolve("options");
226
227        try (PrintWriter out = new PrintWriter(Files.newBufferedWriter(options, CREATE, WRITE, TRUNCATE_EXISTING))) {
228            for (URL url : set) {
229                out.println("-link");
230                out.println(url);
231            }
232
233            for (Map.Entry<URL,List<Artifact>> entry : map.entrySet()) {
234                List<Artifact> artifacts = entry.getValue();
235
236                for (Artifact artifact : artifacts) {
237                    Path location = parent.resolve(ArtifactUtils.versionlessKey(artifact));
238
239                    Files.createDirectories(location);
240
241                    try (JarFile jar = new JarFile(artifact.getFile())) {
242                        List<JarEntry> entries =
243                            jar.stream()
244                            .filter(t -> JAR_ENTRY_PATTERN.matcher(t.getName()).matches())
245                            .collect(toList());
246
247                        if (! entries.isEmpty()) {
248                            for (JarEntry jarEntry : entries) {
249                                try (InputStream in = jar.getInputStream(jarEntry)) {
250                                    Files.copy(in, location.resolve(jarEntry.getName()), REPLACE_EXISTING);
251                                }
252                            }
253
254                            Path packageList = location.resolve("package-list");
255                            Path elementList = location.resolve("element-list");
256
257                            if (! Files.exists(packageList)) {
258                                Files.copy(elementList, packageList);
259                            } else if (! Files.exists(elementList)) {
260                                Files.copy(packageList, elementList);
261                            }
262
263                            out.println("-linkoffline");
264                            out.println(entry.getKey());
265                            out.println(location);
266                        } else {
267                            log.warn("{}: Location directory is empty; skipping...", location);
268                        }
269                    }
270                }
271            }
272        }
273    }
274
275    private class JavadocArtifact extends DefaultArtifact {
276        public JavadocArtifact(String gav) { this(gav.split("[:]")); }
277
278        public JavadocArtifact(String... gav) {
279            super(gav.length > 0 ? gav[0] : EMPTY,
280                  gav.length > 1 ? gav[1] : EMPTY,
281                  gav.length > 2 ? gav[gav.length - 1] : EMPTY,
282                  EMPTY, "jar", "javadoc", manager.getArtifactHandler("jar"));
283        }
284
285        public JavadocArtifact(Artifact artifact) {
286            this(artifact.getGroupId(), artifact.getArtifactId(), artifact.getVersion());
287        }
288    }
