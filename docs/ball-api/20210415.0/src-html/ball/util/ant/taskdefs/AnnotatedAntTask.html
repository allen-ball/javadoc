023import org.apache.tools.ant.BuildException;
024import org.apache.tools.ant.Task;
025
026/**
027 * Interface indicating {@link Task} is annotated with {@link AntTask} and
028 * related annotations.
029 *
030 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
031 * @version $Revision: 7215 $
032 */
033public interface AnnotatedAntTask extends AntTaskLogMethods {
034
035    /**
036     * Method to get {@link AntTask#value()}.
037     *
038     * @return  {@link AntTask#value()} if {@link.this} {@link Task} is
039     *          annotated; {@code null} otherwise.
040     */
041    default String getAntTaskName() {
042        AntTask annotation = getClass().getAnnotation(AntTask.class);
043
044        return (annotation != null) ? annotation.value() : null;
045    }
046
047    /**
048     * Default implementation for {@link Task} subclasses.  Check attributes
049     * annotated with {@link AntTaskAttributeConstraint}.
050     * See {@link AnnotatedAntTaskConfigurationChecker}.
051     *
052     * @throws  BuildException  If a
053     *                          {@link AntTaskAttributeConstraint.Checker}
054     *                          fails.
055     */
056    default void execute() throws BuildException {
057        new AnnotatedAntTaskConfigurationChecker((Task) this).checkErrors();
058    }
059}
