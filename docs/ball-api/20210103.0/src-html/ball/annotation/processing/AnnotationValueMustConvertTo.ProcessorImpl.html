023import ball.annotation.ServiceProviderFor;
024import java.lang.annotation.Documented;
025import java.lang.annotation.Retention;
026import java.lang.annotation.Target;
027import javax.annotation.processing.Processor;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030
031import static java.lang.annotation.ElementType.ANNOTATION_TYPE;
032import static java.lang.annotation.RetentionPolicy.RUNTIME;
033
034/**
035 * {@link java.lang.annotation.Annotation} to test an
036 * {@link java.lang.annotation.Annotation} can be converted to a specified
037 * {@link Class}.
038 *
039 * @see AnnotatedProcessor
040 *
041 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
042 * @version $Revision: 7215 $
043 */
044@Documented
045@Retention(RUNTIME)
046@Target(ANNOTATION_TYPE)
047public @interface AnnotationValueMustConvertTo {
048    Class<?> value();
049    String method() default "";
050    String name() default "value";
051
052    /**
053     * {@link Processor} implementation.
054     */
055    @ServiceProviderFor({ Processor.class })
056    @For({ AnnotationValueMustConvertTo.class })
057    @NoArgsConstructor @ToString
058    public static class ProcessorImpl extends AnnotatedProcessor {
059    }
060}
