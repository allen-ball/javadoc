023import lombok.NoArgsConstructor;
024import lombok.ToString;
025import lombok.extern.log4j.Log4j2;
026import org.webjars.WebJarAssetLocator;
027
028/**
029 * {@link WebJars} Thymeleaf dialect implementation.
030 *
031 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
032 * @version $Revision: 6679 $
033 */
034@NoArgsConstructor @ToString @Log4j2
035public class WebJars {
036    private WebJarAssetLocator locator = new WebJarAssetLocator();
037
038    /**
039     * Method to convert a WebJar resource path to its corresponding CDN
040     * (URI) path.  Typical Thymeleaf usage:
041     * {@code @{${#webjars.cdn(#request.scheme, path)}}}.
042     *
043     * @param   scheme          The {@code URI} scheme.
044     * @param   path            The (possibly partial) local path,
045     *
046     * @return  The CDN URI if one may be constructed; {@code path}
047     *          otherwise.
048     */
049    public String cdn(String scheme, String path) {
050        String uri = null;
051        String groupId = (path != null) ? groupId(path) : null;
052
053        if (groupId != null) {
054            uri =
055                String.format("%s://cdn.jsdelivr.net/webjars/%s/%s",
056                              (scheme != null) ? scheme : "http",
057                              groupId, getLocalPath(path));
058        }
059
060        return (uri != null) ? uri : path;
061    }
062
063    private String getLocalPath(String path) {
064        return locator.getFullPath(path).replaceAll("^.*/webjars/", "");
065    }
066
067    private String groupId(String path) {
068        return locator.groupId(locator.getFullPath(path));
069    }
070}
