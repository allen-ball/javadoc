023import java.text.MessageFormat;
024import java.util.Map;
025import java.util.regex.Pattern;
026
027/**
028 * {@link MessageFormat} subclass that provides {@link #format(String,Map)}
029 * method that translates the pattern {@link String} and named to parameters
030 * in the argument {@link Map} to positional parameters for a call to
031 * {@link #format(String,Object...)}.
032 *
033 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
034 * @version $Revision: 5285 $
035 */
036public abstract class ParameterizedMessageFormat extends MessageFormat {
037    private static final long serialVersionUID = -5750488498631645637L;
038
039    private ParameterizedMessageFormat() { super(null); }
040
041    /**
042     * Method that translates the pattern {@link String} and named to
043     * parameters in the argument {@link Map} to positional parameters for a
044     * call to {@link #format(String,Object...)}.
045     *
046     * @param   pattern         The pattern {@link String}.
047     * @param   map             The parameter {@link Map}.
048     *
049     * @return  The formatted {@link String}.
050     */
051    public static String format(String pattern, Map<String,?> map) {
052        Object[] values = new Object[map.size()];
053        int i = 0;
054
055        for (Map.Entry<String,?> entry : map.entrySet()) {
056            pattern =
057                pattern.replaceAll(Pattern.quote("{" + entry.getKey() + "}"),
058                                   "{" + i + "}");
059            values[i] = entry.getValue();
060
061            i += 1;
062        }
063
064        return format(pattern, values);
065    }
066}
