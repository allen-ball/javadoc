049/**
050 * Class suitable for mapping polymorphic subtypes.  Subclasses can specify
051 * the mapping in a {@link java.util.Properties} resource which will be
052 * automatically loaded on instantiation.
053 *
054 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
055 * @version $Revision: 5285 $
056 */
057public abstract class PolymorphicTypeMap extends TreeMap<Class<?>,Class<?>[]> {
058    private static final long serialVersionUID = -4465979259676184876L;
059
060    /**
061     * Sole constructor.
062     */
063    protected PolymorphicTypeMap() {
064        super(comparing(Class::getName));
065
066        try {
067            PropertiesImpl properties = new PropertiesImpl();
068            String name = getClass().getSimpleName() + ".properties";
069            InputStream in = getClass().getResourceAsStream(name);
070
071            if (in != null) {
072                try {
073                    properties.load(in);
074                } finally {
075                    in.close();
076                }
077            }
078
079            ClassLoader loader = getClass().getClassLoader();
080            Package pkg = getClass().getPackage();
081
082            for (String key : properties.stringPropertyNames()) {
083                TreeSet<Class<?>> value =
084                    new TreeSet<>(comparing(Class::getName));
085
086                for (String substring :
087                         properties.getProperty(key)
088                         .split("[,\\p{Space}]+")) {
089                    substring = substring.trim();
090
091                    if (! StringUtils.isEmpty(substring)) {
092                        value.add(getClassFor(loader, pkg, substring));
093                    }
094                }
095
096                put(getClassFor(loader, pkg, key),
097                    value.toArray(new Class<?>[] { }));
098            }
099        } catch (Exception exception) {
100            throw new ExceptionInInitializerError(exception);
101        }
102    }
103
104    private Class<?> getClassFor(ClassLoader loader,
105                                 Package pkg, String name) throws Exception {
106        Class<?> cls = null;
107
108        try {
109            cls = Class.forName(pkg.getName() + "." + name, true, loader);
110        } catch (Exception exception) {
111            cls = Class.forName(name, true, loader);
112        }
113
114        return cls;
115    }
116
117    /**
118     * Method to get a {@link BeanDeserializerModifier} for this
119     * {@link PolymorphicTypeMap}.
120     *
121     * @return  The {@link BeanDeserializerModifier}.
122     */
123    public BeanDeserializerModifier getBeanDeserializerModifier() {
124        return new DeserializerModifier();
125    }
126
127    /**
128     * Callback from {@link JsonDeserializer} implementation to allow
129     * subclass implementations to initialize the resulting bean with the
130     * parse-tree ({@link JsonNode}).
131     *
132     * @param   object          The {@link Object} to initialize.
133     * @param   codec           The {@link ObjectCodec}.
134     * @param   node            The {@link JsonNode}.
135     *
136     * @throws  IOException     If there is a problem initializing the
137     *                          {@link JSONBean}.
138     */
139    protected void initialize(Object object,
140                              ObjectCodec codec,
141                              JsonNode node) throws IOException {
142    }
143
144    @Override
145    public Class<?>[] put(Class<?> key, Class<?>[] value) {
146        for (Class<?> subtype : value) {
147            if (! key.isAssignableFrom(subtype)) {
148                throw new IllegalArgumentException(subtype.getName()
149                                                   + " is not a subclass of "
150                                                   + key.getName());
151            }
152        }
153
154        return super.put(key, value);
155    }
156
157    private class DeserializerModifier extends BeanDeserializerModifier {
158        public DeserializerModifier() { super(); }
159
160        @Override
161        public JsonDeserializer<?> modifyDeserializer(DeserializationConfig config,
162                                                      BeanDescription description,
163                                                      JsonDeserializer<?> deserializer) {
164            Class<?> key = description.getBeanClass();
165
166            if (containsKey(key)) {
167                deserializer =
168                    (deserializer instanceof BeanDeserializer)
169                        ? new BeanDeserializerImpl((BeanDeserializer) deserializer,
170                                                   key, get(key))
171                        : new AbstractDeserializerImpl(description,
172                                                       key, get(key));
173            }
174
175            return deserializer;
176        }
177
178        @Override
179        public String toString() { return super.toString(); }
180
181        private class BeanInfoList extends ArrayList<BeanInfo> {
182            private static final long serialVersionUID = 1008165295877024242L;
183
184            private final Class<?> supertype;
185
186            public BeanInfoList(Class<?> supertype, Class<?>... subtypes) {
187                super(subtypes.length);
188
189                this.supertype = requireNonNull(supertype, "supertype");
190
191                try {
192                    for (Class<?> subtype : subtypes) {
193                        if (supertype.isAssignableFrom(subtype)) {
194                            add(getBeanInfo(subtype, supertype));
195                        } else {
196                            throw new IllegalArgumentException(subtype.getName()
197                                                               + " is not a subclass of "
198                                                               + supertype.getName());
199                        }
200                    }
201                } catch (Exception exception) {
202                    throw new ExceptionInInitializerError(exception);
203                }
204            }
205
206            public Class<?> supertype() { return supertype; }
207
208            public Class<?> subtypeFor(JsonNode node) {
209                Class<?> subtype = null;
210
211                for (BeanInfo info : this) {
212                    if (hasAll(node, info.getPropertyDescriptors())) {
213                        subtype =
214                            info.getBeanDescriptor()
215                            .getBeanClass()
216                            .asSubclass(supertype);
217                        break;
218                    }
219                }
220
221                return subtype;
222            }
223
224            private boolean hasAll(JsonNode node,
225                                   PropertyDescriptor... properties) {
226                boolean hasAll = true;
227
228                for (PropertyDescriptor property : properties) {
229                    hasAll &= node.has(property.getName());
230
231                    if (! hasAll) {
232                        break;
233                    }
234                }
235
236                return hasAll;
237            }
238        }
239
240        private class JsonParserImpl extends TreeTraversingParser {
241            protected JsonNode node = null;
242
243            public JsonParserImpl(JsonParser parser) throws IOException {
244                this((JsonNode) parser.readValueAsTree(), parser.getCodec());
245            }
246
247            private JsonParserImpl(JsonNode node, ObjectCodec codec) {
248                super(node, codec);
249
250                this.node = requireNonNull(node, "node");
251            }
252
253            @Override
254            public String toString() { return super.toString(); }
255        }
256
257        private class BeanDeserializerImpl extends BeanDeserializer {
258            private static final long serialVersionUID = 3127841504650850219L;
259
260            private final BeanInfoList list;
261
262            public BeanDeserializerImpl(BeanDeserializer deserializer,
263                                        Class<?> supertype,
264                                        Class<?>... subtypes) {
265                super(deserializer);
266
267                list = new BeanInfoList(supertype, subtypes);
268            }
269
270            @Override
271            public Class<?> handledType() { return list.supertype(); }
272
273            @Override
274            public Object deserialize(JsonParser parser,
275                                      DeserializationContext context) throws IOException {
276                Object object = null;
277                ObjectCodec codec = parser.getCodec();
278                JsonNode node = null;
279
280                if (parser instanceof JsonParserImpl) {
281                    node = ((JsonParserImpl) parser).node;
282
283                    if (node != null) {
284                        ((JsonParserImpl) parser).node = null;
285                    }
286                }
287
288                if (node != null) {
289                    Class<?> subtype = list.subtypeFor(node);
290
291                    if (subtype != null) {
292                        object = codec.readValue(parser, subtype);
293                    } else {
294                        object = super.deserialize(parser, context);
295                    }
296
297                    initialize(list.supertype().cast(object), codec, node);
298                } else {
299                    object =
300                        codec.readValue(new JsonParserImpl(parser),
301                                        list.supertype());
302                }
303
304                return object;
305            }
306
307            @Override
308            public String toString() { return super.toString(); }
309        }

363        }
