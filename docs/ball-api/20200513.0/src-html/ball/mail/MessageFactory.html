023import ball.util.Factory;
024import java.io.File;
025import java.io.IOException;
026import java.io.InputStream;
027import java.lang.reflect.InvocationTargetException;
028import java.util.Properties;
029import javax.activation.DataSource;
030import javax.activation.FileDataSource;
031import javax.mail.Message;
032import javax.mail.MessagingException;
033import javax.mail.Session;
034
035import static java.util.Objects.requireNonNull;
036
037/**
038 * {@link Factory} implementation to produce {@link Message}
039 * {@link Object}s.  Manages the {@link Session}.
040 *
041 * @param       <T>             The type of {@link Message} this
042 *                              {@link Factory} will produce.
043 *
044 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
045 * @version $Revision: 5285 $
046 */
047public class MessageFactory<T extends Message> extends Factory<T> {
048    private static final long serialVersionUID = 4173116479658297998L;
049
050    /** @serial */ private final Session session;
051
052    /**
053     * Construct a {@link Message} {@link Factory} with the argument
054     * {@link Session}.
055     *
056     * @param   type            The type ({@link Class}) of {@link Message}
057     *                          this {@link Factory} will produce.
058     * @param   session         The {@link Session}.
059     *
060     * @throws  NullPointerException
061     *                          If any of the arguments are {@code null}.
062     */
063    public MessageFactory(Class<? extends T> type, Session session) {
064        super(type);
065
066        this.session = requireNonNull(session, "session");
067    }
068
069    /**
070     * Construct a {@link Message} {@link Factory}.
071     *
072     * @param   type            The type ({@link Class}) of {@link Message}
073     *                          this {@link Factory} will produce.
074     * @param   properties      The {@link Properties} to use to create the
075     *                          {@link Session}.
076     *
077     * @throws  NullPointerException
078     *                          If any of the arguments are {@code null}.
079     */
080    public MessageFactory(Class<? extends T> type, Properties properties) {
081        this(type, Session.getInstance(properties));
082    }
083
084    /**
085     * Construct a {@link Message} {@link Factory}.
086     *
087     * @param   type            The type ({@link Class}) of {@link Message}
088     *                          this {@link Factory} will produce.
089     *
090     * @throws  NullPointerException
091     *                          If {@code type} is {@code null}.
092     */
093    public MessageFactory(Class<? extends T> type) {
094        this(type, System.getProperties());
095    }
096
097    /**
098     * Method to get the {@link Session} used to create {@link Message}s.
099     *
100     * @return  The {@link Session}.
101     */
102    public Session getSession() { return session; }
103
104    /**
105     * Factory method to load a {@link Message} from a {@link File}.
106     *
107     * @param   file            The {@link File} containing the message.
108     *
109     * @return  The {@link Message} instance.
110     *
111     * @throws  IOException     If the message cannot be read from the
112     *                          {@link File} for any reason.
113     * @throws  MessagingException
114     *                          If the message cannot be constructed from
115     *                          the {@link File} for any reason.
116     * @throws  NullPointerException
117     *                          If {@code file} is {@code null}.
118     */
119    public T load(File file) throws IOException, MessagingException {
120        T message = null;
121        Object[] arguments = new Object[] { getSession(), file };
122
123        if (hasFactoryMethodFor(typesOf(arguments))) {
124            try {
125                message = getInstance(arguments);
126            } catch (Exception exception) {
127                rethrow(exception);
128            }
129        } else {
130            message = load(new FileDataSource(file));
131        }
132
133        return message;
134    }
135
136    /**
137     * Factory method to load a {@link Message} from a {@link DataSource}.
138     *
139     * @param   ds              The {@link DataSource} containing the message.
140     *
141     * @return  The {@link Message} instance.
142     *
143     * @throws  IOException     If the message cannot be read from the
144     *                          {@link DataSource} for any reason.
145     * @throws  MessagingException
146     *                          If the message cannot be constructed from
147     *                          the {@link DataSource} for any reason.
148     * @throws  NullPointerException
149     *                          If {@code ds} is {@code null}.
150     */
151    public T load(DataSource ds) throws IOException, MessagingException {
152        T message = null;
153        Object[] arguments = new Object[] { getSession(), ds };
154
155        if (hasFactoryMethodFor(typesOf(arguments))) {
156            try {
157                message = getInstance(arguments);
158            } catch (Exception exception) {
159                rethrow(exception);
160            }
161        } else {
162            try (InputStream in = ds.getInputStream();) {
163                message = load(in);
164            }
165        }
166
167        return message;
168    }
169
170    /**
171     * Factory method to load a {@link Message} from an {@link InputStream}.
172     *
173     * @param   in              The {@link InputStream} containing the message.
174     *
175     * @return  The {@link Message} instance.
176     *
177     * @throws  IOException     If the message cannot be read from the
178     *                          {@link InputStream} for any reason.
179     * @throws  MessagingException
180     *                          If the message cannot be constructed from
181     *                          the {@link InputStream} for any reason.
182     * @throws  NullPointerException
183     *                          If {@code in} is {@code null}.
184     */
185    public T load(InputStream in) throws IOException, MessagingException {
186        T message = null;
187
188        try {
189            message = getInstance(getSession(), in);
190        } catch (Exception exception) {
191            rethrow(exception);
192        }
193
194        return message;
195    }
196
197    private void rethrow(Throwable throwable) throws IOException,
198                                                     MessagingException {
199        if (throwable instanceof IOException) {
200            throw (IOException) throwable;
201        } else if (throwable instanceof MessagingException) {
202            throw (MessagingException) throwable;
203        } else if (throwable instanceof InvocationTargetException) {
204            if (throwable.getCause() != null) {
205                rethrow(throwable.getCause());
206            } else {
207                throw new MessagingException(throwable.getMessage(),
208                                             (Exception) throwable);
209            }
210        } else if (throwable instanceof RuntimeException) {
211            throw (RuntimeException) throwable;
212        } else if (throwable instanceof Exception) {
213            throw new MessagingException(throwable.getMessage(),
214                                         (Exception) throwable);
215        } else if (throwable instanceof Error) {
216            throw (Error) throwable;
217        } else {
218            throw new IllegalStateException(throwable);
219        }
220    }
221
222    @Override
223    public String toString() { return super.toString(); }
224}
