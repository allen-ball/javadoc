026import java.util.Objects;

040/**
041 * QMail {@link.uri http://www.qmail.org/man/man5/maildir.html Maildir}
042 * {@link Folder} implementation.
043 *
044 * {@bean.info}
045 *
046 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
047 * @version $Revision: 5285 $
048 */
049public class MAILDIRFolder extends Folder
050                           implements Comparable<MAILDIRFolder> {
051    private static final Comparator<? super MAILDIRFolder> COMPARATOR =
052        Comparator.comparing(MAILDIRFolder::getPath);
053
054    /**
055     * Name of subdirectory of {@link MAILDIRFolder} directory.
056     */
057    protected static final String
058        CUR = "cur",
059        NEW = "new",
060        TMP = "tmp";
061
062    /**
063     * {@link Pattern} matching {@link MAILDIRMimeMessage} {@link File}
064     * names.
065     */
066    public static final
067        Pattern PATTERN = Pattern.compile("([^:]+)(:[12],(.*))?");
068
069    /**
070     * See {@link #PATTERN}.
071     */
072    public static final int
073        PATTERN_UNIQ_GROUP = 1,
074        PATTERN_INFO_GROUP = 3;
075
076    private final MAILDIRPath path;
077    private transient boolean isOpen = false;
078    private transient long lastModified = -1;
079    private transient MAILDIRMimeMessage[] messages =
080        new MAILDIRMimeMessage[] { };
081
082    /**
083     * Sole constructor.
084     *
085     * @param   store           The {@link javax.mail.Store} containing the
086     *                          {@link MAILDIRFolder}.
087     * @param   path            The {@link MAILDIRPath} specifying the
088     *                          {@link MAILDIRFolder}.
089     */
090    protected MAILDIRFolder(MAILDIRStore store, MAILDIRPath path) {
091        super(store);
092
093        this.path = requireNonNull(path, "path");
094    }
095
096    @Override
097    public MAILDIRStore getStore() { return (MAILDIRStore) super.getStore(); }
098
099    /**
100     * Method to get this {@link MAILDIRFolder}'s underlying
101     * {@link MAILDIRPath}.
102     *
103     * @return  The underlying {@link MAILDIRPath}.
104     */
105    protected MAILDIRPath getPath() { return path; }
106
107    /**
108     * Method to get the underlying subdirectory ({@link File}) of this
109     * {@link MAILDIRFolder}.
110     *
111     * @param   names           Optional component names of a subdirectory.
112     *
113     * @return  The underlying directory or subdirectory ({@link File}).
114     */
115    protected File getFile(String... names) {
116        File file = new File(getStore().getFile(), getPath().toFilename());
117
118        for (String name : names) {
119            file = new File(file, name);
120        }
121
122        return file;
123    }
124
125    /**
126     * Method to determine if the argumnt {@link MAILDIRPath} matches the
127     * argument {@code pattern} (relative to the this {@link Folder}).
128     *
129     * @param   pattern         The {@code pattern} to match.
130     * @param   path            The {@link MAILDIRPath} to test.
131     *
132     * @return  {@code true} if the {@link MAILDIRPath} matches;
133     *          {@code false} otherwise.
134     */
135    protected boolean matches(String pattern, MAILDIRPath path) {
136        return pattern(pattern).matcher(path.toString()).matches();
137    }
138
139    private Pattern pattern(String pattern) {
140        String name = getFullName();
141
142        if (! name.equals(EMPTY)) {
143            pattern = name + String.valueOf(MAILDIRPath.SEPARATOR) + pattern;
144        }
145
146        pattern = pattern.replaceAll(Pattern.quote("."), "[.]");
147        pattern = pattern.replaceAll(Pattern.quote("*"), ".*");
148        pattern = pattern.replaceAll(Pattern.quote("%"), "[^.]*");
149
150        return Pattern.compile(pattern);
151    }
152
153    @Override
154    public String getName() { return getPath().getName(); }
155
156    @Override
157    public String getFullName() { return getPath().toString(); }
158
159    @Override
160    public MAILDIRFolder getParent() throws MessagingException {
161        return getStore().getFolder(getPath().getParentPath());
162    }
163
164    @Override
165    public boolean exists() { return getFile().isFile(); }
166
167    @Override
168    public MAILDIRFolder[] list(String pattern) throws MessagingException {
169        return getStore().list(this, pattern);
170    }
171
172    @Override
173    public char getSeparator() { return MAILDIRPath.SEPARATOR; }
174
175    @Override
176    public int getType() { return (HOLDS_FOLDERS | HOLDS_MESSAGES); }
177
178    @Override
179    public boolean create(int type) throws MessagingException {
180        MAILDIRFolder parent = getParent();
181
182        if (parent != null) {
183            if (! parent.exists()) {
184                throw new MessagingException("Parent folder does not exist");
185            } else {
186                throw new ReadOnlyFolderException(parent);
187            }
188        }
189
190        boolean created = false;
191
192        if (! exists()) {
193            getFile(CUR).mkdirs();
194            getFile(NEW).mkdirs();
195            getFile(TMP).mkdirs();
196
197            created = exists();
198        } else {
199            created = false;
200        }
201
202        if (created) {
203            notifyFolderListeners(FolderEvent.CREATED);
204        }
205
206        return created;
207    }
208
209    @Override
210    public boolean hasNewMessages() throws MessagingException {
211        return getNewMessageCount() > 0;
212    }
213
214    @Override
215    public MAILDIRFolder getFolder(String name) throws MessagingException {
216        return new MAILDIRFolder(getStore(), new MAILDIRPath(getPath(), name));
217    }
218
219    @Override
220    public boolean delete(boolean recurse) throws MessagingException {
221        throw new ReadOnlyFolderException(this);
222    }
223
224    @Override
225    public boolean renameTo(Folder folder) throws MessagingException {
226        throw new ReadOnlyFolderException(this);
227    }
228
229    @Override
230    public void open(int mode) throws MessagingException {
231        isOpen = exists();
232
233        if (! isOpen()) {
234            throw new FolderNotFoundException(this);
235        }
236
237        notifyConnectionListeners(ConnectionEvent.OPENED);
238    }
239
240    @Override
241    public void close(boolean expunge) throws MessagingException {
242        if (expunge) {
243            throw new ReadOnlyFolderException(this);
244        }
245
246        isOpen = false;
247
248        notifyConnectionListeners(ConnectionEvent.CLOSED);
249    }

252    public boolean isOpen() { return isOpen; }

255    public int getMode() { return READ_ONLY; }

257    @Override
258    public MAILDIRFlags getPermanentFlags() { return new MAILDIRFlags(); }

261    public int getMessageCount() throws MessagingException {
262        return getMessages().length;
263    }

265    @Override
266    public MAILDIRMimeMessage getMessage(int index) throws MessagingException {
267        return getMessages()[index - 1];
268    }
269
270    @Override
271    public MAILDIRMimeMessage[] getMessages() throws MessagingException {
272        if (! exists()) {
273            throw new FolderNotFoundException(this);
274        }
275
276        try {
277            if (lastModified < Math.max(getFile(CUR).lastModified(),
278                                        getFile(NEW).lastModified())) {
279                lastModified =
280                    Math.max(getFile(CUR).lastModified(),
281                             getFile(NEW).lastModified());
282
283                File[] files =
284                    Stream.of(getFile(CUR), getFile(NEW))
285                    .map(t -> t.listFiles())
286                    .filter(t -> t != null)
287                    .flatMap(t -> Stream.of(t))
288                    .filter(t -> t.isFile())
289                    .filter(t -> PATTERN.matcher(t.getName()).matches())
290                    .sorted(Comparator.comparingLong(File::lastModified))
291                    .toArray(File[]::new);
292
293                messages = new MAILDIRMimeMessage[files.length];
294
295                for (int i = 0; i < messages.length; i += 1) {
296                    messages[i] =
297                        new MAILDIRMimeMessage(this, i + 1, files[i]);
298                }
299            }
300        } catch (IOException exception) {
301            throw new MessagingException(exception.getMessage(), exception);
302        }
303
304        return messages;
305    }

310    }

315    }

319        return Objects.compare(this, that, COMPARATOR);
320    }

322    @Override
323    public boolean equals(Object that) {
324        return ((that instanceof MAILDIRFolder)
325                    ? (this.compareTo((MAILDIRFolder) that) == 0)
326                    : super.equals(that));
327    }
328
329    @Override
330    public int hashCode() { return Objects.hash(getPath()); }
