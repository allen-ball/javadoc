059/**
060 * {@link Processor} implementation to scan {@link ball.annotation.Manifest}
061 * {@link java.lang.annotation.Annotation}s and generate a
062 * {@link java.util.jar.Manifest META-INF/MANIFEST.MF}.
063 *
064 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
065 * @version $Revision: 5904 $
066 */
067@ServiceProviderFor({ Processor.class })
068@For({
069        Attribute.class, MainClass.class, Section.class,
070            JavaBean.class, DependsOn.class, DesignTimeOnly.class
071     })
072@NoArgsConstructor @ToString
073public class ManifestProcessor extends AnnotatedProcessor
074                               implements ClassFileProcessor {
075    private static final String META_INF = "META-INF";
076    private static final String MANIFEST_MF = "MANIFEST.MF";
077    private static final String _CLASS = ".class";
078
079    private static abstract class PROTOTYPE {
080        public static void main(String[] argv) { }
081    }
082
083    private static final Method PROTOTYPE =
084        PROTOTYPE.class.getDeclaredMethods()[0];
085
086    private ManifestImpl manifest = null;
087    private HashSet<Element> processed = new HashSet<>();
088
089    @Override
090    public void init(ProcessingEnvironment processingEnv) {
091        super.init(processingEnv);
092
093        try {
094            /*
095             * Load any partially generated files.
096             */
097        } catch (Exception exception) {
098            print(ERROR, exception);
099        }
100    }
101
102    @Override
103    public boolean process(Set<? extends TypeElement> annotations,
104                           RoundEnvironment roundEnv) {
105        boolean result = true;
106
107        try {
108            String path = META_INF + "/" + MANIFEST_MF;
109
110            if (manifest == null) {
111                manifest = new ManifestImpl();
112
113                FileObject file = filer.getResource(CLASS_OUTPUT, EMPTY, path);
114
115                if (file.getLastModified() != 0) {
116                    try (InputStream in = file.openInputStream()) {
117                        manifest.read(in);
118                    } catch (Exception exception) {
119                    }
120                }
121
122                manifest.init();
123            }
124
125            if (! roundEnv.errorRaised()) {
126                result &= super.process(annotations, roundEnv);
127
128                if (roundEnv.processingOver()) {
129                    FileObject file =
130                        filer.createResource(CLASS_OUTPUT, EMPTY, path);
131
132                    try (OutputStream out = file.openOutputStream()) {
133                        manifest.write(out);
134                    }
135                }
136            }
137        } catch (Exception exception) {
138            print(ERROR, exception);
139        }
140
141        return result;
142    }
143
144    @Override
145    protected void process(RoundEnvironment roundEnv,
146                           TypeElement ignore, Element element) {
147        super.process(roundEnv, ignore, element);
148
149        if (! processed.contains(element)) {
150            processed.add(element);
151
152            Attribute attribute = element.getAnnotation(Attribute.class);
153            MainClass main = element.getAnnotation(MainClass.class);
154
155            if (main != null) {
156                switch (element.getKind()) {
157                case CLASS:
158                case INTERFACE:
159                    TypeElement type = (TypeElement) element;
160                    ExecutableElement method = getMethod(type, PROTOTYPE);
161
162                    if (method != null
163                        && (method.getModifiers()
164                            .containsAll(getModifiers(PROTOTYPE)))) {
165                        String name = elements.getBinaryName(type).toString();
166                        String old = manifest.put(main, name);
167
168                        if (old != null && (! name.equals(old))) {
169                            print(WARNING, element,
170                                  "@%s: %s overwrites previous definition of %s",
171                                  main.annotationType().getSimpleName(),
172                                  element.getKind(), old);
173                        }
174                    } else {
175                        print(ERROR, element,
176                              "@%s: %s does not implement '%s'",
177                              main.annotationType().getSimpleName(),
178                              element.getKind(), declaration(PROTOTYPE));
179                    }
180                    break;
181
182                default:
183                    break;
184                }
185            }
186
187            Section section = element.getAnnotation(Section.class);
188
189            if (section != null) {
190                switch (element.getKind()) {
191                case PACKAGE:
192                    manifest.put(asPath((PackageElement) element), section);
193                    break;
194
195                default:
196                    break;
197                }
198            }
199
200            JavaBean bean = element.getAnnotation(JavaBean.class);
201            DependsOn depends = element.getAnnotation(DependsOn.class);
202            DesignTimeOnly design =
203                element.getAnnotation(DesignTimeOnly.class);
204
205            if (bean != null || depends != null || design != null) {
206                switch (element.getKind()) {
207                case CLASS:
208                    manifest.put(asPath((TypeElement) element),
209                                 bean, depends, design);
210                    break;
211
212                default:
213                    break;
214                }
215            }
216        }
217    }
218
219    @Override
220    public void process(Set<Class<?>> set, File destdir) throws IOException {
221        File file = new File(new File(destdir, META_INF), MANIFEST_MF);
222
223        if (manifest == null) {
224            manifest = new ManifestImpl();
225
226            if (file.exists()) {
227                try (FileInputStream in = new FileInputStream(file)) {
228                    manifest.read(in);
229                }
230            }
231
232            manifest.init();
233        }
234
235        for (Class<?> type : set) {
236            Attribute attribute = type.getAnnotation(Attribute.class);
237            MainClass main = type.getAnnotation(MainClass.class);
238
239            if (main != null) {
240                manifest.put(main, type.getName());
241            }
242
243            Section section = type.getAnnotation(Section.class);
244
245            if (section != null) {
246                manifest.put(asPath(type.getPackage()), section);
247            }
248
249            JavaBean bean = type.getAnnotation(JavaBean.class);
250            DependsOn depends = type.getAnnotation(DependsOn.class);
251            DesignTimeOnly design = type.getAnnotation(DesignTimeOnly.class);
252
253            if (bean != null || depends != null || design != null) {
254                manifest.put(asPath(type), bean, depends, design);
255            }
256        }
257
258        Files.createDirectories(file.toPath().getParent());
259
260        try (FileOutputStream out = new FileOutputStream(file)) {
261            manifest.write(out);
262        }
263    }
264
265    @Override
266    protected String asPath(TypeElement element) {
267        return super.asPath(element) + _CLASS;
268    }
269
270    @Override
271    protected String asPath(Class<?> type) {
272        return super.asPath(type) + _CLASS;
273    }
274
275    @NoArgsConstructor @ToString
276    private class ManifestImpl extends Manifest {
277        private static final String MANIFEST_VERSION = "Manifest-Version";
278
279        protected void init() {
280            if (getMainAttributes().getValue(MANIFEST_VERSION) == null) {
281                getMainAttributes().putValue(MANIFEST_VERSION, "1.0");
282            }
283        }
284
285        public Attributes putAttributes(String name, Attributes attributes) {
286            return getEntries().put(name, attributes);
287        }
288
289        public String put(MainClass main, String name) {
290            return (getMainAttributes()
291                    .putValue(getAttributeName(main.getClass()), name));
292        }
293
294        public void put(String path, Section section) {
295            Attributes attributes = getEntries().get(path);
296
297            if (attributes == null) {
298                attributes = new Attributes();
299                getEntries().put(path, attributes);
300            }
301
302            attributes.putValue("Sealed", String.valueOf(section.sealed()));
303        }
304
305        public void put(String path,
306                        JavaBean bean,
307                        DependsOn depends, DesignTimeOnly design) {
308            Attributes attributes = getEntries().get(path);
309
310            if (attributes == null) {
311                attributes = new Attributes();
312                getEntries().put(path, attributes);
313            }
314
315            attributes.putValue(getAttributeName(JavaBean.class),
316                                String.valueOf(true));
317
318            if (depends != null) {
319                attributes.putValue(getAttributeName(depends.getClass()),
320                                    String.join(" ", depends.value()));
321            }
322
323            if (design != null) {
324                attributes.putValue(getAttributeName(design.getClass()),
325                                    String.join(" ", design.value()));
326            }
327        }
328
329        private String getAttributeName(Class<? extends Annotation> type) {
330            return type.getAnnotation(Attribute.class).value();
331        }
332    }
