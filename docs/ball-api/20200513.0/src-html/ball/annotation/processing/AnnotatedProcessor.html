023import java.lang.annotation.Annotation;
024import java.lang.reflect.InvocationTargetException;
025import java.util.Arrays;
026import java.util.List;
027import java.util.Objects;
028import java.util.Set;
029import java.util.stream.Stream;
030import javax.annotation.processing.RoundEnvironment;
031import javax.lang.model.element.AnnotationMirror;
032import javax.lang.model.element.AnnotationValue;
033import javax.lang.model.element.Element;
034import javax.lang.model.element.ElementKind;
035import javax.lang.model.element.ExecutableElement;
036import javax.lang.model.element.Modifier;
037import javax.lang.model.element.TypeElement;
038import javax.lang.model.element.VariableElement;
039import javax.lang.model.type.TypeMirror;
040import lombok.AllArgsConstructor;
041import lombok.NoArgsConstructor;
042import lombok.ToString;
043
044import static java.util.stream.Collectors.toList;
045import static java.util.stream.Collectors.toSet;
046import static javax.tools.Diagnostic.Kind.ERROR;
047import static lombok.AccessLevel.PROTECTED;
048
049/**
050 * Abstract {@link javax.annotation.processing.Processor} base class for
051 * processing {@link Annotation}s specified by @{@link For}.  Provides
052 * built-in support for a number of {@link Annotation} types.
053 *
054 * @see AnnotatedElementMustBe
055 * @see AnnotatedTypeMustExtend
056 * @see AnnotatedTypeMustHaveConstructor
057 * @see AnnotationValueMustConvertTo
058 *
059 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
060 * @version $Revision: 5905 $
061 */
062@NoArgsConstructor(access = PROTECTED) @ToString
063public abstract class AnnotatedProcessor extends AbstractProcessor {
064
065    /**
066     * Method to get the {@link List} of supported {@link Annotation}
067     * {@link Class}es.
068     *
069     * @return  The {@link List} of supported {@link Annotation}
070     *          {@link Class}es.
071     */
072    protected List<Class<? extends Annotation>> getSupportedAnnotationTypeList() {
073        return Arrays.asList(getClass().getAnnotation(For.class).value());
074    }
075
076    @Override
077    public Set<String> getSupportedAnnotationTypes() {
078        Set<String> set =
079            getSupportedAnnotationTypeList()
080            .stream()
081            .map(Class::getCanonicalName)
082            .collect(toSet());
083
084        return set;
085    }
086
087    @Override
088    public boolean process(Set<? extends TypeElement> annotations,
089                           RoundEnvironment roundEnv) {
090        annotations.stream().forEach(t -> process(roundEnv, t));
091
092        return true;
093    }
094
095    private void process(RoundEnvironment roundEnv, TypeElement annotation) {
096        try {
097            roundEnv.getElementsAnnotatedWith(annotation)
098                .stream()
099                .peek(new AnnotatedElementMustBeCheck(annotation))
100                .peek(new AnnotatedTypeMustExtendCheck(annotation))
101                .peek(new AnnotatedTypeMustHaveConstructorCheck(annotation))
102                .peek(new AnnotationValueMustConvertToCheck(annotation))
103                .forEach(t -> process(roundEnv, annotation, t));
104        } catch (Throwable throwable) {
105            print(ERROR, throwable);
106        }
107    }
108
109    /**
110     * Callback method to process an annotated {@link Element}.  Default
111     * implementation does nothing.
112     *
113     * @param   roundEnv        The {@link RoundEnvironment}.
114     * @param   annotation      The annotation {@link TypeElement}.
115     * @param   element         The annotated {@link Element}.
116     */
117    protected void process(RoundEnvironment roundEnv,
118                           TypeElement annotation, Element element) {
119    }
120
121    @AllArgsConstructor @ToString
122    private class AnnotatedElementMustBeCheck extends Check<Element> {
123        private final TypeElement annotation;
124
125        @Override
126        public void accept(Element element) {
127            AnnotationMirror meta =
128                getAnnotationMirror(annotation, AnnotatedElementMustBe.class);
129
130            if (meta != null) {
131                AnnotationValue value = getAnnotationValue(meta, "value");
132                String name =
133                    ((VariableElement) value.getValue()).getSimpleName()
134                    .toString();
135                ElementKind kind = ElementKind.valueOf(name);
136
137                if (! kind.equals(element.getKind())) {
138                    print(ERROR, element,
139                          "@%s: %s is not a %s",
140                          annotation.getSimpleName(), element.getKind(), kind);
141                }
142            }
143        }
144    }
145
146    @AllArgsConstructor @ToString
147    private class AnnotatedTypeMustExtendCheck extends Check<Element> {
148        private final TypeElement annotation;
149
150        @Override
151        public void accept(Element element) {
152            AnnotationMirror meta =
153                getAnnotationMirror(annotation, AnnotatedTypeMustExtend.class);
154
155            if (meta != null) {
156                AnnotationValue value = getAnnotationValue(meta, "value");
157                TypeElement type =
158                    (TypeElement)
159                    types.asElement((TypeMirror) value.getValue());
160
161                if (! types.isAssignable(element.asType(), type.asType())) {
162                    print(ERROR, element,
163                          "@%s: %s does not extend %s",
164                          annotation.getSimpleName(),
165                          element, type.getQualifiedName());
166                }
167            }
168        }
169    }
170
171    @AllArgsConstructor @ToString
172    private class AnnotatedTypeMustHaveConstructorCheck extends Check<Element> {
173        private final TypeElement annotation;
174
175        @Override
176        public void accept(Element element) {
177            AnnotationMirror meta =
178                getAnnotationMirror(annotation,
179                                    AnnotatedTypeMustHaveConstructor.class);
180
181            if (meta != null) {
182                AnnotationValue value = getAnnotationValue(meta, "value");
183                String name =
184                    ((VariableElement) value.getValue()).getSimpleName()
185                    .toString();
186                Modifier modifier = Modifier.valueOf(name);
187                List<TypeMirror> parameters =
188                    Stream.of(getAnnotationValue(meta, "parameters"))
189                    .filter(Objects::nonNull)
190                    .map(t -> (List<?>) t.getValue())
191                    .flatMap(List::stream)
192                    .map(t -> (AnnotationValue) t)
193                    .map(t -> (TypeMirror) t.getValue())
194                    .collect(toList());
195                ExecutableElement constructor =
196                    getConstructor((TypeElement) element, parameters);
197                boolean found =
198                    (constructor != null
199                     && constructor.getModifiers().contains(modifier));
200
201                if (! found) {
202                    print(ERROR, element,
203                          "@%s: No %s matching constructor",
204                          annotation.getSimpleName(), modifier);
205                }
206            }
207        }
208    }
209
210    @AllArgsConstructor @ToString
211    private class AnnotationValueMustConvertToCheck extends Check<Element> {
212        private final TypeElement annotation;
213
214        @Override
215        public void accept(Element element) {
216            AnnotationMirror meta =
217                getAnnotationMirror(annotation,
218                                    AnnotationValueMustConvertTo.class);
219
220            if (meta != null) {
221                AnnotationValue value = getAnnotationValue(meta, "value");
222                TypeElement to =
223                    (TypeElement)
224                    types.asElement((TypeMirror) value.getValue());
225                String method =
226                    (String) getAnnotationValue(meta, "method").getValue();
227                String name =
228                    (String) getAnnotationValue(meta, "name").getValue();
229                AnnotationMirror mirror =
230                    getAnnotationMirror(element, annotation);
231                AnnotationValue from = null;
232
233                try {
234                    from = getAnnotationValue(mirror, name);
235
236                    Class<?> type =
237                        Class.forName(to.getQualifiedName().toString());
238
239                    if (! method.isEmpty()) {
240                        type.getMethod(method, from.getValue().getClass())
241                            .invoke(null, from.getValue());
242                    } else {
243                        type.getConstructor(from.getValue().getClass())
244                            .newInstance(from.getValue());
245                    }
246                } catch (Exception exception) {
247                    Throwable throwable = exception;
248
249                    while (throwable instanceof InvocationTargetException) {
250                        throwable = throwable.getCause();
251                    }
252
253                    print(ERROR, element, mirror,
254                          "Cannot convert %s to %s\n%s",
255                          from, to.getQualifiedName(), throwable.getMessage());
256                }
257            }
258        }
259    }
260}
