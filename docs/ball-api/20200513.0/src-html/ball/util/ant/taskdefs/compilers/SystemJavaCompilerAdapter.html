050/**
051 * {@link CompilerAdapter} implementation for the system
052 * {@link JavaCompiler}.  Subclass implementors will likely only need to
053 * override {@link #compile(StandardJavaFileManager,DiagnosticMap)}.
054 *
055 * @see ToolProvider#getSystemJavaCompiler()
056 *
057 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
058 * @version $Revision: 5855 $
059 */
060@NoArgsConstructor @ToString
061public class SystemJavaCompilerAdapter implements CompilerAdapter,
062                                                  CompilerAdapterExtension {
063    /** {@link #CARET} = {@value #CARET} */
064    protected static final String CARET = "^";
065    /** {@link #SPACE} = {@value #SPACE} */
066    protected static final String SPACE = " ";
067
068    private Javac javac = null;
069    protected JavaCompiler compiler = null;
070    protected JavaCompiler.CompilationTask task = null;
071
072    protected Javac getJavac() { return javac; }
073
074    @Override
075    public void setJavac(Javac javac) {
076        this.javac = javac;
077
078        javac.add(this);
079        javac.setCompiler("modern");
080    }
081
082    @Override
083    public boolean execute() throws BuildException {
084        boolean success = false;
085        DiagnosticMap map = new DiagnosticMap();
086
087        compiler =
088            Objects.requireNonNull(ToolProvider.getSystemJavaCompiler(),
089                                   "No system Java compiler");
090
091        Locale locale = null;
092        Charset charset = null;
093
094        if (getJavac().getEncoding() != null) {
095            charset = Charset.forName(getJavac().getEncoding());
096        }
097
098        try (StandardJavaFileManager fm =
099                 compiler.getStandardFileManager(map, locale, charset)) {
100            fm.setLocation(StandardLocation.PLATFORM_CLASS_PATH,
101                           asFileList(getJavac().getBootclasspath()));
102            fm.setLocation(StandardLocation.CLASS_PATH,
103                           asFileList(getJavac().getClasspath()));
104            fm.setLocation(StandardLocation.CLASS_OUTPUT,
105                           asFileList(getJavac().getDestdir()));
106            fm.setLocation(StandardLocation.SOURCE_PATH,
107                           asFileList(getJavac().getSrcdir()));
108
109            success = compile(fm, map);
110        } catch (BuildException exception) {
111            throw exception;
112        } catch (Throwable throwable) {
113            throw new BuildException(throwable);
114        } finally {
115            log(map);
116        }
117
118        return success;
119    }
120
121    /**
122     * Method to do the compilation work.  Subclass implementations should
123     * override this method by first chaining back to the superclass
124     * implementation.
125     * <p>
126     * The {@link StandardJavaFileManager} parameter will have its
127     * {@link StandardLocation#PLATFORM_CLASS_PATH} set to
128     * {@link Javac#getBootclasspath()},
129     * {@link StandardLocation#CLASS_PATH} set to
130     * {@link Javac#getClasspath()}, {@link StandardLocation#CLASS_OUTPUT}
131     * set to {@link Javac#getDestdir()}, and
132     * {@link StandardLocation#SOURCE_PATH} set to
133     * {@link Javac#getSrcdir()}.
134     *
135     * @param   fm              The {@link StandardJavaFileManager}.
136     * @param   map             The {@link DiagnosticMap}.
137     *
138     * @return  {@code true} if the compilation was successful;
139     *          {@code false} otherwise.
140     *
141     * @throws  Throwable       If any problems are encountered in the
142     *                          compilation.
143     */
144    protected boolean compile(StandardJavaFileManager fm,
145                              DiagnosticMap map) throws Throwable {
146        List<String> options =
147            Arrays.asList(getJavac().getCurrentCompilerArgs());
148        Iterable<? extends JavaFileObject> objects =
149            fm.getJavaFileObjects(getJavac().getFileList());
150
151        task = compiler.getTask(null, fm, map, options, null, objects);
152
153        return task.call();
154    }
155
156    /**
157     * Method to convert the argument {@link Path} to a {@link File}
158     * {@link List}.
159     *
160     * @param   path            The {@link Path}.
161     *
162     * @return  The {@link Path} as a {@link File} {@link List} or
163     *          {@code null} if the argument {@link Path} is {@code null}.
164     */
165    protected List<? extends File> asFileList(Path path) {
166        File[] files = null;
167
168        if (path != null) {
169            File base = getJavac().getProject().getBaseDir();
170            String[] paths = path.list();
171
172            files = new File[paths.length];
173
174            for (int i = 0; i < files.length; i += 1) {
175                files[i] =
176                    FileUtils.getFileUtils().resolveFile(base, paths[i]);
177            }
178        }
179
180        return asFileList(files);
181    }
182
183    /**
184     * Method to convert the argument {@link File} array to a {@link File}
185     * {@link List}.
186     *
187     * @param   files           The {@link File} array.
188     *
189     * @return  The {@link File} array as a {@link File} {@link List} or
190     *          {@code null} if the argument {@link File} array is
191     *          {@code null}.
192     */
193    protected List<? extends File> asFileList(File... files) {
194        return (files != null) ? Arrays.asList(files) : null;
195    }
196
197    /**
198     * See {@link Javac#log(String)}.
199     *
200     * @param   string          The {@link String} to log.
201     */
202    protected void log(String string) { getJavac().log(string); }
203
204    private void log(DiagnosticMap map) {
205        for (Map.Entry<Diagnostic<? extends JavaFileObject>,String> entry :
206                 map.entrySet()) {
207            Diagnostic<? extends JavaFileObject> diagnostic = entry.getKey();
208            String remedy = entry.getValue();
209
210            log(String.valueOf(diagnostic));
211
212            if (diagnostic.getPosition() != Diagnostic.NOPOS) {
213                String source = source(diagnostic);
214
215                if (source != null) {
216                    if (String.valueOf(diagnostic).indexOf(source) < 0) {
217                        log(source);
218                        log(pointer(diagnostic));
219                    }
220                }
221            }
222
223            if (remedy != null) {
224                log(remedy);
225            }
226        }
227
228        for (Map.Entry<Diagnostic.Kind,Integer> entry :
229                 map.getKindCountMap().entrySet()) {
230            String kind = entry.getKey().toString().toLowerCase();
231            int count = entry.getValue();
232
233            log(count + SPACE + kind + ((count == 1) ? EMPTY : "s"));
234        }
235    }
236
237    private String source(Diagnostic<? extends JavaFileObject> diagnostic) {
238        String line = null;
239
240        try (CharSequenceReader reader =
241                 new CharSequenceReader(diagnostic.getSource()
242                                        .getCharContent(true))) {
243            do {
244                line = reader.readLine();
245            } while (line != null
246                     && reader.getLineNumber() < diagnostic.getLineNumber());
247        } catch (IOException exception) {
248        }
249
250        return line;
251    }
252
253    private String pointer(Diagnostic<?> diagnostic) {
254        StringBuilder buffer = new StringBuilder(CARET);
255
256        while (buffer.length() < diagnostic.getColumnNumber()) {
257            buffer.insert(0, SPACE);
258        }
259
260        return buffer.toString();
261    }
262
263    @Override
264    public String[] getSupportedFileExtensions() {
265        return new String[] { "java" };
266    }
