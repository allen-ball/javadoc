023import ball.tools.DiagnosticMap;
024import ball.tools.Remedy;
025import java.net.URI;
026import java.util.Collections;
027import java.util.Map;
028import java.util.SortedSet;
029import java.util.TreeSet;
030import java.util.regex.Pattern;
031import javax.tools.Diagnostic;
032import javax.tools.JavaFileObject;
033import javax.tools.StandardJavaFileManager;
034import lombok.NoArgsConstructor;
035import lombok.ToString;
036
037import static java.util.Comparator.comparing;
038import static javax.tools.JavaFileObject.Kind.CLASS;
039import static javax.tools.StandardLocation.CLASS_OUTPUT;
040import static javax.tools.StandardLocation.CLASS_PATH;
041import static org.apache.commons.lang3.StringUtils.EMPTY;
042
043/**
044 * {@link SystemJavaCompilerAdapter} implementation that provides additional
045 * {@link.man lint(1)}-like analysis.
046 *
047 * @see Remedy
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 * @version $Revision: 5855 $
051 */
052@NoArgsConstructor @ToString
053public class LintAdapter extends SystemJavaCompilerAdapter {
054    private static final URI FILE = URI.create("file:///");
055
056    @Override
057    protected boolean compile(StandardJavaFileManager fm,
058                              DiagnosticMap map) throws Throwable {
059        boolean success = super.compile(fm, map);
060        SortedSet<Class<?>> classes =
061            Collections.unmodifiableSortedSet(new OutputClassSet(fm));
062
063        for (Map.Entry<Diagnostic<? extends JavaFileObject>,String> entry :
064                 map.entrySet()) {
065            Diagnostic<? extends JavaFileObject> key = entry.getKey();
066            String code = key.getCode();
067            Remedy remedy = Remedy.getRemedyMap().get(code);
068
069            if (remedy != null) {
070                entry.setValue(remedy.getRx(key, fm, classes));
071            } else {
072                entry.setValue(code);
073            }
074        }
075
076        return success;
077    }
078
079    private static class OutputClassSet extends TreeSet<Class<?>> {
080        private static final long serialVersionUID = -6071215710420201764L;
081
082        public OutputClassSet(StandardJavaFileManager fm) {
083            super(comparing(Class::getName));
084
085            URI root = fm.getLocation(CLASS_OUTPUT).iterator().next().toURI();
086            ClassLoader loader = fm.getClassLoader(CLASS_PATH);
087            Iterable<JavaFileObject> iterable = null;
088
089            try {
090                iterable =
091                    fm.list(CLASS_OUTPUT, EMPTY,
092                            Collections.singleton(CLASS), true);
093            } catch (Throwable throwable) {
094                iterable = Collections.<JavaFileObject>emptyList();
095            }
096
097            for (JavaFileObject object : iterable) {
098                URI uri = FILE.resolve(object.toUri());
099                String name = root.relativize(uri).toString();
100
101                if (name.toLowerCase().endsWith(CLASS.extension)) {
102                    name =
103                        name.substring(0,
104                                       name.length()
105                                       - CLASS.extension.length());
106                }
107
108                name = name.replaceAll(Pattern.quote("/"), ".");
109
110                try {
111                    add(Class.forName(name, false, loader));
112                } catch (Throwable throwable) {
113                }
114            }
115        }
116    }
117}
