023import ball.lang.reflect.JavaLangReflectMethods;
024import java.util.Collections;
025import java.util.ServiceLoader;
026import java.util.SortedMap;
027import java.util.SortedSet;
028import java.util.TreeMap;
029import javax.tools.Diagnostic;
030import javax.tools.JavaFileObject;
031import javax.tools.StandardJavaFileManager;
032import lombok.NoArgsConstructor;
033import lombok.ToString;
034
035import static lombok.AccessLevel.PRIVATE;
036import static lombok.AccessLevel.PROTECTED;
037
038/**
039 * Abstract base class for {@link Diagnostic} remedies.
040 *
041 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
042 * @version $Revision: 5855 $
043 */
044@NoArgsConstructor(access = PROTECTED) @ToString
045public abstract class Remedy implements JavaLangReflectMethods {
046    @NoArgsConstructor(access = PRIVATE) @ToString
047    private static abstract class Lazy {
048        public static final SortedMap<String,Remedy> MAP;
049
050        static {
051            TreeMap<String,Remedy> map = new TreeMap<>();
052
053            for (Remedy remedy : ServiceLoader.load(Remedy.class)) {
054                for (String code : remedy.getCodes()) {
055                    map.put(code, remedy);
056                }
057            }
058
059            MAP = Collections.unmodifiableSortedMap(map);
060        }
061    }
062
063    /**
064     * Method to get the {@link SortedMap} of available {@link Remedy}s
065     * (keyed by {@link Diagnostic} code).
066     *
067     * @return  The {@link SortedMap}.
068     */
069    public static SortedMap<String,Remedy> getRemedyMap() { return Lazy.MAP; }
070
071    /**
072     * See {@link Codes} and {@link Diagnostic#getCode()}.
073     *
074     * @return  This {@link Remedy}'s codes.
075     */
076    public String[] getCodes() {
077        Codes codes = getClass().getAnnotation(Codes.class);
078
079        return (codes != null) ? codes.value() : null;
080    }
081
082    /**
083     * Method to provide a prescriptive message.
084     *
085     * @param   diagnostic      The {@link Diagnostic} to remedy.
086     * @param   fm              The {@link StandardJavaFileManager}.
087     * @param   classes         The {@link SortedSet} of output classes.
088     *
089     * @return  The remedy ({@link String}).
090     */
091    public abstract String getRx(Diagnostic<? extends JavaFileObject> diagnostic,
092                                 StandardJavaFileManager fm,
093                                 SortedSet<Class<?>> classes);
094
095    /**
096     * Method to get a {@link Class} for the argument {@code name}.
097     *
098     * @param   name            The name of the {@link Class}.
099     * @param   classes         See {@link #getRx(Diagnostic,StandardJavaFileManager,SortedSet)}.
100     *
101     * @return  A {@link Class} corresponding to the name if one could be
102     *          found; {@code null} otherwise.
103     */
104    protected Class<?> getClassForNameFrom(String name,
105                                           SortedSet<Class<?>> classes) {
106        Class<?> type = null;
107
108        if (type == null) {
109            try {
110                type = Class.forName(name);
111            } catch (ClassNotFoundException exception) {
112            }
113        }
114
115        if (type == null) {
116            type = new ClassMap(classes).get(name);
117        }
118
119        if (type == null) {
120            Package[] pkgs = Package.getPackages();
121
122            for (Package pkg : pkgs) {
123                try {
124                    type = Class.forName(pkg.getName() + "." + name);
125                } catch (ClassNotFoundException exception) {
126                }
127
128                if (type != null) {
129                    break;
130                }
131            }
132        }
133
134        return type;
135    }
136
137    /**
138     * {@link TreeMap} implementation to map {@link Class#getSimpleName()}
139     * to {@link Class}.
140     */
141    protected class ClassMap extends TreeMap<String,Class<?>> {
142        private static final long serialVersionUID = 469591372955628535L;
143
144        /**
145         * Sole constructor.
146         *
147         * @param       values          The {@link Iterable} of
148         *                              {@link Class} values.
149         */
150        public ClassMap(Iterable<Class<?>> values) {
151            super();
152
153            for (Class<?> value : values) {
154                String key = value.getSimpleName();
155
156                if (key != null && (! containsKey(key))) {
157                    put(key, value);
158                }
159            }
160        }
161    }
162}
