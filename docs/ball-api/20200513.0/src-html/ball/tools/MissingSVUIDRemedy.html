023import ball.annotation.ServiceProviderFor;
024import java.io.ObjectStreamClass;
025import java.io.Serializable;
026import java.lang.reflect.Field;
027import java.util.Collections;
028import java.util.Locale;
029import java.util.SortedSet;
030import javax.tools.Diagnostic;
031import javax.tools.JavaFileObject;
032import javax.tools.StandardJavaFileManager;
033import javax.tools.StandardLocation;
034import lombok.NoArgsConstructor;
035import lombok.ToString;
036
037/**
038 * compiler.warn.missing.SVUID {@link Remedy}.
039 *
040 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
041 * @version $Revision: 5855 $
042 */
043@ServiceProviderFor({ Remedy.class })
044@Codes({ "compiler.warn.missing.SVUID" })
045@NoArgsConstructor @ToString
046public class MissingSVUIDRemedy extends Remedy implements Serializable {
047    private static final long serialVersionUID = -6900234880629237267L;
048
049    private static abstract class PROTOTYPE {
050        private static final long serialVersionUID = 0L;
051    }
052
053    private static final Field FIELD = PROTOTYPE.class.getDeclaredFields()[0];
054
055    @Override
056    public String getRx(Diagnostic<? extends JavaFileObject> diagnostic,
057                        StandardJavaFileManager fm,
058                        SortedSet<Class<?>> classes) {
059        JavaFileObject source = diagnostic.getSource();
060        String message = diagnostic.getMessage(Locale.ROOT);
061        String name = fm.inferBinaryName(StandardLocation.SOURCE_PATH, source);
062        int start = message.lastIndexOf(name);
063        int end =
064            (start >= 0)
065                ? message.indexOf(" ", start)
066                : message.length();
067
068        if (0 <= start && start < end) {
069            name = message.substring(start, end);
070        }
071
072        Class<?> type = null;
073
074        for (Class<?> member : classes) {
075            if (name.equals(member.getCanonicalName())) {
076                type = member;
077                break;
078            }
079        }
080
081        return getRX(type);
082    }
083
084    protected String getRX(Class<?> type) {
085        String remedy = null;
086
087        if (type != null) {
088            long uid = ObjectStreamClass.lookup(type).getSerialVersionUID();
089
090            remedy = String.format("%s = %dL;", declaration(FIELD), uid);
091        }
092
093        return remedy;
094    }
095}
