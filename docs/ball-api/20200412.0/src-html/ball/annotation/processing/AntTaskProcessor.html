063/**
064 * {@link Processor} implementation to check {@link Class}es annotated with
065 * {@link AntTask}:
066 * <ol>
067 *   <li value="1">Are an instance of {@link Task},</li>
068 *   <li value="2">Concrete, and</li>
069 *   <li value="3">Have a public no-argument constructor</li>
070 * </ol>
071 *
072 * And {@link Package}s annotated with {@link AntLib}.  Generates the
073 * specified resources at the end of annotation processing.
074 *
075 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
076 * @version $Revision: 5285 $
077 */
078@ServiceProviderFor({ Processor.class })
079@For({ AntLib.class, AntTask.class })
080@NoArgsConstructor @ToString
081public class AntTaskProcessor extends AbstractAnnotationProcessor
082                              implements ClassFileProcessor {
083    private static final String ANTLIB_XML = "antlib.xml";
084
085    private static final String NO = "no";
086    private static final String YES = "yes";
087
088    private static final String INDENT_AMOUNT =
089        "{http://xml.apache.org/xslt}indent-amount";
090
091    private static final Transformer TRANSFORMER;
092
093    static {
094        try {
095            TRANSFORMER = TransformerFactory.newInstance().newTransformer();
096            TRANSFORMER.setOutputProperty(OMIT_XML_DECLARATION, NO);
097            TRANSFORMER.setOutputProperty(INDENT, YES);
098            TRANSFORMER.setOutputProperty(INDENT_AMOUNT, String.valueOf(2));
099        } catch (Exception exception) {
100            throw new ExceptionInInitializerError(exception);
101        }
102    }
103
104    private ResourceMap map = new ResourceMap();
105    private LinkedHashSet<String> packages = new LinkedHashSet<>();
106
107    @Override
108    public boolean process(Set<? extends TypeElement> annotations,
109                           RoundEnvironment roundEnv) {
110        boolean result = true;
111
112        try {
113            if (! roundEnv.errorRaised()) {
114                result &= super.process(annotations, roundEnv);
115
116                if (roundEnv.processingOver()) {
117                    for (Map.Entry<String,PropertiesImpl> entry :
118                             map.entrySet()) {
119                        FileObject file =
120                            filer.createResource(CLASS_OUTPUT,
121                                                 EMPTY, entry.getKey());
122
123                        try (OutputStream out = file.openOutputStream()) {
124                            entry.getValue().store(out, entry.getKey());
125                        }
126                    }
127
128                    for (String pkg : packages) {
129                        AntLibXML antlib = new AntLibXML(pkg, map);
130                        FileObject file =
131                            filer.createResource(CLASS_OUTPUT,
132                                                 EMPTY, antlib.getPath());
133
134                        try (OutputStream out = file.openOutputStream()) {
135                            antlib.writeTo(out);
136                        }
137                    }
138                }
139            }
140        } catch (Exception exception) {
141            print(ERROR, null, exception);
142        }
143
144        return result;
145    }
146
147    @Override
148    protected void process(RoundEnvironment env,
149                           TypeElement annotation,
150                           Element element) throws Exception {
151        switch (element.getKind()) {
152        case CLASS:
153            String name = element.getAnnotation(AntTask.class).value();
154            String resource = element.getAnnotation(AntTask.class).resource();
155            Element enclosing = element;
156
157            while (enclosing != null
158                   && enclosing.getKind() != ElementKind.PACKAGE) {
159                enclosing = enclosing.getEnclosingElement();
160            }
161
162            if (enclosing != null) {
163                resource =
164                    URI.create(asPath((PackageElement) enclosing) + resource)
165                    .normalize()
166                    .toString();
167            }
168
169            if (isNotEmpty(name)) {
170                if (isAssignable(element.asType(), Task.class)) {
171                    if (! element.getModifiers().contains(ABSTRACT)) {
172                        if (hasPublicNoArgumentConstructor(element)) {
173                            map.put(resource, name, (TypeElement) element);
174                        } else {
175                            print(ERROR,
176                                  element,
177                                  element.getKind() + " annotated with "
178                                  + AT + annotation.getSimpleName()
179                                  + " but does not have a " + PUBLIC
180                                  + " no-argument constructor");
181                        }
182                    } else {
183                        print(ERROR,
184                              element,
185                              element.getKind() + " annotated with "
186                              + AT + annotation.getSimpleName()
187                              + " but is " + ABSTRACT);
188                    }
189                } else {
190                    /*
191                     * See AntTaskMixInProcessor.
192                     */
193                }
194            } else {
195                print(ERROR,
196                      element,
197                      element.getKind() + " annotated with "
198                      + AT + annotation.getSimpleName()
199                      + " but does not specify value()");
200            }
201            break;
202
203        case PACKAGE:
204            packages.add(((PackageElement) element)
205                         .getQualifiedName().toString());
206            break;
207
208        default:
209            break;
210        }
211    }
212
213    @Override
214    public void process(Set<Class<?>> set, File destdir) throws Exception {
215        for (Class<?> type : set) {
216            AntTask task = type.getAnnotation(AntTask.class);
217
218            if (task != null) {
219                if (Task.class.isAssignableFrom(type)) {
220                    if (! isAbstract(type.getModifiers())) {
221                        String name = task.value();
222                        String resource = task.resource();
223                        Package pkg = type.getPackage();
224
225                        if (pkg != null) {
226                            resource =
227                                URI.create(asPath(pkg) + resource)
228                                .normalize()
229                                .toString();
230                        }
231
232                        map.put(resource, name, type);
233                    }
234                }
235            }
236
237            AntLib lib = type.getAnnotation(AntLib.class);
238
239            if (lib != null) {
240                packages.add(type.getPackage().getName());
241            }
242        }
243
244        for (Map.Entry<String,PropertiesImpl> entry : map.entrySet()) {
245            File file = new File(destdir, entry.getKey());
246
247            Files.createDirectories(file.toPath().getParent());
248
249            try (OutputStream out = new FileOutputStream(file)) {
250                entry.getValue().store(out, entry.getKey());
251            }
252        }
253
254        for (String pkg : packages) {
255            AntLibXML xml = new AntLibXML(pkg, map);
256            File file = new File(destdir, xml.getPath());
257
258            Files.createDirectories(file.toPath().getParent());
259
260            try (OutputStream out = new FileOutputStream(file)) {
261                xml.writeTo(out);
262            }
263        }
264    }
265
266    @NoArgsConstructor
267    private class ResourceMap extends TreeMap<String,PropertiesImpl> {
268        private static final long serialVersionUID = -6734240104662740026L;
269
270        public boolean put(String resource, String name, String task) {
271            PropertiesImpl value =
272                computeIfAbsent(resource, k -> new PropertiesImpl());
273
274            return (value.put(name, task) != task);
275        }
276
277        public boolean put(String resource, String name, Class<?> task) {
278            return put(resource, name, task.getName());
279        }
280
281        public boolean put(String resource, String name, TypeElement task) {
282            return put(resource,
283                       name, elements.getBinaryName(task).toString());
284        }
285    }
286
287    private static class AntLibXML extends TreeMap<String,String> {
288        private static final long serialVersionUID = -4362352118276244430L;
289
290        /** @serial */ private final String path;
291
292        private AntLibXML(String pkg, ResourceMap map) {
293            super();
294
295            this.path = asPath(pkg) + SLASH + ANTLIB_XML;
296
297            map.values()
298                .stream()
299                .flatMap(t -> t.entrySet().stream())
300                .filter(t -> t.getValue().toString().startsWith(pkg + DOT))
301                .forEach(t -> put(t.getKey().toString(),
302                                  t.getValue().toString()));
303        }
304
305        public String getPath() { return path; }
306
307        public FluentDocument asDocument() throws Exception {
308            FluentDocument d =
309                FluentDocumentBuilderFactory.newInstance()
310                .newDocumentBuilder()
311                .newDocument();
312
313            d.add(d.element("antlib",
314                            entrySet().stream()
315                            .map(t -> d.element("taskdef",
316                                                d.attr("name", t.getKey()),
317                                                d.attr("classname", t.getValue())))));
318
319            return d;
320        }
321
322        public void writeTo(OutputStream out) throws Exception {
323            TRANSFORMER.transform(new DOMSource(asDocument()),
324                                  new StreamResult(out));
325        }
326    }
