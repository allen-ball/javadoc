023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.Collections;
026import java.util.regex.Pattern;
027
028import static java.util.stream.Collectors.joining;
029import static org.apache.commons.lang3.StringUtils.EMPTY;
030import static org.apache.commons.lang3.StringUtils.isNotEmpty;
031
032/**
033 * {@link MAILDIRFolder} path implementation.
034 *
035 * {@bean.info}
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 * @version $Revision: 5285 $
039 */
040public class MAILDIRPath extends ArrayList<String>
041                         implements Comparable<MAILDIRPath> {
042    private static final long serialVersionUID = 361116961370030253L;
043
044    /**
045     * The separator {@code char}.
046     */
047    protected static final char SEPARATOR = '.';
048
049    private static final String REGEX =
050        Pattern.quote(String.valueOf(SEPARATOR));
051
052    /**
053     * @param   parent          The containing {@link MAILDIRPath}.
054     * @param   pathnames       The pathnames that make up the
055     *                          {@link MAILDIRPath}.
056     */
057    public MAILDIRPath(MAILDIRPath parent, String... pathnames) {
058        this(parent, Arrays.asList(pathnames));
059    }
060
061    /**
062     * @param   pathnames       The pathnames that make up the
063     *                          {@link MAILDIRPath}.
064     */
065    public MAILDIRPath(String... pathnames) { this(null, pathnames); }
066
067    private MAILDIRPath(MAILDIRPath parent, Iterable<String> iterable) {
068        super();
069
070        if (parent != null) {
071            addAll(parent);
072        }
073
074        if (iterable != null) {
075            for (String string : iterable) {
076                if (isNotEmpty(string)) {
077                    Collections.addAll(this, string.split(REGEX));
078                }
079            }
080        }
081
082        while (remove(null)) {
083        }
084
085        for (int index = 0, count = size(); index < count; index += 1) {
086            set(index, get(index).trim());
087        }
088
089        while (remove(EMPTY)) {
090        }
091    }
092
093    /**
094     * Method to get the parent {@link MAILDIRPath}.
095     *
096     * @return  The parent {@link MAILDIRPath} or <code>null</code> if this
097     *          {@link MAILDIRPath} is the "root."
098     */
099    public MAILDIRPath getParentPath() {
100        return isEmpty() ? null : new MAILDIRPath(null, subList(0, size() - 1));
101    }
102
103    /**
104     * Method to get the name (or final component) of this
105     * {@link MAILDIRPath}.
106     *
107     * @return  The name of this {@link MAILDIRPath} or <code>""</code> if
108     *          this {@link MAILDIRPath} is the "root."
109     */
110    public String getName() { return (! isEmpty()) ? get(size() - 1) : EMPTY; }
111
112    /**
113     * Method to convert this {@link MAILDIRPath} to the {@link MAILDIRFolder}
114     * directory {@link java.io.File} name.
115     *
116     * @return  The {@link MAILDIRFolder} directory {@link java.io.File}
117     *          name.
118     */
119    protected String toFilename() {
120        return String.valueOf(SEPARATOR) + toString();
121    }
122
123    @Override
124    public int compareTo(MAILDIRPath that) {
125        int difference = 0;
126
127        if (difference == 0) {
128            int n = Math.min(this.size(), that.size());
129
130            for (int i = 0; i < n; i += 1) {
131                difference = this.get(i).compareTo(that.get(i));
132
133                if (difference != 0) {
134                    break;
135                }
136            }
137        }
138
139        if (difference == 0) {
140            difference = this.size() - that.size();
141        }
142
143        return difference;
144    }
145
146    @Override
147    public String toString() {
148        return stream().collect(joining(String.valueOf(SEPARATOR)));
149    }
150}
