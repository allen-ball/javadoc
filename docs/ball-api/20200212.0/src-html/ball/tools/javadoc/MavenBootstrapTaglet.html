023import com.sun.tools.doclets.Taglet;
024import java.util.Map;
025import java.util.ServiceLoader;
026
027/**
028 * {@link AbstractTaglet} whose {@link #register(Map)} method loads all
029 * {@link Taglet}s specified through a {@link ServiceLoader}.  Designed to
030 * be specified as {@code <tagletClass/>} in a Maven {@code pom.xml}.
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 * @version $Revision: 5285 $
034 */
035public abstract class MavenBootstrapTaglet extends AbstractTaglet {
036
037    /**
038     * Static method to register all service providers for {@link Taglet}.
039     *
040     * @param   map             The {@link Map} of {@link Taglet}s.
041     */
042    public static void register(Map<Object,Object> map) {
043        try {
044            ServiceLoader.load(Taglet.class,
045                               MavenBootstrapTaglet.class.getClassLoader())
046                .iterator()
047                .forEachRemaining(t -> map.putIfAbsent(t.getName(), t));
048        } catch (Throwable throwable) {
049            throwable.printStackTrace(System.err);
050
051            if (throwable instanceof Error) {
052                throw (Error) throwable;
053            } else {
054                throw new ExceptionInInitializerError(throwable);
055            }
056        }
057    }
058
059    private MavenBootstrapTaglet() {
060        super(false, false, false, false, false, false, false);
061    }
062}
