023import java.io.File;
024import java.io.IOException;
025import java.io.InputStream;
026import java.io.OutputStream;
027import javax.activation.DataHandler;
028import javax.activation.DataSource;
029import javax.activation.FileDataSource;
030import javax.mail.IllegalWriteException;
031import javax.mail.MessagingException;
032import javax.mail.Session;
033import org.apache.commons.io.IOUtils;
034
035import static ball.mail.internet.InternetHeadersImpl.UNMODIFIABLE_HEADERS;
036import static java.util.Objects.requireNonNull;
037
038/**
039 * Extends {@link MimeMessageImpl} to create a "read-mostly" message.
040 *
041 * {@bean.info}
042 *
043 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
044 * @version $Revision: 5285 $
045 */
046public class ReadMostlyMimeMessage extends MimeMessageImpl {
047    private final long contentStart;
048    private final DataSource ds;
049
050    /**
051     * Constructor to create a read-mostly message from a {@link File}.
052     *
053     * @param   session         The {@link Session} object for this
054     *                          message.
055     * @param   file            The {@link File} containing the message.
056     *
057     * @throws  IOException     If the message cannot be read from the
058     *                          {@link File} for any reason.
059     * @throws  MessagingException
060     *                          If the message cannot be constructed from
061     *                          the {@link File} for any reason.
062     * @throws  NullPointerException
063     *                          If any of the arguments are {@code null}.
064     */
065    public ReadMostlyMimeMessage(Session session,
066                                 File file) throws IOException,
067                                                   MessagingException {
068        this(session, new FileDataSource(file));
069    }
070
071    /**
072     * Constructor to create a read-mostly message from a
073     * {@link DataSource}.
074     *
075     * @param   session         The {@link Session} object for this
076     *                          message.
077     * @param   ds              The {@link DataSource} containing this
078     *                          message.
079     *
080     * @throws  IOException     If the message cannot be read from the
081     *                          {@link DataSource} for any reason.
082     * @throws  MessagingException
083     *                          If the message cannot be constructed from
084     *                          the {@link DataSource} for any reason.
085     * @throws  NullPointerException
086     *                          If any of the arguments are {@code null}.
087     */
088    public ReadMostlyMimeMessage(Session session,
089                                 DataSource ds) throws IOException,
090                                                       MessagingException {
091        this(session, new InternetHeadersImpl(ds.getInputStream(), true), ds);
092    }
093
094    private ReadMostlyMimeMessage(Session session,
095                                  InternetHeadersImpl headers, DataSource ds) {
096        super(session);
097
098        this.headers = requireNonNull(headers, "headers");
099        this.contentStart = headers.getInputLength();
100        this.ds = requireNonNull(ds, "ds");
101
102        saveChanges();
103    }
104
105    @Override
106    protected void parse(InputStream in) throws MessagingException {
107        throw new IllegalWriteException();
108    }
109
110    @Override
111    protected InputStream getContentStream() throws MessagingException {
112        InputStream in = null;
113
114        try {
115            in = ds.getInputStream();
116            in.skip(contentStart);
117        } catch (IOException exception) {
118            throw new MessagingException(exception.getMessage(), exception);
119        }
120
121        return in;
122    }
123
124    @Override
125    public void setDataHandler(DataHandler handler) throws MessagingException {
126        throw new IllegalWriteException();
127    }
128
129    @Override
130    public void writeTo(OutputStream out,
131                        String[] ignore) throws IOException,
132                                                MessagingException {
133        ((InternetHeadersImpl) headers).writeTo(out, ignore);
134
135        try (InputStream in = getContentStream()) {
136            IOUtils.copy(in, out);
137        }
138    }
139
140    /**
141     * Method to determine if a header is modifiable.
142     *
143     * @param   name            The name of the header.
144     *
145     * @return  {@code true} if the header is modifiable; {@code false}
146     *          otherwise.
147     */
148    protected boolean isModifiableHeader(String name) {
149        return (! UNMODIFIABLE_HEADERS.contains(name));
150    }
151
152    @Override
153    public void addHeader(String name,
154                          String value) throws MessagingException {
155        if (name == null || isModifiableHeader(name)) {
156            super.addHeader(name, value);
157        } else {
158            throw new IllegalWriteException();
159        }
160    }
161
162    @Override
163    public void setHeader(String name,
164                          String value) throws MessagingException {
165        if (name == null || isModifiableHeader(name)) {
166            super.setHeader(name, value);
167        } else {
168            String[] values = getHeader(name);
169
170            if (! (values != null
171                   && values.length == 1 && values[0].equals(value))) {
172                throw new IllegalWriteException();
173            }
174        }
175    }
176
177    @Override
178    public void removeHeader(String name) throws MessagingException {
179        if (name == null || isModifiableHeader(name)) {
180            super.removeHeader(name);
181        } else {
182            if (getHeader(name) != null) {
183                throw new IllegalWriteException();
184            }
185        }
186    }
187
188    @Override
189    public void addHeaderLine(String line) throws MessagingException {
190        int index = line.indexOf(":");
191        String name = (index >= 0) ? line.substring(0, index) : null;
192
193        if (name == null || isModifiableHeader(name)) {
194            super.addHeaderLine(line);
195        } else {
196            throw new IllegalWriteException();
197        }
198    }
199
200    @Override
201    public void saveChanges() {
202        saved = true;
203        modified = false;
204    }
205}
