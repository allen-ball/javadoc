023import java.util.ArrayList;
024import java.util.Collections;
025import java.util.stream.IntStream;
026
027/**
028 * {@link Integer} {@link java.util.List} implementation useful for testing
029 * {@link ball.util.stream.Combinations} and
030 * {@link ball.util.stream.Permutations} {@link java.util.stream.Stream}s.
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 * @version $Revision: 5285 $
034 */
035public class IntegerList extends ArrayList<Integer> {
036    private static final long serialVersionUID = 5581619270177831511L;
037
038    /**
039     * @param   count           The element count.
040     */
041    public IntegerList(Integer count) {
042        super(count);
043
044        Collections.addAll(this,
045                           IntStream.range(0, count)
046                           .boxed()
047                           .toArray(Integer[]::new));
048    }
049
050    @Override
051    public Integer[] toArray() { return toArray(new Integer[] { }); }
052
053    @Override
054    public IntegerList clone() { return (IntegerList) super.clone(); }
055}
