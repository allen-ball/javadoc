043/**
044 * Extends {@link MimeMessage} by overriding the
045 * {@link MimeMessage#updateHeaders()} method to not modify the
046 * {@value ball.mail.internet.InternetHeadersImpl#MESSAGE_ID}.  Also
047 * provides implementations for {@link javax.mail.Message#getReceivedDate()}
048 * and {@link SMTPMessage#getEnvelopeFrom()} by subclassing
049 * {@link SMTPMessage}.
050 *
051 * {@bean.info}
052 *
053 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
054 * @version $Revision: 5285 $
055 */
056public class MimeMessageImpl extends SMTPMessage {
057    private static final String SPACE = " ";
058
059    protected static final MessageIDFormat MESSAGE_ID_FORMAT =
060        new MessageIDFormat();
061
062    private final MailDateFormat sentDateFormat = new MailDateFormatImpl();
063    private Date receivedDate = null;

079     *                          For any error with the underlying stream.

082                           InputStream in) throws MessagingException {
083        super(session, in);
084        /*
085         * Invoke for side-effect to set MailDateFormat TimeZone.
086         */
087        getSentDate();
088    }

090    /**
091     * @see MimeMessage#MimeMessage(MimeMessage)
092     *
093     * @param   source          The source {@link MimeMessage}.
094     *
095     * @throws  MessagingException
096     *                          For any error with the underlying stream.
097     */
098    public MimeMessageImpl(MimeMessage source) throws MessagingException {
099        super(source);
100
101        setMessageNumber(source.getMessageNumber());
102
103        if (source instanceof SMTPMessage) {
104            SMTPMessage in = (SMTPMessage) source;
105
106            setAllow8bitMIME(in.getAllow8bitMIME());
107            setEnvelopeFrom(in.getEnvelopeFrom());
108            setMailExtension(in.getMailExtension());
109            setNotifyOptions(in.getNotifyOptions());
110            setReturnOption(in.getReturnOption());
111            setSendPartial(in.getSendPartial());
112            setSubmitter(in.getSubmitter());
113        }
114
115        if (source instanceof MimeMessageImpl) {
116            MimeMessageImpl in = (MimeMessageImpl) source;
117
118            setReceivedDate(in.getReceivedDate());
119            setTimeZone(in.getTimeZone());
120        }
121    }
122
123    {
124        if (! (headers instanceof InternetHeadersImpl)) {
125            headers = new InternetHeadersImpl(headers);
126        }
127    }

240    /**
241     * Method to get the
242     * {@value ball.mail.internet.InternetHeadersImpl#REFERENCES}
243     * header value.
244     *
245     * @return  The
246     *          {@value ball.mail.internet.InternetHeadersImpl#REFERENCES}
247     *          header value (or {@code null} if none is defined).
248     *
249     * @throws  MessagingException
250     *                          If the
251     *                          {@value ball.mail.internet.InternetHeadersImpl#REFERENCES}
252     *                          header value cannot be retrieved for any
253     *                          reason.
254     */
255    public LinkedHashSet<String> getReferences() throws MessagingException {
256        LinkedHashSet<String> set = null;
257        String value = getHeader(REFERENCES, SPACE);
258
259        if (value != null) {
260            value = MimeUtility.unfold(value);
261
262            try {
263                value = MimeUtility.decodeText(value);
264            } catch (UnsupportedEncodingException exception) {
265            }
266
267            set = new LinkedHashSet<>();
268
269            for (String string : value.trim().split("[\\p{Blank}]+")) {
270                if (! string.equals(EMPTY)) {
271                    try {
272                        set.add(MESSAGE_ID_FORMAT.parse(string));
273                    } catch (ParseException exception) {
274                    }
275                }
276            }
277        }
278
279        return set;
280    }

282    /**
283     * Method to set the
284     * {@value ball.mail.internet.InternetHeadersImpl#REFERENCES}
285     * header value.
286     *
287     * @param   iterable        The
288     *                          {@value ball.mail.internet.InternetHeadersImpl#REFERENCES}
289     *                          header value.
290     *
291     * @throws  MessagingException
292     *                          If the
293     *                          {@value ball.mail.internet.InternetHeadersImpl#REFERENCES}
294     *                          header value cannot be updated for any
295     *                          reason.
296     */
297    public void setReferences(Iterable<String> iterable)
298            throws MessagingException {
299        StringBuilder buffer = new StringBuilder();
300
301        for (String string : iterable) {
302            if (buffer.length() > 0) {
303                buffer.append(SPACE);
304            }
305
306            synchronized (MESSAGE_ID_FORMAT) {
307                buffer.append(MESSAGE_ID_FORMAT.format(string));
308            }
309        }
310
311        String value = buffer.toString().trim();
312
313        if (value.length() > 0) {
314            setHeader(REFERENCES,
315                      MimeUtility.fold(REFERENCES.length() + 2, value));
316        } else {
317            removeHeader(REFERENCES);
318        }
319    }
