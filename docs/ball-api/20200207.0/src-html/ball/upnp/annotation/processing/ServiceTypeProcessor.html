023import ball.annotation.ServiceProviderFor;
024import ball.annotation.processing.AbstractAnnotationProcessor;
025import ball.annotation.processing.For;
026import ball.upnp.AnnotatedService;
027import ball.upnp.annotation.ServiceType;
028import java.net.URI;
029import javax.annotation.processing.Processor;
030import javax.annotation.processing.RoundEnvironment;
031import javax.lang.model.element.Element;
032import javax.lang.model.element.TypeElement;
033import lombok.NoArgsConstructor;
034import lombok.ToString;
035
036import static javax.tools.Diagnostic.Kind.ERROR;
037import static org.apache.commons.lang3.StringUtils.isNotEmpty;
038
039/**
040 * {@link ServiceType} annotation {@link Processor}.
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 * @version $Revision: 5285 $
044 */
045@ServiceProviderFor({ Processor.class })
046@For({ ServiceType.class })
047@NoArgsConstructor @ToString
048public class ServiceTypeProcessor extends AbstractAnnotationProcessor {
049    @Override
050    protected void process(RoundEnvironment env,
051                           TypeElement annotation,
052                           Element element) throws Exception {
053        String string = element.getAnnotation(ServiceType.class).value();
054
055        if (isNotEmpty(string)) {
056            if (isAssignable(element.asType(), AnnotatedService.class)) {
057                try {
058                    new URI(string);
059                } catch (Exception exception) {
060                    print(ERROR,
061                          element,
062                          element.getKind() + " annotated with "
063                          + AT + annotation.getSimpleName()
064                          + " but cannot convert `" + String.valueOf(string)
065                          + "' to " + URI.class.getName());
066                }
067            } else {
068                print(ERROR,
069                      element,
070                      element.getKind() + " annotated with "
071                      + AT + annotation.getSimpleName()
072                      + " but does not implement "
073                      + AnnotatedService.class.getName());
074            }
075        } else {
076            print(ERROR,
077                  element,
078                  element.getKind() + " annotated with "
079                  + AT + annotation.getSimpleName()
080                  + " but does not specify value()");
081        }
082    }
083}
