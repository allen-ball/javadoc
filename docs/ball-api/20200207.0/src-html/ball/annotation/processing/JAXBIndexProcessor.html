023import ball.annotation.ServiceProviderFor;
024import java.io.File;
025import java.io.IOException;
026import java.lang.annotation.Annotation;
027import java.nio.file.Files;
028import java.util.ArrayList;
029import java.util.Map;
030import java.util.Set;
031import java.util.TreeMap;
032import java.util.TreeSet;
033import java.util.regex.Pattern;
034import javax.annotation.processing.Processor;
035import javax.annotation.processing.RoundEnvironment;
036import javax.lang.model.element.Element;
037import javax.lang.model.element.PackageElement;
038import javax.lang.model.element.TypeElement;
039import javax.tools.FileObject;
040import javax.xml.bind.annotation.XmlRootElement;
041import javax.xml.bind.annotation.XmlType;
042import lombok.NoArgsConstructor;
043import lombok.ToString;
044
045import static java.lang.reflect.Modifier.isAbstract;
046import static javax.tools.Diagnostic.Kind.ERROR;
047import static javax.tools.StandardLocation.CLASS_OUTPUT;
048
049/**
050 * {@link Processor} implementation to generate {@code jaxb.index} files
051 * from {@link Class}es annotated with JAXB annotations.
052 *
053 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
054 * @version $Revision: 5285 $
055 */
056@ServiceProviderFor({ Processor.class })
057@For({ XmlRootElement.class, XmlType.class })
058@NoArgsConstructor @ToString
059public class JAXBIndexProcessor extends AbstractAnnotationProcessor
060                                implements ClassFileProcessor {
061    private static final String JAXB_INDEX = "jaxb.index";
062
063    private MapImpl map = new MapImpl();
064
065    @Override
066    public boolean process(Set<? extends TypeElement> annotations,
067                           RoundEnvironment roundEnv) {
068        boolean result = true;
069
070        try {
071            if (! roundEnv.errorRaised()) {
072                result &= super.process(annotations, roundEnv);
073
074                if (roundEnv.processingOver()) {
075                    for (Map.Entry<String,Set<String>> entry :
076                             map.entrySet()) {
077                        FileObject file =
078                            filer.createResource(CLASS_OUTPUT,
079                                                 entry.getKey(), JAXB_INDEX);
080                        ArrayList<String> lines = new ArrayList<>();
081
082                        lines.add("# " + JAXB_INDEX);
083                        lines.addAll(entry.getValue());
084
085                        Files.createDirectories(toPath(file).getParent());
086                        Files.write(toPath(file), lines, CHARSET);
087                    }
088                }
089            }
090        } catch (Exception exception) {
091            print(ERROR, null, exception);
092        }
093
094        return result;
095    }
096
097    @Override
098    protected void process(RoundEnvironment env,
099                           TypeElement annotation,
100                           Element element) throws Exception {
101        switch (element.getKind()) {
102        case CLASS:
103            map.add((TypeElement) element);
104            break;
105
106        default:
107            break;
108        }
109    }
110
111    @Override
112    public void process(Set<Class<?>> set, File destdir) throws IOException {
113        for (Class<?> type : set) {
114            if (! isAbstract(type.getModifiers())) {
115                for (Class<? extends Annotation> annotation :
116                         getSupportedAnnotationTypeList()) {
117                    if (type.getAnnotation(annotation) != null) {
118                        map.add(type);
119                    }
120                }
121            }
122        }
123
124        for (Map.Entry<String,Set<String>> entry : map.entrySet()) {
125            String pkg = entry.getKey();
126            File file = new File(new File(destdir, asPath(pkg)), JAXB_INDEX);
127
128            Files.createDirectories(file.toPath().getParent());
129
130            ArrayList<String> lines = new ArrayList<>();
131
132            lines.add("# " + JAXB_INDEX);
133            lines.addAll(entry.getValue());
134
135            Files.write(file.toPath(), lines, CHARSET);
136        }
137    }
138
139    @NoArgsConstructor
140    private class MapImpl extends TreeMap<String,Set<String>> {
141        private static final long serialVersionUID = 5693261055822027911L;
142
143        public boolean add(Class<?> type) {
144            return add(type.getPackage().getName(), type.getCanonicalName());
145        }
146
147        public boolean add(TypeElement type) {
148            return add(getPackageElementFor(type)
149                       .getQualifiedName().toString(),
150                       type.getQualifiedName().toString());
151        }
152
153        private boolean add(String pkg, String type) {
154            if (type.startsWith(pkg)) {
155                type = type.substring(pkg.length());
156            }
157
158            if (type.startsWith(DOT)) {
159                type = type.substring(DOT.length());
160            }
161
162            return get(pkg).add(type);
163        }
164
165        @Override
166        public Set<String> get(Object key) {
167            if (! super.containsKey(key)) {
168                super.put((String) key, new TreeSet<>());
169            }
170
171            return super.get(key);
172        }
173    }
174}
