023import ball.annotation.CompileTimeCheck;
024import ball.upnp.annotation.XmlNs;
025import java.util.regex.Matcher;
026import java.util.regex.Pattern;
027import lombok.AllArgsConstructor;
028import lombok.Data;
029import org.springframework.core.annotation.AnnotationUtils;
030
031import static java.util.Objects.requireNonNull;
032import static lombok.AccessLevel.PRIVATE;
033
034/**
035 * {@link Description} support for {@link XmlNs} annotation.
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 * @version $Revision: 6313 $
039 */
040public interface Description {
041
042    /**
043     * {@link Pattern} to parse {@link #getXmlns()} to generate a
044     * {@link SpecVersion}.  Provides "{@code major}" and  "{@code minor}"
045     * matching groups.
046     */
047    @CompileTimeCheck
048    public static final Pattern SPEC_VERSION_PATTERN =
049        Pattern.compile("(?i)^.+-(?<major>[0-9]+)-(?<minor>[0-9]+)$");
050
051    /**
052     * Method to get the {@link XmlNs}'s name.
053     *
054     * @return  The name.
055     */
056    default String getXmlns() {
057        XmlNs annotation =
058            AnnotationUtils.findAnnotation(getClass(), XmlNs.class);
059
060        return (annotation != null) ? annotation.value() : null;
061    }
062
063    /**
064     * Method to generate {@link SpecVersion} from {@link XmlNs}.
065     *
066     * @return  The {@link SpecVersion}.
067     */
068    default SpecVersion getSpecVersion() {
069        SpecVersion version = null;
070        Matcher matcher = SPEC_VERSION_PATTERN.matcher(getXmlns());
071
072        if (matcher.matches()) {
073            version =
074                new SpecVersion(matcher.group("major"),
075                                matcher.group("minor"));
076        }
077
078        return version;
079    }
080
081    /**
082     * See {@link #getSpecVersion()}.
083     *
084     * {@bean.info}
085     */
086    @AllArgsConstructor(access = PRIVATE) @Data
087    public static class SpecVersion {
088        private String major = null;
089        private String minor = null;
090    }
091}
