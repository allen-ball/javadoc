021import java.lang.annotation.Documented;
022import java.lang.annotation.Retention;
023import java.lang.annotation.Target;
024import lombok.NoArgsConstructor;
025import org.apache.tools.ant.Task;
026import org.apache.tools.ant.TaskConfigurationChecker;
027
028import static java.lang.annotation.ElementType.ANNOTATION_TYPE;
029import static java.lang.annotation.ElementType.FIELD;
030import static java.lang.annotation.ElementType.METHOD;
031import static java.lang.annotation.RetentionPolicy.RUNTIME;
032import static lombok.AccessLevel.PROTECTED;
033
034/**
035 * JSR303-inspired Ant Task attribute constraint annotation.
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 */
039@Documented
040@Retention(RUNTIME)
041@Target({ ANNOTATION_TYPE, FIELD, METHOD })
042public @interface AntTaskAttributeConstraint {
043    Class<? extends AntTaskAttributeConstraint.Checker> value();
044
045    /**
046     * {@link #value()} base class
047     */
048    @NoArgsConstructor(access = PROTECTED)
049    public static abstract class Checker {
050
051        /**
052         * Method to validate an attribute value.
053         *
054         * @param   task        The {@link Task} that owns and is validating
055         *                      the attribute.
056         * @param   checker     The {@link TaskConfigurationChecker} to
057         *                      report violations.
058         * @param   name        The name of the attribute.
059         * @param   value       The attribute value.
060         */
061        protected abstract void check(Task task, TaskConfigurationChecker checker, String name, Object value);
062    }
063}
