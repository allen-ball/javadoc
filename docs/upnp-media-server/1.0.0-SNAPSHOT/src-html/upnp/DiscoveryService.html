023import ball.upnp.ssdp.SSDPDiscoveryCache;
024import ball.upnp.ssdp.SSDPDiscoveryService;
025import ball.upnp.ssdp.SSDPMessage;
026import java.io.IOException;
027import javax.annotation.PostConstruct;
028import javax.annotation.PreDestroy;
029import javax.servlet.Filter;
030import javax.servlet.FilterChain;
031import javax.servlet.FilterConfig;
032import javax.servlet.ServletException;
033import javax.servlet.ServletRequest;
034import javax.servlet.ServletResponse;
035import javax.servlet.http.HttpServletResponse;
036import lombok.NoArgsConstructor;
037import lombok.extern.log4j.Log4j2;
038import org.springframework.beans.factory.annotation.Autowired;
039import org.springframework.beans.factory.annotation.Value;
040import org.springframework.boot.context.event.ApplicationReadyEvent;
041import org.springframework.context.event.EventListener;
042import org.springframework.stereotype.Service;
043
044import static java.util.concurrent.TimeUnit.SECONDS;
045
046/**
047 * SSDP discovery {@link Service}.  Implements step #1, "Discovery," in
048 * {@link.uri https://openconnectivity.org/upnp-specs/UPnP-arch-DeviceArchitecture-v2.0-20200417.pdf UPnP Device Architecture 2.0}.
049 *
050 * {@injected.fields}
051 *
052 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
053 * @version $Revision: 7215 $
054 */
055@Service
056@NoArgsConstructor @Log4j2
057public class DiscoveryService extends SSDPDiscoveryCache implements Filter {
058    private static final long serialVersionUID = 1529044646056002588L;
059
060    /** @serial */ @Value("${ssdp.product}") private String product = null;
061    /** @serial */ @Autowired private MediaServer device = null;
062    /** @serial */ private SSDPDiscoveryService service = null;
063
064    @EventListener({ ApplicationReadyEvent.class })
065    public void ready() throws Exception {
066        service =
067            new SSDPDiscoveryService(product)
068            .addListener(this)
069            .advertise(device, device.getMaxAge() / 2);
070    }
071
072    @Override
073    public void init(FilterConfig config) throws ServletException { }
074
075    @Override
076    public void doFilter(ServletRequest request,
077                         ServletResponse response,
078                         FilterChain chain) throws IOException,
079                                                   ServletException {
080        process((HttpServletResponse) response);
081
082        chain.doFilter(request, response);
083    }
084
085    private void process(HttpServletResponse response) {
086        if (service != null) {
087            response.setHeader(SSDPMessage.SERVER, service.getUserAgent());
088        }
089    }
090
091    @Override
092    public void destroy() {
093        try {
094            service.shutdown();
095            service.awaitTermination(2, SECONDS);
096        } catch (Exception exception) {
097            log.error("{}", exception.getMessage(), exception);
098        }
099    }
100}
