023import java.util.NoSuchElementException;
024import javax.annotation.PostConstruct;
025import javax.annotation.PreDestroy;
026import lombok.NoArgsConstructor;
027import lombok.ToString;
028import lombok.extern.log4j.Log4j2;
029import org.springframework.beans.factory.annotation.Autowired;
030import org.springframework.http.HttpStatus;
031import org.springframework.stereotype.Controller;
032import org.springframework.ui.Model;
033import org.springframework.web.bind.annotation.ExceptionHandler;
034import org.springframework.web.bind.annotation.ModelAttribute;
035import org.springframework.web.bind.annotation.PathVariable;
036import org.springframework.web.bind.annotation.RequestMapping;
037import org.springframework.web.bind.annotation.ResponseStatus;
038
039import static org.springframework.http.MediaType.TEXT_XML_VALUE;
040import static org.springframework.web.bind.annotation.RequestMethod.GET;
041
042/**
043 * {@link MediaServer} description {@link Controller}.  Implements step #2,
044 * "Description," in
045 * {@link.uri https://openconnectivity.org/upnp-specs/UPnP-arch-DeviceArchitecture-v2.0-20200417.pdf UPnP Device Architecture 2.0}.
046 *
047 * {@injected.fields}
048 * {@bean.info}
049 *
050 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
051 * @version $Revision: 7215 $
052 */
053@Controller
054@RequestMapping(method = { GET },
055                value = { "/" },
056                produces = { TEXT_XML_VALUE })
057@NoArgsConstructor @ToString @Log4j2
058public class DescriptionController {
059    protected static final String LOCATION = "description.xml";
060
061    @Autowired private MediaServer device = null;
062
063    @PostConstruct
064    public void init() { }
065
066    @PreDestroy
067    public void destroy() { }
068
069    @ModelAttribute("device")
070    public MediaServer device() { return device; }
071
072    @RequestMapping(value = { LOCATION })
073    public String description(Model model) { return device().getName(); }
074
075    @RequestMapping(value = { "scpd/{key}.xml" })
076    public String service(Model model, @PathVariable String key) {
077        var service = device().getServiceByName(key);
078
079        model.addAttribute("service", service);
080
081        return service.getName();
082    }
083
084    @ExceptionHandler({ NoSuchElementException.class })
085    @ResponseStatus(value = HttpStatus.NOT_FOUND, reason = "Resource not found")
086    public void handleNOT_FOUND() { }
087}
