053/**
054 * {@link.uri http://www.upnp.org/ UPnP} {@link MediaServer}
055 * {@link ball.upnp.Device} which implements {@link ContentDirectory} and
056 * {@link ConnectionManager} {@link ball.upnp.Service}s.
057 *
058 * {@bean.info}
059 *
060 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
061 */
062@DeviceType("urn:schemas-upnp-org:device:MediaServer:4")
063@Named @Singleton
064@Log4j2
065public class MediaServer extends AbstractDevice implements RootDevice {
066    private final ServerProperties properties;
067    private final ContentDirectory directory;
068    private final ConnectionManager manager;
069    private UUID uuid = null;
070    private URI location = URI.create(DescriptionController.LOCATION);
071
072    /**
073     * Sole constructor.
074     *
075     * @param   home            The {code ${application.home}}.
076     * @param   properties      The {@link DescriptionController}
077     *                          {@link ServerProperties}.
078     * @param   directory       The {@link ContentDirectory}.
079     * @param   manager         The {@link ConnectionManager}.
080     */
081    @Inject
082    public MediaServer(@Value("${application.home}") String home, ServerProperties properties, ContentDirectory directory, ConnectionManager manager) {
083        var path = Paths.get(home, getName(), "UUID");
084
085        try {
086            Files.createDirectories(path.getParent());
087
088            UUID uuid = null;
089
090            try {
091                var list = Files.readAllLines(path);
092
093                uuid = UUID.fromString(list.get(0));
094            } catch (Throwable throwable) {
095                uuid = UUID.randomUUID();
096
097                Files.write(path, singleton(uuid.toString()));
098            }
099
100            this.uuid = uuid;
101        } catch (Exception exception) {
102            throw new ExceptionInInitializerError(exception);
103        }
104
105        this.properties = requireNonNull(properties);
106        this.directory = requireNonNull(directory);
107        this.manager = requireNonNull(manager);
108    }
109
110    @PostConstruct
111    public void init() throws Exception {
112        log.info("{}", uuid);
113
114        getServiceList().stream().forEach(t -> t.setDevice(this));
115        getDeviceList().stream().forEach(t -> t.setDevice(this));
116
117        InetAddress localhost = null;
118
119        for (var ni : list(NetworkInterface.getNetworkInterfaces())) {
120            if (ni.isUp() && ni.supportsMulticast()) {
121                for (InterfaceAddress address : ni.getInterfaceAddresses()) {
122                    if (address.getAddress() instanceof Inet4Address && (! address.getAddress().isLoopbackAddress())) {
123                        localhost = address.getAddress();
124                        break;
125                    }
126                }
127            }
128        }
129
130        if (localhost == null) {
131            localhost = InetAddress.getLocalHost();
132        }
133
134        if (localhost == null) {
135            localhost = InetAddress.getLoopbackAddress();
136        }
137
138        var host = localhost.getHostAddress();
139
140        location =
141            new URI((properties.getSsl() != null) ? "https" : "http",
142                    null, host, properties.getPort(), "/", null, null)
143            .resolve(location);
144    }
145
146    @PreDestroy
147    public void destroy() { }
148
149    public String getName() { return getClass().getSimpleName(); }
150
151    public MediaServer.Service getServiceByName(String name) {
152        var service =
153            getServiceList().stream()
154            .filter(t -> name.equals(t.getName()))
155            .findFirst().get();
156
157        return service;
158    }
159
160    protected URI getSCPDURL(MediaServer.Service service) {
161        return URI.create(String.format("/scpd/%s.xml", service.getName()));
162    }
163
164    protected URI getControlURL(MediaServer.Service service) {
165        return URI.create(String.format("/control/%s", service.getName()));
166    }
167
168    protected URI getEventSubURL(MediaServer.Service service) {
169        return URI.create(String.format("/event/%s", service.getName()));
170    }
171
172    @Override
173    public UUID getUUID() { return uuid; }
174
175    @Override
176    public List<MediaServer.Service> getServiceList() {
177        return Stream.of(directory, manager).collect(toList());
178    }
179
180    @Override
181    public List<MediaServer.Device> getDeviceList() { return List.of(); }
182
183    @Override
184    public URI getLocation() { return location; }
185
186    @Override
187    public URI getPresentationURL() { return URI.create("/"); }
188
189    /**
190     * {@link MediaServer} {@link ball.upnp.Service}.
191     *
192     * {@bean.info}
193     */
194    @NoArgsConstructor(access = PROTECTED)
195    public static abstract class Service extends AbstractService {
196        @Getter @Setter
197        private MediaServer device = null;
198
199        public String getName() { return getClass().getSimpleName(); }
200
201        public int getConfigId() { return getDevice().getConfigId(); }
202
203        @Override
204        public URI getSCPDURL() { return device.getSCPDURL(this); }
205
206        @Override
207        public URI getControlURL() { return device.getControlURL(this); }
208
209        @Override
210        public URI getEventSubURL() { return device.getEventSubURL(this); }
211    }
212
213    /**
214     * {@link MediaServer} {@link ball.upnp.Device}.
215     *
216     * {@bean.info}
217     */
218    @NoArgsConstructor(access = PROTECTED)
219    public static abstract class Device extends AbstractDevice {
220        @Getter @Setter
221        private MediaServer device = null;
222    }
