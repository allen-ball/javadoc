080/**
081 * {@link Artifact} to {@link LicenseSet}
082 * ({@link org.spdx.rdfparser.license.License}) {@link java.util.Map}
083 * implementation.  The {@link #get(Object)} method transparently calculates
084 * and caches any value.
085 *
086 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
087 * @version $Revision: 8233 $
088 */
089@Named @Singleton
090@Slf4j
091public class ArtifactLicenseCatalog extends TreeMap<Artifact,AnyLicenseInfo> {
092    private static final long serialVersionUID = -7887839577334232433L;
093
094    private static final String CATALOG = "artifact-license-catalog.xml";
095
096    private static final Predicate<String> INCLUDE =
097        Pattern.compile("(?i)^(.*/|)(LICENSE([.][^/]+)?|about.html)$")
098        .asPredicate();
099    private static final Predicate<String> EXCLUDE =
100        Pattern.compile("(?i)^.*[.]class$")
101        .asPredicate().negate();
102
103    private static final Comparator<? super Boolean> TRUTH =
104        (t, u) -> Objects.equals(t, u) ? 0 : (t ? -1 : 1);
105    private static Comparator<? super AnyLicenseInfo> SIEVE =
106        Comparator
107        .<AnyLicenseInfo,Boolean>
108        comparing(t -> (! (t instanceof URLLicenseInfo)), TRUTH)
109        .thenComparing(t -> (! (t instanceof TextLicenseInfo)), TRUTH)
110        .thenComparing(t -> (! (t instanceof ExtractedLicenseInfo)), TRUTH)
111        .thenComparing(LicenseUtilityMethods::isFullySpdxListed, TRUTH)
112        .thenComparing(t -> (t instanceof LicenseSet), TRUTH)
113        .thenComparing(LicenseUtilityMethods::countOf,
114                       Comparator.<Integer>reverseOrder())
115        .thenComparing(t -> (t instanceof WithExceptionOperator), TRUTH)
116        .thenComparing(t -> (t instanceof OrLaterOperator), TRUTH)
117        .thenComparing(LicenseUtilityMethods::isPartiallySpdxListed, TRUTH);

121    /** @serial */ private final LicenseMap map;
122    /** @serial */ private final LicenseResolver resolver;
123    /** @serial */ private final File file;
124    /** @serial */ private final Properties defaults = new Properties();
125    /** @serial */ private final Properties catalog = new Properties(defaults);

127    /**
128     * Sole constructor.
129     *
130     * @param   session         The injected {@link MavenSession}.
131     * @param   cache           The injected {@link ArtifactModelCache}.
132     * @param   map             The injected {@link LicenseMap}.
133     * @param   resolver        The injected {@link LicenseResolver}.
134     */
135    @Inject
136    public ArtifactLicenseCatalog(MavenSession session,
137                                  ArtifactModelCache cache,
138                                  LicenseMap map, LicenseResolver resolver) {
139        super(Comparator.comparing(ArtifactUtils::key,
140                                   String.CASE_INSENSITIVE_ORDER));
141
142        this.session = Objects.requireNonNull(session);
143        this.cache = Objects.requireNonNull(cache);
144        this.map = Objects.requireNonNull(map);
145        this.resolver = Objects.requireNonNull(resolver);
146        this.file =
147            new File(session.getLocalRepository().getBasedir(), CATALOG);
148    }
149
150    protected void load() {
151        try (InputStream in = getClass().getResourceAsStream(CATALOG)) {
152            if (in != null) {
153                defaults.loadFromXML(in);
154            }
155        } catch (IOException exception) {
156        }
157
158        if (file.exists()) {
159            try (FileInputStream in = new FileInputStream(file)) {
160                catalog.loadFromXML(in);
161            } catch (IOException exception) {
162                log.error("Cannot read {}", file);
163            }
164        }
165
166        for (String key : catalog.stringPropertyNames()) {
167            try {
168                put(new KeyArtifact(key),
169                    resolver.parseLicenseString(catalog.getProperty(key)));
170            } catch (Exception exception) {
171                log.error("{}: {}", key, exception.getMessage(), exception);
172            }
173        }
174    }
175
176    protected void flush() {
177        boolean dirty = (! file.exists());
178
179        for (Map.Entry<Artifact,AnyLicenseInfo> entry : entrySet()) {
180            AnyLicenseInfo license = entry.getValue();
181
182            if (isFullySpdxListed(license)) {
183                String key = ArtifactUtils.key(entry.getKey());
184                String value = license.toString();
185
186                dirty |= (! Objects.equals(value, catalog.put(key, value)));
187            }
188        }
189
190        if (dirty) {
191            try (FileOutputStream out = new FileOutputStream(file)) {
192                catalog.storeToXML(out, file.getName());
193            } catch (IOException exception) {
194                log.warn("Cannot write {}", file);
195            }
196        }
197    }
198
199    @PostConstruct
200    public void init() {
201        load();
202    }
203
204    @PreDestroy
205    public void destroy() {
206        flush();
207        log.debug("{}.size() = {}", getClass().getSimpleName(), size());
208    }

210    @Override
211    public AnyLicenseInfo get(Object key) {
212        AnyLicenseInfo value = super.get(key);
213
214        if (value == null) {
215            value = compute((Artifact) key);
216
217            put((Artifact) key, value);
218        }
219
220        return value;
221    }
222
223    private AnyLicenseInfo compute(Artifact artifact) {
224        URL url = toURL(artifact);
225        /*
226         * Licenses specified in the Manifest Bundle-License
227         * or found in the Artifact
228         */
229        List<AnyLicenseInfo> bundle = Collections.emptyList();
230        List<AnyLicenseInfo> scanned = Collections.emptyList();
231
232        try {
233            JarFile jar =
234                ((JarURLConnection) url.openConnection()).getJarFile();
235            Pattern pattern = Pattern.compile("((?<id>.+);link=)?(?<url>.*)");
236
237            bundle =
238                Stream.of(jar.getManifest())
239                .filter(Objects::nonNull)
240                .map(t -> t.getMainAttributes().getValue("Bundle-License"))
241                .filter(StringUtils::isNotBlank)
242                .flatMap(t -> Stream.of(t.split(",")))
243                .map(t -> t.trim())
244                .distinct()
245                .map(pattern::matcher)
246                .filter(Matcher::matches)
247                .map(t -> parse(t.group("id"), resolve(url, t.group("url"))))
248                .collect(toList());
249
250            scanned =
251                jar.stream()
252                .map(JarEntry::getName)
253                .filter(INCLUDE)
254                .filter(EXCLUDE)
255                .map(t -> parse(t, resolve(url, t)))
256                .filter(t -> (! (t instanceof URLLicenseInfo)))
257                .collect(toList());
258        } catch (ZipException exception) {
259        } catch (IOException exception) {
260            log.debug("{}: {}", artifact, exception.getMessage(), exception);
261        }
262
263        Map<String,AnyLicenseInfo> found =
264            scanned.stream()
265            .filter(t -> LicenseUtilityMethods.countOf(t) > 0)
266            .filter(LicenseUtilityMethods::isFullySpdxListed)
267            .collect(toMap(k -> k.toString(), v -> v, (t, u) -> t));
268        /*
269         * Licenses specified in the Artifact's POM
270         */
271        List<AnyLicenseInfo> pom = Collections.emptyList();
272        Model model = cache.get(artifact);
273
274        if (model != null) {
275            pom =
276                Stream.of(model.getLicenses())
277                .filter(Objects::nonNull)
278                .flatMap(List::stream)
279                .distinct()
280                .map(t -> parse((t.getName() != null) ? t.getName().replaceAll(",", "") : null,
281                                resolve(toURL(model.getUrl()), t.getUrl())))
282                .collect(toList());
283        }
284
285        if (bundle.isEmpty() && pom.isEmpty() && found.isEmpty()) {
286            log.warn("{}: No license(s) specified or found", artifact);
287        }
288
289        List<AnyLicenseInfo> licenses = bundle.stream().collect(toList());
290
291        if (! isFullySpecified(licenses)) {
292            if (licenses.isEmpty() || isFullySpecified(pom)) {
293                licenses.clear();
294                licenses.addAll(pom);
295            }
296        }
297
298        if ((! isFullySpecified(licenses)) && found.size() >= licenses.size()) {
299            licenses.clear();
300            licenses.addAll(found.values());
301        }
302
303        if ((! licenses.isEmpty()) && (! isFullySpecified(licenses))) {
304            log.debug("------------------------------------------------------------");
305            log.debug("{}", ArtifactUtils.key(artifact));
306            log.debug("{}", url);
307            log.debug("      Bundle: {}", bundle);
308            log.debug("         POM: {}", pom);
309            log.debug("     Scanned: {}", scanned);
310            log.debug("       Found: {}", found.values());
311            log.debug("  License(s): {}", licenses);
312            log.debug("------------------------------------------------------------");
313        }
314
315        return resolver.toLicense(licenses);
316    }

318    private URL toURL(Artifact artifact) {
319        URL url = null;
320
321        try {
322            url =
323                new URI("jar",
324                        artifact.getFile().toURI().toASCIIString() + "!/",
325                        null)
326                .toURL();
327        } catch(URISyntaxException | MalformedURLException exception) {
328            log.debug("{}: {}", artifact, exception.getMessage(), exception);
329            throw new IllegalStateException(exception);
330        }
331
332        return url;
333    }
334
335    private URL toURL(String string) {
336        URL url = null;
337
338        if (isNotBlank(string)) {
339            try {
340                url = new URI(string).toURL();
341            } catch(URISyntaxException | MalformedURLException exception) {
342                log.debug("{}: {}", string, exception.getMessage(), exception);
343            }
344        }
345
346        return url;
347    }

349    private String[] resolve(URL root, String urls) {
350        ArrayList<String> list = new ArrayList<>();
351
352        if (isNotBlank(urls)) {
353            for (String url : urls.split("[,\\p{Space}]+")) {
354                if (isNotBlank(url)) {
355                    String value = url;
356
357                    if (root != null && (! isAbsolute(value))) {
358                        String string = root.toString();
359
360                        if (! string.endsWith("/")) {
361                            string += "/";
362                        }
363
364                        value = string + value;
365                    }
366
367                    list.add(value);
368                }
369            }
370        }
371
372        return list.toArray(new String[] { });
373    }
374
375    private boolean isAbsolute(String url) {
376        boolean isAbsolute = false;
377
378        try {
379            isAbsolute = new URI(url).isAbsolute();
380        } catch (Exception exception) {
381        }
382
383        return isAbsolute;
384    }

386    private AnyLicenseInfo parse(String id, String... urls) {
387        AnyLicenseInfo option0 = isNotBlank(id) ? map.get(id) : null;
388        AnyLicenseInfo option1 = null;
389
390        try {
391            option1 = isNotBlank(id) ? resolver.parseLicenseString(id) : null;
392        } catch (Exception exception) {
393        }
394
395        AnyLicenseInfo option2 =
396            new URLLicenseInfo(isNotBlank(id) ? id : urls[0], urls);
397        AnyLicenseInfo[] options =
398            Stream.of(option0, option1, option2)
399            .filter(Objects::nonNull)
400            .map(t -> resolver.parse(t))
401            .toArray(AnyLicenseInfo[]::new);
402
403        Arrays.sort(options, SIEVE);
404
405        return (options.length > 0) ? options[0] : null;
406    }
407
408    private int countOf(Collection<? extends AnyLicenseInfo> collection) {
409        return collection.stream().mapToInt(LicenseUtilityMethods::countOf).sum();
410    }
411
412    private boolean isFullySpecified(Collection<AnyLicenseInfo> collection) {
413        return ((! collection.isEmpty())
414                && collection.stream().allMatch(t -> isFullySpdxListed(t)));
415    }

418        public KeyArtifact(String gav) { this(gav.split(":", 3)); }
419
420        private KeyArtifact(String[] gav) {
421            super(gav[0], gav[1], gav[2], EMPTY, "pom", EMPTY, null);
422        }
423    }
