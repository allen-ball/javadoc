077/**
078 * {@link org.apache.maven.plugin.Mojo} to generate LICENSE and DEPENDENCIES
079 * resources.
080 *
081 * {@maven.plugin.fields}
082 *
083 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
084 * @version $Revision: 8235 $
085 */
086@Mojo(name = "generate-license-resources",
087      configurator = "license-mojo-component-configurator",
088      requiresDependencyResolution = TEST,
089      defaultPhase = GENERATE_RESOURCES, requiresProject = true)
090@NoArgsConstructor @ToString @Slf4j
091public class GenerateLicenseResourcesMojo extends AbstractLicenseMojo {
092    private static final String DEPENDENCIES = "DEPENDENCIES";
093
094    private static final String COMPILE = "compile";
095    private static final String PROVIDED = "provided";
096    private static final String RUNTIME = "runtime";
097    private static final String SYSTEM = "system";
098    private static final String TEST = "test";
099    private static final String INCLUDE_SCOPE = COMPILE + "," + RUNTIME;
100    private static final String EXCLUDE_SCOPE = EMPTY;
101
102    private static final TreeSet<String> ARCHIVE_PACKAGING =
103        Stream.of("jar", "maven-plugin", "ejb", "war", "ear", "rar")
104        .collect(toCollection(() -> new TreeSet<>(String.CASE_INSENSITIVE_ORDER)));
105
106    private static final Comparator<AnyLicenseInfo> LICENSE_ORDER =
107        Comparator
108        .<AnyLicenseInfo>comparingInt(t -> LicenseUtilityMethods.countOf(t))
109        .reversed()
110        .thenComparing(t -> toString(t))
111        .thenComparingInt(t -> (t instanceof OrLaterOperator) ? -1 : 1)
112        .thenComparingInt(t -> (t instanceof WithExceptionOperator) ? -1 : 1);
113    private static final Comparator<Model> MODEL_ORDER =
114        Comparator
115        .<Model,String>comparing(t -> Objects.toString(t.getName(), EMPTY))
116        .thenComparing(t -> Objects.toString(t.getUrl(), EMPTY))
117        .thenComparingInt(t -> isBlank(t.getName()) ? Objects.hashCode(t) : 0);
118
119    @Parameter(defaultValue = "${project.build.outputDirectory}",
120               property = "license.resources.directory")
121    private File directory = null;
122
123    @Parameter(defaultValue = "META-INF",
124               property = "license.resources.subdirectory")
125    private String subdirectory = null;
126
127    @Parameter(defaultValue = INCLUDE_SCOPE, property = "license.includeScope")
128    private String includeScope = INCLUDE_SCOPE;
129
130    @Parameter(defaultValue = EXCLUDE_SCOPE, property = "license.excludeScope")
131    private String excludeScope = EXCLUDE_SCOPE;
132
133    @Parameter(required = false)
134    private Selection[] selections = null;
135
136    @Parameter(defaultValue = "true")
137    private boolean skipIfEmpty = true;
138
139    @Inject private MavenProject project = null;
140    @Inject private ArtifactLicenseCatalog catalog = null;
141    @Inject private ArtifactModelCache cache = null;
142    @Inject private ExecutorServiceImpl executor = null;
143
144    @Override
145    public void execute() throws MojoExecutionException, MojoFailureException {
146        try {
147            if (! isSkip()) {
148                String packaging = project.getPackaging();
149
150                if (ARCHIVE_PACKAGING.contains(packaging)) {
151                    if (directory.exists() && (! directory.isDirectory())) {
152                        throw new FileAlreadyExistsException(directory.toString(),
153                                                             null,
154                                                             "Is not a directory");
155                    }
156
157                    Path parent = directory.toPath();
158
159                    if (subdirectory != null && (! subdirectory.isEmpty())) {
160                        Path path = Paths.get(subdirectory);
161
162                        if (path.isAbsolute()) {
163                            throw new IllegalArgumentException(path + " is not relative");
164                        }
165
166                        parent = parent.resolve(path);
167                    }
168                    /*
169                     * Populate the caches.
170                     */
171                    Set<String> scope = getScope();
172                    List<Callable<AnyLicenseInfo>> tasks =
173                        project.getArtifacts()
174                        .stream()
175                        .filter(t -> scope.contains(t.getScope()))
176                        .<Callable<AnyLicenseInfo>>map(t -> (() -> getLicense(t)))
177                        .collect(toList());
178
179                    executor.invokeAll(tasks);
180                    /*
181                     * Get the records for the report.
182                     */
183                    List<Tuple> tuples =
184                        project.getArtifacts()
185                        .stream()
186                        .filter(t -> scope.contains(t.getScope()))
187                        .map(t -> new Tuple(getLicense(t), cache.get(t), t))
188                        .collect(toList());
189                    /*
190                     * LICENSE
191                     */
192                    generateLicense(parent);
193                    /*
194                     * DEPENDENCIES
195                     *
196                     * Report sort order:
197                     *      Licenses (LICENSE_ORDER)
198                     *      Model[name, url] (same if name is not blank)
199                     *      Artifacts (ArtifactModelCache.ORDER)
200                     */
201                    TreeMap<AnyLicenseInfo,Map<Model,List<Tuple>>> report =
202                        tuples.stream()
203                        .collect(groupingBy(Tuple::getLicense,
204                                            () -> new TreeMap<>(LICENSE_ORDER),
205                                            groupingBy(Tuple::getModel,
206                                                       () -> new TreeMap<>(MODEL_ORDER),
207                                                       toList())));
208
209                    warnIfExtractedLicenseInfo(report.keySet().stream());
210
211                    if ((! report.isEmpty()) || (! skipIfEmpty)) {
212                        generateReport(report, parent);
213                    } else {
214                        log.info("Skipping empty {} resource generation",
215                                 DEPENDENCIES);
216                    }
217                } else {
218                    log.warn("Skipping for '{}' packaging", packaging);
219                }
220            } else {
221                log.info("Skipping {} and {} resource generation",
222                         getFile().getName(), DEPENDENCIES);
223            }
224        } catch (IOException exception) {
225            fail(exception.getMessage(), exception);
226        } catch (Throwable throwable) {
227            log.error("{}", throwable.getMessage(), throwable);
228
229            if (throwable instanceof MojoExecutionException) {
230                throw (MojoExecutionException) throwable;
231            } else if (throwable instanceof MojoFailureException) {
232                throw (MojoFailureException) throwable;
233            } else {
234                throw new MojoExecutionException(throwable.getMessage(),
235                                                 throwable);
236            }
237        } finally {
238            catalog.flush();
239        }
240    }
241
242    private Set<String> getScope() {
243        Pattern pattern = Pattern.compile("[\\p{Punct}\\p{Space}]+");
244        TreeSet<String> scope =
245            pattern.splitAsStream(includeScope)
246            .filter(StringUtils::isNotBlank)
247            .collect(toCollection(() -> new TreeSet<>(String.CASE_INSENSITIVE_ORDER)));
248
249        pattern.splitAsStream(excludeScope)
250            .filter(StringUtils::isNotBlank)
251            .forEach(t -> scope.remove(t));
252
253        if (scope.isEmpty()) {
254            log.warn("Specified scope is empty");
255            log.debug("    includeScope = {}", includeScope);
256            log.debug("    excludeScope = {}", excludeScope);
257        }
258
259        return scope;
260    }
261
262    private AnyLicenseInfo getLicense(Artifact artifact) {
263        AnyLicenseInfo value = null;
264
265        if (value == null) {
266            if (selections != null) {
267                for (Selection selection : selections) {
268                    if (selection.include(artifact)) {
269                        value = selection.getLicense();
270                        break;
271                    }
272                }
273            }
274        }
275
276        if (value == null) {
277            value = catalog.get(artifact);
278        }
279
280        return value;
281    }
282
283    private void generateLicense(Path parent) throws Exception {
284        File file = getFile();
285
286        if (file.exists() && file.isDirectory()) {
287            throw new FileAlreadyExistsException(file.toString(),
288                                                 null,
289                                                 "Is not a file or link");
290        }
291
292        Path source = file.toPath();
293        Path target = parent.resolve(source.getFileName());
294
295        Files.createDirectories(parent);
296        Files.copy(file.toPath(), target, REPLACE_EXISTING);
297        Files.setLastModifiedTime(target, Files.getLastModifiedTime(source));
298    }
299
300    private void generateReport(TreeMap<AnyLicenseInfo,Map<Model,List<Tuple>>> report,
301                                Path parent) throws Exception {
302        Path target = parent.resolve(DEPENDENCIES);
303
304        try (PrintWriter out =
305                 new PrintWriter(Files.newBufferedWriter(target,
306                                                         CREATE, WRITE,
307                                                         TRUNCATE_EXISTING))) {
308            String boundary = String.join(EMPTY, Collections.nCopies(78, "-"));
309
310            out.println(boundary);
311            out.println(target.getFileName() + " " + project.getArtifact());
312            out.println(boundary);
313
314            for (Map.Entry<AnyLicenseInfo,Map<Model,List<Tuple>>> section :
315                     report.entrySet()) {
316                out.println(boundary);
317                out.println(toString(section.getKey()));
318                out.println(boundary);
319
320                boolean first = true;
321
322                for (Map.Entry<Model,List<Tuple>> group :
323                         section.getValue().entrySet()) {
324                    if (first) {
325                        first = false;
326                    } else {
327                        out.println();
328                    }
329
330                    if (isNotBlank(group.getKey().getName())) {
331                        out.println(group.getKey().getName());
332                    }
333
334                    for (Tuple tuple : group.getValue()) {
335                        out.println(tuple.getArtifact());
336                    }
337
338                    if (isNotBlank(group.getKey().getUrl())) {
339                        out.println(group.getKey().getUrl());
340                    }
341                }
342
343                out.println(boundary);
344            }
345        }
346    }
347
348    private static String toString(AnyLicenseInfo license) {
349        String string = null;
350
351        if (license instanceof LicenseSet) {
352            AnyLicenseInfo[] members = ((LicenseSet) license).getMembers();
353
354            Arrays.sort(members, Comparator.comparing(Objects::toString));
355
356            string =
357                Stream.of(members)
358                .map(Objects::toString)
359                .collect(joining((license instanceof DisjunctiveLicenseSet) ? " OR " : " AND "));
360        } else {
361            string = Objects.toString(license, EMPTY);
362        }
363
364        return string;
365    }
366
367    @AllArgsConstructor @Getter @ToString
368    private class Tuple {
369        private AnyLicenseInfo license = null;
370        private Model model = null;
371        private Artifact artifact = null;
372    }
