023import java.util.Objects;
024import lombok.NoArgsConstructor;
025import lombok.ToString;
026import org.codehaus.plexus.component.configurator.ComponentConfigurationException;
027import org.codehaus.plexus.component.configurator.converters.basic.AbstractBasicConverter;
028import org.spdx.rdfparser.license.AnyLicenseInfo;
029import org.spdx.rdfparser.license.LicenseInfoFactory;
030
031/**
032 * {@link AnyLicenseInfo}
033 * {@link org.codehaus.plexus.component.configurator.converters.ConfigurationConverter}.
034 *
035 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
036 * @version $Revision: 7215 $
037 */
038@NoArgsConstructor @ToString
039public class LicenseConverter extends AbstractBasicConverter {
040    private static final Class<?> TYPE = AnyLicenseInfo.class;
041
042    @Override
043    public boolean canConvert(Class<?> type) {
044        return Objects.equals(TYPE, type);
045    }
046
047    @Override
048    public Object fromString(String string) throws ComponentConfigurationException {
049        Object object = null;
050
051        try {
052            object =
053                TYPE.cast(LicenseInfoFactory.parseSPDXLicenseString(string));
054        } catch (Exception exception) {
055            String message =
056                "Unable to convert '" + string + "' to " + TYPE.getName();
057
058            throw new ComponentConfigurationException(message, exception);
059        }
060
061        return object;
062    }
063}
