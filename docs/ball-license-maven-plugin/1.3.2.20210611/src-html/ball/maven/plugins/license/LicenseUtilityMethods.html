023import java.util.Arrays;
024import java.util.Collection;
025import java.util.Collections;
026import java.util.Iterator;
027import java.util.Objects;
028import java.util.Spliterator;
029import java.util.Spliterators.AbstractSpliterator;
030import java.util.function.Consumer;
031import java.util.function.Function;
032import java.util.function.Supplier;
033import java.util.stream.Stream;
034import java.util.stream.StreamSupport;
035import lombok.NoArgsConstructor;
036import lombok.ToString;
037import org.spdx.rdfparser.license.AnyLicenseInfo;
038import org.spdx.rdfparser.license.License;
039import org.spdx.rdfparser.license.LicenseSet;
040import org.spdx.rdfparser.license.OrLaterOperator;
041import org.spdx.rdfparser.license.WithExceptionOperator;
042
043import static lombok.AccessLevel.PRIVATE;
044
045/**
046 * {@link AnyLicenseInfo} utility methods.
047 *
048 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
049 * @version $Revision: 7215 $
050 */
051@NoArgsConstructor(access = PRIVATE) @ToString
052public abstract class LicenseUtilityMethods {
053
054    /**
055     * Static method to create a {@link Stream} to walk the tree rooted at
056     * {@code root}.
057     *
058     * @param   root            The {@link AnyLicenseInfo} representing the
059     *                          root of the tree.
060     *
061     * @return  A {@link Stream}.
062     */
063    public static Stream<AnyLicenseInfo> walk(AnyLicenseInfo root) {
064        return walk(root, LicenseUtilityMethods::childrenOf);
065    }
066
067    /**
068     * Static method to get the children of a {@link AnyLicenseInfo}.
069     *
070     * @param   node            The {@link AnyLicenseInfo}.
071     *
072     * @return  A {@link Collection} of {@link AnyLicenseInfo} for a
073     *          {@link LicenseSet}, {@link OrLaterOperator}, or
074     *          {@link WithExceptionOperator} node; an empty
075     *          {@link Collection} otherwise.
076     */
077    public static Collection<AnyLicenseInfo> childrenOf(AnyLicenseInfo node) {
078        AnyLicenseInfo[] children = null;
079
080        if (node instanceof LicenseSet) {
081            children = ((LicenseSet) node).getMembers();
082        } else if (node instanceof OrLaterOperator) {
083            children =
084                new AnyLicenseInfo[] {
085                    ((OrLaterOperator) node).getLicense()
086                };
087        } else if (node instanceof WithExceptionOperator) {
088            children =
089                new AnyLicenseInfo[] {
090                    ((WithExceptionOperator) node).getLicense()
091                };
092        } else {
093            children = new AnyLicenseInfo[] { };
094        }
095
096        return Arrays.asList(children);
097    }
098
099    /**
100     * Returns the count of licenses represented by this
101     * {@link AnyLicenseInfo}: The number of members in a
102     * {@link LicenseSet} or {@code 1}.
103     *
104     * @param   license         The {@link AnyLicenseInfo}.
105     *
106     * @return  The license count.
107     */
108    public static int countOf(AnyLicenseInfo license) {
109        int count = 0;
110
111        if (license instanceof LicenseSet) {
112            count =
113                Stream.of(((LicenseSet) license).getMembers())
114                .mapToInt(t -> countOf(t))
115                .sum();
116        } else if (license != null) {
117            count = 1;
118        }
119
120        return count;
121    }
122
123    /**
124     * Method to determine if an {@link AnyLicenseInfo} is "empty" (defined
125     * as the license consists solely of empty containers).
126     *
127     * @param   license         The {@link AnyLicenseInfo}.
128     *
129     * @return  {@code true} if the {@code license} is empty; {@code false}
130     *          otherwise.
131     */
132    public static boolean isEmpty(AnyLicenseInfo license) {
133        return countOf(license) == 0;
134    }
135
136    /**
137     * Static method to test if an {@link AnyLicenseInfo} is a branch.
138     *
139     * @param   node            The {@link AnyLicenseInfo}.
140     *
141     * @return  {@code true} if a branch; {@code false} otherwise.
142     */
143    public static boolean isBranch(AnyLicenseInfo node) {
144        return (isLicenseSet(node) || isOperator(node));
145    }
146
147    /**
148     * Static method to test if an {@link AnyLicenseInfo} is a leaf.
149     *
150     * @param   node            The {@link AnyLicenseInfo}.
151     *
152     * @return  {@code true} if a leaf; {@code false} otherwise.
153     */
154    public static boolean isLeaf(AnyLicenseInfo node) {
155        return (! isBranch(node));
156    }
157
158    /**
159     * Static method to test if an {@link AnyLicenseInfo}
160     * is a {@link LicenseSet}.
161     *
162     * @param   node            The {@link AnyLicenseInfo}.
163     *
164     * @return  {@code true} if a {@code node} is a {@link LicenseSet};
165     *          {@code false} otherwise.
166     */
167    public static boolean isLicenseSet(AnyLicenseInfo node) {
168        return (node instanceof LicenseSet);
169    }
170
171    /**
172     * Static method to test if an {@link AnyLicenseInfo} is an operator
173     * node ({@link OrLaterOperator} or {@link WithExceptionOperator}).
174     *
175     * @param   node            The {@link AnyLicenseInfo}.
176     *
177     * @return  {@code true} if an operator node; {@code false} otherwise.
178     */
179    public static boolean isOperator(AnyLicenseInfo node) {
180        return (node instanceof OrLaterOperator
181                || node instanceof WithExceptionOperator);
182    }
183
184    /**
185     * Method to test if an {@link AnyLicenseInfo} is fully specified by
186     * SPDX license(s).
187     *
188     * @param   license         The {@link AnyLicenseInfo}.
189     *
190     * @return  {@code true} if fully specified; {@code false} otherwise.
191     */
192    public static boolean isFullySpdxListed(AnyLicenseInfo license) {
193        boolean fullySpecified =
194             LicenseUtilityMethods.walk(license)
195            .filter(LicenseUtilityMethods::isLeaf)
196            .map(t -> (t instanceof License))
197            .reduce(Boolean::logicalAnd).orElse(false);
198
199        return fullySpecified;
200    }
201
202    /**
203     * Method to test if an {@link AnyLicenseInfo} is partially specified by
204     * SPDX license(s).
205     *
206     * @param   license         The {@link AnyLicenseInfo}.
207     *
208     * @return  {@code true} if partially specified; {@code false}
209     *          otherwise.
210     */
211    public static boolean isPartiallySpdxListed(AnyLicenseInfo license) {
212        boolean partiallySpecified =
213            LicenseUtilityMethods.walk(license)
214            .filter(LicenseUtilityMethods::isLeaf)
215            .map(t -> (t instanceof License))
216            .reduce(Boolean::logicalOr).orElse(false);
217
218        return partiallySpecified;
219    }
220
221    private static <T> Stream<T> walk(T root,
222                                      Function<? super T,Collection<? extends T>> childrenOf) {
223        Walker<T> walker =
224            new Walker<>(Collections.singleton(root), childrenOf);
225
226        return StreamSupport.stream(walker, false);
227    }
228
229    @ToString
230    private static class Walker<T> extends AbstractSpliterator<T> {
231        private final Stream<Supplier<Walker<T>>> stream;
232        private Iterator<Supplier<Walker<T>>> iterator = null;
233        private Spliterator<? extends T> spliterator = null;
234
235        private Walker(Collection<? extends T> nodes,
236                       Function<? super T,Collection<? extends T>> childrenOf) {
237            super(Long.MAX_VALUE, IMMUTABLE | NONNULL);
238
239            stream =
240                nodes.stream()
241                .filter(Objects::nonNull)
242                .map(childrenOf)
243                .filter(Objects::nonNull)
244                .map(t -> (() -> new Walker<T>(t, childrenOf)));
245            spliterator =
246                nodes.stream()
247                .filter(Objects::nonNull)
248                .spliterator();
249        }
250
251        @Override
252        public boolean tryAdvance(Consumer<? super T> consumer) {
253            boolean accepted = false;
254
255            if (spliterator != null) {
256                accepted = spliterator.tryAdvance(consumer);
257            }
258
259            if (! accepted) {
260                if (iterator == null) {
261                    iterator = stream.iterator();
262                }
263
264                if (iterator.hasNext()) {
265                    spliterator = iterator.next().get();
266                    accepted = tryAdvance(consumer);
267                }
268            }
269
270            return accepted;
271        }
272    }
273}
