078/**
079 * {@link org.apache.maven.plugin.Mojo} to generate LICENSE and DEPENDENCIES
080 * resources.
081 *
082 * {@maven.plugin.fields}
083 *
084 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
085 * @version $Revision: 6761 $
086 */
087@Mojo(name = "generate-license-resources",
088      configurator = "license-mojo-component-configurator",
089      requiresDependencyResolution = TEST,
090      defaultPhase = GENERATE_RESOURCES, requiresProject = true)
091@NoArgsConstructor @ToString @Slf4j
092public class GenerateLicenseResourcesMojo extends AbstractLicenseMojo {
093    private static final String DEPENDENCIES = "DEPENDENCIES";
094
095    private static final String COMPILE = "compile";
096    private static final String PROVIDED = "provided";
097    private static final String RUNTIME = "runtime";
098    private static final String SYSTEM = "system";
099    private static final String TEST = "test";
100    private static final String INCLUDE_SCOPE = COMPILE + "," + RUNTIME;
101    private static final String EXCLUDE_SCOPE = EMPTY;
102
103    private static final TreeSet<String> ARCHIVE_PACKAGING =
104        Stream.of("jar", "maven-plugin", "ejb", "war", "ear", "rar")
105        .collect(toCollection(() -> new TreeSet<>(String.CASE_INSENSITIVE_ORDER)));
106
107    private static final Comparator<AnyLicenseInfo> LICENSE_ORDER =
108        Comparator
109        .<AnyLicenseInfo>comparingInt(t -> LicenseUtilityMethods.countOf(t))
110        .reversed()
111        .thenComparing(t -> toString(t))
112        .thenComparingInt(t -> (t instanceof OrLaterOperator) ? -1 : 1)
113        .thenComparingInt(t -> (t instanceof WithExceptionOperator) ? -1 : 1);
114    private static final Comparator<Model> MODEL_ORDER =
115        Comparator
116        .<Model,String>comparing(t -> Objects.toString(t.getName(), EMPTY))
117        .thenComparing(t -> Objects.toString(t.getUrl(), EMPTY))
118        .thenComparingInt(t -> isBlank(t.getName()) ? Objects.hashCode(t) : 0);
119
120    @Parameter(defaultValue = "${project.build.outputDirectory}",
121               property = "license.resources.directory")
122    private File directory = null;
123
124    @Parameter(defaultValue = "META-INF",
125               property = "license.resources.subdirectory")
126    private String subdirectory = null;
127
128    @Parameter(defaultValue = INCLUDE_SCOPE, property = "license.includeScope")
129    private String includeScope = INCLUDE_SCOPE;
130
131    @Parameter(defaultValue = EXCLUDE_SCOPE, property = "license.excludeScope")
132    private String excludeScope = EXCLUDE_SCOPE;
133
134    @Parameter(required = false)
135    private Selection[] selections = null;
136
137    @Parameter(defaultValue = "true")
138    private boolean skipIfEmpty = true;
139
140    @Inject private MavenProject project = null;
141    @Inject private ArtifactLicenseCatalog catalog = null;
142    @Inject private ArtifactModelCache cache = null;
143    @Inject private ExecutorServiceImpl executor = null;
144
145    @Override
146    public void execute() throws MojoExecutionException, MojoFailureException {
147        try {
148            if (! isSkip()) {
149                String packaging = project.getPackaging();
150
151                if (ARCHIVE_PACKAGING.contains(packaging)) {
152                    if (directory.exists() && (! directory.isDirectory())) {
153                        throw new FileAlreadyExistsException(directory.toString(),
154                                                             null,
155                                                             "Is not a directory");
156                    }
157
158                    Path parent = directory.toPath();
159
160                    if (subdirectory != null && (! subdirectory.isEmpty())) {
161                        Path path = Paths.get(subdirectory);
162
163                        if (path.isAbsolute()) {
164                            throw new IllegalArgumentException(path + " is not relative");
165                        }
166
167                        parent = parent.resolve(path);
168                    }
169                    /*
170                     * Populate the caches.
171                     */
172                    Selections selections = new Selections();
173                    Set<String> scope = getScope();
174                    List<Callable<AnyLicenseInfo>> tasks =
175                        project.getArtifacts()
176                        .stream()
177                        .filter(t -> scope.contains(t.getScope()))
178                        .<Callable<AnyLicenseInfo>>map(t -> (() -> selections.get(t)))
179                        .collect(toList());
180
181                    executor.invokeAll(tasks);
182                    /*
183                     * Get the records for the report.
184                     */
185                    List<Tuple> tuples =
186                        project.getArtifacts()
187                        .stream()
188                        .filter(t -> scope.contains(t.getScope()))
189                        .map(t -> new Tuple(selections.get(t), cache.get(t), t))
190                        .collect(toList());
191                    /*
192                     * LICENSE
193                     */
194                    generateLicense(parent);
195                    /*
196                     * DEPENDENCIES
197                     *
198                     * Report sort order:
199                     *      Licenses (LICENSE_ORDER)
200                     *      Model[name, url] (same if name is not blank)
201                     *      Artifacts (ArtifactModelCache.ORDER)
202                     */
203                    TreeMap<AnyLicenseInfo,Map<Model,List<Tuple>>> report =
204                        tuples.stream()
205                        .collect(groupingBy(Tuple::getLicense,
206                                            () -> new TreeMap<>(LICENSE_ORDER),
207                                            groupingBy(Tuple::getModel,
208                                                       () -> new TreeMap<>(MODEL_ORDER),
209                                                       toList())));
210
211                    warnIfExtractedLicenseInfo(report.keySet().stream());
212
213                    if ((! report.isEmpty()) || (! skipIfEmpty)) {
214                        generateReport(report, parent);
215                    } else {
216                        log.info("Skipping empty {} resource generation",
217                                 DEPENDENCIES);
218                    }
219                } else {
220                    log.warn("Skipping for '{}' packaging", packaging);
221                }
222            } else {
223                log.info("Skipping {} and {} resource generation",
224                         getFile().getName(), DEPENDENCIES);
225            }
226        } catch (IOException exception) {
227            fail(exception.getMessage(), exception);
228        } catch (Throwable throwable) {
229            log.error("{}", throwable.getMessage(), throwable);
230
231            if (throwable instanceof MojoExecutionException) {
232                throw (MojoExecutionException) throwable;
233            } else if (throwable instanceof MojoFailureException) {
234                throw (MojoFailureException) throwable;
235            } else {
236                throw new MojoExecutionException(throwable.getMessage(),
237                                                 throwable);
238            }
239        } finally {
240            catalog.flush();
241        }
242    }
243
244    private Set<String> getScope() {
245        Pattern pattern = Pattern.compile("[\\p{Punct}\\p{Space}]+");
246        TreeSet<String> scope =
247            pattern.splitAsStream(includeScope)
248            .filter(StringUtils::isNotBlank)
249            .collect(toCollection(() -> new TreeSet<>(String.CASE_INSENSITIVE_ORDER)));
250
251        pattern.splitAsStream(excludeScope)
252            .filter(StringUtils::isNotBlank)
253            .forEach(t -> scope.remove(t));
254
255        if (scope.isEmpty()) {
256            log.warn("Specified scope is empty");
257            log.debug("    includeScope = {}", includeScope);
258            log.debug("    excludeScope = {}", excludeScope);
259        }
260
261        return scope;
262    }
263
264    private void generateLicense(Path parent) throws Exception {
265        File file = getFile();
266
267        if (file.exists() && file.isDirectory()) {
268            throw new FileAlreadyExistsException(file.toString(),
269                                                 null,
270                                                 "Is not a file or link");
271        }
272
273        Path source = file.toPath();
274        Path target = parent.resolve(source.getFileName());
275
276        Files.createDirectories(parent);
277        Files.copy(file.toPath(), target, REPLACE_EXISTING);
278        Files.setLastModifiedTime(target, Files.getLastModifiedTime(source));
279    }
280
281    private void generateReport(TreeMap<AnyLicenseInfo,Map<Model,List<Tuple>>> report,
282                                Path parent) throws Exception {
283        Path target = parent.resolve(DEPENDENCIES);
284
285        try (PrintWriter out =
286                 new PrintWriter(Files.newBufferedWriter(target,
287                                                         CREATE, WRITE,
288                                                         TRUNCATE_EXISTING))) {
289            String boundary = String.join(EMPTY, Collections.nCopies(78, "-"));
290
291            out.println(boundary);
292            out.println(target.getFileName() + " " + project.getArtifact());
293            out.println(boundary);
294
295            for (Map.Entry<AnyLicenseInfo,Map<Model,List<Tuple>>> section :
296                     report.entrySet()) {
297                out.println(boundary);
298                out.println(toString(section.getKey()));
299                out.println(boundary);
300
301                boolean first = true;
302
303                for (Map.Entry<Model,List<Tuple>> group :
304                         section.getValue().entrySet()) {
305                    if (first) {
306                        first = false;
307                    } else {
308                        out.println();
309                    }
310
311                    if (isNotBlank(group.getKey().getName())) {
312                        out.println(group.getKey().getName());
313                    }
314
315                    for (Tuple tuple : group.getValue()) {
316                        out.println(tuple.getArtifact());
317                    }
318
319                    if (isNotBlank(group.getKey().getUrl())) {
320                        out.println(group.getKey().getUrl());
321                    }
322                }
323
324                out.println(boundary);
325            }
326        }
327    }
328
329    private static String toString(AnyLicenseInfo license) {
330        String string = null;
331
332        if (license instanceof LicenseSet) {
333            AnyLicenseInfo[] members = ((LicenseSet) license).getMembers();
334
335            Arrays.sort(members, Comparator.comparing(Objects::toString));
336
337            string =
338                Stream.of(members)
339                .map(Objects::toString)
340                .collect(joining((license instanceof ConjunctiveLicenseSet) ? " AND " : " OR "));
341        } else {
342            string = Objects.toString(license, EMPTY);
343        }
344
345        return string;
346    }

348    /**
349     * {@link Selection}s @{@link Parameter} {@link Map}.  Dispatches to
350     * {@link #catalog} if no license is specified.
351     */
352    @NoArgsConstructor
353    private class Selections extends TreeMap<String,AnyLicenseInfo> {
354        private static final long serialVersionUID = 4229926081116865138L;
355
356        {
357            if (selections != null) {
358                Stream.of(selections)
359                    .forEach(t -> put(t.getKey(), t.getValue()));
360            }
361        }
362
363        @Override
364        public AnyLicenseInfo get(Object key) {
365            AnyLicenseInfo value = null;
366
367            if (key instanceof Artifact) {
368                Artifact artifact = (Artifact) key;
369
370                if (value == null) {
371                    value = get(ArtifactUtils.key(artifact));
372                }
373
374                if (value == null) {
375                    value = get(ArtifactUtils.versionlessKey(artifact));
376                }
377
378                if (value == null) {
379                    value = catalog.get(artifact);
380                }
381            } else {
382                value = super.get(key);
383            }
384
385            return value;
386        }
387    }
388
389    @AllArgsConstructor @Getter @ToString
390    private class Tuple {
391        private AnyLicenseInfo license = null;
392        private Model model = null;
393        private Artifact artifact = null;
394    }
