051/**
052 * Abstract base class for license {@link org.apache.maven.plugin.Mojo}s.
053 *
054 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
055 * @version $Revision: 7215 $
056 */
057@NoArgsConstructor(access = PROTECTED) @Getter @ToString @Slf4j
058public abstract class AbstractLicenseMojo extends AbstractMojo {
059    @Parameter(defaultValue = "false", property = "license.skip")
060    private boolean skip = false;
061
062    @Parameter(defaultValue = "${basedir}/LICENSE",
063               property = "license.file", readonly = true)
064    private File file = null;
065
066    protected void warnIfExtractedLicenseInfo(Stream<AnyLicenseInfo> stream) {
067        Set<ExtractedLicenseInfo> extracted =
068            stream
069            .flatMap(t -> LicenseUtilityMethods.walk(t))
070            .filter(t -> (t instanceof ExtractedLicenseInfo))
071            .map(t -> (ExtractedLicenseInfo) t)
072            .collect(toSet());
073
074        if (! extracted.isEmpty()) {
075            log.warn("Cannot match to SPDX license(s)");
076
077            for (ExtractedLicenseInfo license : extracted) {
078                String id = license.getLicenseId();
079
080                log.warn("    '{}'", license.getLicenseId());
081
082                String[] seeAlso = license.getSeeAlso();
083
084                if (seeAlso != null) {
085                    for (String string : seeAlso) {
086                        if (! string.equals(license.getLicenseId())) {
087                            log.warn("        {}", string);
088                        }
089                    }
090                }
091            }
092        }
093    }
094
095    /**
096     * Method to copy from {@link URL} ({@link String} representation) to
097     * local {@link Path}.
098     *
099     * @param   from            The source {@link URL} ({@link String}
100     *                          representation).
101     * @param   to              The target {@link Path}.
102     * @param   overwrite       Whether or not to overwrite an existing
103     *                          {@link Path}.
104     *
105     * @throws  MojoExecutionException
106     *                          See
107     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
108     * @throws  MojoFailureException
109     *                          See
110     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
111     */
112    protected void copy(String from, Path to,
113                        boolean overwrite) throws MojoExecutionException,
114                                                  MojoFailureException {
115        try {
116            copy(new URL(from), to, overwrite);
117        } catch (MalformedURLException exception) {
118            fail(from, exception);
119        }
120    }
121
122    /**
123     * Method to copy from {@link URL} ({@link String} representation) to
124     * local {@link Path}.
125     *
126     * @param   from            The source {@link URL}.
127     * @param   to              The target {@link Path}.
128     * @param   overwrite       Whether or not to overwrite an existing
129     *                          {@link Path}.
130     *
131     * @throws  MojoExecutionException
132     *                          See
133     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
134     * @throws  MojoFailureException
135     *                          See
136     *                          {@link org.apache.maven.plugin.Mojo#execute()}.
137     */
138    protected void copy(URL from, Path to,
139                        boolean overwrite) throws MojoExecutionException,
140                                                  MojoFailureException {
141        String message = null;
142
143        try {
144            message = to.toString() + " <- " + from.toString();
145
146            if (Files.isDirectory(to)) {
147                throw new FileAlreadyExistsException(to.toString(), null,
148                                                     "Is not a file or link");
149            }
150
151            FileTime local = FileTime.fromMillis(0);
152
153            if (Files.exists(to)) {
154                local = Files.getLastModifiedTime(to);
155            }
156
157            URLConnection connection = from.openConnection();
158            FileTime remote =
159                FileTime.fromMillis(connection.getLastModified());
160            boolean isNewer =
161                (! Files.exists(to))
162                || remote.toMillis() == 0
163                || remote.compareTo(local) > 0;
164
165            if (isNewer) {
166                CopyOption[] options =
167                    Stream.of(StandardCopyOption.REPLACE_EXISTING)
168                    .filter(t -> isNewer & overwrite)
169                    .toArray(CopyOption[]::new);
170
171                try (InputStream in = connection.getInputStream()) {
172                    Files.copy(in, to, options);
173                }
174
175                if (remote.toMillis() != 0) {
176                    Files.setLastModifiedTime(to, remote);
177                }
178
179                log.info("{}", message);
180            } else {
181                log.info("{} is up-to-date", to);
182            }
183        } catch (IOException exception) {
184            fail(message, exception);
185        } catch (Throwable throwable) {
186            log.error("{}", throwable.getMessage(), throwable);
187
188            if (throwable instanceof MojoExecutionException) {
189                throw (MojoExecutionException) throwable;
190            } else if (throwable instanceof MojoFailureException) {
191                throw (MojoFailureException) throwable;
192            } else {
193                throw new MojoExecutionException(throwable.getMessage(),
194                                                 throwable);
195            }
196        }
197    }
198
199    /**
200     * Log and throw a {@link MojoFailureException}.
201     *
202     * @param   message         The message {@link String}.
203     *
204     * @throws  MojoFailureException
205     *                          Always.
206     */
207    protected void fail(String message) throws MojoFailureException {
208        log.error("{}", message);
209        throw new MojoFailureException(message);
210    }
211
212    /**
213     * Log and throw a {@link MojoFailureException}.
214     *
215     * @param   message         The message {@link String}.
216     * @param   reason          The reason {@link Throwable}.
217     *
218     * @throws  MojoFailureException
219     *                          Always.
220     */
221    protected void fail(String message,
222                        Throwable reason) throws MojoFailureException {
223        log.error("{}", message, reason);
224        throw new MojoFailureException(message, reason);
225    }
226}
