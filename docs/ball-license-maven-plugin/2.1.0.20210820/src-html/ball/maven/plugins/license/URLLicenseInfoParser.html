074/**
075 * {@link URL} ({@link String} representation) to {@link AnyLicenseInfo}
076 * {@link java.util.Map} implementation.  The {@link #get(Object)} method
077 * transparently calculates and caches any value.

079 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
080 * @version $Revision: 8122 $

082@Named @Singleton

084public class URLLicenseInfoParser extends ConcurrentSkipListMap<String,AnyLicenseInfo> implements DefaultMethods {
085    private static final long serialVersionUID = -1086861055455360074L;

089    private static final Set<Integer> REDIRECT_CODES =
090        Stream.of(HttpURLConnection.HTTP_MOVED_TEMP,
091                  HttpURLConnection.HTTP_MOVED_PERM,
092                  HttpURLConnection.HTTP_SEE_OTHER)
093        .collect(toSet());
094
095    private static final Pattern CANONICAL =
096        Pattern.compile("<([^>]*)>; rel=\"canonical\"");
097
098    /** @serial */ private final LicenseMap map;
099    /** @serial */ private final Map<Pattern,String> redirects;

101    /**
102     * Sole constructor.
103     *
104     * @param   map             The injected {@link LicenseMap}.
105     */
106    @Inject
107    public URLLicenseInfoParser(LicenseMap map) {
108        super(String.CASE_INSENSITIVE_ORDER);
109
110        this.map = Objects.requireNonNull(map);
111
112        try {
113            for (AnyLicenseInfo value : map.values()) {
114                if (value instanceof License) {
115                    String id = ((License) value).getLicenseId();
116
117                    put(String.format("https://opensource.org/licenses/%s", id),
118                        value);
119                    put(String.format("https://spdx.org/licenses/%s.html", id),
120                        value);
121                }
122            }
123
124            for (JsonNode node :
125                     LicenseMap.LICENSES_FULL_JSON.at("/licenses")) {
126                AnyLicenseInfo value =
127                    map.get(node.at("/identifiers/spdx/0").asText());
128
129                if (value != null) {
130                    for (JsonNode uri : node.at("/uris")) {
131                        String key = uri.asText();
132
133                        putIfAbsent(key, value);
134                    }
135                }
136            }
137
138            Properties seeds = getXMLProperties("seeds");
139
140            for (String id : seeds.stringPropertyNames()) {
141                AnyLicenseInfo value =
142                    LicenseInfoFactory.parseSPDXLicenseString(id);
143
144                if (! isFullySpdxListed(value)) {
145                    throw new IllegalArgumentException(id);
146                }
147
148                Stream.of(seeds.getProperty(id).split("[\\p{Space}]+"))
149                    .filter(StringUtils::isNotBlank)
150                    .forEach(t -> putIfAbsent(t.trim(), value));
151            }
152
153            for (AnyLicenseInfo value : map.values()) {
154                if (value instanceof License) {
155                    for (String key : ((License) value).getSeeAlso()) {
156                        if (isNotBlank(key)) {
157                            putIfAbsent(key.trim(), value);
158                        }
159                    }
160                }
161            }
162
163            Set<String> set =
164                keySet().stream()
165                .filter(t -> t.startsWith("https:"))
166                .collect(toSet());
167
168            set.stream()
169                .forEach(t -> computeIfAbsent(t.replace("https:", "http:"),
170                                              k -> get(t)));
171
172            redirects =
173                getXMLProperties("redirects").entrySet()
174                .stream()
175                .collect(toMap(k -> Pattern.compile(k.getKey().toString()),
176                               v -> v.getValue().toString().trim()));
177        } catch (Exception exception) {
178            log.error("{}", exception.getMessage(), exception);
179            throw new ExceptionInInitializerError(exception);
180        }
181    }

191    /**
192     * Entry-point that first attempts to parse the specified license ID as
193     * an SPDX license ID and, if that fails, continues to analyze the
194     * document at specified URL.
195     *
196     * @param   resolver        The {@link LicenseResolver}.
197     * @param   key             The {@link URLLicenseInfo}.
198     *
199     * @return  {@link AnyLicenseInfo} reppresenting the results of the
200     *          parse (may be {@code key}).
201     */
202    public AnyLicenseInfo parse(LicenseResolver resolver, URLLicenseInfo key) {
203        String name = key.getLicenseId();
204        AnyLicenseInfo value = isNotBlank(name) ? map.get(name) : null;
205
206        if (value == null) {
207            Set<AnyLicenseInfo> set =
208                Stream.of(key.getSeeAlso())
209                .map(t -> computeIfAbsent(t, k -> compute(resolver, k)))
210                .collect(toSet());
211
212            if (! set.isEmpty()) {
213                value = resolver.toLicense(set);
214            }
215        }
216
217        return (value != null) ? value : key;
218    }

220    private AnyLicenseInfo compute(LicenseResolver resolver, String url) {
221        AnyLicenseInfo value = null;
222        URLConnection connection = null;
223
224        try {
225            connection = new URL(url).openConnection();
226
227            if (connection instanceof HttpURLConnection) {
228                ((HttpURLConnection) connection)
229                    .setInstanceFollowRedirects(false);
230            }
231
232            if (connection instanceof HttpsURLConnection) {
233                ((HttpsURLConnection) connection).setHostnameVerifier(NONE);
234            }
235
236            String canonicalURL = getCanonicalURL(connection);
237
238            if (isNotBlank(canonicalURL) && (! equals(canonicalURL, url))) {
239                if (value == null) {
240                    value = get(canonicalURL);
241                }
242            }
243
244            String redirectURL = getRedirectURL(connection);
245
246            if (isNotBlank(redirectURL)) {
247                if (! equals(redirectURL, url)) {
248                    if (value != null) {
249                        put(redirectURL, value);
250                    } else {
251                        value =
252                            computeIfAbsent(redirectURL,
253                                            k -> compute(resolver, k));
254                    }
255                }
256            }
257
258            if (value == null) {
259                ContentType type =
260                    ContentType.parse(connection.getContentType());
261                Charset charset =
262                    (type.getCharset() != null) ? type.getCharset() : UTF_8;
263
264                if (type.getMimeType().matches("(?i).*(html|xml).*")) {
265                    try (InputStream in = connection.getInputStream()) {
266                        Document document =
267                            Jsoup.parse(in, charset.name(), url);
268
269                        document.outputSettings()
270                            .syntax(Document.OutputSettings.Syntax.xml);
271                        /*
272                         * Heuristic: Look for a "canonical" <link/> with a
273                         * known href.
274                         */
275                        if (value == null) {
276                            value =
277                                document.select("head>link[rel='canonical'][href]")
278                                .stream()
279                                .map(t -> t.attr("abs:href"))
280                                .filter(StringUtils::isNotBlank)
281                                .map(t -> get(t))
282                                .filter(Objects::nonNull)
283                                .findFirst().orElse(null);
284                        }
285                        /*
286                         * Heuristics to consider:
287                         *
288                         * Search for SPDX-License-Identifier
289                         *
290                         * If a single href of
291                         * http://opensource.org/licenses/([^/]) is found
292                         * and $1 is in the LicenseMap
293                         */
294                        /*
295                         * Parse the document if the heuristics fail.
296                         */
297                        if (value == null) {
298                            value =
299                                Stream.of("content, .content, #content",
300                                          "main, .main, #main",
301                                          "body, .body, #body")
302                                .map(t -> document.select(t))
303                                .flatMap(Elements::stream)
304                                .filter(Element::hasText)
305                                .map(t -> t.text())
306                                .distinct()
307                                .map(t -> new TextLicenseInfo(url, t, url))
308                                .map(t -> resolver.parse(t))
309                                .filter(Objects::nonNull)
310                                .filter(t -> (! (t instanceof ExtractedLicenseInfo)))
311                                .findFirst()
312                                .orElse(null);
313                        }
314
315                        if (value == null) {
316                            value =
317                                new TextLicenseInfo(url,
318                                                    document.wholeText(), url);
319                            value = resolver.parse(value);
320                        }
321                    }
322                } else {
323                    try (InputStream in = connection.getInputStream();
324                         Scanner scanner = new Scanner(in, charset.name())) {
325                        String text = scanner.useDelimiter("\\A").next();
326
327                        value = new TextLicenseInfo(url, text, url);
328                        value = resolver.parse(value);
329                    }
330                }
331            }
332        } catch (FileNotFoundException exception) {
333            log.debug("File not found: {}", url);
334        } catch (Exception exception) {
335            log.warn("Cannot read {}", url);
336
337            if (connection != null) {
338                connection.getHeaderFields().entrySet().stream()
339                    .forEach(t -> log.debug("{}", t));
340            }
341        } finally {
342            if (value == null) {
343                value = new TextLicenseInfo(url, EMPTY, url);
344            }
345        }
346
347        if (value instanceof ExtractedLicenseInfo) {
348            TextLicenseInfo.addSeeAlso((ExtractedLicenseInfo) value, url);
349        }
350
351        return value;
352    }

354    private boolean equals(String left, String right) {
355        return (Objects.compare(left, right, comparator()) == 0
356                && Objects.compare(right, left, comparator()) == 0);
357    }

359    private String getCanonicalURL(URLConnection connection) {
360        String url = null;
361
362        if (connection instanceof HttpURLConnection) {
363            List<String> list = connection.getHeaderFields().get("Link");
364
365            if (list != null) {
366                url =
367                    list.stream()
368                    .map(t -> CANONICAL.matcher(t))
369                    .filter(t -> t.find())
370                    .map(t -> t.group(1))
371                    .filter(StringUtils::isNotBlank)
372                    .map(t -> resolve((HttpURLConnection) connection, t))
373                    .map(t -> t.toASCIIString())
374                    .findFirst().orElse(null);
375            }
376        }
377
378        return url;
379    }
380
381    private String getRedirectURL(URLConnection connection) throws IOException {
382        String redirectURL = null;
383
384        if (connection instanceof HttpURLConnection) {
385            String url = connection.getURL().toString();
386            int code = ((HttpURLConnection) connection).getResponseCode();
387
388            if (REDIRECT_CODES.contains(code)) {
389                String location = connection.getHeaderField("Location");
390
391                if (isNotBlank(location)) {
392                    redirectURL =
393                        resolve((HttpURLConnection) connection, location)
394                        .toASCIIString();
395                }
396            }
397
398            if (isBlank(redirectURL)) {
399                for (Map.Entry<Pattern,String> entry : redirects.entrySet()) {
400                    Matcher matcher = entry.getKey().matcher(url);
401
402                    if (matcher.matches()) {
403                        redirectURL = matcher.replaceFirst(entry.getValue());
404
405                        if (isNotBlank(redirectURL)) {
406                            break;
407                        } else {
408                            redirectURL = null;
409                        }
410                    }
411                }
412            }
413        }
414
415        return redirectURL;
416    }
417
418    private URI resolve(HttpURLConnection connection, String location) {
419        URI uri = URI.create(location);
420
421        if (! uri.isAbsolute()) {
422            try {
423                uri = connection.getURL().toURI().resolve(uri);
424            } catch (Exception exception) {
425                log.debug("{}", exception.getMessage(), exception);
426            }
427        }
428
429        return uri;
430    }
