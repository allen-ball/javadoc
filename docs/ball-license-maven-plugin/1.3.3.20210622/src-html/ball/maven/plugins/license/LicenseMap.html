023import com.fasterxml.jackson.databind.JsonNode;
024import com.fasterxml.jackson.databind.ObjectMapper;
025import java.util.Objects;
026import java.util.Properties;
027import java.util.TreeMap;
028import java.util.stream.Stream;
029import javax.annotation.PostConstruct;
030import javax.annotation.PreDestroy;
031import javax.inject.Inject;
032import javax.inject.Named;
033import javax.inject.Singleton;
034import lombok.extern.slf4j.Slf4j;
035import org.apache.commons.lang3.StringUtils;
036import org.spdx.rdfparser.license.AnyLicenseInfo;
037import org.spdx.rdfparser.license.License;
038import org.spdx.rdfparser.license.LicenseInfoFactory;
039/* import org.spdx.rdfparser.license.ListedExceptions; */
040import org.spdx.rdfparser.license.ListedLicenses;
041
042import static ball.maven.plugins.license.LicenseUtilityMethods.isFullySpdxListed;
043import static org.apache.commons.lang3.StringUtils.SPACE;
044
045/**
046 * {@link java.util.Map} implementation that relates SPDX ID and known
047 * aliases to {@link License} sourced from
048 * {@link ListedLicenses#getListedLicenses()}.
049 *
050 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
051 * @version $Revision: 8122 $
052 */
053@Named @Singleton
054@Slf4j
055public class LicenseMap extends TreeMap<String,AnyLicenseInfo> implements DefaultMethods {
056    private static final long serialVersionUID = 8959458091372664080L;
057
058    private static final String ONLY_USE_LOCAL_LICENSES = "SPDXParser.OnlyUseLocalLicenses";
059
060    /**
061     * {@code resources/licenses-full.json} read from SPDX artifacts.
062     */
063    public static final JsonNode LICENSES_FULL_JSON;
064
065    static {
066        if (System.getProperty(ONLY_USE_LOCAL_LICENSES) == null) {
067            System.setProperty(ONLY_USE_LOCAL_LICENSES, String.valueOf(true));
068        }
069
070        try {
071            LICENSES_FULL_JSON =
072                new ObjectMapper()
073                .readTree(LicenseMap.class
074                          .getClassLoader()
075                          .getResource("resources/licenses-full.json"));
076        } catch (Exception exception) {
077            throw new ExceptionInInitializerError(exception);
078        }
079    }
080
081    /**
082     * Sole constructor.
083     */
084    @Inject
085    public LicenseMap() { super(String.CASE_INSENSITIVE_ORDER); }
086
087    @PostConstruct
088    public void init() throws Exception {
089        ListedLicenses licenses = ListedLicenses.getListedLicenses();
090        /* ListedExceptions exceptions = ListedExceptions.getListedExceptions(); */
091
092        for (String key : licenses.getSpdxListedLicenseIds()) {
093            License license = licenses.getListedLicenseById(key);
094
095            put(key, license);
096            putIfAbsent(license.getName(), license);
097        }
098
099        for (JsonNode node : LICENSES_FULL_JSON.at("/licenses")) {
100            String spdx = node.at("/identifiers/spdx/0").asText();
101
102            if (containsKey(spdx)) {
103                Stream.of(node.at("/id").asText(),
104                          node.at("/name").asText())
105                    .map(t -> t.trim())
106                    .map(t -> t.replaceAll("[\\p{Space}]+", SPACE))
107                    .filter(StringUtils::isNotBlank)
108                    .forEach(t -> computeIfAbsent(t, k -> get(spdx)));
109            }
110        }
111
112        Properties aliases = getXMLProperties("aliases");
113
114        for (String id : aliases.stringPropertyNames()) {
115            AnyLicenseInfo value =
116                LicenseInfoFactory.parseSPDXLicenseString(id);
117
118            if (! isFullySpdxListed(value)) {
119                throw new IllegalArgumentException(id);
120            }
121
122            Stream.of(aliases.getProperty(id).split("\\R+"))
123                .map(String::trim)
124                .filter(StringUtils::isNotBlank)
125                .forEach(t -> putIfAbsent(t, value));
126        }
127    }
128
129    @PreDestroy
130    public void destroy() {
131        log.debug("{}.size() = {}", getClass().getSimpleName(), size());
132    }
133
134    @Override
135    public AnyLicenseInfo get(Object key) {
136        AnyLicenseInfo value = super.get(key);
137
138        if (value == null) {
139            if (key instanceof String) {
140                String string = ((String) key).trim();
141
142                value =
143                    Stream.of(string.replaceAll("(?i)[\\p{Space}]+", "-"),
144                              string.replaceAll("(?i)(V(ERSION)?)[\\p{Space}]?([\\p{Digit}])", "$3"))
145                    .filter(t -> (! string.equals(t)))
146                    .map(t -> get(t))
147                    .filter(Objects::nonNull)
148                    .findFirst().orElse(null);
149            }
150        }
151
152        return value;
153    }
154}
