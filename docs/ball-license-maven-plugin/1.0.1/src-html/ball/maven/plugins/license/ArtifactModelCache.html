024import java.io.File;
025import java.util.Comparator;
026import java.util.Objects;
027import java.util.TreeMap;
028import javax.annotation.PostConstruct;
029import javax.annotation.PreDestroy;
030import javax.inject.Inject;
031import javax.inject.Named;
032import javax.inject.Singleton;
033import lombok.extern.slf4j.Slf4j;
034import org.apache.maven.artifact.Artifact;
035import org.apache.maven.execution.MavenSession;
036import org.apache.maven.model.Model;
037import org.apache.maven.model.io.ModelReader;
038import org.apache.maven.project.DefaultProjectBuildingRequest;
039import org.apache.maven.project.ProjectBuilder;
040import org.apache.maven.project.ProjectBuildingRequest;
041
042import static org.apache.maven.model.building.ModelBuildingRequest.VALIDATION_LEVEL_MINIMAL;
043
044/**
045 * {@link Artifact} to {@link Model} {@link java.util.Map}
046 * implementation.  The {@link #get(Object)} method transparently calculates
047 * and caches any value.
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 * @version $Revision: 5736 $
051 */
052@Named @Singleton
053@Slf4j
054public class ArtifactModelCache extends TreeMap<Artifact,Model> {
055    public static final Comparator<Artifact> ORDER =
056        Comparator
057        .comparing(Artifact::getGroupId)
058        .thenComparing(Artifact::getArtifactId)
059        .thenComparing(Artifact::getVersion);
060
061    /** @serial */ private final MavenSession session;
062    /** @serial */ private final ProjectBuilder builder;
063    /** @serial */ private final ModelReader reader;
064
065    /**
066     * Sole constructor.
067     *
068     * @param   session         The injected {@link MavenSession}.
069     * @param   builder         The injected {@link ProjectBuilder}.
070     * @param   reader          The injected {@link ModelReader}.
071     */
072    @Inject
073    public ArtifactModelCache(MavenSession session,
074                              ProjectBuilder builder, ModelReader reader) {
075        super(ORDER);
076
077        this.session = Objects.requireNonNull(session);
078        this.builder = Objects.requireNonNull(builder);
079        this.reader = Objects.requireNonNull(reader);
080    }
081
082    @PostConstruct
083    public void init() { }
084
085    @PreDestroy
086    public void destroy() {
087        log.debug(getClass().getSimpleName() + ".size() = " + size());
088    }
089
090    @Override
091    public Model get(Object key) {
092        Model value = super.get(key);
093
094        if (value == null) {
095            value = compute((Artifact) key);
096
097            put((Artifact) key, value);
098        }
099
100        return value;
101    }
102
103    private Model compute(Artifact artifact) {
104        File file =
105            new File(artifact.getFile().getParentFile(),
106                     artifact.getArtifactId()
107                     + "-" + artifact.getVersion() + ".pom");
108        Model model =
109            session.getProjects()
110            .stream()
111            .filter(t -> ORDER.compare(t.getArtifact(), artifact) == 0)
112            .map(t -> t.getModel())
113            .filter(Objects::nonNull)
114            .findFirst().orElse(null);
115
116        if (model == null) {
117            try {
118                model = reader.read(file, null);
119            } catch (Exception exception) {
120                log.debug("Cannot read POM for " + artifact);
121                /* log.debug(exception.getMessage(), exception); */
122            }
123
124            if (model != null
125                && (model.getLicenses() == null
126                    || model.getLicenses().isEmpty())) {
127                try {
128                    ProjectBuildingRequest request =
129                        new DefaultProjectBuildingRequest(session.getProjectBuildingRequest())
130                        .setValidationLevel(VALIDATION_LEVEL_MINIMAL)
131                        .setResolveDependencies(false)
132                        .setProcessPlugins(false);
133
134                    model =
135                        builder.build(file, request)
136                        .getProject().getModel();
137                } catch (Exception exception) {
138                    log.debug("Cannot load POM for " + artifact);
139                    /* log.debug(exception.getMessage(), exception); */
140                }
141            }
142        }
143
144        return model;
145    }
146}
