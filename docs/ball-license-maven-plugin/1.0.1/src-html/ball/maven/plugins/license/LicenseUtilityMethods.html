024import java.util.Arrays;
025import java.util.Collection;
026import java.util.Collections;
027import java.util.Iterator;
028import java.util.Objects;
029import java.util.Spliterator;
030import java.util.Spliterators.AbstractSpliterator;
031import java.util.function.Consumer;
032import java.util.function.Function;
033import java.util.function.Supplier;
034import java.util.stream.Stream;
035import java.util.stream.StreamSupport;
036import lombok.NoArgsConstructor;
037import lombok.ToString;
038import org.spdx.rdfparser.license.AnyLicenseInfo;
039import org.spdx.rdfparser.license.License;
040import org.spdx.rdfparser.license.LicenseSet;
041import org.spdx.rdfparser.license.OrLaterOperator;
042import org.spdx.rdfparser.license.WithExceptionOperator;
043
044import static lombok.AccessLevel.PRIVATE;
045
046/**
047 * {@link AnyLicenseInfo} utility methods.
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 * @version $Revision: 5888 $
051 */
052@NoArgsConstructor(access = PRIVATE) @ToString
053public abstract class LicenseUtilityMethods {
054
055    /**
056     * Static method to create a {@link Stream} to walk the tree rooted at
057     * {@code root}.
058     *
059     * @param   root            The {@link AnyLicenseInfo} representing the
060     *                          root of the tree.
061     *
062     * @return  A {@link Stream}.
063     */
064    public static Stream<AnyLicenseInfo> walk(AnyLicenseInfo root) {
065        return walk(root, LicenseUtilityMethods::childrenOf);
066    }
067
068    /**
069     * Static method to get the children of a {@link AnyLicenseInfo}.
070     *
071     * @param   node            The {@link AnyLicenseInfo}.
072     *
073     * @return  A {@link Collection} of {@link AnyLicenseInfo} for a
074     *          {@link LicenseSet}, {@link OrLaterOperator}, or
075     *          {@link WithExceptionOperator} node; an empty
076     *          {@link Collection} otherwise.
077     */
078    public static Collection<AnyLicenseInfo> childrenOf(AnyLicenseInfo node) {
079        AnyLicenseInfo[] children = null;
080
081        if (node instanceof LicenseSet) {
082            children = ((LicenseSet) node).getMembers();
083        } else if (node instanceof OrLaterOperator) {
084            children =
085                new AnyLicenseInfo[] {
086                    ((OrLaterOperator) node).getLicense()
087                };
088        } else if (node instanceof WithExceptionOperator) {
089            children =
090                new AnyLicenseInfo[] {
091                    ((WithExceptionOperator) node).getLicense()
092                };
093        } else {
094            children = new AnyLicenseInfo[] { };
095        }
096
097        return Arrays.asList(children);
098    }
099
100    /**
101     * Returns the count of licenses represented by this
102     * {@link AnyLicenseInfo}: The number of members in a
103     * {@link LicenseSet} or {@code 1}.
104     *
105     * @param   license         The {@link AnyLicenseInfo}.
106     *
107     * @return  The license count.
108     */
109    public static int countOf(AnyLicenseInfo license) {
110        int count = 0;
111
112        if (license instanceof LicenseSet) {
113            count =
114                Stream.of(((LicenseSet) license).getMembers())
115                .mapToInt(t -> countOf(t))
116                .sum();
117        } else if (license != null) {
118            count = 1;
119        }
120
121        return count;
122    }
123
124    /**
125     * Method to determine if an {@link AnyLicenseInfo} is "empty" (defined
126     * as the license consists solely of empty containers).
127     *
128     * @param   license         The {@link AnyLicenseInfo}.
129     *
130     * @return  {@code true} if the {@code license} is empty; {@code false}
131     *          otherwise.
132     */
133    public static boolean isEmpty(AnyLicenseInfo license) {
134        return countOf(license) == 0;
135    }
136
137    /**
138     * Static method to test if an {@link AnyLicenseInfo} is a branch.
139     *
140     * @param   node            The {@link AnyLicenseInfo}.
141     *
142     * @return  {@code true} if a branch; {@code false} otherwise.
143     */
144    public static boolean isBranch(AnyLicenseInfo node) {
145        return (isLicenseSet(node) || isOperator(node));
146    }
147
148    /**
149     * Static method to test if an {@link AnyLicenseInfo} is a leaf.
150     *
151     * @param   node            The {@link AnyLicenseInfo}.
152     *
153     * @return  {@code true} if a leaf; {@code false} otherwise.
154     */
155    public static boolean isLeaf(AnyLicenseInfo node) {
156        return (! isBranch(node));
157    }
158
159    /**
160     * Static method to test if an {@link AnyLicenseInfo}
161     * is a {@link LicenseSet}.
162     *
163     * @param   node            The {@link AnyLicenseInfo}.
164     *
165     * @return  {@code true} if a {@code node} is a {@link LicenseSet};
166     *          {@code false} otherwise.
167     */
168    public static boolean isLicenseSet(AnyLicenseInfo node) {
169        return (node instanceof LicenseSet);
170    }
171
172    /**
173     * Static method to test if an {@link AnyLicenseInfo} is an operator
174     * node ({@link OrLaterOperator} or {@link WithExceptionOperator}).
175     *
176     * @param   node            The {@link AnyLicenseInfo}.
177     *
178     * @return  {@code true} if an operator node; {@code false} otherwise.
179     */
180    public static boolean isOperator(AnyLicenseInfo node) {
181        return (node instanceof OrLaterOperator
182                || node instanceof WithExceptionOperator);
183    }
184
185    /**
186     * Method to test if an {@link AnyLicenseInfo} is fully specified by
187     * SPDX license(s).
188     *
189     * @param   license         The {@link AnyLicenseInfo}.
190     *
191     * @return  {@code true} if fully specified; {@code false} otherwise.
192     */
193    public static boolean isFullySpdxListed(AnyLicenseInfo license) {
194        boolean fullySpecified =
195             LicenseUtilityMethods.walk(license)
196            .filter(LicenseUtilityMethods::isLeaf)
197            .map(t -> (t instanceof License))
198            .reduce(Boolean::logicalAnd).orElse(false);
199
200        return fullySpecified;
201    }
202
203    /**
204     * Method to test if an {@link AnyLicenseInfo} is partially specified by
205     * SPDX license(s).
206     *
207     * @param   license         The {@link AnyLicenseInfo}.
208     *
209     * @return  {@code true} if partially specified; {@code false}
210     *          otherwise.
211     */
212    public static boolean isPartiallySpdxListed(AnyLicenseInfo license) {
213        boolean partiallySpecified =
214            LicenseUtilityMethods.walk(license)
215            .filter(LicenseUtilityMethods::isLeaf)
216            .map(t -> (t instanceof License))
217            .reduce(Boolean::logicalOr).orElse(false);
218
219        return partiallySpecified;
220    }
221
222    private static <T> Stream<T> walk(T root,
223                                      Function<? super T,Collection<? extends T>> childrenOf) {
224        Walker<T> walker =
225            new Walker<>(Collections.singleton(root), childrenOf);
226
227        return StreamSupport.stream(walker, false);
228    }
229
230    private static class Walker<T> extends AbstractSpliterator<T> {
231        private final Stream<Supplier<Walker<T>>> stream;
232        private Iterator<Supplier<Walker<T>>> iterator = null;
233        private Spliterator<? extends T> spliterator = null;
234
235        private Walker(Collection<? extends T> nodes,
236                       Function<? super T,Collection<? extends T>> childrenOf) {
237            super(Long.MAX_VALUE, IMMUTABLE | NONNULL);
238
239            stream =
240                nodes.stream()
241                .filter(Objects::nonNull)
242                .map(childrenOf)
243                .filter(Objects::nonNull)
244                .map(t -> (() -> new Walker<T>(t, childrenOf)));
245            spliterator =
246                nodes.stream()
247                .filter(Objects::nonNull)
248                .spliterator();
249        }
250
251        @Override
252        public boolean tryAdvance(Consumer<? super T> consumer) {
253            boolean accepted = false;
254
255            if (spliterator != null) {
256                accepted = spliterator.tryAdvance(consumer);
257            }
258
259            if (! accepted) {
260                if (iterator == null) {
261                    iterator = stream.iterator();
262                }
263
264                if (iterator.hasNext()) {
265                    spliterator = iterator.next().get();
266                    accepted = tryAdvance(consumer);
267                }
268            }
269
270            return accepted;
271        }
272    }
273}
