080/**
081 * {@link Artifact} to {@link LicenseSet}
082 * ({@link org.spdx.rdfparser.license.License}) {@link java.util.Map}
083 * implementation.  The {@link #get(Object)} method transparently calculates
084 * and caches any value.
085 *
086 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
087 * @version $Revision: 5736 $
088 */
089@Named @Singleton
090@Slf4j
091public class ArtifactLicenseCatalog extends TreeMap<Artifact,AnyLicenseInfo> {
092    private static final String CATALOG = "artifact-license-catalog.xml";
093
094    private static final Pattern INCLUDE =
095        Pattern.compile("(?i)^(.*/|)(LICENSE([.][^/]+)?|about.html)$");
096    private static final Pattern EXCLUDE =
097        Pattern.compile("(?i)^.*[.]class$");
098
099    private static final Comparator<? super Boolean> TRUTH =
100        (t, u) -> Objects.equals(t, u) ? 0 : (t ? -1 : 1);
101    private static Comparator<? super AnyLicenseInfo> SIEVE =
102        Comparator
103        .<AnyLicenseInfo,Boolean>
104        comparing(t -> (! (t instanceof URLLicenseInfo)), TRUTH)
105        .thenComparing(t -> (! (t instanceof TextLicenseInfo)), TRUTH)
106        .thenComparing(t -> (! (t instanceof ExtractedLicenseInfo)), TRUTH)
107        .thenComparing(LicenseUtilityMethods::isFullySpdxListed, TRUTH)
108        .thenComparing(t -> (t instanceof LicenseSet), TRUTH)
109        .thenComparing(LicenseUtilityMethods::countOf,
110                       Comparator.<Integer>reverseOrder())
111        .thenComparing(t -> (t instanceof WithExceptionOperator), TRUTH)
112        .thenComparing(t -> (t instanceof OrLaterOperator), TRUTH)
113        .thenComparing(LicenseUtilityMethods::isPartiallySpdxListed, TRUTH);

116    /** @serial */ private final ArtifactModelCache cache;
117    /** @serial */ private final LicenseMap map;
118    /** @serial */ private final LicenseResolver resolver;
119    /** @serial */ private final File file;
120    /** @serial */ private final Properties catalog = new Properties();

122    /**
123     * Sole constructor.
124     *
125     * @param   session         The injected {@link MavenSession}.
126     * @param   cache           The injected {@link ArtifactModelCache}.
127     * @param   map             The injected {@link LicenseMap}.
128     * @param   resolver        The injected {@link LicenseResolver}.
129     */
130    @Inject
131    public ArtifactLicenseCatalog(MavenSession session,
132                                  ArtifactModelCache cache,
133                                  LicenseMap map, LicenseResolver resolver) {
134        super(Comparator.comparing(ArtifactUtils::key,
135                                   String.CASE_INSENSITIVE_ORDER));
136
137        this.session = Objects.requireNonNull(session);
138        this.cache = Objects.requireNonNull(cache);
139        this.map = Objects.requireNonNull(map);
140        this.resolver = Objects.requireNonNull(resolver);
141        this.file =
142            new File(session.getLocalRepository().getBasedir(), CATALOG);
143    }
144
145    @PostConstruct
146    public void init() {
147        if (file.exists()) {
148            try (FileInputStream in = new FileInputStream(file)) {
149                catalog.loadFromXML(in);
150            } catch (IOException exception) {
151                log.error("Cannot read " + file);
152            }
153        } else {
154            try (InputStream in = getClass().getResourceAsStream(CATALOG)) {
155                if (in != null) {
156                    catalog.loadFromXML(in);
157                }
158            } catch (IOException exception) {
159            }
160        }
161
162        for (String key : catalog.stringPropertyNames()) {
163            try {
164                put(new KeyArtifact(key),
165                    resolver.parseLicenseString(catalog.getProperty(key)));
166            } catch (Exception exception) {
167                log.error(key + ": " + exception.getMessage(), exception);
168            }
169        }
170    }
171
172    @PreDestroy
173    public void destroy() {
174        log.debug(getClass().getSimpleName() + ".size() = " + size());
175
176        boolean changed = (! file.exists());
177
178        for (Map.Entry<Artifact,AnyLicenseInfo> entry : entrySet()) {
179            AnyLicenseInfo license = entry.getValue();
180
181            if (isFullySpdxListed(license)) {
182                String key = ArtifactUtils.key(entry.getKey());
183                String value = license.toString();
184
185                changed |= (! Objects.equals(value, catalog.put(key, value)));
186            }
187        }
188
189        if (changed) {
190            try (FileOutputStream out = new FileOutputStream(file)) {
191                catalog.storeToXML(out, file.getName());
192            } catch (IOException exception) {
193                log.warn("Cannot write " + file);
194            }
195        }
196    }
197
198    @Override
199    public AnyLicenseInfo get(Object key) {
200        AnyLicenseInfo value = super.get(key);
201
202        if (value == null) {
203            value = compute((Artifact) key);
204
205            put((Artifact) key, value);
206        }
207
208        return value;
209    }
210
211    private AnyLicenseInfo compute(Artifact artifact) {
212        URL url = toURL(artifact);
213        /*
214         * Licenses specified in the Manifest Bundle-License
215         * or found in the Artifact
216         */
217        List<AnyLicenseInfo> bundle = Collections.emptyList();
218        List<AnyLicenseInfo> scanned = Collections.emptyList();
219
220        try (JarFile jar =
221                 ((JarURLConnection) url.openConnection()).getJarFile()) {
222            bundle =
223                Stream.of(jar.getManifest())
224                .filter(Objects::nonNull)
225                .map(t -> t.getMainAttributes().getValue("Bundle-License"))
226                .filter(StringUtils::isNotBlank)
227                .flatMap(t -> Stream.of(t.split(",")))
228                .map(t -> t.trim())
229                .distinct()
230                .map(t -> Pattern.compile("((.+);link=)?(.*)").matcher(t))
231                .filter(t -> t.matches())
232                .map(t -> parse(t.group(2), resolve(url, t.group(3))))
233                .collect(toList());
234
235            scanned =
236                jar.stream()
237                .map(JarEntry::getName)
238                .filter(t -> INCLUDE.matcher(t).matches())
239                .filter(t -> (! EXCLUDE.matcher(t).matches()))
240                .map(t -> parse(t, resolve(url, t)))
241                .filter(t -> (! (t instanceof URLLicenseInfo)))
242                .collect(toList());
243        } catch (ZipException exception) {
244        } catch (IOException exception) {
245            log.debug(artifact + ": " + exception.getMessage(), exception);
246        }
247
248        Map<String,AnyLicenseInfo> found =
249            scanned.stream()
250            .filter(t -> LicenseUtilityMethods.countOf(t) > 0)
251            .filter(LicenseUtilityMethods::isFullySpdxListed)
252            .collect(toMap(k -> k.toString(), v -> v, (t, u) -> t));
253        /*
254         * Licenses specified in the Artifact's POM
255         */
256        Model model = cache.get(artifact);
257        List<AnyLicenseInfo> pom =
258            Stream.of(model.getLicenses())
259            .filter(Objects::nonNull)
260            .flatMap(List::stream)
261            .distinct()
262            .map(t -> parse(t.getName().replaceAll(",", ""),
263                            resolve(toURL(model.getUrl()), t.getUrl())))
264            .collect(toList());
265
266        if (bundle.isEmpty() && pom.isEmpty() && found.isEmpty()) {
267            log.warn(artifact + ": No license(s) specified or found");
268        }
269
270        List<AnyLicenseInfo> licenses = bundle.stream().collect(toList());
271
272        if (! isFullySpecified(licenses)) {
273            if (licenses.isEmpty() || isFullySpecified(pom)) {
274                licenses.clear();
275                licenses.addAll(pom);
276            }
277        }
278
279        if ((! isFullySpecified(licenses)) && found.size() >= licenses.size()) {
280            licenses.clear();
281            licenses.addAll(found.values());
282        }
283
284        if ((! licenses.isEmpty()) && (! isFullySpecified(licenses))) {
285            log.debug("------------------------------------------------------------");
286            log.debug(ArtifactUtils.key(artifact));
287            log.debug(String.valueOf(url));
288            log.debug("      Bundle: " + bundle);
289            log.debug("         POM: " + pom);
290            log.debug("     Scanned: " + scanned);
291            log.debug("       Found: " + found.values());
292            log.debug("  License(s): " + licenses);
293            log.debug("------------------------------------------------------------");
294        }
295
296        return resolver.toLicense(licenses);
297    }

299    private URL toURL(Artifact artifact) {
300        URL url = null;
301
302        try {
303            url =
304                new URI("jar",
305                        artifact.getFile().toURI().toASCIIString() + "!/",
306                        null)
307                .toURL();
308        } catch(URISyntaxException | MalformedURLException exception) {
309            log.debug(artifact + ": " + exception.getMessage(), exception);
310            throw new IllegalStateException(exception);
311        }
312
313        return url;
314    }
315
316    private URL toURL(String string) {
317        URL url = null;
318
319        if (isNotBlank(string)) {
320            try {
321                url = new URI(string).toURL();
322            } catch(URISyntaxException | MalformedURLException exception) {
323                log.debug(string + ": " + exception.getMessage(), exception);
324            }
325        }
326
327        return url;
328    }

330    private String[] resolve(URL root, String urls) {
331        ArrayList<String> list = new ArrayList<>();
332
333        if (isNotBlank(urls)) {
334            for (String url : urls.split("[,\\p{Space}]+")) {
335                if (isNotBlank(url)) {
336                    String value = url;
337
338                    if (root != null && (! isAbsolute(value))) {
339                        String string = root.toString();
340
341                        if (! string.endsWith("/")) {
342                            string += "/";
343                        }
344
345                        value = string + value;
346                    }
347
348                    list.add(value);
349                }
350            }
351        }
352
353        return list.toArray(new String[] { });
354    }
355
356    private boolean isAbsolute(String url) {
357        boolean isAbsolute = false;
358
359        try {
360            isAbsolute = new URI(url).isAbsolute();
361        } catch (Exception exception) {
362        }
363
364        return isAbsolute;
365    }

367    private AnyLicenseInfo parse(String id, String... urls) {
368        AnyLicenseInfo option0 = isNotBlank(id) ? map.get(id) : null;
369        AnyLicenseInfo option1 = null;
370
371        try {
372            option1 = isNotBlank(id) ? resolver.parseLicenseString(id) : null;
373        } catch (Exception exception) {
374        }
375
376        AnyLicenseInfo option2 =
377            new URLLicenseInfo(isNotBlank(id) ? id : urls[0], urls);
378        AnyLicenseInfo[] options =
379            Stream.of(option0, option1, option2)
380            .filter(Objects::nonNull)
381            .map(t -> resolver.parse(t))
382            .toArray(AnyLicenseInfo[]::new);
383
384        Arrays.sort(options, SIEVE);
385
386        return (options.length > 0) ? options[0] : null;
387    }
388
389    private int countOf(Collection<? extends AnyLicenseInfo> collection) {
390        return collection.stream().mapToInt(LicenseUtilityMethods::countOf).sum();
391    }
392
393    private boolean isFullySpecified(Collection<AnyLicenseInfo> collection) {
394        return ((! collection.isEmpty())
395                && collection.stream().allMatch(t -> isFullySpdxListed(t)));
396    }

399        public KeyArtifact(String gav) { this(gav.split(":", 3)); }
400
401        private KeyArtifact(String[] gav) {
402            super(gav[0], gav[1], gav[2], EMPTY, "pom", EMPTY, null);
403        }
