024import com.fasterxml.jackson.databind.JsonNode;
025import com.fasterxml.jackson.databind.ObjectMapper;
026import java.util.Objects;
027import java.util.TreeMap;
028import java.util.stream.Stream;
029import javax.annotation.PostConstruct;
030import javax.annotation.PreDestroy;
031import javax.inject.Inject;
032import javax.inject.Named;
033import javax.inject.Singleton;
034import lombok.extern.slf4j.Slf4j;
035import org.apache.commons.lang3.StringUtils;
036import org.spdx.rdfparser.license.License;
037/* import org.spdx.rdfparser.license.ListedExceptions; */
038import org.spdx.rdfparser.license.ListedLicenses;
039
040import static org.apache.commons.lang3.StringUtils.SPACE;
041
042/**
043 * {@link java.util.Map} implementation that relates SPDX ID and known
044 * aliases to {@link License} sourced from
045 * {@link ListedLicenses#getListedLicenses()}.
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 * @version $Revision: 5736 $
049 */
050@Named @Singleton
051@Slf4j
052public class LicenseMap extends TreeMap<String,License> {
053    private static final ListedLicenses LICENSES =
054        ListedLicenses.getListedLicenses();
055    /*
056     * private static final ListedExceptions EXCEPTIONS =
057     *     ListedExceptions.getListedExceptions();
058     */
059
060    /**
061     * {@code resources/licenses-full.json} read from SPDX artifacts.
062     */
063    public static final JsonNode LICENSES_FULL_JSON;
064
065    static {
066        try {
067            LICENSES_FULL_JSON =
068                new ObjectMapper()
069                .readTree(LicenseMap.class
070                          .getClassLoader()
071                          .getResource("resources/licenses-full.json"));
072        } catch (Exception exception) {
073            throw new ExceptionInInitializerError(exception);
074        }
075    }
076
077    /**
078     * Sole constructor.
079     */
080    @Inject
081    public LicenseMap() {
082        super(String.CASE_INSENSITIVE_ORDER);
083
084        try {
085            for (String key : LICENSES.getSpdxListedLicenseIds()) {
086                License license = LICENSES.getListedLicenseById(key);
087
088                put(key, license);
089                putIfAbsent(license.getName(), license);
090            }
091
092            for (JsonNode node : LICENSES_FULL_JSON.at("/licenses")) {
093                String spdx = node.at("/identifiers/spdx/0").asText();
094
095                if (containsKey(spdx)) {
096                    Stream.of(node.at("/id").asText(),
097                              node.at("/name").asText())
098                        .map(t -> t.trim())
099                        .map(t -> t.replaceAll("[\\p{Space}]+", SPACE))
100                        .filter(StringUtils::isNotBlank)
101                        .forEach(t -> computeIfAbsent(t, k -> get(spdx)));
102                }
103            }
104        } catch (Exception exception) {
105            throw new ExceptionInInitializerError(exception);
106        }
107    }
108
109    @PostConstruct
110    public void init() { }
111
112    @PreDestroy
113    public void destroy() {
114        log.debug(getClass().getSimpleName() + ".size() = " + size());
115    }
116
117    @Override
118    public License get(Object key) {
119        License value = super.get(key);
120
121        if (value == null) {
122            if (key instanceof String) {
123                String string = ((String) key).trim();
124
125                value =
126                    Stream.of(string.replaceAll("(?i)[\\p{Space}]+", "-"),
127                              string.replaceAll("(?i)(V(ERSION)?)[\\p{Space}]?([\\p{Digit}])", "$3"))
128                    .filter(t -> (! string.equals(t)))
129                    .map(t -> get(t))
130                    .filter(Objects::nonNull)
131                    .findFirst().orElse(null);
132            }
133        }
134
135        return value;
136    }
137}
