021import ball.annotation.ServiceProviderFor;
022import ball.xml.FluentNode;
023import com.sun.javadoc.ClassDoc;
024import com.sun.javadoc.Tag;
025import com.sun.tools.doclets.Taglet;
026import java.lang.annotation.Annotation;
027import java.lang.annotation.Retention;
028import java.lang.reflect.Field;
029import java.util.HashSet;
030import java.util.Map;
031import java.util.Set;
032import java.util.stream.Stream;
033import lombok.NoArgsConstructor;
034import lombok.ToString;
035
036import static org.apache.commons.lang3.StringUtils.isNotEmpty;
037
038/**
039 * Inline {@link Taglet} to provide a report of members whose values are
040 * injected.
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 */
044@TagletName("injected.fields")
045@ServiceProviderFor({ Taglet.class })
046@NoArgsConstructor @ToString
047public class InjectedFieldsTaglet extends AbstractInlineTaglet implements SunToolsInternalToolkitTaglet {
048    private static final InjectedFieldsTaglet INSTANCE = new InjectedFieldsTaglet();
049
050    public static void register(Map<Object,Object> map) {
051        register(map, INSTANCE);
052    }
053
054    private static final String[] NAMES = new String[] {
055        javax.annotation.Resource.class.getName(),
056        javax.annotation.Resources.class.getName(),
057        "javax.inject.Inject",
058        "javax.inject.Named",
059        "org.springframework.beans.factory.annotation.Autowired",
060        "org.springframework.beans.factory.annotation.Value",
061        "org.springframework.boot.web.server.LocalServerPort"
062    };
063
064    @Override
065    public FluentNode toNode(Tag tag) throws Throwable {
066        ClassDoc doc = null;
067        String[] argv = tag.text().trim().split("[\\p{Space}]+", 2);
068
069        if (isNotEmpty(argv[0])) {
070            doc = getClassDocFor(tag, argv[0]);
071        } else {
072            doc = containingClass(tag);
073        }
074
075        Set<Class<? extends Annotation>> set = new HashSet<>();
076
077        for (String name : NAMES) {
078            Class<? extends Annotation> annotation = null;
079
080            try {
081                annotation = Class.forName(name, false, getClassLoader()).asSubclass(Annotation.class);
082            } catch (Exception exception) {
083            }
084
085            if (annotation != null) {
086                Retention retention = annotation.getAnnotation(Retention.class);
087
088                if (retention == null) {
089                    throw new IllegalStateException(annotation.getCanonicalName()
090                                                    + " does not specify a retention policy");
091                }
092
093                switch (retention.value()) {
094                case RUNTIME:
095                    break;
096
097                case CLASS:
098                case SOURCE:
099                default:
100                    throw new IllegalStateException(annotation.getCanonicalName()
101                                                    + " specifies " + retention.value() + " retention policy");
102                    /* break; */
103                }
104
105                set.add(annotation);
106            }
107        }
108
109        if (set.isEmpty()) {
110            throw new IllegalStateException("No annotations to map");
111        }
112
113        return div(attr("class", "summary"),
114                   h3("Injected Field Summary"),
115                   table(tag, getClassFor(doc), set));
116    }
117
118    private FluentNode table(Tag tag, Class<?> type, Set<Class<? extends Annotation>> set) {
119        return table(thead(tr(th("Annotation(s)"), th("Field"))),
120                     tbody(Stream.of(type.getDeclaredFields())
121                           .filter(t -> (Stream.of(t.getAnnotations())
122                                         .filter(a -> set.contains(a.annotationType()))
123                                         .findFirst().isPresent()))
124                           .map(t -> tr(tag, t, set))));
125    }
126
127    private FluentNode tr(Tag tag, Field field, Set<Class<? extends Annotation>> set) {
128        return tr(td(fragment(Stream.of(field.getAnnotations())
129                              .filter(t -> set.contains(t.annotationType()))
130                              .map(t -> annotation(tag, t)))),
131                  td(declaration(tag, field)));
132    }
133}
