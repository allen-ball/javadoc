109    /**
110     * Sole constructor.
111     *
112     * @param   path            The path ({@link String}) to the DVD
113     *                          device.
114     *
115     * @throws  IOException     If the device cannot be opened.
116     */
117    public DVDNav(String path) throws IOException {
118        this.path = Objects.requireNonNull(path);
119
120        var selfp = new long[] { 0L };
121
122        if (dvdnav_open(selfp, path) == DVDNAV_STATUS_OK) {
123            self = selfp[0];
124        } else {
125            throw new IOException("Cannot open " + path);
126        }
127
128        if (set_readahead_flag((byte) 0) != DVDNAV_STATUS_OK) {
129            throw new IOException(err_to_string());
130        }
131    }
132
133    protected int set_readahead_flag(byte flag) {
134        return dvdnav_set_readahead_flag(self, flag);
135    }
136
137    protected String err_to_string() { return dvdnav_err_to_string(self); }

139    @Synchronized
140    @Override
141    public void close() {
142        if (! closed) {
143            if (self != 0L) {
144                dvdnav_close(self);
145            }
146        }
147    }

169    public String getTitle() {
170        if (title == null) {
171            var titlep = new long[] { 0L };
172
173            if (dvdnav_get_title_string(self, titlep) == DVDNAV_STATUS_OK) {
174                title = strdup(titlep[0]);
175            }
176        }
177
178        return title;
179    }

193    @Synchronized
194    public Block getNextBlock() throws IOException {
195        Block block = null;
196
197        if (! finished) {
198            var buf = new byte[DVD_VIDEO_LB_LEN];
199            var eventp = new int[] { 0 };
200            var lenp = new int[] { 0 };
201
202            if (get_next_block(buf, eventp, lenp) == DVDNAV_STATUS_OK) {
203                block = new Block(buf, eventp[0], lenp[0]);
204                finished |= (block.getEvent() == DVDNAV_STOP);
205            } else {
206                throw new IOException(err_to_string());
207            }
208        }
209
210        return block;
211    }
212
213    protected int get_next_block(byte[] buf, int[] eventp, int[] lenp) {
214        return dvdnav_get_next_block(self, buf, eventp, lenp);
215    }

470    private String strdup(long pointer) {
471        String string = null;
472
473        if (pointer != 0L) {
474            int length = strlen(pointer);
475            byte[] bytes = new byte[length];
476
477            memmove(bytes, pointer, bytes.length);
478
479            try {
480                string = new String(bytes, UTF_8.name());
481            } catch (UnsupportedEncodingException exception) {
482                throw new IllegalStateException(exception);
483            }
484        }
485
486        return string;
487    }
