106    /**
107     * Sole constructor.
108     *
109     * @param   path            The path ({@link String}) to the DVD
110     *                          device.
111     *
112     * @throws  IOException     If the device cannot be opened.
113     */
114    public DVDNav(String path) throws IOException {
115        this.path = Objects.requireNonNull(path);
116
117        var selfp = new long[] { 0L };
118
119        if (dvdnav_open(selfp, path) == DVDNAV_STATUS_OK) {
120            self = selfp[0];
121        } else {
122            throw new IOException("Cannot open " + path);
123        }
124
125        if (set_readahead_flag((byte) 0) != DVDNAV_STATUS_OK) {
126            throw new IOException(err_to_string());
127        }
128    }
129
130    protected int set_readahead_flag(byte flag) {
131        return dvdnav_set_readahead_flag(self, flag);
132    }
133
134    protected String err_to_string() { return dvdnav_err_to_string(self); }

136    @Synchronized
137    @Override
138    public void close() {
139        if (! closed) {
140            if (self != 0L) {
141                dvdnav_close(self);
142            }
143        }
144    }

166    public String getTitle() {
167        if (title == null) {
168            var titlep = new long[] { 0L };
169
170            if (dvdnav_get_title_string(self, titlep) == DVDNAV_STATUS_OK) {
171                title = strdup(titlep[0]);
172            }
173        }
174
175        return title;
176    }

190    @Synchronized
191    public Block getNextBlock() throws IOException {
192        Block block = null;
193
194        if (! finished) {
195            var buf = new byte[DVD_VIDEO_LB_LEN];
196            var eventp = new int[] { 0 };
197            var lenp = new int[] { 0 };
198
199            if (get_next_block(buf, eventp, lenp) == DVDNAV_STATUS_OK) {
200                block = new Block(buf, eventp[0], lenp[0]);
201                finished |= (block.getEvent() == DVDNAV_STOP);
202            } else {
203                throw new IOException(err_to_string());
204            }
205        }
206
207        return block;
208    }
209
210    protected int get_next_block(byte[] buf, int[] eventp, int[] lenp) {
211        return dvdnav_get_next_block(self, buf, eventp, lenp);
212    }

418    private String strdup(long pointer) {
419        String string = null;
420
421        if (pointer != 0L) {
422            int length = strlen(pointer);
423            byte[] bytes = new byte[length];
424
425            memmove(bytes, pointer, bytes.length);
426
427            try {
428                string = new String(bytes, UTF_8.name());
429            } catch (UnsupportedEncodingException exception) {
430                throw new IllegalStateException(exception);
431            }
432        }
433
434        return string;
435    }
