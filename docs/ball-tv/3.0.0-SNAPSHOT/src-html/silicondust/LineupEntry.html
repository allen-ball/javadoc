023import ball.databind.JSONBean;
024import lombok.NoArgsConstructor;
025
026/**
027 * {@link HDHRPrimeProtocol#getLineup(String)} {@link java.util.List}
028 * element.
029 *
030 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
031 * @version $Revision: 7215 $
032 */
033@NoArgsConstructor
034public class LineupEntry extends JSONBean {
035    private static final long serialVersionUID = -963053630857365297L;
036
037    /** @serial */ private int guideNumber = -1;
038    /** @serial */ private String guideName = null;
039    /** @serial */ private String videoCodec = null;
040    /** @serial */ private String audioCodec = null;
041    /** @serial */ private boolean drm = false;
042    /** @serial */ private String url = null;
043
044    public int getGuideNumber() { return guideNumber; }
045    public String getGuideName() { return guideName; }
046    public String getVideoCodec() { return videoCodec; }
047    public String getAudioCodec() { return audioCodec; }
048    public boolean getDRM() { return drm; }
049    public String getURL() { return url; }
050}
