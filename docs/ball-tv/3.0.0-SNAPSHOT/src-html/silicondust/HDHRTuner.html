126    /**
127     * Static method to determine the HDHomeRun tuners on the network.
128     *
129     * @return  The {@link Map} of IDs and {@link HDHRTuner}s.
130     */
131    public static Map<Integer,HDHRTuner> discover() {
132        var map = new HashMap<Integer,HDHRTuner>();
133
134        for (var reply : new ArrayList<DiscoverReply>(thread().map().values())) {
135            if (! map.containsKey(reply.getID())) {
136                try {
137                    map.put(reply.getID(), new HDHRTuner(reply));
138                } catch (Exception exception) {
139                }
140            }
141
142            if (! map.containsKey(reply.getID())) {
143                thread().map().remove(reply.getInetAddress());
144            }
145        }
146
147        return map;
148    }

165        return thread;
166    }

184    private static DiscoverReply ping(InetAddress address) {
185        DiscoverReply reply = null;
186
187        try (var socket = new DatagramSocket()) {
188            socket.setBroadcast(true);
189            socket.setReuseAddress(true);
190            socket.setSoTimeout(1000);
191
192            new DiscoverRequest().send(socket, address, DISCOVER_UDP_PORT);
193
194            reply = new DiscoverReply();
195            reply.receive(socket);
196        } catch (Exception exception) {
197            reply = null;
198        }
199
200        return reply;
201    }

289    /**
290     * Method to obtain a lock on a tuner.
291     *
292     * @param   tuner           The tuner number.
293     *
294     * @return  The lock key value.
295     *
296     * @throws  IOException     If the tuner cannot be locked.
297     */
298    public int lock(int tuner) throws IOException {
299        int key = 0;
300
301        try {
302            while (key == 0) {
303                key = Math.abs(random.nextInt()) & 0xFFFFFFFF;
304            }
305
306            var name = String.format("/tuner%d/lockkey", tuner);
307            var value = String.format("%d", key);
308            var reply = new GetSetReply();
309
310            new GetSetRequest(0, name, value).send(socket);
311            reply.receive(socket);
312
313            if (reply.getErrorMessage() != null) {
314                throw new IOException(reply.getErrorMessage());
315            }
316
317            var map = reply.map();
318
319            if (NetworkInterface.getByInetAddress(InetAddress.getByName(map.get(name))) == null) {
320                throw new IOException("Cannot lock " + name);
321            }
322        } catch (IOException exception) {
323            throw exception;
324        } catch (Exception exception) {
325            throw new IOException(exception);
326        }
327
328        return key;
329    }

389    /**
390     * Method to set an option value for this {@link HDHRTuner}.
391     *
392     * @param   key             The tuner lockkey (if set).
393     * @param   name            The option name.
394     * @param   value           The option value.
395     * @param   pairs           Optional name/value pairs.
396     *
397     * @throws  IOException     If the option(s) cannot be set.
398     */
399    public void set(int key, String name, String value, String... pairs) throws IOException {
400        var reply = new GetSetReply();
401
402        new GetSetRequest(key, name, value, pairs).send(socket);
403        reply.receive(socket);
404
405        if (reply.getErrorMessage() != null) {
406            throw new IOException(name + ": " + reply.getErrorMessage());
407        }
408
409        var map = reply.map();
410
411        check(name, value, map);
412
413        for (var i = 0; i < pairs.length; i += 2) {
414            check(pairs[i], pairs[i + 1], map);
415        }
416    }
417
418    private void check(String name, String value, Map<String,String> map) throws IOException {
419        if (! value.equals(map.get(name))) {
420            throw new IOException("Could not set " + name + " to " + value);
421        }
422    }

425     * Method to get the configuration options supported by this
426     * {@link HDHRTuner}.
427     *
428     * @return  The {@link SortedSet} of options.
429     *
430     * @throws  IOException     If the options cannot be retrieved.
431     */
432    public SortedSet<String> options() throws IOException {
433        var set = new TreeSet<String>();
434        var string = get(0, "help");
435
436        for (var line : string.split(LF)) {
437            if (line.startsWith(SLASH)) {
438                var option = line.split(SPACE, 2)[0].trim();
439
440                for (int i = 0, n = getCount(); i < n; i += 1) {
441                    set.add(option.replaceAll("<n>", String.valueOf(i)));
442                }
443            }
444        }
445
446        return set;
447    }

449    /**
450     * Method to scan the first tuner available and return a {@link List} of
451     * {@link HDHRProgram}s found.
452     *
453     * @return  The {@link List} of {@link HDHRProgram}s found.
454     *
455     * @throws  IOException     If the {@link HDHRProgram}s cannot be
456     *                          retrieved.
457     */
458    public List<HDHRProgram> scan() throws IOException {
459        List<HDHRProgram> list = null;
460        IOException thrown = null;
461
462        for (int tuner = 0, n = getCount(); tuner < n; tuner += 1) {
463            var key = 0;
464
465            try {
466                key = lock(tuner);
467                list = scan(tuner, key);
468                thrown = null;
469                break;
470            } catch (IOException exception) {
471                thrown = exception;
472                continue;
473            } finally {
474                unlock(tuner, key);
475            }
476        }
477
478        if (thrown != null) {
479            throw thrown;
480        }
481
482        return list;
483    }

485    /**
486     * Method to scan the specified tuner and return a {@link List} of
487     * {@link HDHRProgram}s found.
488     *
489     * @param   tuner           The tuner to scan.
490     * @param   key             The key value.
491     *
492     * @return  The {@link List} of {@link HDHRProgram}s found.
493     *
494     * @throws  IOException     If the {@link HDHRProgram}s cannot be
495     *                          retrieved.
496     */
497    public List<HDHRProgram> scan(int tuner, int key) throws IOException {
498        if (! (0 <= tuner && tuner < getCount())) {
499            throw new IllegalArgumentException("tuner=" + tuner);
500        }
501
502        var list = new ArrayList<HDHRProgram>();
503
504        for (var frequency : new TreeSet<Integer>(Channel.MAP.keySet())) {
505            StatusMap status = null;
506
507            try {
508                channel(tuner, key, frequency);
509                status = new StatusMap(tuner, key);
510            } catch (IOException exception) {
511                continue;
512            }
513
514            var deadline = System.currentTimeMillis() + 2500;
515
516            do {
517                status = new StatusMap(tuner, key);
518
519                if (status.hasLock() && status.getSS() > 45) {
520                    break;
521                } else {
522                    sleep(250);
523                }
524            } while (System.currentTimeMillis() < deadline);
525
526            if (status.hasLock()) {
527                deadline = System.currentTimeMillis() + 5000;
528
529                do {
530                    status = new StatusMap(tuner, key);
531
532                    if (status.getSEQ() >= 100) {
533                        break;
534                    } else {
535                        sleep(250);
536                    }
537                } while (System.currentTimeMillis() < deadline);
538
539                if (status.getSEQ() >= 100) {
540                    sleep(5000);
541
542                    var streaminfo = new StreaminfoMap(tuner, key, frequency);
543
544                    if (streaminfo.getTSID() != null) {
545                        list.addAll(streaminfo.values());
546                    }
547                }
548            }
549        }
550
551        target(tuner, key, null);
552
553        return list;
554    }

716    private static final int TIMEOUT = 10 * 1000;
717
718    private static class DiscoveryThread extends Thread {
719        private final DatagramSocket socket;
720        private final Set<InetAddress> addresses;
721        private final ConcurrentHashMap<InetAddress,DiscoverReply> map = new ConcurrentHashMap<>();
722
723        public DiscoveryThread() {
724            super();
725
726            try {
727                socket = new DatagramSocket();
728                socket.setBroadcast(true);
729                socket.setReuseAddress(true);
730                socket.setSoTimeout(TIMEOUT);
731
732                var list = new ArrayList<InterfaceAddress>();
733
734                Collections.list(NetworkInterface.getNetworkInterfaces()).stream()
735                    .forEach(t -> list.addAll(t.getInterfaceAddresses()));
736
737                addresses =
738                    list.stream()
739                    .filter(Objects::nonNull)
740                    .map(t -> t.getBroadcast())
741                    .filter(Objects::nonNull)
742                    .collect(toSet());
743            } catch (Exception exception) {
744                throw new ExceptionInInitializerError(exception);
745            }
746
747            setDaemon(true);
748            setName(getClass().getName());
749        }
750
751        public Map<InetAddress,DiscoverReply> map() { return map; }
752
753        @Override
754        public void run() {
755            try {
756                var list = List.of(new Receiver(), new Sender());
757
758                for (Thread thread : list) {
759                    thread.start();
760                }
761
762                for (Thread thread : list) {
763                    try {
764                        thread.join();
765                    } catch (InterruptedException exception) {
766                        continue;
767                    }
768                }
769            } finally {
770                if (socket != null) {
771                    socket.close();
772                }
773            }
774        }
775
776        private class Receiver extends Thread {
777            public Receiver() {
778                super();
779
780                setName(getClass().getName());
781            }
782
783            @Override
784            public void run() {
785                var map = new HashMap<InetAddress,DiscoverReply>();
786
787                for (;;) {
788                    try {

983        private static final SortedMap<Short,String> MAP = new TreeMap<>();
984
985        static {
986            try {
987                var list =
988                    FieldUtils.getAllFieldsList(HDHRTuner.class).stream()
989                    .filter(t -> Modifier.isStatic(t.getModifiers()))

1002        private final ByteBuffer payload;
1003        private transient String message = null;

1019            this.buffer.getShort();
1020            this.buffer.getShort();

1024

1046            payload.put(tag);
1047
1048            var bytes = value.getBytes(UTF_8);

1051            payload.put(bytes);
1052            payload.put((byte) 0);

1055        private void putLen(int len) {
1056            if (len <= 127) {
1057                payload.put((byte) (len & 0xFF));
1058            } else {
1059                payload.put((byte) ((len | 0x80) & 0xFF));
1060                payload.put((byte) ((len >> 7)   & 0xFF));
1061            }

1076        protected void seal(int length) {
1077            setPayloadLength(length);
1078            buffer.position(getCRCOffset());
1079
1080            var crc = new CRC32();
1081
1082            crc.update(buffer.array(), 0, buffer.position());
1083
1084            buffer.order(LITTLE_ENDIAN);
1085            buffer.putInt((int) (crc.getValue() & 0xFFFFFFFF));
1086            buffer.order(BIG_ENDIAN);
1087            buffer.flip();
1088        }

1091            setPayloadLength(getPayloadLength());
1092
1093            var payload = payload();
1094
1095            while (payload.hasRemaining()) {
1096                byte tag = payload.get();
1097                int len = payload.get();
1098
1099                if ((len & 0x80) != 0) {
1100                    len &= 0x7F;
1101                    len |= ((int) payload.get()) << 7;
1102                }
1103
1104                var value = payload.slice();
1105
1106                value.limit(len);
1107
1108                payload.position(payload.position() + len);
1109
1110                parse(tag, value);
1111            }
1112        }
