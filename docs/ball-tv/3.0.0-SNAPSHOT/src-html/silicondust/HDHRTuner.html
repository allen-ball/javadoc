130     * Static method to determine the HDHomeRun tuners on the network.
131     *
132     * @return  The {@link Map} of IDs and {@link HDHRTuner}s.
133     */
134    public static Map<Integer,HDHRTuner> discover() {
135        var map = new HashMap<Integer,HDHRTuner>();
136
137        for (var reply : new ArrayList<DiscoverReply>(thread().map().values())) {
138            if (! map.containsKey(reply.getID())) {
139                try {
140                    map.put(reply.getID(), new HDHRTuner(reply));
141                } catch (Exception exception) {
142                }
143            }
144
145            if (! map.containsKey(reply.getID())) {
146                thread().map().remove(reply.getInetAddress());
147            }
148        }
149
150        return map;
151    }

168        return thread;
169    }

187    private static DiscoverReply ping(InetAddress address) {
188        DiscoverReply reply = null;
189
190        try (var socket = new DatagramSocket()) {
191            socket.setBroadcast(true);
192            socket.setReuseAddress(true);
193            socket.setSoTimeout(1000);
194
195            new DiscoverRequest().send(socket, address, DISCOVER_UDP_PORT);
196
197            reply = new DiscoverReply();
198            reply.receive(socket);
199        } catch (Exception exception) {
200            reply = null;
201        }
202
203        return reply;
204    }

292    /**
293     * Method to obtain a lock on a tuner.
294     *
295     * @param   tuner           The tuner number.
296     *
297     * @return  The lock key value.
298     *
299     * @throws  IOException     If the tuner cannot be locked.
300     */
301    public int lock(int tuner) throws IOException {
302        int key = 0;
303
304        try {
305            while (key == 0) {
306                key = Math.abs(random.nextInt()) & 0xFFFFFFFF;
307            }
308
309            var name = String.format("/tuner%d/lockkey", tuner);
310            var value = String.format("%d", key);
311            var reply = new GetSetReply();
312
313            new GetSetRequest(0, name, value).send(socket);
314            reply.receive(socket);
315
316            if (reply.getErrorMessage() != null) {
317                throw new IOException(reply.getErrorMessage());
318            }
319
320            var map = reply.map();
321
322            if (NetworkInterface.getByInetAddress(InetAddress.getByName(map.get(name))) == null) {
323                throw new IOException("Cannot lock " + name);
324            }
325        } catch (IOException exception) {
326            throw exception;
327        } catch (Exception exception) {
328            throw new IOException(exception);
329        }
330
331        return key;
332    }

392    /**
393     * Method to set an option value for this {@link HDHRTuner}.
394     *
395     * @param   key             The tuner lockkey (if set).
396     * @param   name            The option name.
397     * @param   value           The option value.
398     * @param   pairs           Optional name/value pairs.
399     *
400     * @throws  IOException     If the option(s) cannot be set.
401     */
402    public void set(int key, String name, String value, String... pairs) throws IOException {
403        var reply = new GetSetReply();
404
405        new GetSetRequest(key, name, value, pairs).send(socket);
406        reply.receive(socket);
407
408        if (reply.getErrorMessage() != null) {
409            throw new IOException(name + ": " + reply.getErrorMessage());
410        }
411
412        var map = reply.map();
413
414        check(name, value, map);
415
416        for (var i = 0; i < pairs.length; i += 2) {
417            check(pairs[i], pairs[i + 1], map);
418        }
419    }
420
421    private void check(String name, String value, Map<String,String> map) throws IOException {
422        if (! value.equals(map.get(name))) {
423            throw new IOException("Could not set " + name + " to " + value);
424        }
425    }

430     *
431     * @return  The {@link SortedSet} of options.
432     *
433     * @throws  IOException     If the options cannot be retrieved.
434     */
435    public SortedSet<String> options() throws IOException {
436        var set = new TreeSet<String>();
437        var string = get(0, "help");
438
439        for (var line : string.split(LF)) {
440            if (line.startsWith(SLASH)) {
441                var option = line.split(SPACE, 2)[0].trim();
442
443                for (int i = 0, n = getCount(); i < n; i += 1) {
444                    set.add(option.replaceAll("<n>", String.valueOf(i)));
445                }
446            }
447        }
448
449        return set;
450    }

452    /**
453     * Method to scan the first tuner available and return a {@link List} of
454     * {@link HDHRProgram}s found.
455     *
456     * @return  The {@link List} of {@link HDHRProgram}s found.
457     *
458     * @throws  IOException     If the {@link HDHRProgram}s cannot be
459     *                          retrieved.
460     */
461    public List<HDHRProgram> scan() throws IOException {
462        List<HDHRProgram> list = null;
463        IOException thrown = null;
464
465        for (int tuner = 0, n = getCount(); tuner < n; tuner += 1) {
466            var key = 0;
467
468            try {
469                key = lock(tuner);
470                list = scan(tuner, key);
471                thrown = null;
472                break;
473            } catch (IOException exception) {
474                thrown = exception;
475                continue;
476            } finally {
477                unlock(tuner, key);
478            }
479        }
480
481        if (thrown != null) {
482            throw thrown;
483        }
484
485        return list;
486    }

488    /**
489     * Method to scan the specified tuner and return a {@link List} of
490     * {@link HDHRProgram}s found.
491     *
492     * @param   tuner           The tuner to scan.
493     * @param   key             The key value.
494     *
495     * @return  The {@link List} of {@link HDHRProgram}s found.
496     *
497     * @throws  IOException     If the {@link HDHRProgram}s cannot be
498     *                          retrieved.
499     */
500    public List<HDHRProgram> scan(int tuner, int key) throws IOException {
501        if (! (0 <= tuner && tuner < getCount())) {
502            throw new IllegalArgumentException("tuner=" + tuner);
503        }
504
505        var list = new ArrayList<HDHRProgram>();
506
507        for (var frequency : new TreeSet<Integer>(Channel.MAP.keySet())) {
508            StatusMap status = null;
509
510            try {
511                channel(tuner, key, frequency);
512                status = new StatusMap(tuner, key);
513            } catch (IOException exception) {
514                continue;
515            }
516
517            var deadline = System.currentTimeMillis() + 2500;
518
519            do {
520                status = new StatusMap(tuner, key);
521
522                if (status.hasLock() && status.getSS() > 45) {
523                    break;
524                } else {
525                    sleep(250);
526                }
527            } while (System.currentTimeMillis() < deadline);
528
529            if (status.hasLock()) {
530                deadline = System.currentTimeMillis() + 5000;
531
532                do {
533                    status = new StatusMap(tuner, key);
534
535                    if (status.getSEQ() >= 100) {
536                        break;
537                    } else {
538                        sleep(250);
539                    }
540                } while (System.currentTimeMillis() < deadline);
541
542                if (status.getSEQ() >= 100) {
543                    sleep(5000);
544
545                    var streaminfo = new StreaminfoMap(tuner, key, frequency);
546
547                    if (streaminfo.getTSID() != null) {
548                        list.addAll(streaminfo.values());
549                    }
550                }
551            }
552        }
553
554        target(tuner, key, null);
555
556        return list;
557    }

721    private static class DiscoveryThread extends Thread {
722        private final DatagramSocket socket;
723        private final Set<InetAddress> addresses;
724        private final ConcurrentHashMap<InetAddress,DiscoverReply> map = new ConcurrentHashMap<>();
725
726        public DiscoveryThread() {
727            super();
728
729            try {
730                socket = new DatagramSocket();
731                socket.setBroadcast(true);
732                socket.setReuseAddress(true);
733                socket.setSoTimeout(TIMEOUT);
734
735                var list = new ArrayList<InterfaceAddress>();
736
737                Collections.list(NetworkInterface.getNetworkInterfaces()).stream()
738                    .forEach(t -> list.addAll(t.getInterfaceAddresses()));
739
740                addresses =
741                    list.stream()
742                    .filter(Objects::nonNull)
743                    .map(t -> t.getBroadcast())
744                    .filter(Objects::nonNull)
745                    .collect(toSet());
746            } catch (Exception exception) {
747                throw new ExceptionInInitializerError(exception);
748            }
749
750            setDaemon(true);
751            setName(getClass().getName());
752        }
753
754        public Map<InetAddress,DiscoverReply> map() { return map; }
755
756        @Override
757        public void run() {
758            try {
759                var list = List.of(new Receiver(), new Sender());
760
761                for (Thread thread : list) {
762                    thread.start();
763                }
764
765                for (Thread thread : list) {
766                    try {
767                        thread.join();
768                    } catch (InterruptedException exception) {
769                        continue;
770                    }
771                }
772            } finally {
773                if (socket != null) {
774                    socket.close();
775                }
776            }

986    private static class Pkt {
987        private static final SortedMap<Short,String> MAP = new TreeMap<>();
988
989        static {
990            try {
991                var list =
992                    FieldUtils.getAllFieldsList(HDHRTuner.class).stream()
993                    .filter(t -> Modifier.isStatic(t.getModifiers()))
994                    .filter(t -> Short.class.isAssignableFrom(t.getType()))
995                    .collect(toList());
996
997                for (var field : list) {
998                    MAP.put(field.getShort(null), field.getName());
999                }
1000            } catch (Exception exception) {
1001                throw new ExceptionInInitializerError(exception);
1002            }
1003        }
1004
1005        private final ByteBuffer buffer;
1006        private final ByteBuffer payload;
1007        private transient String message = null;
1008
1009        public Pkt() { this((short) 0); }
1010
1011        public Pkt(short type) {
1012            this(ByteBuffer.allocate(MAX_PACKET_SIZE));
1013
1014            buffer.rewind();
1015            payload.rewind();
1016
1017            setType(type);
1018            setPayloadLength(MAX_PAYLOAD_SIZE);
1019        }
1020
1021        private Pkt(ByteBuffer buffer) {
1022            this.buffer = buffer.order(BIG_ENDIAN);
1023            this.buffer.getShort();
1024            this.buffer.getShort();
1025
1026            this.payload = buffer.slice();
1027        }
1028
1029        protected ByteBuffer buffer() { return buffer; }
1030        protected ByteBuffer payload() { return payload; }
1031
1032        public int getLength() { return getCRCOffset() + 4; }
1033
1034        public short getType() { return buffer.getShort(0); }
1035        protected void setType(short type) { buffer.putShort(0, type); }
1036
1037        protected void put(byte tag, byte value) {
1038            payload.put(tag); putLen(1); payload.put(value);
1039        }
1040
1041        protected void put(byte tag, short value) {
1042            payload.put(tag); putLen(2); payload.putShort(value);
1043        }
1044
1045        protected void put(byte tag, int value) {
1046            payload.put(tag); putLen(4); payload.putInt(value);
1047        }
1048
1049        protected void put(byte tag, String value) {
1050            payload.put(tag);
1051
1052            var bytes = value.getBytes(UTF_8);
1053
1054            putLen(bytes.length + 1);
1055            payload.put(bytes);
1056            payload.put((byte) 0);
1057        }
1058
1059        private void putLen(int len) {
1060            if (len <= 127) {
1061                payload.put((byte) (len & 0xFF));
1062            } else {
1063                payload.put((byte) ((len | 0x80) & 0xFF));
1064                payload.put((byte) ((len >> 7)   & 0xFF));
1065            }
1066        }
1067
1068        protected String toString(ByteBuffer buffer) {
1069            return new String(buffer.array(), buffer.arrayOffset(), buffer.limit() - 1, UTF_8);
1070        }
1071
1072        public int getPayloadLength() { return buffer.getShort(2); }
1073        protected void setPayloadLength(int length) {
1074            buffer.putShort(2, (short) (length & 0xFFFF));
1075            payload.limit(getPayloadLength());
1076        }
1077
1078        public int getCRCOffset() { return 4 + getPayloadLength(); }
1079
1080        protected void seal(int length) {
1081            setPayloadLength(length);
1082            buffer.position(getCRCOffset());
1083
1084            var crc = new CRC32();
1085
1086            crc.update(buffer.array(), 0, buffer.position());
1087
1088            buffer.order(LITTLE_ENDIAN);
1089            buffer.putInt((int) (crc.getValue() & 0xFFFFFFFF));
1090            buffer.order(BIG_ENDIAN);
1091            buffer.flip();
1092        }
1093
1094        protected void parse() {
1095            setPayloadLength(getPayloadLength());
1096
1097            var payload = payload();
1098
1099            while (payload.hasRemaining()) {
1100                byte tag = payload.get();
1101                int len = payload.get();
1102
1103                if ((len & 0x80) != 0) {
1104                    len &= 0x7F;
1105                    len |= ((int) payload.get()) << 7;
1106                }
1107
1108                var value = payload.slice();
1109
1110                value.limit(len);
1111
1112                payload.position(payload.position() + len);
1113
1114                parse(tag, value);
1115            }
1116        }
1117
1118        protected void parse(byte tag, ByteBuffer value) {
1119            switch (tag) {
1120            case TAG_ERROR_MESSAGE:
1121                message = toString(value);
1122                break;
1123
1124            default:
1125                break;
1126            }
1127        }
1128
1129        public String getErrorMessage() { return message; }
1130
1131        public void receive(DatagramSocket socket) throws IOException {
1132            buffer().clear();
1133
1134            var packet = new DatagramPacket(buffer().array(), getLength());
1135
1136            socket.receive(packet);
1137
1138            receive(packet);
1139        }
1140
1141        protected void receive(DatagramPacket packet) { parse(); }
1142
1143        public void receive(Socket socket) throws IOException {
1144            buffer().clear();
1145
1146            int length = socket.getInputStream().read(buffer.array(), 0, buffer.limit());
1147
1148            buffer.limit(length);
1149
1150            parse();
1151        }
1152
1153        public void send(DatagramSocket socket, InetAddress address, int port) throws IOException {
1154            var packet = new DatagramPacket(buffer().array(), getLength());
1155
1156            packet.setAddress(address);
1157            packet.setPort(port);
1158
1159            socket.send(packet);
1160        }
1161
1162        public void send(Socket socket) throws IOException {
1163            socket.getOutputStream().write(buffer().array(), 0, getLength());
1164        }
1165
1166        @Override
1167        public String toString() {
1168            var type = MAP.get(getType());
1169
1170            return (LB + ((type != null) ? type : String.valueOf(getType()))
1171                    + COLON + getPayloadLength() + RB);
1172        }
