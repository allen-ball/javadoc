021import java.util.Collections;
022import java.util.Comparator;
023import java.util.Objects;
024import java.util.SortedMap;
025import java.util.SortedSet;
026import java.util.TreeMap;
027import java.util.TreeSet;
028
029/**
030 * TV {@link Channel}.
031 *
032 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
033 */
034public class Channel implements Comparable<Channel> {
035    private static final Comparator<? super Channel> COMPARATOR =
036        Comparator
037        .comparingInt(Channel::getNumber)
038        .thenComparingInt(Channel::getFrequency)
039        .thenComparing(Channel::getMapName);
040
041    /**
042     * Frequency/{@link Channel} {@link SortedMap} of all known
043     * {@link Channel}s.
044     */
045    public static final SortedMap<Integer,SortedSet<Channel>> MAP;
046
047    static {
048        var map = new TreeMap<Integer,SortedSet<Channel>>();
049
050        for (var value : ChannelRangeMap.MAP.values()) {
051            for (var channel : value.channelSet()) {
052                map.computeIfAbsent(channel.getFrequency(), k -> new TreeSet<>())
053                    .add(channel);
054            }
055        }
056
057        for (int key : new TreeSet<Integer>(map.keySet())) {
058            map.put(key, Collections.unmodifiableSortedSet(map.get(key)));
059        }
060
061        MAP = Collections.unmodifiableSortedMap(map);
062    }
063
064    private final String map;
065    private final int number;
066    private final int frequency;
067
068    /**
069     * Sole constructor.
070     *
071     * @param   map             The {@link Channel} map name.
072     * @param   number          The {@link Channel} number.
073     * @param   frequency       The {@link Channel} frequency.
074     */
075    public Channel(String map, int number, int frequency) {
076        this.map = map;
077        this.number = number;
078        this.frequency = frequency;
079    }
080
081    /**
082     * @return  The {@link Channel} map name.
083     */
084    public String getMapName() { return map; }
085
086    /**
087     * @return  The {@link Channel} number.
088     */
089    public int getNumber() { return number; }
090
091    /**
092     * @return  {@link Channel} (digital center) frequency.
093     */
094    public int getFrequency() { return frequency; }
095
096    @Override
097    public int compareTo(Channel that) {
098        return Objects.compare(this, that, COMPARATOR);
099    }
100
101    @Override
102    public boolean equals(Object object) {
103        return ((object instanceof Channel)
104                    ? (object != null && this.compareTo((Channel) object) == 0)
105                    : super.equals(object));
106    }
107
108    @Override
109    public int hashCode() {
110        return Objects.hash(getNumber(), getFrequency(), getMapName());
111    }
112
113    @Override
114    public String toString() {
115        return getMapName() + ":" +  getNumber() + ":" + getFrequency();
116    }
117}
