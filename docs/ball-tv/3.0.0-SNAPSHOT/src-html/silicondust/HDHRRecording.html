021import ball.activation.DataSourceDefaultMethods;
022import ball.activation.ReaderWriterDataSource;
023import ball.annotation.MatcherGroup;
024import ball.annotation.PatternRegex;
025import ball.util.PatternMatcherBean;
026import com.fasterxml.jackson.databind.JsonNode;
027import java.io.EOFException;
028import java.io.File;
029import java.io.IOException;
030import java.io.InputStream;
031import java.io.OutputStream;
032import java.nio.ByteBuffer;
033import java.text.SimpleDateFormat;
034import java.util.TimeZone;
035import javax.activation.FileDataSource;
036import lombok.ToString;
037
038import static java.nio.charset.StandardCharsets.UTF_8;
039import static org.apache.commons.lang3.StringUtils.isBlank;
040
041/**
042 * {@link.uri http://www.silicondust.com/ SiliconDust} HDHomeRun recording.
043 *
044 * {@bean.info}
045 *
046 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
047 */
048@PatternRegex("(?is)^(?<title>.+)( (?<episodeNumber>S[0-9]+E[0-9]+))( (?<origAirDate>[0-9]{8}))( \\[(?<startTime>[0-9]{8}-[0-9]{4})\\])[.]mpg$")
049public class HDHRRecording extends FileDataSource implements DataSourceDefaultMethods, PatternMatcherBean {
050    private static final SimpleDateFormat YYYYMMDD = new SimpleDateFormatImpl("yyyyMMdd");
051    private static final SimpleDateFormat YYYYMMDD_HHMM = new SimpleDateFormatImpl("yyyyMMdd-HHmm");
052
053    private final long offset;
054    private final JsonNode json;
055    private String title = null;
056    private String episodeTitle = null;
057    private String episodeNumber = null;
058    private Long origAirDate = null;
059    private Long startTime = null;
060
061    /**
062     * Sole constructor.
063     *
064     * @param   file            The recording {@link java.io.File}.
065     *
066     * @throws  IOException     If the recording metadata cannot be read.
067     */
068    public HDHRRecording(File file) throws IOException {
069        super(file);
070
071        PatternMatcherBean.super.initialize(getName());
072
073        long offset = 0;
074        var json = new ReaderWriterDataSource(null, null, UTF_8);
075        var buffer = ByteBuffer.allocate(188);
076
077        try (var in = super.getInputStream();
078             var out = json.getOutputStream()) {
079            for (;;) {
080                try {
081                    buffer.clear();
082                    buffer.limit(in.read(buffer.array()));
083
084                    if (buffer.limit() == buffer.capacity()) {
085                        if (buffer.get(0) == (byte) 'G'
086                            && (buffer.get(1) == (byte) '_' || buffer.get(1) == (byte) 0037)
087                            && buffer.get(2) == (byte) 0372) {
088                            offset += buffer.limit();
089
090                            buffer.position(buffer.position() + 4);
091
092                            out.write(buffer.array(), buffer.position(), buffer.remaining());
093                        } else {
094                            break;
095                        }
096                    } else {
097                        break;
098                    }
099                } catch (IllegalArgumentException exception) {
100                    throw new EOFException("Expected " + buffer.capacity() + " bytes");
101                }
102            }
103        }
104
105        this.offset = offset;
106        this.json = ObjectMapperConfiguration.MAPPER.readTree(json.toString());
107    }
108
109    @MatcherGroup(1)
110    protected void setTitle(String string) {
111        if (! isBlank(string)) {
112            title = string;
113        }
114    }
115
116    protected void setEpisodeTitle(String string) {
117        if (! isBlank(string)) {
118            episodeTitle = string;
119        }
120    }
121
122    @MatcherGroup(3)
123    protected void setEpisodeNumber(String string) {
124        if (! isBlank(string)) {
125            episodeNumber = string;
126        }
127    }
128
129    @MatcherGroup(5)
130    protected void setOrigAirDate(String string) {
131        try {
132            if (! isBlank(string)) {
133                origAirDate = YYYYMMDD.parse(string).getTime();
134            }
135        } catch (RuntimeException exception) {
136            throw exception;
137        } catch (Exception exception) {
138            throw new IllegalArgumentException(String.valueOf(string),
139                                               exception);
140        }
141    }
142
143    @MatcherGroup(7)
144    protected void setStartTime(String string) {
145        try {
146            if (! isBlank(string)) {
147                startTime = YYYYMMDD_HHMM.parse(string).getTime();
148            }
149        } catch (RuntimeException exception) {
150            throw exception;
151        } catch (Exception exception) {
152            throw new IllegalArgumentException(String.valueOf(string),
153                                               exception);
154        }
155    }
156
157    public long getOffset() { return offset; }
158
159    @Override
160    public InputStream getInputStream() throws IOException {
161        var in = super.getInputStream();
162
163        in.skip(getOffset());
164
165        return in;
166    }
167
168    @Override
169    public OutputStream getOutputStream() throws IOException {
170        return DataSourceDefaultMethods.super.getOutputStream();
171    }
172
173    @Override
174    public String toString() { return String.valueOf(json); }
175
176    @ToString
177    private static class SimpleDateFormatImpl extends SimpleDateFormat {
178        private static final long serialVersionUID = -8532683766075038547L;
179
180        private static final TimeZone UTC = TimeZone.getTimeZone("UTC");
181
182        public SimpleDateFormatImpl(String pattern) {
183            super(pattern);
184
185            setLenient(true);
186            setTimeZone(UTC);
187        }
188    }
189}
