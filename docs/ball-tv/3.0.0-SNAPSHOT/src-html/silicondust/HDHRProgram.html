021import ball.annotation.CompileTimeCheck;
022import ball.annotation.ConstantValueMustConvertTo;
023import java.util.LinkedHashSet;
024import java.util.List;
025import java.util.Set;
026import java.util.regex.Pattern;
027
028import static org.apache.commons.lang3.StringUtils.EMPTY;
029
030/**
031 * {@link.uri http://www.silicondust.com/ SiliconDust} HDHomeRun program
032 * (sub-channel).
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 */
036public class HDHRProgram {
037
038    /**
039     * {@link #REGEX} = {@value #REGEX}
040     */
041    @ConstantValueMustConvertTo(value = Pattern.class, method = "compile")
042    protected static final String REGEX =
043        "(?<number>[\\p{Digit}]+)"
044        + ":[\\p{Space}]+(?<virtual>[\\p{Digit}]+([.][\\p{Digit}]+)?)"
045        + "([\\p{Space}]+(?<name>[^()]+))?"
046        + "([\\p{Space}]+([(](?<flags>.*)[)]))?";
047
048    /**
049     * Compiled {@link #REGEX}.
050     */
051    @CompileTimeCheck
052    protected static final Pattern PATTERN = Pattern.compile(REGEX);
053
054    private final int frequency;
055    private final String modulation;
056    private final Integer tsid;
057    private final int number;
058    private final String virtual;
059    private final String name;
060    private final LinkedHashSet<String> flags = new LinkedHashSet<>();
061
062    private static final String COLON = ":";
063    private static final String COMMA = ",";
064
065    /**
066     * Sole constructor.
067     *
068     * @param   frequency       The {@link Channel} frequency.
069     * @param   modulation      The modulation.
070     * @param   tsid            The transport stream ID.
071     * @param   line            The line to parse.
072     */
073    protected HDHRProgram(int frequency, String modulation, Integer tsid, String line) {
074        this.frequency = frequency;
075        this.modulation = modulation;
076        this.tsid = tsid;
077
078        var matcher = PATTERN.matcher(line);
079
080        if (matcher.matches()) {
081            this.number = Integer.decode(matcher.group("number"));
082            this.virtual = matcher.group("virtual");
083            this.name = matcher.group("name");
084
085            var string = matcher.group("flags");
086
087            if (string != null) {
088                this.flags.addAll(List.of(string.split(COMMA)));
089                this.flags.remove(EMPTY);
090            }
091        } else {
092            throw new ExceptionInInitializerError(line);
093        }
094    }
095
096    public int getFrequency() { return frequency; }
097    public String getModulation() { return modulation; }
098    public Integer getTSID() { return tsid; }
099    public int getNumber() { return number; }
100    public String getVirtual() { return virtual; }
101    public String getName() { return name; }
102    public Set<String> getFlags() { return flags; }
103
104    @Override
105    public String toString() {
106        return (getModulation()
107                + COLON + String.valueOf(getFrequency())
108                + COLON + String.valueOf(getNumber())
109                + COLON + getVirtual()
110                + COLON + ((getName() != null) ? getName() : EMPTY)
111                + COLON + String.valueOf(getFlags()));
112    }
113}
