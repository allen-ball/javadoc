023import ball.http.ant.taskdefs.HTTPTask;
024import ball.util.ant.taskdefs.AntTask;
025import ball.util.ant.taskdefs.NotNull;
026import java.io.File;
027import java.io.FileOutputStream;
028import java.net.InetAddress;
029import lombok.Getter;
030import lombok.NoArgsConstructor;
031import lombok.Setter;
032import lombok.ToString;
033import org.apache.tools.ant.BuildException;
034import silicondust.HDHRPrimeClient;
035
036import static lombok.AccessLevel.PROTECTED;
037
038/**
039 * Abstract {@link.uri http://ant.apache.org/ Ant} base
040 * {@link org.apache.tools.ant.Task} for tasks that invoke
041 * {@link HDHRPrimeClient} methods.
042 *
043 * {@ant.task}
044 *
045 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
046 * @version $Revision: 8378 $
047 */
048@NoArgsConstructor(access = PROTECTED)
049public abstract class HDHRPrimeClientTask extends HTTPTask {
050    @NotNull @Getter
051    private InetAddress address = null;
052    protected HDHRPrimeClient client = null;
053
054    public void setAddress(String address) throws BuildException {
055        try {
056            this.address = InetAddress.getByName(address);
057        } catch (BuildException exception) {
058            throw exception;
059        } catch (RuntimeException exception) {
060            throw exception;
061        } catch (Throwable throwable) {
062            throw new BuildException(throwable);
063        }
064    }
065
066    @Override
067    public void execute() throws BuildException {
068        super.execute();
069
070        try {
071            client = new HDHRPrimeClientImpl();
072        } catch (BuildException exception) {
073            throw exception;
074        } catch (RuntimeException exception) {
075            throw exception;
076        } catch (Throwable throwable) {
077            throw new BuildException(throwable);
078        }
079    }
080
081    /**
082     * {@link.uri http://ant.apache.org/ Ant}
083     * {@link org.apache.tools.ant.Task} to invoke
084     * {@link HDHRPrimeClient#getLineup()}.
085     *
086     * {@ant.task}
087     */
088    @AntTask("hdhr-lineup")
089    @NoArgsConstructor @ToString
090    public static class Lineup extends HDHRPrimeClientTask {
091        @Override
092        public void execute() throws BuildException {
093            super.execute();
094
095            try {
096                client.getLineup().stream()
097                    .map(String::valueOf)
098                    .forEach(this::log);
099            } catch (BuildException exception) {
100                throw exception;
101            } catch (RuntimeException exception) {
102                throw exception;
103            } catch (Throwable throwable) {
104                throw new BuildException(throwable);
105            }
106        }
107    }
108
109    /**
110     * {@link.uri http://ant.apache.org/ Ant}
111     * {@link org.apache.tools.ant.Task} to invoke
112     * {@link HDHRPrimeClient#stream(Integer,Integer,Integer)} and save
113     * the stream to a {@link File}.
114     *
115     * {@ant.task}
116     */
117    @AntTask("hdhr-record")
118    @NoArgsConstructor @ToString
119    public static class Record extends HDHRPrimeClientTask {
120        @Getter @Setter
121        private Integer tuner = null;
122        @NotNull @Getter @Setter
123        private Integer channel = null;
124        @NotNull @Getter @Setter
125        private Integer duration = null;
126        @NotNull @Getter @Setter
127        private File file = null;
128
129        @Override
130        public void execute() throws BuildException {
131            super.execute();
132
133            try (var out = new FileOutputStream(getFile())) {
134                client
135                    .stream(getTuner(), getChannel(), getDuration())
136                    .writeTo(out);
137            } catch (BuildException exception) {
138                throw exception;
139            } catch (RuntimeException exception) {
140                throw exception;
141            } catch (Throwable throwable) {
142                throw new BuildException(throwable);
143            }
144        }
145    }
146
147    private class HDHRPrimeClientImpl extends HDHRPrimeClient {
148        public HDHRPrimeClientImpl() {
149            super(builder(), HDHRPrimeClientTask.this.getAddress().getHostAddress());
150        }
151    }
152}
