021import ball.http.ant.taskdefs.HTTPTask;
022import ball.util.ant.taskdefs.AntTask;
023import ball.util.ant.taskdefs.NotNull;
024import java.io.File;
025import java.io.FileOutputStream;
026import java.net.InetAddress;
027import lombok.Getter;
028import lombok.NoArgsConstructor;
029import lombok.Setter;
030import lombok.ToString;
031import org.apache.tools.ant.BuildException;
032import silicondust.HDHRPrimeClient;
033
034import static lombok.AccessLevel.PROTECTED;
035
036/**
037 * Abstract {@link.uri http://ant.apache.org/ Ant} base
038 * {@link org.apache.tools.ant.Task} for tasks that invoke
039 * {@link HDHRPrimeClient} methods.
040 *
041 * {@ant.task}
042 *
043 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
044 */
045@NoArgsConstructor(access = PROTECTED)
046public abstract class HDHRPrimeClientTask extends HTTPTask {
047    @NotNull @Getter
048    private InetAddress address = null;
049    protected HDHRPrimeClient client = null;
050
051    public void setAddress(String address) throws BuildException {
052        try {
053            this.address = InetAddress.getByName(address);
054        } catch (BuildException exception) {
055            throw exception;
056        } catch (RuntimeException exception) {
057            throw exception;
058        } catch (Throwable throwable) {
059            throw new BuildException(throwable);
060        }
061    }
062
063    @Override
064    public void execute() throws BuildException {
065        super.execute();
066
067        try {
068            client = new HDHRPrimeClientImpl();
069        } catch (BuildException exception) {
070            throw exception;
071        } catch (RuntimeException exception) {
072            throw exception;
073        } catch (Throwable throwable) {
074            throw new BuildException(throwable);
075        }
076    }
077
078    /**
079     * {@link.uri http://ant.apache.org/ Ant}
080     * {@link org.apache.tools.ant.Task} to invoke
081     * {@link HDHRPrimeClient#getLineup()}.
082     *
083     * {@ant.task}
084     */
085    @AntTask("hdhr-lineup")
086    @NoArgsConstructor @ToString
087    public static class Lineup extends HDHRPrimeClientTask {
088        @Override
089        public void execute() throws BuildException {
090            super.execute();
091
092            try {
093                client.getLineup().stream()
094                    .map(String::valueOf)
095                    .forEach(this::log);
096            } catch (BuildException exception) {
097                throw exception;
098            } catch (RuntimeException exception) {
099                throw exception;
100            } catch (Throwable throwable) {
101                throw new BuildException(throwable);
102            }
103        }
104    }
105
106    /**
107     * {@link.uri http://ant.apache.org/ Ant}
108     * {@link org.apache.tools.ant.Task} to invoke
109     * {@link HDHRPrimeClient#stream(Integer,Integer,Integer)} and save
110     * the stream to a {@link File}.
111     *
112     * {@ant.task}
113     */
114    @AntTask("hdhr-record")
115    @NoArgsConstructor @ToString
116    public static class Record extends HDHRPrimeClientTask {
117        @Getter @Setter
118        private Integer tuner = null;
119        @NotNull @Getter @Setter
120        private Integer channel = null;
121        @NotNull @Getter @Setter
122        private Integer duration = null;
123        @NotNull @Getter @Setter
124        private File file = null;
125
126        @Override
127        public void execute() throws BuildException {
128            super.execute();
129
130            try (var out = new FileOutputStream(getFile())) {
131                client
132                    .stream(getTuner(), getChannel(), getDuration())
133                    .writeTo(out);
134            } catch (BuildException exception) {
135                throw exception;
136            } catch (RuntimeException exception) {
137                throw exception;
138            } catch (Throwable throwable) {
139                throw new BuildException(throwable);
140            }
141        }
142    }
143
144    private class HDHRPrimeClientImpl extends HDHRPrimeClient {
145        public HDHRPrimeClientImpl() {
146            super(builder(), HDHRPrimeClientTask.this.getAddress().getHostAddress());
147        }
148    }
149}
