021import ball.swing.table.ArrayListTableModel;
022import ball.util.ant.taskdefs.AnnotatedAntTask;
023import ball.util.ant.taskdefs.AntTask;
024import ball.util.ant.taskdefs.ClasspathDelegateAntTask;
025import ball.util.ant.taskdefs.ConfigurableAntTask;
026import lombok.Getter;
027import lombok.NoArgsConstructor;
028import lombok.Setter;
029import lombok.ToString;
030import lombok.experimental.Accessors;
031import org.apache.tools.ant.BuildException;
032import org.apache.tools.ant.Task;
033import org.apache.tools.ant.util.ClasspathUtils;
034import silicondust.HDHRTuner;
035
036/**
037 * {@link.uri http://ant.apache.org/ Ant} {@link Task} to discover HDHomeRun
038 * tuners on the network.
039 *
040 * {@ant.task}
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 */
044@AntTask("hdhr-discover")
045@NoArgsConstructor @ToString
046public class HDHRDiscoverTask extends Task implements AnnotatedAntTask, ClasspathDelegateAntTask, ConfigurableAntTask {
047    @Getter @Setter @Accessors(chain = true, fluent = true)
048    private ClasspathUtils.Delegate delegate = null;
049
050    @Override
051    public void init() throws BuildException {
052        super.init();
053        ClasspathDelegateAntTask.super.init();
054        ConfigurableAntTask.super.init();
055    }
056
057    @Override
058    public void execute() throws BuildException {
059        super.execute();
060        AnnotatedAntTask.super.execute();
061
062        try {
063            var map = HDHRTuner.discover();
064
065            if (! map.values().isEmpty()) {
066                log();
067                log(new TableModelImpl(map.values()));
068            }
069        } catch (BuildException exception) {
070            throw exception;
071        } catch (Throwable throwable) {
072            throw new BuildException(throwable);
073        }
074    }
075
076    private class TableModelImpl extends ArrayListTableModel<HDHRTuner> {
077        private static final long serialVersionUID = -8623509340568397843L;
078
079        public TableModelImpl(Iterable<HDHRTuner> iterable) {
080            super(iterable, "URI", "ID", "Model", "Count");
081        }
082
083        @Override
084        protected Object getValueAt(HDHRTuner row, int x) {
085            Object object = null;
086
087            switch (x) {
088            case 0:
089            default:
090                object = row.getURI();
091                break;
092
093            case 1:
094                object = Integer.toString(row.getID(), 16);
095                break;
096
097            case 2:
098                object = row.getModel();
099                break;
100
101            case 3:
102                object = row.getCount();
103                break;
104            }
105
106            return object;
107        }
108    }
109}
