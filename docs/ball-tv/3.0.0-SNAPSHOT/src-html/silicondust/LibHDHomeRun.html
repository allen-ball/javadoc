021import lombok.NoArgsConstructor;
022import lombok.ToString;
023import org.fusesource.hawtjni.runtime.JniArg;
024import org.fusesource.hawtjni.runtime.JniClass;
025import org.fusesource.hawtjni.runtime.JniField;
026import org.fusesource.hawtjni.runtime.JniMethod;
027import org.fusesource.hawtjni.runtime.Library;
028
029import static org.fusesource.hawtjni.runtime.FieldFlag.CONSTANT;
030import static org.fusesource.hawtjni.runtime.MethodFlag.CONSTANT_INITIALIZER;
031
032/**
033 * {@link LibHDHomeRun} is the Java entry-point to
034 * {@link.uri https://github.com/Silicondust/libhdhomerun libhdhomerun}.
035 *
036 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
037 */
038@JniClass
039@NoArgsConstructor @ToString
040public class LibHDHomeRun {
041    private static final Library LIBRARY = new Library("silicondust", LibHDHomeRun.class);
042
043    @JniMethod(flags = { CONSTANT_INITIALIZER })
044    private static final native void init();
045
046    static { LIBRARY.load(); init(); }
047}
