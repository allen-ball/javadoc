044/**
045 * Abstract {@link.uri http://ant.apache.org/ Ant} base
046 * {@link org.apache.tools.ant.Task} for tasks that invoke
047 * {@link SDClient} methods.
048 *
049 * {@ant.task}
050 *
051 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
052 * @version $Revision: 8378 $
053 */
054@NoArgsConstructor(access = PROTECTED)
055public abstract class SDClientTask extends HTTPTask implements Callable<Object> {
056    protected SDClient client = null;
057
058    @Override
059    public void execute() throws BuildException {
060        super.execute();
061
062        try {
063            client = new SDClientImpl();
064
065            var result = call();
066
067            if (result instanceof Map) {
068                log(new MapTableModel((Map<?,?>) result));
069            } else if (result instanceof JsonNode) {
070                log(MAPPER.writeValueAsString((JsonNode) result));
071            } else if (result instanceof Iterable) {
072                for (Object object : (Iterable<?>) result) {
073                    log(String.valueOf(object));
074                }
075            } else if (result != Void.class) {
076                log(String.valueOf(result));
077            }
078        } catch (BuildException exception) {
079            throw exception;
080        } catch (RuntimeException exception) {
081            throw exception;
082        } catch (Throwable throwable) {
083            throw new BuildException(throwable);
084        }
085    }
086
087    @Override
088    public abstract Object call() throws Exception;
089
090    protected List<String> split(String string) {
091        return List.of(string.trim().split("[\\p{Space}]+"));
092    }

117        }
