042/**
043 * Abstract {@link.uri http://ant.apache.org/ Ant} base
044 * {@link org.apache.tools.ant.Task} for tasks that invoke
045 * {@link SDClient} methods.
046 *
047 * {@ant.task}
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 */
051@NoArgsConstructor(access = PROTECTED)
052public abstract class SDClientTask extends HTTPTask implements Callable<Object> {
053    protected SDClient client = null;
054
055    @Override
056    public void execute() throws BuildException {
057        super.execute();
058
059        try {
060            client = new SDClientImpl();
061
062            var result = call();
063
064            if (result instanceof Map) {
065                log(new MapTableModel((Map<?,?>) result));
066            } else if (result instanceof JsonNode) {
067                log(MAPPER.writeValueAsString((JsonNode) result));
068            } else if (result instanceof Iterable) {
069                for (Object object : (Iterable<?>) result) {
070                    log(String.valueOf(object));
071                }
072            } else if (result != Void.class) {
073                log(String.valueOf(result));
074            }
075        } catch (BuildException exception) {
076            throw exception;
077        } catch (RuntimeException exception) {
078            throw exception;
079        } catch (Throwable throwable) {
080            throw new BuildException(throwable);
081        }
082    }
083
084    @Override
085    public abstract Object call() throws Exception;
086
087    protected List<String> split(String string) {
088        return List.of(string.trim().split("[\\p{Space}]+"));
089    }

099    @NoArgsConstructor @ToString
100    public static class Intialize extends SDClientTask {
101        @Getter @Setter
102        private String username = null;
103        @Getter @Setter
104        private String password = null;
105
106        @Override
107        public Object call() throws Exception {
108            var username = getUsername();
109            var password = getPassword();
110
111            client.initialize(username, password);
112
113            return Void.class;
114        }
115    }
