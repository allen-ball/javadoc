023import ball.tv.epg.entity.Schedule;
024import java.util.List;
025import java.util.Set;
026import org.springframework.data.jpa.repository.JpaRepository;
027import org.springframework.data.jpa.repository.Query;
028import org.springframework.stereotype.Repository;
029import org.springframework.transaction.annotation.Transactional;
030
031/**
032 * {@link Schedule} {@link JpaRepository}.
033 *
034 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
035 * @version $Revision: 7215 $
036 */
037@Repository
038@Transactional(readOnly = true)
039public interface ScheduleRepository extends JpaRepository<Schedule,Schedule.PK> {
040    @Transactional
041    public long deleteByStationIDAndAirDateTimeGreaterThanEqual(int stationID, long dateTime);
042
043    @Query("SELECT DISTINCT schedule.programID FROM #{#entityName} schedule")
044    public Set<String> findAllProgramIDs();
045
046    @Query("SELECT schedule FROM #{#entityName} schedule WHERE ?1 BETWEEN schedule.airDateTime AND (schedule.airDateTime + schedule.duration)")
047    public List<Schedule> findAiringAt(long dateTime);
048}
