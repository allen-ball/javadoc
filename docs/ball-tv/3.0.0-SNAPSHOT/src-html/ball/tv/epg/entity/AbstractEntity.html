023import ball.persistence.entity.JSONEntity;
024import ball.tv.ObjectMapperConfiguration;
025import com.fasterxml.jackson.databind.JsonNode;
026import javax.persistence.MappedSuperclass;
027import lombok.NoArgsConstructor;
028
029import static lombok.AccessLevel.PROTECTED;
030
031/**
032 * Abstract {@link ball.tv.epg.entity} {@link JSONEntity} base class.
033 *
034 * {@bean.info}
035 *
036 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
037 * @version $Revision: 8377 $
038 */
039@MappedSuperclass
040@NoArgsConstructor(access = PROTECTED)
041public abstract class AbstractEntity extends JSONEntity {
042    private static final long serialVersionUID = 8798994064011298347L;
043
044    { mapper = ObjectMapperConfiguration.MAPPER; }
045
046    @Override
047    public JsonNode asJsonNode() {
048        var node = super.asJsonNode();
049
050        if (node == null) {
051            var json = getJson();
052
053            if (json != null) {
054                try {
055                    node = mapper.readTree(json);
056                    super.node = node;
057                } catch (Exception exception) {
058                }
059            }
060        }
061
062        return node;
063    }
064}
