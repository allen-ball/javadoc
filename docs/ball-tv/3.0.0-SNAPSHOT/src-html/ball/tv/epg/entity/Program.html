023import javax.persistence.Column;
024import javax.persistence.Entity;
025import javax.persistence.Id;
026import javax.persistence.Table;
027import lombok.EqualsAndHashCode;
028import lombok.Getter;
029import lombok.NoArgsConstructor;
030import lombok.Setter;
031
032/**
033 * {@link Program} {@link Entity}.
034 *
035 * {@bean.info}
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 * @version $Revision: 7215 $
039 */
040@Entity
041@Table(catalog = "epg", name = "programs")
042@NoArgsConstructor @EqualsAndHashCode(callSuper = false)
043public class Program extends AbstractEntity {
044    private static final long serialVersionUID = -6619166576558605886L;
045
046    /** @serial */
047    @Id @Column(length = 16, nullable = false)
048    @Getter @Setter
049    private String programID = null;
050
051    /** @serial */
052    @Column(length = 24, nullable = false)
053    @Getter @Setter
054    private String md5 = null;
055}
