021import java.util.Set;
022import java.util.TreeMap;
023import javax.persistence.Access;
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.Id;
027import javax.persistence.Table;
028import lombok.EqualsAndHashCode;
029import lombok.Getter;
030import lombok.NoArgsConstructor;
031import lombok.Setter;
032
033import static java.util.stream.Collectors.toSet;
034import static javax.persistence.AccessType.PROPERTY;
035import static lombok.AccessLevel.PROTECTED;
036
037/**
038 * {@link Lineup} {@link Entity}.
039 *
040 * {@bean.info}
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 */
044@Entity
045@Table(catalog = "epg", name = "lineups")
046@NoArgsConstructor @EqualsAndHashCode(callSuper = false)
047public class Lineup extends AbstractEntity {
048    private static final long serialVersionUID = -3653917021345933188L;
049
050    /** @serial */
051    @Id @Column(length = 32, nullable = false) @Access(PROPERTY)
052    @Setter
053    private String lineup = null;
054
055    public String getLineup() {
056        if (lineup == null) {
057            lineup = textAt("/metadata/lineup");
058        }
059
060        return lineup;
061    }
062
063    public Set<Channel> asChannelSet() {
064        var map = new TreeMap<String,Channel>();
065
066        try {
067            var iterator = nodeAt("/map").elements();
068
069            while (iterator.hasNext()) {
070                var entry = mapper.treeToValue(iterator.next(), MapEntry.class);
071                var channel = new Channel();
072
073                channel.setLineup(getLineup());
074                channel.setChannel(entry.channel().toString());
075                channel.setStationID(entry.getStationID());
076
077                map.put(channel.getChannel(), channel);
078            }
079        } catch (RuntimeException exception) {
080            throw exception;
081        } catch (Exception exception) {
082            throw new IllegalStateException(exception);
083        }
084
085        return map.values().stream().collect(toSet());
086    }
087
088    public Set<Station> asStationSet() {
089        var map = new TreeMap<Integer,Station>();
090
091        try {
092            var iterator = nodeAt("/stations").elements();
093
094            while (iterator.hasNext()) {
095                var station = mapper.treeToValue(iterator.next(), Station.class);
096
097                map.put(station.getStationID(), station);
098            }
099        } catch (RuntimeException exception) {
100            throw exception;
101        } catch (Exception exception) {
102            throw new IllegalStateException(exception);
103        }
104
105        return map.values().stream().collect(toSet());
106    }
107
108    /**
109     * Abstract {@link MapEntry} {@link Entity}.
110     *
111     * {@bean.info}
112     */
113    @NoArgsConstructor(access = PROTECTED) @EqualsAndHashCode(callSuper = false)
114    protected static abstract class MapEntry extends AbstractEntity {
115        private static final long serialVersionUID = -151173051125744327L;
116
117        /** @serial */ @Getter private int stationID = -1;
118
119        public abstract Object channel();
120    }
121
122    /**
123     * {@link Cable} {@link Entity}.
124     *
125     * {@bean.info}
126     */
127    @NoArgsConstructor @EqualsAndHashCode(callSuper = true)
128    public static class Cable extends MapEntry {
129        private static final long serialVersionUID = -1574882754578381978L;
130
131        /** @serial */ @Getter private int channel = -1;
132
133        @Override
134        public Integer channel() { return getChannel(); }
135    }
136
137    /**
138     * {@link OTA} {@link Entity}.
139     *
140     * {@bean.info}
141     */
142    @NoArgsConstructor @EqualsAndHashCode(callSuper = true)
143    public static class OTA extends MapEntry {
144        private static final long serialVersionUID = 626722624913278995L;
145
146        /* @Getter * private int uhfVhf = -1; */
147        /** @serial */ @Getter private int atscMajor = -1;
148        /** @serial */ @Getter private int atscMinor = -1;
149
150        @Override
151        public String channel() {
152            return getAtscMajor() + "." + getAtscMinor();
153        }
154    }
155}
