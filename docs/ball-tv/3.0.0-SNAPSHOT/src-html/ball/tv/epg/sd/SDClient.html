102    /**
103     * Method to initialize the user configuration.
104     *
105     * @param   username        The user's name.
106     * @param   password        The user's password.
107     *
108     * @throws  Exception       If the hierarchy cannot be initialized.
109     */
110    public void initialize(String username, String password) throws Exception {
111        var node = mapper.createObjectNode();
112
113        node.put("username", username);
114        node.put("password", hash(password));
115
116        Files.createDirectories(credentials.getParent());
117        mapper.writeValue(credentials.toFile(), node);
118    }

120    /**
121     * Method to invoke {@link SDProtocol#postToken(File)}.
122     *
123     * @return  The token {@link String} if successful; {@code null}
124     *          otherwise.
125     *
126     * @throws  Exception       If an exception is encountered invoking
127     *                          {@link SDProtocol#postToken(File)}.
128     */
129    public String getToken() throws Exception {
130        String value = null;
131
132        synchronized (this) {
133            if (Files.exists(token) && ageOf(token).toMinutes() > (24 * 60)) {
134                Files.deleteIfExists(token);
135            }
136
137            if (Files.exists(token)) {
138                value = new String(Files.readAllBytes(token), UTF_8);
139            } else {
140                var response = proxy().postToken(credentials.toFile());
141
142                if (response.getCode() == 0) {
143                    value = response.getToken();
144
145                    if (value != null) {
146                        Files.write(token, value.getBytes(UTF_8));
147                    }
148                } else {
149                    throw new ProtocolException(response.getMessage());
150                }
151            }
152        }
153
154        return value;
155    }
156
157    private Duration ageOf(Path path) throws Exception {
158        return Duration.between(Files.getLastModifiedTime(path).toInstant(), Instant.now());
159    }

368    /**
369     * {@link #getSchedules(Collection)} and
370     * {@link #getSchedulesMD5(Collection)} parameter: Use {@link #values()}
371     * as argument.
372     */
373    public static class PostSchedulesMap extends LinkedHashMap<Object,Map<String,Object>> {
374        private static final long serialVersionUID = 6106598221772498302L;
375
376        /**
377         * Sole constructor.
378         *
379         * @param   ids             The station IDs.
380         */
381        public PostSchedulesMap(Collection<?> ids) {
382            super();
383
384            for (var key : ids) {
385                computeIfAbsent(key, k -> new LinkedHashMap<>())
386                    .put("stationID", key);
387            }
388        }
389    }

431     *                          {@link SDProtocol#getImage(String)}.

433    public HttpEntity getImage(String string) throws Exception {
434        HttpEntity entity = null;
435        var uri = URI.create(string);
436        var response = uri.isAbsolute() ? proxy().getImage(uri) : proxy().getImage(string);
437
438        if (response instanceof HttpEntityContainer) {
439            entity = ((HttpEntityContainer) response).getEntity();
440        }
441
442        return entity;
443    }

458    /**
459     * Static method to generate a 40-digit {@code SHA1} hash of a user
460     * password.
461     *
462     * @param   password        The user password.
463     *
464     * @return  The hash (as a 40-{@link Character} {@link String}.
465     */
466    public static String hash(String password) {
467        BigInteger hash = null;
468
469        try {
470            hash = new BigInteger(1, MessageDigest.getInstance("SHA1").digest(password.getBytes(UTF_8)));
471        } catch (Exception exception) {
472            throw new Error(exception);
473        }
474
475        return String.format("%040x", hash);
476    }
