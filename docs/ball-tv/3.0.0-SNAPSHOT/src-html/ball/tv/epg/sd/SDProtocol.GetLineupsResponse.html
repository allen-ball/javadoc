434    /**
435     * {@link #postSchedules(String,Collection)} return type.
436     *
437     * {@bean.info}
438     */
439    @NoArgsConstructor @Getter
440    public static class Schedules extends JSONBean {
441        private static final long serialVersionUID = -7884152874678564916L;
442
443        /** @serial */ private int stationID = -1;
444        /** @serial */ private List<Schedule> programs = null;
445
446        public List<Schedule> getPrograms() {
447            if (programs != null) {
448                programs.stream().forEach(s -> s.setStationID(stationID));
449            }
450
451            return programs;
452        }
453    }
