631@AntTask("solve-riddle-2020-01-03")
632@NoArgsConstructor @ToString
633public class SolveRiddle20200103Task extends AbstractTask {
634    private static final Character S = Character.valueOf('S');
635
636    @Override
637    public void execute() throws BuildException {
638        super.execute();
639
640        try {
641            URI uri = getClass().getResource("enable1.txt").toURI();
642            List<String> list =
643                Files.lines(Paths.get(uri))
644                .map(t -> t.split("#", 2)[0].trim())
645                .filter(t -> isNotBlank(t))
646                .map(t -> t.toUpperCase())
647                .collect(toList());
648            Set<String> words =
649                list.stream()
650                .filter(t -> t.indexOf(S) == -1)
651                .filter(t -> score(t) > 0)
652                .filter(t -> toCharacterSet(t).size() <= 7)
653                .collect(toSet());
654            Set<Set<Character>> sets =
655                words.stream()
656                .map(t -> toCharacterSet(t))
657                .filter(t -> t.size() == 7)
658                .collect(toSet());
659            Map<Set<Character>,Set<String>> map =
660                new TreeMap<>(Comparator.comparing(Object::toString));
661
662            for (Set<Character> key : sets) {
663                Set<String> value =
664                    words.stream()
665                    .filter(t -> key.containsAll(toCharacterSet(t)))
666                    .collect(toSet());
667
668                map.put(key, value);
669            }
670
671            Map<List<Character>,Set<String>> solutions =
672                new TreeMap<>(Comparator.comparing(Object::toString));
673            Map<List<Character>,Integer> scores =
674                new TreeMap<>(Comparator.comparing(Object::toString));
675
676            for (Map.Entry<Set<Character>,Set<String>> entry :
677                     map.entrySet()) {
678                for (int i = 0, n = entry.getKey().size(); i < n; i += 1) {
679                    List<Character> key =
680                        entry.getKey().stream().collect(toList());
681
682                    key.add(0, key.remove(i));
683
684                    Set<String> value =
685                        entry.getValue()
686                        .stream()
687                        .filter(t -> t.indexOf(key.get(0)) >= 0)
688                        .collect(toSet());
689
690                    solutions.put(key, value);
691                    scores.put(key, score(value));
692                }
693            }
694
695            log(new SimpleTableModel(new Object[][] { }, 2)
696                .row("List:", list.size())
697                .row("Scoring Words:", words.size())
698                .row("Honeycomb Sets:", sets.size())
699                .row("Solutions:", solutions.size()));
700
701            int maxScore =
702                scores.values()
703                .stream()
704                .mapToInt(Integer::intValue)
705                .max().getAsInt();
706
707            for (List<Character> key : scores.keySet()) {
708                if (scores.get(key).intValue() == maxScore) {
709                    log();
710
711                    List<String> solution =
712                        solutions.get(key).stream().collect(toList());
713
714                    solution.sort(Comparator
715                                  .<String>comparingInt(t -> score(t))
716                                  .reversed()
717                                  .thenComparing(Comparator.naturalOrder()));
718
719                    log(new SolutionTableModel(key, solution));
720                }
721            }
722        } catch (BuildException exception) {
723            throw exception;
724        } catch (Throwable throwable) {
725            throwable.printStackTrace();
726            throw new BuildException(throwable);
727        }
728    }
729
730    private int score(String string) {
731        int score = 0;
732
733        switch (string.length()) {
734        case 0:
735        case 1:
736        case 2:
737        case 3:
738            score = 0;
739            break;
740
741        case 4:
742            score = 1;
743            break;
744
745        default:
746            score = string.length();
747
748            if (string.length() >= 7) {
749                if (toCharacterSet(string).size() == 7) {
750                    score += 7;
751                }
752            }
753            break;
754        }
755
756        return score;
757    }
758
759    private int score(Collection<String> collection) {
760        return collection.stream().mapToInt(t -> score(t)).sum();
761    }
762
763    private List<Character> toCharacterList(String string) {
764        return (string.chars()
765                .mapToObj(c -> Character.valueOf((char) c))
766                .collect(toList()));
767    }
768
769    private Set<Character> toCharacterSet(String string) {
770        return toCharacterList(string).stream().collect(toSet());
771    }

773    private class SolutionTableModel extends ArrayListTableModel<String> {
774        private static final long serialVersionUID = -5945141277843892156L;
775
776        private final List<Character> honeycomb;
777
778        public SolutionTableModel(List<Character> honeycomb,
779                                  List<String> solution) {
780            super(solution,
781                  new String[] {
782                      EMPTY, EMPTY, EMPTY,
783                      String.valueOf(solution.size()),
784                      String.valueOf(score(solution))
785                  });
786
787            this.honeycomb = honeycomb;
788        }
789
790        @Override
791        public Object getValueAt(int y, int x) {
792            Object value = EMPTY;
793
794            if (y == 0 && x == 1) {
795                value = honeycomb.get(2);
796            } else if (y == 1 && x == 0) {
797                value = honeycomb.get(1);
798            } else if (y == 1 && x == 2) {
799                value = honeycomb.get(3);
800            } else if (y == 2 && x == 1) {
801                value = honeycomb.get(0);
802            } else if (y == 3 && x == 0) {
803                value = honeycomb.get(6);
804            } else if (y == 3 && x == 2) {
805                value = honeycomb.get(4);
806            } else if (y == 4 && x == 1) {
807                value = honeycomb.get(5);
808            } else {
809                value = super.getValueAt(y, x);
810            }
811
812            return value;
813        }
814
815        @Override
816        protected Object getValueAt(String row, int x) {
817            Object value = EMPTY;
818
819            switch (x) {
820            case 3:     value = row;            break;
821            case 4:     value = score(row);     break;
822            }
823
824            return value;
825        }
826    }
