023import ball.util.ant.taskdefs.AntTask;
024import java.util.ArrayList;
025import java.util.Arrays;
026import java.util.List;
027import java.util.Random;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import org.apache.tools.ant.BuildException;
031
032/**
033 * {@link.uri http://ant.apache.org/ Ant} {@link org.apache.tools.ant.Task}
034 * to solve
035 * {@link.uri
036 * http://fivethirtyeight.com/features/how-long-will-your-smartphone-distract-you-from-family-dinner/
037 * How Long Will Your Smartphone Distract You From Family Dinner?}
038 * <p>
039 * You’ve just finished unwrapping your holiday presents. You and your
040 * sister got brand-new smartphones, opening them at the same moment. You
041 * immediately both start doing important tasks on the Internet, and each
042 * task you do takes one to five minutes. (All tasks take exactly one, two,
043 * three, four or five minutes, with an equal probability of each). After
044 * each task, you have a brief moment of clarity. During these, you remember
045 * that you and your sister are supposed to join the rest of the family for
046 * dinner and that you promised each other you’d arrive together. You ask if
047 * your sister is ready to eat, but if she is still in the middle of a task,
048 * she asks for time to finish it. In that case, you now have time to kill,
049 * so you start a new task (again, it will take one, two, three, four or
050 * five minutes, exactly, with an equal probability of each). If she asks
051 * you if it’s time for dinner while you’re still busy, you ask for time to
052 * finish up and she starts a new task and so on. From the moment you first
053 * open your gifts, how long on average does it take for both of you to be
054 * between tasks at the same time so you can finally eat? (You can assume
055 * the “moments of clarity” are so brief as to take no measurable time at
056 * all.)
057 * </p>
058 * Solution uses the Monte Carlo method.
059 *
060 * {@ant.task}
061 *
062 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
063 * @version $Revision: 5285 $
064 */
065@AntTask("solve-riddle-2015-12-22")
066@NoArgsConstructor @ToString
067public class SolveRiddle20151222Task extends AbstractSimulationTask {
068    @Override
069    public void execute() throws BuildException {
070        super.execute();
071
072        try {
073            ArrayList<Try> tries = new ArrayList<>(getCount());
074
075            for (int i = 0, n = getCount(); i < n; i += 1) {
076                tries.add(new Try());
077            }
078
079            int min = tries.get(0).duration();
080            int max = tries.get(0).duration();
081            int sum = 0;
082
083            for (Try tryN : tries) {
084                int duration = tryN.duration();
085
086                min = Math.min(min, duration);
087                max = Math.max(max, duration);
088                sum += duration;
089            }
090
091            log("count: " + tries.size());
092            log("min: " + min);
093            log("max: " + max);
094            log("average: " + (((float) sum) / tries.size()));
095        } catch (BuildException exception) {
096            throw exception;
097        } catch (Throwable throwable) {
098            throwable.printStackTrace();
099            throw new BuildException(throwable);
100        }
101    }
102
103    private static final Random RANDOM = new Random();
104    private static final List<Integer> DURATIONS =
105        Arrays.asList(1, 2, 3, 4 ,5);
106
107    @ToString
108    private class Try {
109        private List<Integer> you = new ArrayList<>();
110        private List<Integer> sister = new ArrayList<>();
111
112        public Try() {
113            for (;;) {
114                if (sum(you) < sum(sister)) {
115                    you.add(DURATIONS.get(RANDOM.nextInt(DURATIONS.size())));
116                } else {
117                    sister.add(DURATIONS.get(RANDOM.nextInt(DURATIONS.size())));
118                }
119
120                if (sum(you) == sum(sister)) {
121                    break;
122                }
123            }
124        }
125
126        public int duration() { return sum(you); }
127
128        private int sum(List<Integer> list) {
129            int total = 0;
130
131            for (Integer element : list) {
132                total += element;
133            }
134
135            return total;
136        }
137    }
138}
