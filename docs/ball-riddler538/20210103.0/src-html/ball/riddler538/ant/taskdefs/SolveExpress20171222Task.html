040/**
041 * {@link.uri http://ant.apache.org/ Ant} {@link org.apache.tools.ant.Task}
042 * to solve
043 * {@link.uri https://fivethirtyeight.com/features/hark-two-holiday-puzzles/ Riddler Express}
044 * <p>
045 * From Jeffrey Hope, open up your junk drawer and pull out those decks of
046 * cards:
047 * </p>
048 * <p>
049 * Your challenge is to take any 50 cards from a standard 52-card deck and
050 * arrange them into 10 poker hands, one of each type from a royal flush
051 * down to a lowly high card.2 The hands you build always rank as the
052 * highest type of hand possible,3 and a card may not be reused across the
053 * hands. (This means, for example, that a four-of-a-kind better than four
054 * nines is impossible because it’d use a card you’d need to make a royal
055 * flush.) As in actual poker, it doesn’t matter what order the cards are
056 * arranged within a hand.
057 * </p>
058 * <p>
059 * Sounds easy enough, and indeed there is more than one solution. But:
060 * Exactly how many solutions are there?
061 * </p>
062 *
063 * {@ant.task}
064 *
065<pre>
066Ranking       Hand
067----------------------------------------
068RoyalFlush    [A-♤, K-♤, Q-♤, J-♤, 10-♤]
069StraightFlush [K-♡, Q-♡, J-♡, 10-♡, 9-♡]
070FourOfAKind   [8-♤, 8-♡, 8-♢, 8-♧, J-♧]
071FullHouse     [A-♡, A-♢, A-♧, K-♢, K-♧]
072Flush         [Q-♢, J-♢, 10-♢, 9-♢, 7-♢]
073Straight      [7-♤, 6-♤, 5-♤, 4-♤, 3-♤]
074ThreeOfAKind  [6-♡, 6-♢, 6-♧, 10-♧, 5-♧]
075TwoPair       [9-♤, 9-♧, 7-♡, 7-♧, 4-♡]
076Pair          [5-♡, 5-♢, 4-♢, 4-♧, 3-♡]
077HighCard      [Q-♧, 3-♢, 3-♧, 2-♤, 2-♡]
078Remaining     [2-♢, 2-♧]
079</pre>

084@AntTask("solve-express-2017-12-22")
085@NoArgsConstructor @ToString
086public class SolveExpress20171222Task extends AbstractTask {
087    @Override
088    public void execute() throws BuildException {
089        super.execute();
090
091        try {
092            SimpleTableModel model =
093                new SimpleTableModel(new Object[][] { },
094                                     "Ranking", "Hand");
095            TreeMap<Ranking,List<Card>> map = new TreeMap<>();
096            List<Card> deck = new Deck();
097
098            deck.sort(Evaluator.CARD);
099            Collections.reverse(deck);
100
101            List<Ranking> rankings = Arrays.asList(Ranking.values());
102
103            Collections.reverse(rankings);
104
105            for (Ranking ranking : rankings) {
106                List<Card> hand = ranking.find(deck);
107
108                if (! hand.isEmpty()) {
109                    hand = hand.subList(0, ranking.required());
110
111                    map.put(ranking, hand);
112
113                    deck.removeAll(hand);
114                }
115            }
116
117            for (Ranking ranking : rankings) {
118                List<Card> hand = map.get(ranking);
119
120                if (hand != null) {
121                    if (hand.size() < 5) {
122                        hand = new LinkedList<>(hand);
123
124                        List<Card> draw = deck.subList(0, 5 - hand.size());
125
126                        hand.addAll(draw);
127                        draw.clear();
128                    }
129
130                    model.row(ranking, hand);
131                }
132            }
133
134            model.row("Remaining", deck);
135
136            log(model);
137        } catch (BuildException exception) {
138            throw exception;
139        } catch (Throwable throwable) {
140            throwable.printStackTrace();
141            throw new BuildException(throwable);
142        }
143    }
144}
