023import ball.swing.table.MapsTableModel;
024import ball.swing.table.SimpleTableModel;
025import ball.util.ant.taskdefs.AntTask;
026import java.util.ArrayList;
027import java.util.Arrays;

035import static org.apache.commons.lang3.StringUtils.EMPTY;

037/**
038 * {@link.uri http://ant.apache.org/ Ant} {@link org.apache.tools.ant.Task}
039 * to solve
040 * {@link.uri
041 * http://fivethirtyeight.com/features/can-you-win-this-hot-new-game-show/
042 * Can You Win This Hot New Game Show?}
043 * <p>
044 * Two players go on a hot new game show "Higher Number Wins."  The two go
045 * into separate booths, and each presses a button, and a random number
046 * between zero and one appears on a screen.  (At this point, neither knows
047 * the s number, but they do know the numbers are chosen from a standard
048 * uniform distribution.)  They can choose to keep that first number, or to
049 * press the button again to discard the first number and get a second
050 * random number, which they must keep. Then, they come out of their booths
051 * and see the final number for each player on the wall.  The lavish grand
052 * prize -- a case full of bullion gold -- is awarded to the player who kept
053 * the higher number.  Which number is the optimal cutoff for players to
054 * discard their first number and choose another?  Put another way, within
055 * which range should they choose to keep the first number, and within which
056 * range should they reject it and try their luck with a second number?
057 * </p>
058 * Solution uses the Monte Carlo method.
059 *
060 * {@ant.task}
061 *
062 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
063 * @version $Revision: 7215 $
064 */
065@AntTask("solve-riddle-2016-03-04")
066@NoArgsConstructor @ToString
067public class SolveRiddle20160304Task extends AbstractSimulationTask {
068    @Override
069    public void execute() throws BuildException {
070        super.execute();
071
072        try {
073            ArrayList<Simulation> simulations = new ArrayList<>(getCount());
074
075            for (int i = 0, n = getCount(); i < n; i += 1) {
076                simulations.add(new Simulation());
077            }
078
079            ArrayList<BucketMap> maps = new ArrayList<>();
080
081            maps.add(new BucketMap());
082            maps.add(new BucketMap());
083
084            for (Simulation simulation : simulations) {
085                double[] winner = simulation.getWinnerPicks();
086                double key = maps.get(0).tailMap(winner[0]).firstKey();
087
088                maps.get(0).put(key, maps.get(0).get(key) + 1);
089
090                double[] loser = simulation.getLoserPicks();
091
092                if (winner[0] > max(loser)) {
093                    maps.get(1).put(key, maps.get(1).get(key) + 1);
094                }
095            }
096
097            log(new SimpleTableModel(new Object[][] { }, 3)
098                .row("count:", simulations.size(), EMPTY));
099            log();
100            log(new MapsTableModel(maps,
101                                   "first pick", "total wins",
102                                   "on first pick"));
103        } catch (BuildException exception) {
104            throw exception;
105        } catch (Throwable throwable) {
106            throwable.printStackTrace();
107            throw new BuildException(throwable);
108        }
109    }
110
111    private double max(double... array) { return array[greatest(array)]; }
112
113    private int greatest(double... array) {
114        int greatest = 0;
115
116        for (int i = 1; i < array.length; i += 1) {
117            if (array[i] > array[greatest]) {
118                greatest = i;
119            }
120        }
121
122        return greatest;
123    }
124
125    private static final Random RANDOM = new Random();
126
127    @ToString
128    private class Simulation {
129        private final double[][] p;
130
131        public Simulation() {
132            p =
133                new double[][] {
134                    new double[] { RANDOM.nextDouble(), RANDOM.nextDouble() },
135                    new double[] { RANDOM.nextDouble(), RANDOM.nextDouble() }
136                };
137        }
138
139        public int getWinner() { return greatest(max(p[0]), max(p[1])); }
140
141        public double[] getWinnerPicks() { return p[getWinner()]; }
142
143        public int getLoser() { return (getWinner() == 0) ? 1 : 0; }
144
145        public double[] getLoserPicks() { return p[getLoser()]; }
146    }
147
148    private class BucketMap extends TreeMap<Double,Integer> {
149        private static final long serialVersionUID = -5756858279184775705L;
150
151        public BucketMap() {
152            super();
153
154            for (int i = 0; i <= 100; i += 1) {
155                put((double) i / 100, 0);
156            }
157        }
158    }
159}
