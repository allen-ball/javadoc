021import ball.tv.epg.sd.SDClient;
022import java.nio.file.Path;
023import lombok.NoArgsConstructor;
024import lombok.ToString;
025import lombok.extern.log4j.Log4j2;
026import org.springframework.beans.factory.annotation.Value;
027import org.springframework.boot.autoconfigure.domain.EntityScan;
028import org.springframework.context.annotation.Bean;
029import org.springframework.context.annotation.Configuration;
030import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
031import org.springframework.transaction.annotation.EnableTransactionManagement;
032
033/**
034 * EPG {@link Configuration}.
035 *
036 * {@injected.fields}
037 *
038 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
039 */
040@Configuration
041@EntityScan(basePackageClasses = { ball.tv.epg.entity.Entities.class })
042@EnableJpaRepositories(basePackageClasses = { ball.tv.epg.repository.Repositories.class })
043@EnableTransactionManagement
044@NoArgsConstructor @ToString @Log4j2
045public class EPGConfiguration {
046    @Value("${epg.home}/sd") private Path home;
047
048    @Bean
049    public SDClient client() { return new SDClient(home); }
050}
