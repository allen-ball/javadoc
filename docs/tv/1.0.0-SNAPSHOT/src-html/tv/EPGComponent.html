023import ball.tv.epg.entity.Schedule;
024import ball.tv.epg.repository.ChannelRepository;
025import ball.tv.epg.repository.LineupRepository;
026import ball.tv.epg.repository.ProgramRepository;
027import ball.tv.epg.repository.ScheduleRepository;
028import ball.tv.epg.repository.StationRepository;
029import ball.tv.epg.sd.SDClient;
030import ball.tv.epg.sd.SDProtocol;
031import com.fasterxml.jackson.databind.JsonNode;
032import java.io.IOException;
033import java.nio.file.Files;
034import java.nio.file.Path;
035import java.time.Duration;
036import java.time.Instant;
037import java.util.List;
038import java.util.Objects;
039import javax.annotation.PostConstruct;
040import javax.annotation.PreDestroy;
041import lombok.NoArgsConstructor;
042import lombok.ToString;
043import lombok.extern.log4j.Log4j2;
044import org.springframework.beans.factory.annotation.Autowired;
045import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
046import org.springframework.scheduling.annotation.Scheduled;
047import org.springframework.stereotype.Component;
048
049import static ball.tv.ObjectMapperConfiguration.MAPPER;
050import static java.util.stream.Collectors.toList;
051
052/**
053 * EPG {@link Component}.
054 *
055 * {@injected.fields}
056 *
057 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
058 * @version $Revision: 8102 $
059 */
060@Component
061@ConditionalOnProperty(name = "epg.enable-updates", havingValue = "true", matchIfMissing = true)
062@NoArgsConstructor @ToString @Log4j2
063public class EPGComponent {
064    @Autowired private SDClient client = null;
065    @Autowired private ChannelRepository channelRepository = null;
066    @Autowired private LineupRepository lineupRepository = null;
067    @Autowired private ProgramRepository programRepository = null;
068    @Autowired private ScheduleRepository scheduleRepository = null;
069    @Autowired private StationRepository stationRepository = null;
070    private Path statusPath = null;
071    private List<SDProtocol.LineupElement> lineups = List.of();
072
073    @PostConstruct
074    public void init() {
075        statusPath = client.getParent().resolve("STATUS");
076    }
077
078    @PreDestroy
079    public void destroy() { }
080
081    /**
082     * Update {@link ball.tv.epg.entity.Lineup}s,
083     * {@link ball.tv.epg.entity.Station}s, and
084     * {@link ball.tv.epg.entity.Channel}s.
085     */
086    @Scheduled(fixedDelayString = "PT1M", initialDelayString = "PT0S")
087    public void updateLineups() throws Exception {
088        if (Files.exists(statusPath) && ageOf(statusPath).toMinutes() > 30) {
089            Files.deleteIfExists(statusPath);
090
091            lineups = List.of();
092        }
093
094        if (Files.notExists(statusPath)) {
095            if (lineups.isEmpty()) {
096                var response = client.getStatus();
097
098                if (response.getCode() == 0) {
099                    if (! Objects.equals(response.getLineups(), lineups)) {
100                        for (var element : response.getLineups()) {
101                            var lineup = client.getLineup(element.getLineup());
102
103                            stationRepository.saveAll(lineup.asStationSet());
104
105                            channelRepository.deleteByLineup(lineup.getLineup());
106                            channelRepository.saveAll(lineup.asChannelSet());
107
108                            lineupRepository.save(lineup);
109
110                            log.info("{}: {} channels",
111                                     lineup.getLineup(), lineup.asChannelSet().size());
112                        }
113
114                        lineups = response.getLineups();
115                    }
116                } else {
117                    saveTo(response.getSystemStatus(), statusPath);
118                }
119            }
120        }
121    }
122
123    /**
124     * Update {@link Schedule}s.
125     */
126    @Scheduled(fixedDelayString = "PT4H", initialDelayString = "PT30S")
127    public void updateSchedules() throws Exception {
128        if (Files.notExists(statusPath)) {
129            for (var schedules : client.getSchedules(stationRepository.findAllStationIDs())) {
130                var list = schedules.getPrograms();
131
132                if (list != null && (! list.isEmpty())) {
133                    long min =
134                        list.stream()
135                        .mapToLong(Schedule::getAirDateTime)
136                        .min().orElse(0);
137
138                    if (min > 0) {
139                        scheduleRepository.deleteByStationIDAndAirDateTimeGreaterThanEqual(schedules.getStationID(), min);
140                    }
141
142                    scheduleRepository.saveAll(list);
143                }
144            }
145        }
146    }
147
148    /**
149     * Update {@link ball.tv.epg.entity.Program}s.
150     */
151    @Scheduled(fixedDelayString = "PT1M", initialDelayString = "PT1M")
152    public void updatePrograms() throws Exception {
153        if (Files.notExists(statusPath)) {
154            var set = scheduleRepository.findAllProgramIDs();
155
156            set.removeAll(programRepository.findAllProgramIDs());
157
158            var list = set.stream().collect(toList());
159
160            while (! list.isEmpty()) {
161                var sublist = list.subList(0, Math.min(list.size(), 500));
162
163                list = list.subList(sublist.size(), list.size());
164
165                programRepository.saveAll(client.getPrograms(sublist));
166            }
167        }
168    }
169
170    private Duration ageOf(Path path) throws Exception {
171        return Duration.between(Files.getLastModifiedTime(path).toInstant(),
172                                Instant.now());
173    }
174
175    private void saveTo(JsonNode node, Path path) throws IOException {
176        MAPPER.writeValue(path.toFile(), node);
177    }
178}
