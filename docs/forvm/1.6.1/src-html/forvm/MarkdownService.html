108    private static final MutableDataHolder OPTIONS;

110    static {
111        ArrayList<Extension> extensions = new ArrayList<>();
112
113        try {
114            for (Class<? extends Extension> type : EXTENSIONS) {
115                Extension extension =
116                    (Extension) type.getMethod("create").invoke(null);
117
118                extensions.add(extension);
119            }
120        } catch (Exception exception) {
121            throw new ExceptionInInitializerError(exception);
122        }
123
124        OPTIONS =
125            new MutableDataSet()
126            .setFrom(ParserEmulationProfile.MULTI_MARKDOWN)
127            .set(Parser.EXTENSIONS, extensions)
128            .set(HtmlRenderer.GENERATE_HEADER_ID, true);

131    private static final String README_MD = "README.md";
132    private static final String SLASH = "/";
133    private static final URI ROOT = URI.create(SLASH);

135    @Autowired private Tika tika;

137    /**
138     * Method to compile the parameters into an {@link Author}.
139     *
140     * @param   name            The name of the input file.
141     * @param   contents        The contents of the input file.
142     * @param   slug            The {@link Author}'s slug.
143     * @param   author          The {@link Author}'s entity to update.
144     *
145     * @throws  Exception       If any {@link Exception} is encountered.
146     */
147    public void compile(String name, byte[] contents,
148                        String slug, Author author) throws Exception {
149        ZipFile zip = null;
150
151        try {
152            String markdown = null;
153
154            switch (String.valueOf(tika.detect(contents, name))) {
155            case "application/zip":
156                zip = new ZipFileImpl(name, contents);
157                markdown = getEntryAsUTF8String(zip, README_MD);
158                break;
159
160            default:
161                markdown = new String(contents, UTF_8);
162                break;
163            }
164
165            Document document = parse(markdown);
166            Map<String,List<String>> yaml = getYamlFrom(document);
167
168            if (yaml.containsKey("email")) {
169                String string =
170                    yaml.get("email").stream().collect(Collectors.joining());
171
172                author.setEmail(string);
173            }
174
175            author.setSlug(slug);
176
177            if (yaml.containsKey("name")) {
178                author.setName(yaml.get("name")
179                               .stream().collect(Collectors.joining()));
180            }
181
182            author.setMarkdown(markdown);
183
184            if (zip != null) {
185                for (ZipArchiveEntry entry :
186                         Collections.list(zip.getEntries())
187                         .stream()
188                         .filter(t -> (! t.isDirectory()))
189                         .filter(t -> (! t.getName().equals(README_MD)))
190                         .collect(Collectors.toList())) {
191                    throw new IllegalArgumentException(entry.getName());
192                }
193            }
194
195            author.setHtml(htmlRender(document, null).toString());
196        } finally {
197            if (zip != null) {
198                zip.close();
199            }
200        }
201    }
202
203    private String getEntryAsUTF8String(ZipFile zip,
204                                        String name) throws Exception {
205        ZipArchiveEntry entry = zip.getEntry(name);
206        Scanner scanner = new Scanner(zip.getInputStream(entry), UTF_8.name());
207
208        return scanner.useDelimiter("\\A").next();
209
210    }

212    /**
213     * Method to compile the parameters into an {@link Article}.
214     *
215     * @param   name            The name of the input file.
216     * @param   contents        The contents of the input file.
217     * @param   prefix          The {@link Article}'s path prefix.
218     * @param   slug            The {@link Article}'s slug.
219     * @param   article         The {@link Article}'s entity to update.
220     *
221     * @throws  Exception       If any {@link Exception} is encountered.
222     */
223    public void compile(String name, byte[] contents,
224                        String prefix, String slug,
225                        Article article) throws Exception {
226        ZipFile zip = null;
227
228        try {
229            String markdown = null;
230
231            switch (String.valueOf(tika.detect(contents, name))) {
232            case "application/zip":
233                zip = new ZipFileImpl(name, contents);
234                markdown = getEntryAsUTF8String(zip, README_MD);
235                break;
236
237            default:
238                markdown = new String(contents, UTF_8);
239                break;
240            }
241
242            article.setSlug(slug);
243
244            Document document = parse(markdown);
245            Map<String,List<String>> yaml = getYamlFrom(document);
246            String title =
247                yaml.get("title").stream().collect(Collectors.joining());
248
249            article.setTitle(title);
250            article.setMarkdown(markdown);
251            article.getAttachments().clear();
252
253            if (zip != null) {
254                for (ZipArchiveEntry entry :
255                         Collections.list(zip.getEntries())
256                         .stream()
257                         .filter(t -> (! t.isDirectory()))
258                         .filter(t -> (! t.getName().equals(README_MD)))
259                         .collect(Collectors.toList())) {
260                    String path =
261                        ROOT.resolve(entry.getName()).normalize().getPath();
262                    byte[] content =
263                        IOUtils.toByteArray(zip.getInputStream(entry));
264                    Attachment attachment = new Attachment();
265
266                    attachment.setArticle(article);
267                    attachment.setPath(path);
268                    attachment.setContent(content);
269
270                    article.getAttachments().add(attachment);
271                }
272            }
273
274            String path = prefix + "/" + article.getSlug() + "/";
275            URI uri = URI.create(path.replaceAll("[/]+", "/"));
276            CharSequence html = htmlRender(document, uri);
277
278            article.setHtml(html.toString());
279        } finally {
280            if (zip != null) {
281                zip.close();
282            }
283        }
284    }

314    /**
315     * See {@link HtmlRenderer#render(Node,Appendable)}.  Rewrites relative
316     * and {@code FILE} URIs to match those that are uploaded as part of the
317     * {@code markdown} document.
318     *
319     * @param   document        The {@link Document} to render.
320     * @param   prefix          The prefix (as a {@link URI}).
321     *
322     * @return  The {@link CharSequence} representing the rendered
323     *          {@code HTML}.
324     */
325    public CharSequence htmlRender(Document document, URI prefix) {
326        StringBuilder html = new StringBuilder();
327        HtmlRenderer.Builder builder = HtmlRenderer.builder(OPTIONS);
328
329        if (prefix != null) {
330            builder.linkResolverFactory(new LinkResolverFactoryImpl(prefix));
331        }
332
333        builder.build().render(document, html);
334
335        return html;
336    }

338    /**
339     * Method to get the YAML front matter from the document.
340     *
341     * @param   document        The {@link Document}.
342     *
343     * @return  The {@link Map} representing the parsed YAML.
344     */
345    public Map<String,List<String>> getYamlFrom(Document document) {
346        AbstractYamlFrontMatterVisitor visitor =
347            new AbstractYamlFrontMatterVisitor() { };
348
349        visitor.visit(document);
350
351        return visitor.getData();
352    }

356            super(new SeekableInMemoryByteChannel(contents),

364    private class LinkResolverFactoryImpl implements LinkResolverFactory {
365        private final URI prefix;
366
367        public LinkResolverFactoryImpl(URI prefix) {
368            this.prefix = Objects.requireNonNull(prefix);
369        }
370
371        @Override
372        public Set<Class<?>> getAfterDependents() { return null; }
373
374        @Override
375        public Set<Class<?>> getBeforeDependents() { return null; }
376
377        @Override
378        public boolean affectsGlobalScope() { return false; }
379
380        @Override
381        public LinkResolver apply(LinkResolverBasicContext context) {
382            return new LinkResolverImpl(context, prefix);
383        }
384
385        @Override
386        public String toString() { return super.toString(); }
387    }
388
389    private class LinkResolverImpl implements LinkResolver {
390        private static final String FILE = "file";
391        private static final String SLASH = "/";
392
393        private final URI prefix;
394
395        public LinkResolverImpl(LinkResolverBasicContext context, URI prefix) {
396            this.prefix = Objects.requireNonNull(prefix);
397        }
398
399        @Override
400        public ResolvedLink resolveLink(Node node,
401                                        LinkResolverBasicContext context,
402                                        ResolvedLink link) {
403            URI uri = URI.create(link.getUrl()).normalize();
404
405            if ((! uri.isAbsolute()) || FILE.equals(uri.getScheme())) {
406                if (! uri.getPath().startsWith(SLASH)) {
407                    link =
408                        link
409                        .withStatus(LinkStatus.VALID)
410                        .withUrl(prefix.resolve(uri).toString());
411                }
412            }
413
414            return link;
415        }
416
417        @Override
418        public String toString() { return super.toString(); }
419    }
