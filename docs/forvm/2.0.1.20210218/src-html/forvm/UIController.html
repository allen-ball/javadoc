065@Controller
066@NoArgsConstructor @ToString @Log4j2
067public class UIController extends AbstractController {
068    private static final String ERRORS = "errors";
069    private static final String FORM = "form";
070    private static final String PAGE = "page";
071
072    @Autowired private AuthorRepository authorRepository = null;
073    @Autowired private ArticleRepository articleRepository = null;
074    @Autowired private CredentialRepository credentialRepository = null;
075    @Autowired private MarkdownService service = null;
076    @Autowired private PasswordEncoder encoder = null;
077
078    private int page_size = 8;
079
080    @RequestMapping(value = { "/" })
081    @PreAuthorize("permitAll()")
082    public String root() { return "redirect:/index.html"; }
083
084    @RequestMapping(value = { "/index.html", "/index.htm", "/index" })
085    @PreAuthorize("permitAll()")
086    public String index() { return "redirect:/articles/"; }
087
088    @RequestMapping(value = {
089                        "/articles", "/article/{slug}",
090                        "/authors",
091                        "/preview", "/preview/{slug}"
092                    })
093    @PreAuthorize("permitAll()")
094    public String redirect(HttpServletRequest request) {
095        return "redirect:" + request.getServletPath() + "/";
096    }
097
098    @RequestMapping(method = { GET }, value = { "/articles/" })
099    @PreAuthorize("permitAll()")
100    public String articles(Model model,
101                           HttpServletRequest request,
102                           @RequestParam Optional<String> author,
103                           @RequestParam Optional<Integer> page,
104                           RedirectAttributes redirect) {
105        var view = getViewName();
106
107        if (page.isPresent()) {
108            var pr = PageRequest.of(page.get() - 1, page_size, Sort.Direction.DESC, "slug");
109            Page<?> articles = null;
110
111            if (author.isPresent()) {
112                var slug = author.get();
113
114                articles =
115                    articleRepository
116                    .findByAuthor(pr, authorRepository.findBySlug(slug).get());
117            } else {
118                articles = articleRepository.findAll(pr);
119            }
120
121            model.addAttribute("articles", articles);
122            model.addAttribute(PAGE, articles);
123        } else {
124            redirect.addAttribute(PAGE, String.valueOf(1));
125            redirect.mergeAttributes(request.getParameterMap());
126
127            view = "redirect:" + request.getServletPath();
128        }
129
130        return view;
131    }
132

134    @PreAuthorize("permitAll()")
135    public String article(Model model, @PathVariable String slug) {
136        model

142    @RequestMapping(method = { GET }, value = { "/authors/" })
143    @PreAuthorize("permitAll()")
144    public String authors(Model model,
145                          HttpServletRequest request,
146                          @RequestParam Optional<Integer> page,
147                          RedirectAttributes redirect) {
148        var view = getViewName();
149
150        if (page.isPresent()) {
151            var  pr = PageRequest.of(page.get() - 1, page_size);
152            var authors = authorRepository.findAll(pr);
153
154            model.addAttribute("authors", authors);
155            model.addAttribute(PAGE, authors);
156        } else {
157            redirect.addAttribute(PAGE, String.valueOf(1));
158            redirect.mergeAttributes(request.getParameterMap());
159
160            view = "redirect:" + request.getServletPath();
161        }
162
163        return view;
164    }

166    @RequestMapping(method = { GET }, value = { "/preview/" })

169        model.addAttribute(FORM, new PreviewForm());
170
171        return getViewName();
172    }
173
174    @RequestMapping(method = { POST }, value = { "/preview/" })
175    @PreAuthorize("hasAuthority('AUTHOR')")
176    public String previewPOST(Model model,
177                              Principal principal, HttpSession session,
178                              HttpServletRequest request,
179                              @Valid PreviewForm form, BindingResult result) {
180        var view = getViewName();
181
182        try {
183            if (! result.hasErrors()) {
184                var email = principal.getName();
185                var name = form.getFile().getOriginalFilename();
186                var  slug = FilenameUtils.getBaseName(name);
187                var article = new Article();
188
189                article.setAuthor(authorRepository.findById(email).get());
190                article.setEmail(article.getAuthor().getEmail());
191
192                service.compile(name, form.getFile().getBytes(),
193                                request.getServletPath(), slug, article);
194
195                session.setAttribute(article.getSlug(), article);
196
197                view =
198                    "redirect:" + request.getServletPath()
199                    + article.getSlug() + "/";
200            } else {
201                model.addAttribute(FORM, form);
202            }
203        } catch (Exception exception) {
204            model.addAttribute(ERRORS, exception.getMessage());
205        }
206
207        return view;
208    }

237
238    @RequestMapping(method = { POST }, value = { "/password" })
239    @PreAuthorize("isAuthenticated()")
240    public String passwordPOST(Model model, Principal principal,
241                               @Valid ChangePasswordForm form, BindingResult result) {
242        var credential =
243            credentialRepository.findById(principal.getName())
244            .orElseThrow(() -> new AuthorizationServiceException("Unauthorized"));
245
246        try {
247            if (result.hasErrors()) {
248                throw new RuntimeException(String.valueOf(result.getAllErrors()));
249            }
250
251            if (! (Objects.equals(form.getUsername(), principal.getName())
252                   && encoder.matches(form.getPassword(), credential.getPassword()))) {
253                throw new AccessDeniedException("Invalid user name and password");
254            }
255
256            if (! (form.getNewPassword() != null
257                   && Objects.equals(form.getNewPassword(), form.getRepeatPassword()))) {
258                throw new RuntimeException("Repeated password does not match new password");
259            }
260
261            if (encoder.matches(form.getNewPassword(), credential.getPassword())) {
262                throw new RuntimeException("New password must be different than old");
263            }
264
265            credential.setPassword(encoder.encode(form.getNewPassword()));
266            credentialRepository.save(credential);
267        } catch (Exception exception) {
268            model.addAttribute(FORM, form);
269            model.addAttribute(ERRORS, exception.getMessage());
270        }
271
272        return getViewName();
273    }
