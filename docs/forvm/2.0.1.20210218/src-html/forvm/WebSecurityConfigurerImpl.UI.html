023import lombok.NoArgsConstructor;
024import lombok.ToString;
025import lombok.extern.log4j.Log4j2;
026import org.springframework.beans.factory.annotation.Autowired;
027import org.springframework.context.annotation.Configuration;
028import org.springframework.core.annotation.Order;
029import org.springframework.http.HttpStatus;
030import org.springframework.security.config.annotation.authentication.builders.AuthenticationManagerBuilder;
031import org.springframework.security.config.annotation.method.configuration.EnableGlobalMethodSecurity;
032import org.springframework.security.config.annotation.web.builders.HttpSecurity;
033import org.springframework.security.config.annotation.web.builders.WebSecurity;
034import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
035import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
036import org.springframework.security.core.userdetails.UserDetailsService;
037import org.springframework.security.crypto.password.PasswordEncoder;
038import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserService;
039import org.springframework.security.oauth2.client.registration.ClientRegistrationRepository;
040import org.springframework.security.web.authentication.HttpStatusEntryPoint;
041import org.springframework.security.web.util.matcher.AntPathRequestMatcher;
042
043import static lombok.AccessLevel.PRIVATE;
044
045/**
046 * {@link org.springframework.security.config.annotation.web.WebSecurityConfigurer}
047 * abstract base class.
048 *
049 * {@injected.fields}
050 *
051 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
052 * @version $Revision: 7215 $
053 */
054@Configuration
055@EnableWebSecurity
056@EnableGlobalMethodSecurity(prePostEnabled = true)
057@NoArgsConstructor(access = PRIVATE) @Log4j2
058public abstract class WebSecurityConfigurerImpl extends WebSecurityConfigurerAdapter {
059    @Autowired private UserDetailsService userDetailsService = null;
060    @Autowired private PasswordEncoder passwordEncoder = null;
061
062    @Override
063    public void configure(AuthenticationManagerBuilder auth) throws Exception {
064        auth.userDetailsService(userDetailsService)
065            .passwordEncoder(passwordEncoder);
066    }
067
068    /**
069     * {@link org.springframework.security.config.annotation.web.WebSecurityConfigurer}
070     * implementation for the API.
071     */
072    @Configuration
073    @Order(1)
074    @NoArgsConstructor @ToString
075    public static class API extends WebSecurityConfigurerImpl {
076        @Override
077        protected void configure(HttpSecurity http) throws Exception {
078            http.antMatcher("/api/**")
079                .authorizeRequests(t -> t.anyRequest().authenticated())
080                .csrf(t -> t.disable())
081                .httpBasic(t -> t.authenticationEntryPoint(new HttpStatusEntryPoint(HttpStatus.FORBIDDEN)));
082        }
083    }
084
085    /**
086     * {@link org.springframework.security.config.annotation.web.WebSecurityConfigurer}
087     * implementation for the UI.
088     */
089    @Configuration
090    @Order(2)
091    @NoArgsConstructor @ToString
092    public static class UI extends WebSecurityConfigurerImpl {
093        private static final String[] IGNORE = {
094            "/css/**", "/js/**", "/images/**", "/webjars/**", "/webjarsjs"
095        };
096
097        @Autowired private OidcUserService oidcUserService = null;
098
099        @Override
100        public void configure(WebSecurity web) {
101            web.ignoring().antMatchers(IGNORE);
102        }
103
104        @Override
105        protected void configure(HttpSecurity http) throws Exception {
106            http.antMatcher("/**")
107                .authorizeRequests(t -> t.anyRequest().permitAll())
108                .formLogin(t -> t.loginPage("/login").permitAll())
109                .logout(t -> t.logoutRequestMatcher(new AntPathRequestMatcher("/logout"))
110                              .logoutSuccessUrl("/").permitAll());
111
112            try {
113                var repository =
114                    getApplicationContext().getBean(ClientRegistrationRepository.class);
115
116                if (repository != null) {
117                    http.oauth2Login(t -> t.clientRegistrationRepository(repository)
118                                           .userInfoEndpoint(u -> u.oidcUserService(oidcUserService))
119                                           .loginPage("/login").permitAll());
120                }
121            } catch (Exception exception) {
122            }
123        }
124    }
125}
