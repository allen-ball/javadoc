109    private static final MutableDataHolder OPTIONS;

111    static {
112        var extensions = new ArrayList<Extension>();
113
114        try {
115            for (Class<? extends Extension> type : EXTENSIONS) {
116                var extension = (Extension) type.getMethod("create").invoke(null);
117
118                extensions.add(extension);
119            }
120        } catch (Exception exception) {
121            throw new ExceptionInInitializerError(exception);
122        }
123
124        OPTIONS =
125            new MutableDataSet()
126            .setFrom(ParserEmulationProfile.MULTI_MARKDOWN)
127            .set(Parser.EXTENSIONS, extensions)
128            .set(HtmlRenderer.GENERATE_HEADER_ID, true);

131    private static final String README_MD = "README.md";
132    private static final String SLASH = "/";
133    private static final URI ROOT = URI.create(SLASH);

135    @Autowired private Tika tika = null;

137    /**
138     * Method to compile the parameters into an {@link Author}.
139     *
140     * @param   name            The name of the input file.
141     * @param   contents        The contents of the input file.
142     * @param   slug            The {@link Author}'s slug.
143     * @param   author          The {@link Author}'s entity to update.
144     *
145     * @throws  Exception       If any {@link Exception} is encountered.
146     */
147    public void compile(String name, byte[] contents,
148                        String slug, Author author) throws Exception {
149        ZipFile zip = null;
150
151        try {
152            String markdown = null;
153
154            switch (String.valueOf(tika.detect(contents, name))) {
155            case "application/zip":
156                zip = new ZipFileImpl(name, contents);
157                markdown = getEntryAsUTF8String(zip, README_MD);
158                break;
159
160            default:
161                markdown = new String(contents, UTF_8);
162                break;
163            }
164
165            var document = parse(markdown);
166            var yaml = getYamlFrom(document);
167
168            if (yaml.containsKey("email")) {
169                var string = yaml.get("email").stream().collect(joining());
170
171                author.setEmail(string);
172            }
173
174            author.setSlug(slug);
175
176            if (yaml.containsKey("name")) {
177                author.setName(yaml.get("name").stream().collect(joining()));
178            }
179
180            author.setMarkdown(markdown);
181
182            if (zip != null) {
183                for (var entry :
184                         Collections.list(zip.getEntries())
185                         .stream()
186                         .filter(t -> (! t.isDirectory()))
187                         .filter(t -> (! t.getName().equals(README_MD)))
188                         .collect(toList())) {
189                    throw new IllegalArgumentException(entry.getName());
190                }
191            }
192
193            author.setHtml(htmlRender(document, null).toString());
194        } finally {
195            if (zip != null) {
196                zip.close();
197            }
198        }
199    }
200
201    private String getEntryAsUTF8String(ZipFile zip, String name) throws Exception {
202        ZipArchiveEntry entry = zip.getEntry(name);
203        Scanner scanner = new Scanner(zip.getInputStream(entry), UTF_8.name());
204
205        return scanner.useDelimiter("\\A").next();
206
207    }

209    /**
210     * Method to compile the parameters into an {@link Article}.
211     *
212     * @param   name            The name of the input file.
213     * @param   contents        The contents of the input file.
214     * @param   prefix          The {@link Article}'s path prefix.
215     * @param   slug            The {@link Article}'s slug.
216     * @param   article         The {@link Article}'s entity to update.
217     *
218     * @throws  Exception       If any {@link Exception} is encountered.
219     */
220    public void compile(String name, byte[] contents,
221                        String prefix, String slug,
222                        Article article) throws Exception {
223        ZipFile zip = null;
224
225        try {
226            String markdown = null;
227
228            switch (String.valueOf(tika.detect(contents, name))) {
229            case "application/zip":
230                zip = new ZipFileImpl(name, contents);
231                markdown = getEntryAsUTF8String(zip, README_MD);
232                break;
233
234            default:
235                markdown = new String(contents, UTF_8);
236                break;
237            }
238
239            article.setSlug(slug);
240
241            var document = parse(markdown);
242            var yaml = getYamlFrom(document);
243            var title = yaml.get("title").stream().collect(joining());
244
245            article.setTitle(title);
246            article.setMarkdown(markdown);
247            article.getAttachments().clear();
248
249            if (zip != null) {
250                for (var entry :
251                         Collections.list(zip.getEntries())
252                         .stream()
253                         .filter(t -> (! t.isDirectory()))
254                         .filter(t -> (! t.getName().equals(README_MD)))
255                         .collect(toList())) {
256                    var path = ROOT.resolve(entry.getName()).normalize().getPath();
257                    var content = IOUtils.toByteArray(zip.getInputStream(entry));
258                    var attachment = new Attachment();
259
260                    attachment.setArticle(article);
261                    attachment.setPath(path);
262                    attachment.setContent(content);
263
264                    article.getAttachments().add(attachment);
265                }
266            }
267
268            var path = prefix + "/" + article.getSlug() + "/";
269            var uri = URI.create(path.replaceAll("[/]+", "/"));
270            var html = htmlRender(document, uri);
271
272            article.setHtml(html.toString());
273        } finally {
274            if (zip != null) {
275                zip.close();
276            }
277        }
278    }

308    /**
309     * See {@link HtmlRenderer#render(Node,Appendable)}.  Rewrites relative
310     * and {@code FILE} URIs to match those that are uploaded as part of the
311     * {@code markdown} document.
312     *
313     * @param   document        The {@link Document} to render.
314     * @param   prefix          The prefix (as a {@link URI}).
315     *
316     * @return  The {@link CharSequence} representing the rendered
317     *          {@code HTML}.
318     */
319    public CharSequence htmlRender(Document document, URI prefix) {
320        var html = new StringBuilder();
321        var builder = HtmlRenderer.builder(OPTIONS);
322
323        if (prefix != null) {
324            builder.linkResolverFactory(new LinkResolverFactoryImpl(prefix));
325        }
326
327        builder.build().render(document, html);
328
329        return html;
330    }

332    /**
333     * Method to get the YAML front matter from the document.
334     *
335     * @param   document        The {@link Document}.
336     *
337     * @return  The {@link Map} representing the parsed YAML.
338     */
339    public Map<String,List<String>> getYamlFrom(Document document) {
340        var visitor = new AbstractYamlFrontMatterVisitor() { };
341
342        visitor.visit(document);
343
344        return visitor.getData();
345    }

355    @ToString
356    private class LinkResolverFactoryImpl implements LinkResolverFactory {
357        private final URI prefix;
358
359        public LinkResolverFactoryImpl(URI prefix) {
360            this.prefix = Objects.requireNonNull(prefix);
361        }
362
363        @Override
364        public Set<Class<?>> getAfterDependents() { return null; }
365
366        @Override
367        public Set<Class<?>> getBeforeDependents() { return null; }
368
369        @Override
370        public boolean affectsGlobalScope() { return false; }
371
372        @Override
373        public LinkResolver apply(LinkResolverBasicContext context) {
374            return new LinkResolverImpl(context, prefix);
375        }
376    }
377
378    @ToString
379    private class LinkResolverImpl implements LinkResolver {
380        private static final String FILE = "file";
381        private static final String SLASH = "/";
382
383        private final URI prefix;
384
385        public LinkResolverImpl(LinkResolverBasicContext context, URI prefix) {
386            this.prefix = Objects.requireNonNull(prefix);
387        }
388
389        @Override
390        public ResolvedLink resolveLink(Node node,
391                                        LinkResolverBasicContext context,
392                                        ResolvedLink link) {
393            var uri = URI.create(link.getUrl()).normalize();
394
395            if ((! uri.isAbsolute()) || FILE.equals(uri.getScheme())) {
396                if (! uri.getPath().startsWith(SLASH)) {
397                    link =
398                        link
399                        .withStatus(LinkStatus.VALID)
400                        .withUrl(prefix.resolve(uri).toString());
401                }
402            }
403
404            return link;
405        }
406    }
