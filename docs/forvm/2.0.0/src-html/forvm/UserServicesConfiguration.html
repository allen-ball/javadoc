023import forvm.repository.AuthorRepository;
024import forvm.repository.CredentialRepository;
025import forvm.repository.SubscriberRepository;
026import java.util.HashSet;
027import java.util.Set;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import org.springframework.beans.factory.annotation.Autowired;
032import org.springframework.context.annotation.Bean;
033import org.springframework.context.annotation.Configuration;
034import org.springframework.security.core.GrantedAuthority;
035import org.springframework.security.core.authority.SimpleGrantedAuthority;
036import org.springframework.security.core.userdetails.User;
037import org.springframework.security.core.userdetails.UserDetails;
038import org.springframework.security.core.userdetails.UserDetailsService;
039import org.springframework.security.core.userdetails.UsernameNotFoundException;
040import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserRequest;
041import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserService;
042import org.springframework.security.oauth2.client.userinfo.DefaultOAuth2UserService;
043import org.springframework.security.oauth2.client.userinfo.OAuth2UserRequest;
044import org.springframework.security.oauth2.client.userinfo.OAuth2UserService;
045import org.springframework.security.oauth2.core.OAuth2AuthenticationException;
046import org.springframework.security.oauth2.core.oidc.user.DefaultOidcUser;
047import org.springframework.security.oauth2.core.oidc.user.OidcUser;
048import org.springframework.security.oauth2.core.user.DefaultOAuth2User;
049import org.springframework.security.oauth2.core.user.OAuth2User;
050import org.springframework.transaction.annotation.Transactional;
051
052/**
053 * User details {@link Configuration}.
054 *
055 * {@injected.fields}
056 *
057 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
058 * @version $Revision: 7082 $
059 */
060@Configuration
061@NoArgsConstructor @ToString @Log4j2
062public class UserServicesConfiguration {
063    @Autowired private CredentialRepository credentialRepository = null;
064    @Autowired private AuthorRepository authorRepository = null;
065    @Autowired private SubscriberRepository subscriberRepository = null;
066
067
068    @Bean
069    public UserDetailsService userDetailsService() {
070        return new UserDetailsServiceImpl();
071    }
072
073    @Bean
074    public OAuth2UserService<OAuth2UserRequest,OAuth2User> oAuth2UserService() {
075        return new OAuth2UserServiceImpl();
076    }
077
078    @Bean
079    public OidcUserService oidcUserService() {
080        return new OidcUserServiceImpl();
081    }
082
083    private Set<GrantedAuthority> getGrantedAuthoritySet(String username) {
084        var set = new HashSet<GrantedAuthority>();
085
086        subscriberRepository.findById(username)
087            .ifPresent(t -> set.add(new SimpleGrantedAuthority("SUBSCRIBER")));
088
089        authorRepository.findById(username)
090            .ifPresent(t -> set.add(new SimpleGrantedAuthority("AUTHOR")));
091
092        return set;
093    }
094
095    @NoArgsConstructor @ToString
096    private class UserDetailsServiceImpl implements UserDetailsService {
097        @Override
098        @Transactional(readOnly = true)
099        public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
100            User user = null;
101
102            try {
103                var credential = credentialRepository.findById(username);
104
105                user = new
106                    User(username,
107                         credential.get().getPassword(),
108                         getGrantedAuthoritySet(username));
109            } catch (UsernameNotFoundException exception) {
110                throw exception;
111            } catch (Exception exception) {
112                throw new UsernameNotFoundException(username);
113            }
114
115            return user;
116        }
117    }
118
119    @NoArgsConstructor @ToString
120    private class OAuth2UserServiceImpl extends DefaultOAuth2UserService {
121        private final DefaultOAuth2UserService delegate = new DefaultOAuth2UserService();
122
123        @Override
124        @Transactional(readOnly = true)
125        public OAuth2User loadUser(OAuth2UserRequest request) throws OAuth2AuthenticationException {
126            var attribute =
127                request.getClientRegistration().getProviderDetails()
128                .getUserInfoEndpoint().getUserNameAttributeName();
129            var user = delegate.loadUser(request);
130
131            try {
132                user =
133                    new DefaultOAuth2User(getGrantedAuthoritySet(user.getName()),
134                                          user.getAttributes(), attribute);
135            } catch (OAuth2AuthenticationException exception) {
136                throw exception;
137            } catch (Exception exception) {
138                log.warn("{}", request, exception);
139            }
140
141            return user;
142        }
143    }
144
145    @NoArgsConstructor @ToString
146    private class OidcUserServiceImpl extends OidcUserService {
147        { setOauth2UserService(oAuth2UserService()); }
148
149        @Override
150        @Transactional(readOnly = true)
151        public OidcUser loadUser(OidcUserRequest request) throws OAuth2AuthenticationException {
152            var attribute =
153                request.getClientRegistration().getProviderDetails()
154                .getUserInfoEndpoint().getUserNameAttributeName();
155            var user = super.loadUser(request);
156
157            try {
158                user =
159                    new DefaultOidcUser(getGrantedAuthoritySet(user.getName()),
160                                        user.getIdToken(), user.getUserInfo(),
161                                        attribute);
162            } catch (OAuth2AuthenticationException exception) {
163                throw exception;
164            } catch (Exception exception) {
165                log.warn("{}", request, exception);
166            }
167
168            return user;
169        }
170    }
171}
