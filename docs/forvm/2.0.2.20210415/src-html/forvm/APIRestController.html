023import forvm.entity.Article;
024import forvm.entity.Author;
025import forvm.repository.ArticleRepository;
026import forvm.repository.AuthorRepository;
027import java.security.Principal;
028import java.util.NoSuchElementException;
029import java.util.Optional;
030import lombok.NoArgsConstructor;
031import lombok.ToString;
032import lombok.extern.log4j.Log4j2;
033import org.apache.commons.io.FilenameUtils;
034import org.springframework.beans.factory.annotation.Autowired;
035import org.springframework.http.HttpMethod;
036import org.springframework.http.HttpStatus;
037import org.springframework.http.ResponseEntity;
038import org.springframework.security.access.AccessDeniedException;
039import org.springframework.security.access.prepost.PreAuthorize;
040import org.springframework.transaction.annotation.Transactional;
041import org.springframework.web.bind.annotation.ExceptionHandler;
042import org.springframework.web.bind.annotation.PathVariable;
043import org.springframework.web.bind.annotation.RequestMapping;
044import org.springframework.web.bind.annotation.RequestParam;
045import org.springframework.web.bind.annotation.ResponseStatus;
046import org.springframework.web.bind.annotation.RestController;
047import org.springframework.web.multipart.MultipartFile;
048
049import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
050import static org.springframework.web.bind.annotation.RequestMethod.DELETE;
051import static org.springframework.web.bind.annotation.RequestMethod.POST;
052import static org.springframework.web.bind.annotation.RequestMethod.PUT;
053
054/**
055 * API {@link RestController} implementation.
056 *
057 * {@injected.fields}
058 *
059 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
060 * @version $Revision: 7215 $
061 */
062@RestController
063@RequestMapping(value = { "/api/v1" }, produces = APPLICATION_JSON_VALUE)
064@NoArgsConstructor @ToString @Log4j2
065public class APIRestController {
066    @Autowired private AuthorRepository authorRepository = null;
067    @Autowired private ArticleRepository articleRepository = null;
068    @Autowired private MarkdownService service = null;
069
070    @RequestMapping(method = { PUT }, value = { "/author", "/author/{slug}" })
071    @PreAuthorize("hasAuthority('AUTHOR')")
072    @Transactional
073    public ResponseEntity<String> author(Principal principal,
074                                         HttpMethod method,
075                                         @PathVariable Optional<String> slug,
076                                         @RequestParam MultipartFile file) throws Exception {
077        var email = principal.getName();
078        Author author = null;
079
080        switch (method) {
081        case PUT:
082            author = authorRepository.findById(email).get();
083            break;
084
085        default:
086            throw new MethodNotAllowedException(String.valueOf(method));
087        }
088
089        var name = file.getOriginalFilename();
090
091        if (! slug.isPresent()) {
092            slug = Optional.of(FilenameUtils.getBaseName(name));
093        }
094
095        service.compile(name, file.getBytes(), slug.get(), author);
096
097        if (email.equals(author.getEmail())) {
098            authorRepository.save(author);
099        } else {
100            throw new ForbiddenException(slug.get());
101        }
102
103        return new ResponseEntity<>(HttpStatus.OK);
104    }
105
106    @RequestMapping(method = { DELETE }, value = { "/article/{slug}" })
107    @PreAuthorize("hasAuthority('AUTHOR')")
108    @Transactional
109    public ResponseEntity<String> article(Principal principal,
110                                          HttpMethod method,
111                                          @PathVariable String slug) throws Exception {
112        var email = principal.getName();
113
114        switch (method) {
115        case DELETE:
116            var article = articleRepository.findBySlug(slug).get();
117
118            if (email.equals(article.getAuthor().getEmail())) {
119                articleRepository.delete(article);
120            } else {
121                throw new ForbiddenException(slug);
122            }
123            break;
124
125        default:
126            throw new MethodNotAllowedException(String.valueOf(method));
127        }
128
129        return new ResponseEntity<>(HttpStatus.OK);
130    }
131
132    @RequestMapping(method = { POST, PUT },
133                    value = { "/article", "/article/{slug}" })
134    @PreAuthorize("hasAuthority('AUTHOR')")
135    @Transactional
136    public ResponseEntity<String> article(Principal principal,
137                                          HttpMethod method,
138                                          @PathVariable Optional<String> slug,
139                                          @RequestParam MultipartFile file) throws Exception {
140        var email = principal.getName();
141        var name = file.getOriginalFilename();
142
143        if (! slug.isPresent()) {
144            slug = Optional.of(FilenameUtils.getBaseName(name));
145        }
146
147        Article article = null;
148
149        switch (method) {
150        case POST:
151            if (articleRepository.findBySlug(slug.get()).isPresent()) {
152                throw new ConflictException(slug.get());
153            }
154
155            article = new Article();
156            article.setAuthor(authorRepository.findById(email).get());
157            article.setEmail(article.getAuthor().getEmail());
158            break;
159
160        case PUT:
161            article = articleRepository.findBySlug(slug.get()).get();
162            break;
163
164        default:
165            throw new MethodNotAllowedException(String.valueOf(method));
166        }
167
168        service.compile(name, file.getBytes(),
169                        "/article", slug.get(), article);
170
171        if (email.equals(article.getAuthor().getEmail())) {
172            articleRepository.save(article);
173        } else {
174            throw new ForbiddenException(slug.get());
175        }
176
177        return new ResponseEntity<>(HttpStatus.OK);
178    }
179
180    @ExceptionHandler({ NoSuchElementException.class })
181    @ResponseStatus(value = HttpStatus.NOT_FOUND, reason = "Resource not found")
182    public void handleNOT_FOUND() { }
183
184    @ExceptionHandler({ AccessDeniedException.class, SecurityException.class })
185    @ResponseStatus(value = HttpStatus.FORBIDDEN, reason = "Forbidden")
186    public void handleFORBIDDEN() { }
187
188    @ResponseStatus(value = HttpStatus.CONFLICT)
189    public static class ConflictException extends RuntimeException {
190        private static final long serialVersionUID = -3019016947330828859L;
191
192        private ConflictException(String message) { super(message, null); }
193    }
194
195    @ResponseStatus(value = HttpStatus.FORBIDDEN)
196    public static class ForbiddenException extends RuntimeException {
197        private static final long serialVersionUID = -3698565165488957902L;
198
199        private ForbiddenException(String message) { super(message, null); }
200    }
201
202    @ResponseStatus(value = HttpStatus.METHOD_NOT_ALLOWED)
203    public static class MethodNotAllowedException extends RuntimeException {
204        private static final long serialVersionUID = 8201840514907306612L;
205
206        private MethodNotAllowedException(String message) {
207            super(message, null);
208        }
209    }
210}
