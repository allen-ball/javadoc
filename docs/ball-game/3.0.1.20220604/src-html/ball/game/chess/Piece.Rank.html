021import lombok.Data;
022import lombok.NonNull;
023
024/**
025 * Chess {@link Piece}.
026 *
027 * {@bean.info}
028 *
029 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
030 */
031@Data
032public class Piece {
033    private static final String[][] UNICODE = {
034        /*         White,    Black   */
035        /* K */ { "\u2654", "\u265a" },
036        /* Q */ { "\u2655", "\u265b" },
037        /* R */ { "\u2656", "\u265c" },
038        /* B */ { "\u2657", "\u265d" },
039        /* N */ { "\u2658", "\u265e" },
040        /* P */ { "\u2659", "\u265f" }
041    };
042
043    @NonNull private final Rank rank;
044    @NonNull private final Color color;
045
046    @Override
047    public String toString() {
048        return UNICODE[rank.ordinal()][color.ordinal()];
049    }
050
051    /**
052     * {@link Piece} {@link Rank}.
053     */
054    public enum Rank { K, Q, R, B, N, P; }
055}
