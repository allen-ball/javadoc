023import java.math.BigInteger;
024import java.util.LinkedList;
025
026/**
027 * Life {@link Game}.
028 *
029 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
030 * @version $Revision: 7215 $
031 */
032public class Game extends LinkedList<BigInteger> {
033    private static final long serialVersionUID = 7650642647319679493L;
034
035    /** @serial */ private final Automata automata;
036
037    /**
038     * @param   height          The extent of the y-axis.
039     * @param   width           The extent of the x-axis.
040     * @param   state0          The initial state.
041     */
042    public Game(int height, int width, BigInteger state0) {
043        super();
044
045        automata = new Automata(height, width);
046
047        addLast(state0);
048    }
049
050    /**
051     * Method to get the {@link Automata} for this {@link Game}.
052     *
053     * @return  The {@link Automata}.
054     */
055    public Automata automata() { return automata; }
056}
