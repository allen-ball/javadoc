101    /**
102     * Private constructor for {@link Spliterator} implmentation.
103     *
104     * @param   parent          The source {@link Puzzle}.
105     * @param   solution        The {@link Solution} to update.
106     * @param   sequence        The {@link CharSequence} for the
107     *                          {@link Solution}.
108     */
109    private Puzzle(Puzzle parent, Solution solution, CharSequence sequence) {
110        this(parent);
111
112        try {
113            for (var coordinate : solution) {
114                put(coordinate, parent.get(coordinate).clone());
115            }
116
117            solution.setSolution(this, sequence);
118        } catch (Exception exception) {
119            throw new ExceptionInInitializerError(exception);
120        }
121    }

137    /**
138     * Private constructor for {@link #load(String)}.
139     *
140     * @param   headers         The {@link List} of header {@link String}s.
141     * @param   grid            The {@link List} of grid row
142     *                          {@link String}s.
143     * @param   clues           The {@link List} of clue {@link String}s.
144     * @param   notes           The {@link List} of note {@link String}s.
145     */
146    private Puzzle(List<String> headers, List<String> grid,
147                   List<String> clues, List<String> notes) {
148        this(null,
149             new OrderedHeaders(headers),
150             new ArrayList<>(), new TreeMap<>(), new IdentityHashMap<>(),
151             new TreeMap<>(), (notes != null) ? notes : new LinkedList<>());
152        /*
153         * Populate the grid.
154         */
155        for (var line : grid) {
156            var row = Cell.getRowFrom(line.trim());
157
158            if ((getColumnCount() * getRowCount()) > 0) {
159                if (getColumnCount() != row.size()) {
160                    throw new IllegalArgumentException(line + " does not have "
161                                                       + getColumnCount()
162                                                       + " columns");
163                }
164            }
165
166            resize(getRowCount() + 1, row.size());
167            copy(row(getRowCount() - 1).asList(), row);
168        }
169        /*
170         * Find the Coordinate groups that represent solutions.
171         */
172        List<CoordinateMap<Cell>> groups = new LinkedList<>();
173        var stream = Stream.concat(rows().stream(), columns().stream());
174
175        for (var line : (Iterable<CoordinateMap<Cell>>) stream::iterator) {
176            groups.add(new CoordinateMap<Cell>());
177
178            for (var entry : line.entrySet()) {
179                if (! entry.getValue().isBlock()) {
180                    groups.get(groups.size() - 1)
181                        .put(entry.getKey(), entry.getValue());
182                } else {
183                    groups.add(new CoordinateMap<Cell>());
184                }
185            }
186        }
187
188        groups.removeIf(t -> ! (t.size() > 1));
189        /*
190         * Calculate the Label indices.
191         */
192        var indices =
193            groups.stream()
194            .map(t -> t.firstKey())
195            .collect(toCollection(TreeSet::new));
196
197        this.indices.addAll(indices);
198        /*
199         * Populate the solutions and xref maps.
200         */
201        var solutions =
202            groups.stream()
203            .collect(toMap(k -> labelFor(k),
204                           v -> new Solution(v.keySet()),
205                           (v0, v1) -> v0,
206                           TreeMap::new));
207
208        this.solutions.putAll(solutions);
209
210        var values = this.solutions.values();
211        var xref =
212            values.stream()
213            .collect(toMap(k -> k,
214                           v -> (values.stream()
215                                 .filter(t -> t != v)
216                                 .filter(t -> (! disjoint(t, v)))
217                                 .collect(toList())),
218                           (v0, v1) -> v0, IdentityHashMap::new));
219
220        this.xref.putAll(xref);
221        /*
222         * Populate the clues map.
223         */
224        if (clues != null) {
225            clues.stream()
226                .filter(t -> isNotBlank(t))
227                .map(t -> t.split("[. ]+", 2))
228                .forEach(t -> this.clues.put(Label.parse(t[0]), t[1]));
229        }
230
231        var labels = this.clues.keySet().stream().collect(toList());;
232
233        labels.removeAll(this.solutions.keySet());
234
235        for (var label : labels) {
236            throw new IllegalArgumentException("`" + label
237                                               + "' not found in grid'");
238        }
239
240        for (Map.Entry<Label,String> entry : this.clues.entrySet()) {
241            var strings = entry.getValue().split("[~]", 2);
242
243            if (strings.length > 1) {
244                entry.setValue(strings[0].trim());
245
246                if (isNotBlank(strings[1])) {
247                    solutions
248                        .get(entry.getKey())
249                        .setSolution(this, strings[1].trim());
250                }
251            }
252        }
253
254        this.solutions.keySet()
255            .stream()
256            .forEach(t -> this.clues.computeIfAbsent(t, k -> "TBD"));
257    }

314    /**
315     * Method to set the solution in a {@link Puzzle}.
316     *
317     * @param   coordinate      The {@link Coordinate} of the {@link Cell}..
318     * @param   character       The solution {@link Character}.
319     *
320     * @throws  IllegalArgumentException
321     *                          If any of the {@link Cell}s are already with
322     *                          a different {@link Character} than specified
323     *                          in the {@link String}.
324     */
325    public void setSolution(Coordinate coordinate, Character character) {
326        var cell = get(coordinate);
327
328        if (cell == null) {
329            throw new IllegalStateException();
330        }
331
332        if (! cell.isSolved()) {
333            cell.setSolution(character);
334        } else {
335            if (! character.equals(cell.getSolution())) {
336                throw new IllegalArgumentException(coordinate.toString()
337                                                   + ": "
338                                                   + cell.toString()
339                                                   + " -> "
340                                                   + character.toString());
341            }
342        }
343    }

383    /**
384     * Method to write {@link Puzzle} to a {@link PrintWriter} in
385     * {@link.uri https://github.com/century-arcade/xd target=newtab .xd}
386     * format.
387     *
388     * @param   out             The {@link OutputStream}.
389     *
390     * @throws  IOException     If the {@link Puzzle} cannot be written.
391     */
392    public void writeTo(PrintWriter out) throws IOException {
393        for (var entry : headers().entrySet()) {
394            if (isNotBlank(entry.getValue())) {
395                out.println(entry.getKey() + COLON + SPACE + entry.getValue());
396            }
397        }
398
399        out.println(EMPTY);
400        out.println(EMPTY);
401
402        for (var row : rows()) {
403            for (var cell : row.values()) {
404                out.print(cell);
405            }
406
407            out.println();
408        }
409
410        out.println(EMPTY);
411
412        Direction last = null;
413
414        for (var entry : clues.entrySet()) {
415            if (! entry.getKey().getDirection().equals(last)) {
416                out.println(EMPTY);
417            }
418
419            last = entry.getKey().getDirection();
420
421            out.println(entry.getKey().toString() + DOT
422                        + SPACE + entry.getValue()
423                        + SPACE + TILDE + SPACE
424                        + solutions.get(entry.getKey()).getSolution(this));
425        }
426
427        if (! notes.isEmpty()) {
428            out.println(EMPTY);
429            out.println(EMPTY);
430
431            notes.stream().forEach(out::println);
432        }
433    }

450    private Map<Coordinate,Cell> changed() {
451        var map =
452            entrySet().stream()
453            .filter(t -> parent != null)
454            .filter(t -> t.getValue() != parent.get(t.getKey()))
455            .collect(toMap(Map.Entry::getKey,
456                           Map.Entry::getValue,
457                           (v0, v1) -> v0,
458                           TreeMap::new));
459
460        return map;
461    }

468    /**
469     * Static method to load an
470     * {@link.uri https://github.com/century-arcade/xd target=newtab .xd}
471     * resource or {@link FileInputStream File} into a {@link Puzzle}.
472     *
473     * @param   path            The resource path.
474     *
475     * @return  The {@link Puzzle}.
476     *
477     * @throws  IOException     If the path cannot be parsed.
478     */
479    public static Puzzle load(String path) throws IOException {
480        Puzzle puzzle = null;
481        InputStream in = null;
482
483        try {
484            in = Puzzle.class.getResourceAsStream(path);
485
486            if (in == null) {
487                in = new FileInputStream(path);
488            }
489
490            var lines =
491                new BufferedReader(new InputStreamReader(in, UTF_8))
492                .lines()
493                .map(t -> t.trim())
494                .collect(toList());
495            TreeMap<Integer,List<String>> sections = new TreeMap<>();
496
497            while (! lines.isEmpty()) {
498                var index = indexOfSubList(lines, BOUNDARY);
499
500                if (! (index < 0)) {
501                    sections.put(sections.size(), lines.subList(0, index));
502                    lines = lines.subList(index + BOUNDARY.size(), lines.size());
503                } else {
504                    sections.put(sections.size(), lines);
505                    break;
506                }
507            }
508
509            puzzle =
510                new Puzzle(sections.get(0), sections.get(1),
511                           sections.get(2), sections.get(3));
512        } catch (UncheckedIOException exception) {
513            throw exception.getCause();
514        } finally {
515            if (in != null) {
516                in.close();
517            }
518        }
519
520        return puzzle;
521    }

523    private static class OrderedHeaders extends LinkedHashMap<String,String> {
524        private static final long serialVersionUID = -6640158167374093950L;
525
526        private static final List<String> HEADERS =
527            List.of("Title", "Author", "Editor", "Special", "Rebus", "Date");
528
529        public OrderedHeaders(Collection<String> lines) {
530            super();
531
532            HEADERS.stream().forEach(t -> put(t, EMPTY));
533
534            if (lines != null) {
535                lines.stream()
536                    .filter(t -> isNotBlank(t))
537                    .map(t -> t.split(Pattern.quote(COLON), 2))
538                    .filter(t -> isNotBlank(t[0]))
539                    .forEach(t -> put(t[0].trim(),
540                                      (t.length > 1) ? t[1].trim() : EMPTY));
541            }
542
543            values().removeIf(t -> isBlank(t));
544        }
545    }

547    /**
548     * {@link Puzzle} {@link Solution Solution}.
549     */
550    public static class Solution extends ArrayList<Coordinate> {
551        private static final long serialVersionUID = -146821930642639986L;
552
553        private Solution(Collection<Coordinate> collection) {
554            super(collection);
555        }
556
557        /**
558         * Method to get the {@link CharSequence} of a {@link Puzzle}
559         * {@link Solution Solution}. 
560         *
561         * @param       puzzle  The {@link Puzzle}.
562         *
563         * @return      The {@link Solution} {@link CharSequence}.
564         */
565        public CharSequence getSolution(Puzzle puzzle) {
566            return (stream()
567                    .map(t -> puzzle.get(t))
568                    .map(Object::toString)
569                    .collect(joining()));
570        }
571
572        private void setSolution(Puzzle puzzle, CharSequence solution) {
573            if (solution.length() != size()) {
574                throw new IllegalArgumentException();
575            }
576
577            for (var i = 0; i < size(); i += 1) {
578                puzzle.setSolution(get(i), solution.charAt(i));
579            }
580        }
581
582        /**
583         * Method to determine if {@link.this} {@link Solution Solution} is
584         * solved.
585         *
586         * @param       puzzle  The {@link Puzzle}.
587         *
588         * @return      {@code true} if the {@link Solution} is complete;
589         *              {@code false} otherwise.
590         */
591        public boolean isSolved(Puzzle puzzle) {
592            return stream().map(t -> puzzle.get(t)).allMatch(Cell::isSolved);
593        }
594
595        private Pattern asPattern(Puzzle puzzle) {
596            var string = getSolution(puzzle).toString().toUpperCase();
597
598            return Pattern.compile(string);
599        }
600
601        private List<Coordinate> unsolved(Puzzle puzzle) {
602            return (stream()
603                    .filter(t -> (! puzzle.get(t).isSolved()))
604                    .collect(toList()));
605        }
606    }

608    @ToString
609    private static class Solver extends DispatchSpliterator<Puzzle> {
610        private final Puzzle parent;
611        private final Set<CharSequence> dictionary;
612
613        public Solver(Puzzle parent, Set<CharSequence> dictionary) {
614            super(Integer.MAX_VALUE, Spliterator.NONNULL);
615
616            this.parent = Objects.requireNonNull(parent);
617            this.dictionary = Objects.requireNonNull(dictionary);
618        }
619
620        public Comparator<Solution> prioritization() {
621            var comparator =
622                Comparator
623                .<Solution>comparingInt(t -> (t.stream()
624                                              .map(c -> parent.get(c))
625                                              .anyMatch(Cell::isSolved) ? +1 : -1))
626                .thenComparingInt(t -> t.unsolved(parent).size())
627                .thenComparingInt(Solution::size)
628                .reversed();
629
630            return comparator;
631        }
632
633        @Override
634        protected Spliterator<Supplier<Spliterator<Puzzle>>> spliterators() {
635            List<Supplier<Spliterator<Puzzle>>> list = new LinkedList<>();
636            var solution =
637                parent.solutions().values()
638                .stream()
639                .filter(t -> (! t.isSolved(parent)))
640                .sorted(prioritization())
641                .findFirst().orElse(null);
642
643            if (solution != null) {
644                var pattern = solution.asPattern(parent);
645                var sequences =
646                    dictionary.stream()
647                    .filter(t -> pattern.matcher(t).matches())
648                    .collect(toList());
649                var used =
650                    parent.solutions().values()
651                    .stream()
652                    .filter(t -> t.isSolved(parent))
653                    .map(t -> t.getSolution(parent))
654                    .collect(toList());
655
656                sequences.removeAll(used);
657
658                for (var sequence : sequences) {
659                    var child = new Puzzle(parent, solution, sequence);
660                    var changed = child.changed().keySet();
661                    var verified =
662                        child.xref.get(solution)
663                        .stream()
664                        .filter(t -> (! disjoint(t, changed)))
665                        .filter(t -> t.isSolved(child))
666                        .allMatch(t -> dictionary.contains(t.getSolution(child)));
667
668                    if (verified) {
669                        list.add(() -> new Solver(child, dictionary));
670                    }
671                }
672            }
673
674            if (list.isEmpty()) {
675                list.add(() -> Stream.of(parent).spliterator());
676            }
677
678            return list.spliterator();
679        }
680    }
