023import ball.game.card.Card.Rank;
024import ball.game.card.Card.Suit;
025import ball.game.card.Card;
026import ball.util.ListOrderComparator;
027import ball.util.stream.Combinations;
028import java.util.ArrayList;
029import java.util.Collection;
030import java.util.Collections;
031import java.util.Comparator;
032import java.util.List;
033import java.util.function.Consumer;
034import java.util.function.Predicate;
035import java.util.stream.IntStream;
036
037/**
038 * Poker hand {@link Evaluator}.
039 *
040 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
041 * @version $Revision: 8369 $
042 */
043public class Evaluator implements Predicate<List<Card>>, Consumer<List<Card>> {
044
045    /**
046     * {@link Card} {@link Comparator}.
047     */
048    public static final Comparator<Card> CARD =
049        Comparator
050        .comparing(Card::getRank, new ListOrderComparator<>(Rank.ACE_HIGH))
051        .thenComparing(Card::getSuit, new ListOrderComparator<>(Suit.values()));
052
053    /**
054     * Hand ({@link Card} {@link List}) {@link Comparator}.
055     */
056    public static final Comparator<List<Card>> HAND =
057        (l, r) -> (IntStream.range(0, Math.min(l.size(), r.size()))
058                   .map(t -> CARD.compare(l.get(t), r.get(t)))
059                   .filter(t -> t != 0)
060                   .findFirst().orElse(Integer.compare(l.size(), r.size())));
061
062    private final List<Card> hand;
063    private final List<Ranking> orBetter;
064    private Ranking ranking = Ranking.Empty;
065    private List<Card> scoring = List.of();
066
067    /**
068     * Sole public constructor.
069     *
070     * @param   collection      The {@link Collection} of {@link Card}s to
071     *                          evaluate.
072     */
073    public Evaluator(Collection<Card> collection) {
074        this(collection, Ranking.values());
075    }
076
077    /**
078     * Protected constructor to search for specific {@link Ranking}(s).
079     *
080     * @param   collection      The {@link Collection} of {@link Card}s to
081     *                          evaluate.
082     * @param   rankings        The {@link Ranking}s to look for.
083     */
084    protected Evaluator(Collection<Card> collection, Ranking... rankings) {
085        hand = new ArrayList<>(collection);
086        hand.sort(CARD.reversed());
087
088        orBetter = new ArrayList<>(List.of(rankings));
089        Collections.reverse(orBetter);
090
091        var size = Math.min(5, hand.size());
092
093        orBetter.removeIf(t -> t.required() > size);
094
095        Combinations.of(size, size, this, hand)
096            .forEach(this);
097
098        for (int i = 0, n = scoring.size(); i < n; i += 1) {
099            Collections.swap(hand, i, hand.indexOf(scoring.get(i)));
100        }
101
102        hand.subList(scoring.size(), hand.size()).sort(CARD.reversed());
103    }
104
105    /**
106     * Method to get this hand as an unmodifiable {@link List} sorted
107     * according to its {@link Ranking}.
108     *
109     * @return  The sorted {@link List}.
110     */
111    public List<Card> getHand() { return Collections.unmodifiableList(hand); }
112
113    /**
114     * Method to get this hand's {@link Ranking}.
115     *
116     * @return  The {@link Ranking}.
117     */
118    public Ranking getRanking() { return ranking; }
119
120    /**
121     * Method to get this hand's scoring {@link Card}s as an unmodifiable
122     * {@link List}.
123     *
124     * @return  The {@link List} of scoring {@link Card}s.
125     */
126    public List<Card> getScoring() {
127        return Collections.unmodifiableList(scoring);
128    }
129
130    @Override
131    public boolean test(List<Card> prefix) {
132        return orBetter.stream().anyMatch(t -> t.possible().test(prefix));
133    }
134
135    @Override
136    public void accept(List<Card> list) {
137        var ranking =
138            orBetter.stream()
139            .filter(t -> t.test(list))
140            .findFirst().orElse(Ranking.Empty);
141
142        var scoring = list.subList(0, ranking.required());
143        var comparison = Ranking.COMPARATOR.compare(ranking, this.ranking);
144
145        if (comparison > 0) {
146            this.ranking = ranking;
147            this.scoring = scoring;
148
149            var index = orBetter.indexOf(ranking);
150
151            if (! (index < 0)) {
152                orBetter.subList(index + 1, orBetter.size()).clear();
153            }
154        } else if (comparison == 0) {
155            if (HAND.compare(scoring, this.scoring) > 0) {
156                this.scoring = scoring;
157            }
158        }
159    }
160
161    @Override
162    public String toString() {
163        return getRanking().name() + ":" + getScoring() + orBetter;
164    }
165}
