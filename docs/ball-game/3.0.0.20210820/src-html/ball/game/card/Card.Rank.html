042/**
043 * Playing {@link Card}.
044 *
045 * {@bean.info}
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 * @version $Revision: 8370 $
049 */
050public class Card implements Comparable<Card> {
051    private static final Comparator<? super Card> COMPARATOR =
052        Comparator
053        .<Card>comparingInt(t -> t.getSuit().ordinal())
054        .thenComparingInt(t -> t.getRank().ordinal());
055
056    private final Suit suit;
057    private final Rank rank;
058    private final transient String string;
059
060    /**
061     * Sole protected constructor.
062     *
063     * @param   suit            The {@link Card} {@link Suit}.
064     * @param   rank            The {@link Card} {@link Rank}.
065     */
066    @ConstructorProperties({ "suit", "rank" })
067    protected Card(Suit suit, Rank rank) {
068        this.rank = rank;
069        this.suit = suit;
070
071        if (rank.equals(Rank.JOKER)) {
072            if (suit != null) {
073                throw new IllegalArgumentException("suit=" + suit +
074                                                   ",rank=" + rank);
075            }
076        }
077
078        switch (rank) {
079        case JOKER:
080            this.string = rank.toString();
081            break;
082
083        default:
084            this.string = rank.toString() + "-" + suit.toString();
085            break;
086        }
087    }
088
089    /**
090     * Method to get the {@link Card} {@link Suit}.
091     *
092     * @return  {@link Suit}
093     */
094    public Suit getSuit() { return suit; }
095
096    /**
097     * Method to get the {@link Card} {@link Rank}.
098     *
099     * @return  {@link Rank}
100     */
101    public Rank getRank() { return rank; }
102
103    /**
104     * Method to get the {@link Card} {@link Color}.
105     *
106     * @return  {@link Suit#getColor()}
107     */
108    public Color getColor() {
109        Suit suit = getSuit();
110
111        return (suit != null) ? suit.getColor() : null;
112    }
113
114    @Override
115    public int compareTo(Card that) {
116        return Objects.compare(this, that, COMPARATOR);
117    }
118
119    @Override
120    public boolean equals(Object object) {
121        return ((object instanceof Card)
122                    ? (this.compareTo((Card) object) == 0)
123                    : super.equals(object));
124    }
125
126    @Override
127    public int hashCode() { return Objects.hash(getSuit(), getRank()); }
128
129    @Override
130    public String toString() { return string; }
131
132    /**
133     * Static method to parse a {@link String} consistent with
134     * {@link #toString} to a {@link Card}.
135     *
136     * @param   string          The {@link String} to parse.
137     *
138     * @return  The {@link Card}.
139     */
140    public static Card parse(String string) {
141        Card card = null;
142
143        try {
144            var substrings = string.split(Pattern.quote("-"), 2);
145
146            card =
147                new Card((substrings.length > 1)
148                             ? Suit.parse(substrings[1])
149                             : null,
150                         Rank.parse(substrings[0]));
151        } catch (Exception exception) {
152            throw new IllegalArgumentException(string, exception);
153        }
154
155        return card;
156    }
157
158    private static <T> Predicate<List<T>> same(Function<T,Predicate<T>> mapper) {
159        return t -> ((! t.isEmpty()) && t.stream().allMatch(mapper.apply(t.get(0))));
160    }
161
162    private static <T,R> List<R> listOf(Collection<T> collection, Function<T,R> mapper) {
163        return collection.stream().map(mapper).collect(toList());
164    }

166    /**
167     * {@link Card} rank {@link Enum} type.
168     */
169    public enum Rank implements Predicate<Card> {
170        JOKER,
171        ACE,
172        TWO, THREE, FOUR, FIVE, SIX, SEVEN, EIGHT, NINE,
173        TEN, JACK, QUEEN, KING;
174
175        private transient String string = null;
176
177        @Override
178        public boolean test(Card card) {
179            return is(this).test(card.getRank());
180        }
181
182        @Override
183        public String toString() {
184            if (string == null) {
185                switch (this) {
186                case JOKER:
187                    string = super.toString();
188                    break;
189
190                case ACE:
191                case JACK:
192                case QUEEN:
193                case KING:
194                    string = name().substring(0, 1);
195                    break;
196
197                default:
198                    string = String.valueOf(ordinal());
199                    break;
200                }
201            }
202
203            return string;
204        }
205
206        /**
207         * {@include #ACE_HIGH}
208         */
209        public static final List<Rank> ACE_HIGH  =
210            List.of(JOKER,
211                    TWO, THREE, FOUR, FIVE,
212                    SIX, SEVEN, EIGHT, NINE,
213                    TEN, JACK, QUEEN, KING, ACE);
214
215        /**
216         * {@include #ACE_LOW}
217         */
218        public static final List<Rank> ACE_LOW = List.of(values());
219
220        private static final Map<String,Rank> MAP;
221        private static final List<List<Rank>> SEQUENCES;
222
223        static {
224            var map = new TreeMap<String,Rank>(String.CASE_INSENSITIVE_ORDER);
225
226            for (var rank : values()) {
227                map.put(rank.name(), rank);
228                map.put(rank.toString(), rank);
229            }
230
231            MAP = unmodifiableMap(map);
232
233            var high = new ArrayList<Rank>(Rank.ACE_HIGH);
234            var low = new ArrayList<Rank>(Rank.ACE_LOW);
235
236            reverse(high);
237            reverse(low);
238
239            SEQUENCES = List.of(unmodifiableList(high), unmodifiableList(low));
240        }
241
242        /**
243         * {@link Predicate} to test all {@link Card}s are the same
244         * {@link Rank}.
245         */
246        public static final Predicate<List<Card>> SAME = same(Card::getRank);
247
248        /**
249         * {@link Predicate} to test the {@link Card}s make up a sequence.
250         */
251        public static final Predicate<List<Card>> SEQUENCE =
252            t -> ((! t.isEmpty()) && sequence(listOf(t, Card::getRank)));
253
254        private static boolean sequence(List<Rank> list) {
255            return SEQUENCES.stream().anyMatch(t -> indexOfSubList(t, list) >= 0);
256        }
257
258        /**
259         * Method to return a {@link Predicate} to test if a {@link Rank} is
260         * the specified {@link Rank}.
261         *
262         * @param       rank    The {@link Rank}.
263         *
264         * @return      {@link Predicate}
265         */
266        public static Predicate<Rank> is(Rank rank) {
267            return t -> Objects.equals(rank, t);
268        }
269
270        /**
271         * Static method to parse a {@link String} consistent with
272         * {@link #name()} and {@link #toString()} to a {@link Rank}.
273         *
274         * @param       string  The {@link String} to parse.
275         *
276         * @return      The {@link Rank}.
277         */
278        public static Rank parse(String string) {
279            var rank = MAP.get(string);
280
281            if (rank == null) {
282                rank = Enum.valueOf(Rank.class, string);
283            }
284
285            return rank;
286        }
287    }

319    /**
320     * {@link Card} suit {@link Enum} type.
321     */
322    public enum Suit implements Predicate<Card> {
323        CLUBS(Color.BLACK, "\u2667" /* U+2663 */),
324        DIAMONDS(Color.RED, "\u2662" /* U+2666 */),
325        HEARTS(Color.RED, "\u2661" /* U+2665 */),
326        SPADES(Color.BLACK, "\u2664" /* U+2660 */);
327
328        private static final Map<String,Suit> MAP;
329
330        static {
331            var map = new TreeMap<String,Suit>(String.CASE_INSENSITIVE_ORDER);
332
333            for (var suit : values()) {
334                map.put(suit.name(), suit);
335                map.put(suit.name().substring(0, 1), suit);
336                map.put(suit.toString(), suit);
337            }
338
339            MAP = unmodifiableMap(map);
340        }
341
342        private final Color color;
343        private final String string;
344
345        @ConstructorProperties({ "color", EMPTY })
346        private Suit(Color color, String string) {
347            this.color = color;
348            this.string = string;
349        }
350
351        /**
352         * Method to get the {@link Suit} {@link Color}.
353         *
354         * @return      {@link Color}
355         */
356        public Color getColor() { return color; }
357
358        @Override
359        public boolean test(Card card) {
360            return is(this).test(card.getSuit());
361        }
362
363        @Override
364        public String toString() { return string; }
365
366        /**
367         * {@link Predicate} to test all {@link Card}s are the same
368         * {@link Suit}.
369         */
370        public static final Predicate<List<Card>> SAME = same(Card::getSuit);
371
372        /**
373         * Method to return a {@link Predicate} to test if a {@link Suit} is
374         * the specified {@link Suit}.
375         *
376         * @param       suit    The {@link Suit}.
377         *
378         * @return      {@link Predicate}
379         */
380        public static Predicate<Suit> is(Suit suit) {
381            return t -> Objects.equals(suit, t);
382        }
383
384        /**
385         * Static method to parse a {@link String} consistent with
386         * {@link #name()} and {@link #toString()} to a {@link Suit}.
387         *
388         * @param       string  The {@link String} to parse.
389         *
390         * @return      The {@link Suit}.
391         */
392        public static Suit parse(String string) {
393            var suit = MAP.get(string);
394
395            if (suit == null) {
396                suit = Enum.valueOf(Suit.class, string);
397            }
398
399            return suit;
400        }
401    }
