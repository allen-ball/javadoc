023import ball.annotation.ServiceProviderFor;
024import ball.annotation.processing.AnnotatedProcessor;
025import ball.annotation.processing.For;
026import javax.annotation.processing.Processor;
027import javax.annotation.processing.RoundEnvironment;
028import javax.lang.model.element.Element;
029import javax.lang.model.element.TypeElement;
030import lombok.NoArgsConstructor;
031import lombok.ToString;
032
033import static java.util.stream.Collectors.toSet;
034import static javax.tools.Diagnostic.Kind.ERROR;
035
036/**
037 * {@link Processor} implementation to check {@link Class}es annotated with
038 * {@link LetterPremium} or {@link WordPremium}:
039 * <ol>
040 *   <li value="1">Are an instance of {@link SQ},</li>
041 *   <li value="2">Concrete, and</li>
042 *   <li value="3">
043 *      Are annotated with only one of {@link LetterPremium} or
044 *      {@link WordPremium}
045 *   </li>
046 * </ol>
047 *
048 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
049 * @version $Revision: 8365 $
050 */
051@ServiceProviderFor({ Processor.class })
052@For({ LetterPremium.class, WordPremium.class })
053@NoArgsConstructor @ToString
054public class PremiumProcessor extends AnnotatedProcessor {
055    @Override
056    public void process(RoundEnvironment roundEnv,
057                        TypeElement annotation, Element element) {
058        super.process(roundEnv, annotation, element);
059
060        switch (element.getKind()) {
061        case CLASS:
062            var set =
063                getSupportedAnnotationTypeList().stream()
064                .filter(t -> element.getAnnotation(t) != null)
065                .map(t -> asTypeElement(t))
066                .collect(toSet());
067
068            set.remove(annotation);
069
070            if (! set.isEmpty()) {
071                print(ERROR, element,
072                      "%s annotated with @%s but is also annotated with @%s",
073                      element.getKind(),
074                      annotation.getSimpleName(),
075                      set.iterator().next().getSimpleName());
076            }
077            break;
078
079        default:
080            break;
081        }
082    }
083}
