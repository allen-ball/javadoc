024/**
025 * Bridge position {@link Enum} type.
026 *
027 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
028 * @version $Revision: 8365 $
029 */
030public enum Position {
031    SOUTH, WEST, NORTH, EAST;
032
033    /**
034     * Method to determine the {@link Position} to the left of the argument
035     * {@link Position}.
036     *
037     * @param   right           The right {@link Position}.
038     *
039     * @return  The {@link Position} to the left.
040     */
041    public static Position leftOf(Position right) {
042        var values = values();
043
044        return values[(right.ordinal() + 1) % values.length];
045    }
046
047    /**
048     * Method to determine the {@link Position} to the right of the argument
049     * {@link Position}.
050     *
051     * @param   left            The left {@link Position}.
052     *
053     * @return  The {@link Position} to the left.
054     */
055    public static Position rightOf(Position left) {
056        var values = values();
057
058        return values[(left.ordinal() + values.length - 1) % values.length];
059    }
060}
