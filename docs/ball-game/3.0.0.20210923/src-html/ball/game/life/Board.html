023import ball.swing.table.AbstractTableModelImpl;
024
025/**
026 * Life {@link Board}.
027 *
028 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
029 * @version $Revision: 8365 $
030 */
031public class Board extends AbstractTableModelImpl {
032    private static final long serialVersionUID = -8272117924997665864L;
033
034    /** @serial */ private final Game game;
035
036    /**
037     * @param   game            The {@link Game}.
038     */
039    public Board(Game game) {
040        super(game.automata().getWidth());
041
042        this.game = game;
043    }
044
045    public Game game() { return game; }
046
047    @Override
048    public int getRowCount() { return game().automata().getHeight(); }
049
050    @Override
051    public String getValueAt(int y, int x) {
052        var game = game();
053
054        return game.automata().get(game.getLast(), y, x) ? "+" : "-";
055    }
056}
