023import lombok.Data;
024import lombok.NonNull;
025
026/**
027 * Chess {@link Piece}.
028 *
029 * {@bean.info}
030 *
031 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
032 * @version $Revision: 7215 $
033 */
034@Data
035public class Piece {
036    private static final String[][] UNICODE = {
037        /*         White,    Black   */
038        /* K */ { "\u2654", "\u265a" },
039        /* Q */ { "\u2655", "\u265b" },
040        /* R */ { "\u2656", "\u265c" },
041        /* B */ { "\u2657", "\u265d" },
042        /* N */ { "\u2658", "\u265e" },
043        /* P */ { "\u2659", "\u265f" }
044    };
045
046    @NonNull private final Rank rank;
047    @NonNull private final Color color;
048
049    @Override
050    public String toString() {
051        return UNICODE[rank.ordinal()][color.ordinal()];
052    }
053
054    /**
055     * {@link Piece} {@link Rank}.
056     */
057    public enum Rank { K, Q, R, B, N, P; }
058}
