023import java.util.List;
024import lombok.NoArgsConstructor;
025import lombok.ToString;
026import lombok.extern.slf4j.Slf4j;
027import org.apache.maven.plugin.MojoExecutionException;
028import org.apache.maven.plugin.MojoFailureException;
029import org.apache.maven.plugins.annotations.Mojo;
030import org.apache.maven.plugins.annotations.Parameter;
031
032import static org.apache.maven.plugins.annotations.LifecyclePhase.PACKAGE;
033
034/**
035 * {@link org.apache.maven.plugin.Mojo} to attach (zero or more) artifacts
036 * to a project.
037 *
038 * {@maven.plugin.fields}
039 *
040 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
041 * @version $Revision: 7215 $
042 */
043@Mojo(name = "attach", defaultPhase = PACKAGE, requiresProject = true)
044@NoArgsConstructor @ToString @Slf4j
045public class ArtifactAttachMojo extends AbstractArtifactMojo {
046    @Parameter(property = "artifacts")
047    private List<Artifact> artifacts = null;
048
049    @Override
050    public void execute() throws MojoExecutionException, MojoFailureException {
051        try {
052            if (isConfigured()) {
053                attach(this);
054            }
055
056            if (artifacts != null) {
057                for (Artifact artifact : artifacts) {
058                    if (artifact.isConfigured()) {
059                        attach(artifact);
060                    }
061                }
062            }
063        } catch (Exception exception) {
064            if (exception instanceof MojoExecutionException) {
065                throw (MojoExecutionException) exception;
066            } else if (exception instanceof MojoFailureException) {
067                throw (MojoFailureException) exception;
068            } else {
069                String message = exception.getMessage();
070
071                throw new MojoFailureException(exception, message, message);
072            }
073        }
074    }
075
076    /**
077     * {@code <artifact/>} parameter.
078     */
079    @NoArgsConstructor @ToString
080    public static class Artifact extends AbstractAttachedArtifact {
081    }
082}
