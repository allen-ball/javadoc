023import java.io.File;
024import java.util.Objects;
025import javax.inject.Inject;
026import lombok.Getter;
027import lombok.NoArgsConstructor;
028import lombok.Setter;
029import lombok.ToString;
030import lombok.extern.slf4j.Slf4j;
031import org.apache.maven.plugin.AbstractMojo;
032import org.apache.maven.plugins.annotations.Parameter;
033import org.apache.maven.project.MavenProject;
034import org.apache.maven.project.MavenProjectHelper;
035
036import static lombok.AccessLevel.PROTECTED;
037
038/**
039 * {@link org.apache.maven.plugin.Mojo}.
040 *
041 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
042 * @version $Revision: 6330 $
043 */
044@NoArgsConstructor(access = PROTECTED) @ToString @Slf4j
045public abstract class AbstractArtifactMojo extends AbstractMojo
046                                           implements AttachedArtifact {
047    protected static final String JAR = "jar";
048
049    @Inject
050    private MavenProjectHelper helper = null;
051
052    @Parameter(defaultValue = "${project}")
053    private MavenProject project = null;
054
055    @Parameter(defaultValue = "${project.build.directory}")
056    private File directory = null;
057
058    @Parameter(defaultValue = "${project.build.finalName}")
059    private String name = null;
060
061    @Parameter(property = "type")
062    @Getter
063    private String type = null;
064
065    @Parameter(property = "classifier")
066    @Getter
067    private String classifier = null;
068
069    @Parameter(property = "file")
070    @Getter
071    private File file = null;
072
073    @Override
074    public boolean isConfigured() {
075        return isNotEmpty(type) || isNotEmpty(classifier) || file != null;
076    }
077
078    /**
079     * Method to set the main artifact.
080     *
081     * @param   artifact        The {@link AttachedArtifact}.
082     */
083    protected void set(AttachedArtifact artifact) {
084        File file = getArtifactFile(artifact);
085
086        log.info("{} <-- {}",
087                 label(project.getArtifact().getClassifier(),
088                       project.getArtifact().getType()), file);
089        project.getArtifact().setFile(file);
090    }
091
092    /**
093     * Method to attach an artifact.  See
094     * {@link MavenProjectHelper#attachArtifact(MavenProject,String,String,File)}.
095     *
096     * @param   artifact        The {@link AttachedArtifact}.
097     */
098    protected void attach(AttachedArtifact artifact) {
099        File file = getArtifactFile(artifact);
100
101        log.info("{} <-- {}", label(artifact), file);
102        helper.attachArtifact(project,
103                              defaultString(artifact.getType(), JAR),
104                              artifact.getClassifier(),
105                              file);
106    }
107
108    /**
109     * Method to get the default build {@link File}.
110     *
111     * @param   artifact        The {@link AttachedArtifact}.
112     *
113     * @return  The argument {@link File} if non-{@code null}, the default
114     *          {@link File} otherwise.
115     */
116    protected File getArtifactFile(AttachedArtifact artifact) {
117        String type = defaultString(artifact.getType(), JAR);
118        String classifier = artifact.getClassifier();
119        File file = artifact.getFile();
120
121        if (file == null) {
122            file =
123                new File(directory,
124                         name
125                         + (isNotEmpty(classifier) ? ("-" + classifier) : "")
126                         + "." + type);
127        }
128
129        return file;
130    }
131
132    private String label(AttachedArtifact artifact) {
133        return label(artifact.getType(), artifact.getClassifier());
134    }
135
136    private String label(String type, String classifier) {
137        return (defaultString(classifier, "")
138                + ((isNotEmpty(classifier) && isNotEmpty(type)) ? ":" : "")
139                + defaultString(type, ""));
140    }
141
142    private static boolean isEmpty(CharSequence sequence) {
143        return Objects.isNull(sequence) || sequence.length() == 0;
144    }
145
146    private static boolean isNotEmpty(CharSequence sequence) {
147        return (! isEmpty(sequence));
148    }
149
150    private static String defaultString(String string, String defaultString) {
151        return (string != null) ? string : defaultString;
152    }
153
154    /**
155     * {@code <artifact/>} parameter abstract base class.
156     */
157    @NoArgsConstructor(access = PROTECTED) @ToString
158    protected static class AbstractAttachedArtifact
159                           implements AttachedArtifact {
160        @Getter @Setter private String type = null;
161        @Getter @Setter private String classifier = null;
162        @Getter @Setter private File file = null;
163
164        @Override
165        public boolean isConfigured() {
166            return isNotEmpty(type) || isNotEmpty(classifier) || file != null;
167        }
168    }
169}
