181    /**
182     * Dispatches call to {@link #declaration(Tag,Field)} or
183     * {@link #declaration(Tag,Method)} as appropriate.
184     *
185     * @param   tag             The {@link Tag}.
186     * @param   member          The target {@link Member}.
187     *
188     * @return  {@link org.w3c.dom.DocumentFragment}
189     */
190    public default FluentNode declaration(Tag tag, Member member) {
191        FluentNode node = null;
192
193        if (member instanceof Field) {
194            node = declaration(tag, (Field) member);
195        } else if (member instanceof Method) {
196            node = declaration(tag, (Method) member);
197        } else {
198            throw new IllegalArgumentException(String.valueOf(member));
199        }
200
201        return node;
202    }

204    /**
205     * Method to generate a {@link Field} declaration with javadoc
206     * hyperlinks.
207     *
208     * @param   tag             The {@link Tag}.
209     * @param   field           The target {@link Field}.
210     *
211     * @return  {@link org.w3c.dom.DocumentFragment}
212     */
213    public default FluentNode declaration(Tag tag, Field field) {
214        return fragment(modifiers(field.getModifiers()),
215                        type(tag, field.getGenericType()),
216                        code(SPACE),
217                        a(tag, field, (String) null));
218    }

220    /**
221     * Method to generate a {@link Method} declaration with javadoc
222     * hyperlinks.
223     *
224     * @param   tag             The {@link Tag}.
225     * @param   method          The target {@link Method}.
226     *
227     * @return  {@link org.w3c.dom.DocumentFragment}
228     */
229    public default FluentNode declaration(Tag tag, Method method) {
230        FluentNode node =
231            fragment(modifiers(method.getModifiers()),
232                     type(tag, method.getGenericReturnType()),
233                     code(SPACE),
234                     a(tag, method, (String) null));
235
236        Parameter[] parameters = method.getParameters();
237
238        node.add(code("("));
239
240        for (int i = 0; i < parameters.length; i += 1) {
241            if (i > 0) {
242                node.add(code(", "));
243            }
244
245            node.add(declaration(tag, parameters[i]));
246        }
247
248        node.add(code(")"));
249
250        return node;
251    }

283     *
284     * @return  {@link org.w3c.dom.Element}
285     */
286    public default FluentNode annotation(Tag tag, Annotation annotation) {
287        Class<?> type = annotation.annotationType();
288        String string =
289            String.valueOf(annotation)
290            .replace(type.getCanonicalName(), type.getSimpleName());
291
292        return fragment().add(a(href(tag, type), code(string)));
293    }

295    /**
296     * Method to generate modifiers for {@code declaration()} methods.
297     *
298     * @param   modifiers       See {@link Modifier}.
299     *
300     * @return  {@link org.w3c.dom.DocumentFragment}
301     */
302    public default FluentNode modifiers(int modifiers) {
303        FluentNode node = fragment();
304        String string = Modifier.toString(modifiers);
305
306        if (isNotEmpty(string)) {
307            node.add(code(string + SPACE));
308        }
309
310        return node;
311    }

313    /**
314     * Method to generate types for {@code declaration()} methods.
315     *
316     * @param   tag             The {@link Tag}.
317     * @param   type            The target {@link Type}.
318     *
319     * @return  {@link org.w3c.dom.DocumentFragment}
320     */
321    public default FluentNode type(Tag tag, Type type) {
322        FluentNode node = null;
323
324        if (type instanceof ParameterizedType) {
325            node =
326                fragment(type(tag, ((ParameterizedType) type).getRawType()));
327
328            Type[] types = ((ParameterizedType) type).getActualTypeArguments();
329
330            node = node.add(code("<"));
331
332            for (int i = 0; i < types.length; i += 1) {
333                if (i > 0) {
334                    node.add(code(","));
335                }
336
337                node.add(type(tag, types[i]));
338            }
339
340            node.add(code(">"));
341        } else if (type instanceof Class<?>) {
342            node = a(tag, (Class<?>) type);
343        } else {
344            node = code(type.getTypeName());
345        }
346
347        return node;
348    }

366    /**
367     * {@code <table>}{@link TableModel model}{@code </table>}
368     *
369     * @param   tag             The {@link Tag}.
370     * @param   model           The {@link TableModel} to use to create the
371     *                          new table {@link org.w3c.dom.Element}.
372     * @param   nodes           The {@link Node}s to append to the newly
373     *                          created
374     *                          {@link org.w3c.dom.Element}.
375     *
376     * @return  {@link org.w3c.dom.Element}
377     */
378    public default FluentNode table(Tag tag, TableModel model, Node... nodes) {
379        FluentNode table = table();
380        String[] names =
381            IntStream.range(0, model.getColumnCount())
382            .boxed()
383            .map(model::getColumnName)
384            .toArray(String[]::new);
385
386        if (! isAllBlank(names)) {
387            table.add(thead(tr(Stream.of(names).map(this::th))));
388        }
389
390        table
391            .add(tbody(IntStream.range(0, model.getRowCount())
392                       .boxed()
393                       .map(y -> tr(IntStream.range(0, names.length)
394                                    .boxed()
395                                    .map(x -> td(toHTML(tag,
396                                                        model.getValueAt(y, x))))))));
397
398        return table.add(nodes);
399    }

401    /**
402     * Method to get a Javadoc HTML representation of an {@link Object}.
403     *
404     * @param   tag             The {@link Tag}.
405     * @param   object          The target {@link Object}.
406     *
407     * @return  {@link org.w3c.dom.Node}
408     */
409    public default FluentNode toHTML(Tag tag, Object object) {
410        FluentNode node = null;
411
412        if (object instanceof byte[]) {
413            node =
414                text(Stream.of(ArrayUtils.toObject((byte[]) object))
415                     .map (t -> String.format("0x%02X", t))
416                     .collect(Collectors.joining(", ", "[", "]")));
417        } else if (object instanceof boolean[]) {
418            node = text(Arrays.toString((boolean[]) object));
419        } else if (object instanceof double[]) {
420            node = text(Arrays.toString((double[]) object));
421        } else if (object instanceof float[]) {
422            node = text(Arrays.toString((float[]) object));
423        } else if (object instanceof int[]) {
424            node = text(Arrays.toString((int[]) object));
425        } else if (object instanceof long[]) {
426            node = text(Arrays.toString((long[]) object));
427        } else if (object instanceof Object[]) {
428            node = toHTML(tag, Arrays.asList((Object[]) object));
429        } else if (object instanceof Type) {
430            node = type(tag, (Type) object);
431        } else if (object instanceof Enum<?>) {
432            node = a(tag, (Enum<?>) object);
433        } else if (object instanceof Field) {
434            node = a(tag, (Field) object);
435        } else if (object instanceof Constructor) {
436            node = a(tag, (Constructor) object);
437        } else if (object instanceof Method) {
438            node = a(tag, (Method) object);
439        } else if (object instanceof Collection<?>) {
440            List<Node> nodes =
441                ((Collection<?>) object)
442                .stream()
443                .map(t -> toHTML(tag, t))
444                .collect(Collectors.toList());
445
446            for (int i = nodes.size() - 1; i > 0; i -= 1) {
447                nodes.add(i, text(", "));
448            }
449
450            node =
451                fragment()
452                .add(text("["))
453                .add(nodes.stream())
454                .add(text("]"));
455        } else {
456            node = text(String.valueOf(object));
457        }
458
459        return node;
460    }
