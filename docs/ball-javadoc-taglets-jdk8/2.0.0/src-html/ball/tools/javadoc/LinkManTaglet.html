023import ball.annotation.MatcherGroup;
024import ball.annotation.PatternRegex;
025import ball.util.PatternMatcherBean;
026import ball.xml.FluentNode;
027import com.sun.javadoc.Tag;
028import java.io.File;
029import java.util.Map;
030import lombok.NoArgsConstructor;
031import lombok.ToString;
032
033/**
034 * Inline {@link com.sun.tools.doclets.Taglet} providing links to
035 * {@link.man man(1)} pages.
036 *
037 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
038 * @version $Revision: 6635 $
039 */
040@TagletName("link.man")
041@NoArgsConstructor @ToString
042@PatternRegex("(?is)(?<name>.+)[(](?<section>[\\p{Alnum}]+)[)]")
043public class LinkManTaglet extends AbstractInlineTaglet
044                           implements SunToolsInternalToolkitTaglet,
045                                      PatternMatcherBean {
046    private static final LinkManTaglet INSTANCE = new LinkManTaglet();
047
048    public static void register(Map<Object,Object> map) {
049        register(map, INSTANCE);
050    }
051
052    private String name = null;
053    private String section = null;
054
055    @MatcherGroup(1)
056    protected void setName(String string) { name = string; }
057
058    @MatcherGroup(2)
059    protected void setSection(String string) { section = string; }
060
061    @Override
062    public FluentNode toNode(Tag tag) throws Throwable {
063        PatternMatcherBean.super.initialize(tag.text().trim());
064
065        File path = new File(File.separator);
066
067        path = new File(path, "usr");
068        path = new File(path, "share");
069        path = new File(path, "man");
070        path = new File(path, "htmlman" + section);
071        path = new File(path, name + "." + section + ".html");
072
073        return a(path.toURI(), code(name + "(" + section + ")"));
074    }
075}
