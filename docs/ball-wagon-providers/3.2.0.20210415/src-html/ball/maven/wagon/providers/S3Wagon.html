068@Named("s3")
069@NoArgsConstructor @ToString @Slf4j
070public class S3Wagon extends AbstractWagonProvider {
071    @Getter @Setter private String profile = null;
072    @Getter @Setter private String region = null;
073    private volatile Bucket bucket = null;
074    private TransferManager manager = null;
075
076    @Override
077    protected void openConnectionInternal() throws AuthenticationException {
078        try {
079            if (bucket == null) {
080                synchronized (this) {
081                    if (bucket == null) {
082                        AmazonS3 client =
083                            AmazonS3ClientBuilder.standard()
084                            .withCredentials(new CredentialsProviderChain())
085                            .withRegion(new RegionProviderChain().getRegion())
086                            .build();
087                        String name = getHost();
088
089                        bucket =
090                            client.listBuckets()
091                            .stream()
092                            .filter(t -> name.equals(t.getName()))
093                            .findFirst()
094                            .orElseThrow(() -> new ResourceDoesNotExistException(getRepository().toString()));
095                        manager =
096                            TransferManagerBuilder.standard()
097                            .withS3Client(client)
098                            .build();
099                    }
100                }
101            }
102        } catch (Exception exception) {
103            if (exception instanceof AuthenticationException) {
104                throw (AuthenticationException) exception;
105            } else {
106                throw new AuthenticationException(getRepository().toString(),
107                                                  exception);
108            }
109        }
110    }
111
112    @Override
113    public void closeConnection() {
114        TransferManager manager = null;
115
116        synchronized (this) {
117            manager = this.manager;
118            this.manager = null;
119        }
120
121        if (manager != null) {
122            manager.shutdownNow();
123        }
124    }
125
126    @Override
127    public void get(String source,
128                    File target) throws TransferFailedException,
129                                        ResourceDoesNotExistException,
130                                        AuthorizationException {
131        Resource resource = new Resource(source);
132
133        fireGetInitiated(resource, target);
134        createParentDirectories(target);
135        fireGetStarted(resource, target);
136
137        try {
138            if (resourceExists(source)) {
139                manager.download(bucket.getName(), prefix() + source, target)
140                    .waitForCompletion();
141            } else {
142                throw new ResourceDoesNotExistException(source);
143            }
144        } catch (Exception exception) {
145            if (exception instanceof TransferFailedException) {
146                throw (TransferFailedException) exception;
147            } else if (exception instanceof ResourceDoesNotExistException) {
148                throw (ResourceDoesNotExistException) exception;
149            } else if (exception instanceof AuthorizationException) {
150                throw (AuthorizationException) exception;
151            } else {
152                throw new TransferFailedException(target + " <- " + source,
153                                                  exception);
154            }
155        }
156
157        postProcessListeners(resource, target, TransferEvent.REQUEST_GET);
158        fireGetCompleted(resource, target);
159    }
160
161    @Override
162    public boolean getIfNewer(String source, File target,
163                              long timestamp) throws TransferFailedException,
164                                                     ResourceDoesNotExistException,
165                                                     AuthorizationException {
166        boolean newer = false;
167        ObjectMetadata metadata =
168            manager.getAmazonS3Client()
169            .getObjectMetadata(bucket.getName(), prefix() + source);
170
171        if (metadata != null) {
172            newer = metadata.getLastModified().getTime() > timestamp;
173        }
174
175        if (newer) {
176            get(source, target);
177        }
178
179        return newer;
180    }
181
182    @Override
183    public void put(File source,
184                    String target) throws TransferFailedException,
185                                          ResourceDoesNotExistException,
186                                          AuthorizationException {
187        Resource resource = new Resource(target);
188
189        resource.setContentLength(source.length());
190        resource.setLastModified(source.lastModified());
191
192        firePutInitiated(resource, source);
193        firePutStarted(resource, source);
194
195        try {
196            manager.upload(bucket.getName(), prefix() + target, source)
197                .waitForCompletion();
198        } catch (Exception exception) {
199            if (exception instanceof TransferFailedException) {
200                throw (TransferFailedException) exception;
201            } else if (exception instanceof ResourceDoesNotExistException) {
202                throw (ResourceDoesNotExistException) exception;
203            } else if (exception instanceof AuthorizationException) {
204                throw (AuthorizationException) exception;
205            } else {
206                throw new TransferFailedException(source + " -> " + target,
207                                                  exception);
208            }
209        }
210
211        postProcessListeners(resource, source, TransferEvent.REQUEST_PUT);
212        firePutCompleted(resource, source);
213    }
214
215    @Override
216    public boolean resourceExists(String name) throws TransferFailedException,
217                                                      AuthorizationException {
218        boolean exists = false;
219
220        try {
221            exists =
222                manager.getAmazonS3Client()
223                .doesObjectExist(bucket.getName(), prefix() + name);
224        } catch (Exception exception) {
225            if (exception instanceof TransferFailedException) {
226                throw (TransferFailedException) exception;
227            } else if (exception instanceof AuthorizationException) {
228                throw (AuthorizationException) exception;
229            } else {
230                throw new TransferFailedException(name, exception);
231            }
232        }
233
234        return exists;
235    }
236
237    @Override
238    public List<String> getFileList(String name) throws TransferFailedException,
239                                                        ResourceDoesNotExistException,
240                                                        AuthorizationException {
241        TreeSet<String> set = new TreeSet<>();
242        String prefix = prefix();
243
244        name = strip(name, DELIMITER);
245
246        if (isNotEmpty(name)) {
247            prefix += name + DELIMITER;
248        }
249
250        try {
251            ObjectListing listing =
252                manager.getAmazonS3Client()
253                .listObjects(bucket.getName(), defaultIfEmpty(prefix, null));
254
255            for (;;) {
256                for (S3ObjectSummary summary : listing.getObjectSummaries()) {
257                    String key = summary.getKey().substring(prefix.length());
258                    String[] entries = key.split(Pattern.quote(DELIMITER), 2);
259
260                    set.add(entries[0]
261                            + ((entries.length > 1) ? DELIMITER : EMPTY));
262                }
263
264                if (listing.isTruncated()) {
265                    listing =
266                        manager.getAmazonS3Client()
267                        .listNextBatchOfObjects(listing);
268                } else {
269                    break;
270                }
271            }
272        } catch (Exception exception) {
273            if (exception instanceof TransferFailedException) {
274                throw (TransferFailedException) exception;
275            } else if (exception instanceof AuthorizationException) {
276                throw (AuthorizationException) exception;
277            } else {
278                throw new TransferFailedException(name, exception);
279            }
280        }
281
282        return set.stream().collect(toList());
283    }

285    @ToString
286    private class CredentialsProviderChain extends AWSCredentialsProviderChain {
287        public CredentialsProviderChain() {
288            super(new ProfileCredentialsProvider(S3Wagon.this.profile),
289                  DefaultAWSCredentialsProviderChain.getInstance());
290        }
291    }
292
293    @ToString
294    private class RegionProviderChain extends AwsRegionProviderChain {
295        public RegionProviderChain() {
296            super(new RegionProvider(),
297                  new AwsProfileRegionProvider(S3Wagon.this.profile),
298                  new DefaultAwsRegionProviderChain());
299        }
300    }
301
302    @ToString
303    private class RegionProvider extends AwsRegionProvider {
304        public RegionProvider() { super(); }
305
306        @Override
307        public String getRegion() { return S3Wagon.this.region; }
308    }
