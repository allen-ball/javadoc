023import java.io.File;
024import java.io.IOException;
025import java.net.URI;
026import java.nio.file.spi.FileTypeDetector;
027import java.util.Iterator;
028import java.util.ServiceLoader;
029import lombok.NoArgsConstructor;
030import lombok.extern.slf4j.Slf4j;
031import org.apache.maven.wagon.AbstractWagon;
032import org.apache.maven.wagon.ResourceDoesNotExistException;
033import org.apache.maven.wagon.TransferFailedException;
034import org.apache.maven.wagon.authorization.AuthorizationException;
035
036import static lombok.AccessLevel.PROTECTED;
037import static org.apache.commons.lang3.StringUtils.EMPTY;
038import static org.apache.commons.lang3.StringUtils.isNotEmpty;
039import static org.apache.commons.lang3.StringUtils.strip;
040
041/**
042 * Abstract {@link org.apache.maven.wagon.Wagon} which provides a
043 * {@link #probeContentType(File)} method.
044 *
045 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
046 * @version $Revision: 7215 $
047 */
048@NoArgsConstructor(access = PROTECTED) @Slf4j
049public abstract class AbstractWagonProvider extends AbstractWagon {
050
051    /**
052     * {@link #DELIMITER} = {@value #DELIMITER}
053     */
054    protected static final String DELIMITER = "/";
055
056    private URI uri = null;
057    private String prefix = null;
058    private ServiceLoader<FileTypeDetector> loader =
059        ServiceLoader.load(FileTypeDetector.class,
060                           getClass().getClassLoader());
061
062    private URI getURI() {
063        if (uri == null) {
064            URI ssp = URI.create(getRepository().getUrl());
065
066            if (ssp.isOpaque()) {
067                ssp = URI.create(ssp.getSchemeSpecificPart());
068            }
069
070            uri = ssp;
071        }
072
073        return uri;
074    }
075
076    /**
077     * Method to get host from a possibly opaque
078     * {@code getRepository().getHost()}.
079     *
080     * @return  The host.
081     */
082    protected String getHost() { return getURI().getHost(); }
083
084    /**
085     * Method to get path from a possibly opaque
086     * {@code getRepository().getBasedir()}.
087     *
088     * @return  The path.
089     */
090    protected String getPath() { return getURI().getPath(); }
091
092    /**
093     * Method to calculate bucket key prefix from
094     * {@code getRepository().getBasedir()}.
095     *
096     * @return  The bucket key prefix.
097     */
098    protected String prefix() {
099        if (prefix == null) {
100            String basedir = strip(getPath(), DELIMITER);
101
102            prefix = isNotEmpty(basedir) ? (basedir + DELIMITER) : EMPTY;
103        }
104
105        return prefix;
106    }
107
108    /**
109     * Provides functionality similar to
110     * {@link java.nio.file.Files#probeContentType(Path)} but searches this
111     * {@code classpath} for {@link FileTypeDetector} implementations.
112     *
113     * @param   file            The {@link File} to probe.
114     *
115     * @return  The content type of the file, or null if the content type
116     *          cannot be determined.
117     */
118    protected String probeContentType(File file) {
119        String type = null;
120        Iterator<FileTypeDetector> iterator = loader.iterator();
121
122        while (iterator.hasNext()) {
123            try {
124                type = iterator.next().probeContentType(file.toPath());
125
126                if (type != null) {
127                    break;
128                }
129            } catch (IOException exception) {
130                continue;
131            }
132        }
133
134        return type;
135    }
136
137    @Override
138    public boolean supportsDirectoryCopy() { return false; }
139
140    @Override
141    public void putDirectory(File source,
142                             String target) throws TransferFailedException,
143                                                   ResourceDoesNotExistException,
144                                                   AuthorizationException {
145        throw new IllegalStateException();
146    }
147}
