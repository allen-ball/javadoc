058/**
059 * Abstract base class for inline {@link Taglet}s that load
060 * {@link.uri https://maven.apache.org/index.html Maven} artifacts.
061 *
062 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
063 */
064@NoArgsConstructor(access = PROTECTED)
065public abstract class MavenTaglet extends AbstractInlineTaglet {
066    private static final XPath XPATH = XPathFactory.newInstance().newXPath();
067
068    private static final String POM_XML = "pom.xml";
069    private static final String DEPENDENCY = "dependency";
070    private static final String GROUP_ID = "groupId";
071    private static final String ARTIFACT_ID = "artifactId";
072    private static final String VERSION = "version";
073
074    protected XPathExpression compile(String format, Object... argv) {
075        XPathExpression expression = null;
076
077        try {
078            expression = XPATH.compile(String.format(format, argv));
079        } catch (Exception exception) {
080            throw new IllegalStateException(exception);
081        }
082
083        return expression;
084    }
085
086    /**
087     * Method to locate the POM from an {@link Element}.
088     *
089     * @param   tag             The {@link UnknownInlineTagTree}.
090     * @param   context         The element to which the enclosing comment
091     *                          belongs.
092     *
093     * @return  The POM {@link File}.
094     *
095     * @throws  Exception       If the POM {@link File} cannot be found.
096     */
097    protected File getPomFileFor(UnknownInlineTagTree tag, Element context) throws Exception {
098        var name = defaultIfBlank(getText(tag).trim(), POM_XML);
099        var parent = new File(trees.getPath(context).getCompilationUnit().getSourceFile().toUri()).getParentFile();
100        var file = new File(parent, name);
101
102        while (parent != null) {
103            file = new File(parent, name);
104
105            if (file.isFile()) {
106                break;
107            } else {
108                file = null;
109            }
110
111            parent = parent.getParentFile();
112        }
113
114        if (! (file != null && file.isFile())) {
115            throw new FileNotFoundException(name);
116        }
117
118        return file;
119    }

121    /**
122     * Inline {@link Taglet} to provide a report of fields whose values are
123     * configured by the
124     * {@link.uri https://maven.apache.org/index.html Maven}
125     * {@link.uri https://maven.apache.org/plugin-developers/index.html Plugin}
126     * {@code plugin.xml}.
127     */
128    @TagletName("maven.plugin.fields")
129    @ServiceProviderFor({ Taglet.class })
130    @NoArgsConstructor @ToString
131    public static class PluginFields extends MavenTaglet {
132        private static final String PLUGIN_XML = "META-INF/maven/plugin.xml";
133
134        @Override
135        public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
136            TypeElement type = null;
137            var argv = getText(tag).trim().split("[\\p{Space}]+", 2);
138
139            if (isNotEmpty(argv[0])) {
140                type = getTypeElementFor(context, argv[0]);
141            } else {
142                type = getEnclosingTypeElement(context);
143            }
144
145            var url = getResourceURLOf(asClass(type));
146            var protocol = Protocol.of(url);
147            Document document = null;
148
149            switch (protocol) {
150            case FILE:
151                var root = url.getPath().replaceAll(Pattern.quote(getResourcePathOf(asClass(type))), EMPTY);
152
153                document =
154                    DocumentBuilderFactory.newInstance()
155                    .newDocumentBuilder()
156                    .parse(root + PLUGIN_XML);
157                break;
158
159            case JAR:
160                var jar = protocol.getJarFile(url);
161                var entry = jar.getEntry(PLUGIN_XML);
162
163                try (var in = jar.getInputStream(entry)) {
164                    document =
165                        DocumentBuilderFactory.newInstance()
166                        .newDocumentBuilder()
167                        .parse(in);
168                }
169                break;
170            }
171
172            if (document == null) {
173                throw new IllegalStateException("Cannot find " + PLUGIN_XML);
174            }
175
176            var mojo =
177                (Node)
178                compile("/plugin/mojos/mojo[implementation='%s']", type.getQualifiedName())
179                .evaluate(document, NODE);
180
181            return div(attr("class", "summary"),
182                       h3("Maven Plugin Parameter Summary"),
183                       table(tag, context, asClass(type), mojo,
184                             asStream((NodeList) compile("parameters/parameter").evaluate(mojo, NODESET))));
185        }
186
187        private FluentNode table(UnknownInlineTagTree tag, Element context, Class<?> type, Node mojo, Stream<Node> parameters) {
188            return table(thead(tr(th(EMPTY), th("Field"),
189                                  th("Default"), th("Property"),
190                                  th("Required"), th("Editable"),
191                                  th("Description"))),
192                         tbody(parameters.map(t -> tr(tag, context, type, mojo, t))));
193        }
194
195        private FluentNode tr(UnknownInlineTagTree tag, Element context, Class<?> type, Node mojo, Node parameter) {
196            var tr = fragment();
197
198            try {
199                var name = compile("name").evaluate(parameter);
200                var field = FieldUtils.getField(type, name, true);
201
202                if (field != null) {
203                    tr =
204                        tr(td((! type.equals(field.getDeclaringClass()))
205                                  ? type(tag, context, field.getDeclaringClass())
206                                  : text(EMPTY)),
207                           td(declaration(tag, context, field)),
208                           td(code(compile("configuration/%s/@default-value", name).evaluate(mojo))),
209                           td(code(compile("configuration/%s", name).evaluate(mojo))),
210                           td(code(compile("required").evaluate(parameter))),
211                           td(code(compile("editable").evaluate(parameter))),
212                           td(p(compile("description").evaluate(parameter))));
213                }
214            } catch (RuntimeException exception) {
215                throw exception;
216            } catch (Exception exception) {
217                throw new IllegalStateException(exception);
218            }
219
220            return tr;
221        }
222    }

224    /**
225     * Inline {@jdk.javadoc.doclet.link Taglet} to include generated
226     * {@link.uri https://maven.apache.org/index.html Maven}
227     * {@link.uri https://maven.apache.org/plugin-developers/index.html Plugin}
228     * help documentation.
229     */
230    @TagletName("maven.plugin.help")
231    @ServiceProviderFor({ Taglet.class })
232    @NoArgsConstructor @ToString
233    public static class PluginHelp extends MavenTaglet {
234        private static final String NAME = "plugin-help.xml";
235        @CompileTimeCheck
236        private static final Pattern PATTERN =
237            Pattern.compile("META-INF/maven/(?<g>[^/]+)/(?<a>[^/]+)/" + Pattern.quote(NAME));
238
239        @Override
240        public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
241            Class<?> type = null;
242
243            if (context instanceof PackageElement) {
244                type = asPackageInfoClass((PackageElement) context);
245            } else {
246                type = asClass(getEnclosingTypeElement(context));
247            }
248
249            var url = getResourceURLOf(type);
250            var protocol = Protocol.of(url);
251            Document document = null;
252
253            switch (protocol) {
254            case FILE:
255                var root = Paths.get(url.getPath().replaceAll(Pattern.quote(getResourcePathOf(type)), EMPTY));
256                var path =
257                    Files.walk(root, Integer.MAX_VALUE)
258                    .filter(Files::isRegularFile)
259                    .filter(t -> PATTERN.matcher(root.relativize(t).toString()).matches())
260                    .findFirst().orElse(null);
261
262                document =
263                    DocumentBuilderFactory.newInstance()
264                    .newDocumentBuilder()
265                    .parse(path.toFile());
266                break;
267
268            case JAR:
269                try (var jar = protocol.getJarFile(url)) {
270                    var entry =
271                        jar.stream()
272                        .filter(t -> PATTERN.matcher(t.getName()).matches())
273                        .findFirst().orElse(null);
274
275                    try (var in = jar.getInputStream(entry)) {
276                        document =
277                            DocumentBuilderFactory.newInstance()
278                            .newDocumentBuilder()
279                            .parse(in);
280                    }
281                }
282                break;
283            }
284
285            if (document == null) {
286                throw new IllegalStateException("Cannot find " + NAME);
287            }
288
289            return div(attr("class", "summary"),
290                       h3(compile("/plugin/name").evaluate(document)),
291                       p(compile("/plugin/description").evaluate(document)),
292                       table(tag, context,
293                             asStream((NodeList)compile("/plugin/mojos/mojo").evaluate(document, NODESET))));
294        }
295
296        private FluentNode table(UnknownInlineTagTree tag, Element context, Stream<Node> mojos) {
297            return table(thead(tr(th("Goal"), th("Phase"), th("Description"))),
298                         tbody(mojos.map(t -> tr(tag, context, t))));
299        }
300
301        private FluentNode tr(UnknownInlineTagTree tag, Element context, Node mojo) {
302            var tr = fragment();
303
304            try {
305                tr =
306                    tr(td(a(tag, context,
307                            compile("implementation").evaluate(mojo),
308                            code(compile("goal").evaluate(mojo)))),
309                       td(code(compile("phase").evaluate(mojo))),
310                       td(p(code(compile("description").evaluate(mojo)))));
311            } catch (RuntimeException exception) {
312                throw exception;
313            } catch (Exception exception) {
314                throw new IllegalStateException(exception);
315            }
316
317            return tr;
318        }
319    }

321    /**
322     * {@link jdk.javadoc.doclet.Taglet} to provide
323     * {@link.uri https://maven.apache.org/pom.html POM} coordinates as
324     * a {@code <dependency/>} element to include this documented
325     * {@link Class} or {@link Package}.
326     *
327     * <p>For example:</p>
328     *
329     * {@pom.coordinates}
330     */
331    @TagletName("pom.coordinates")
332    @ServiceProviderFor({ Taglet.class })
333    @NoArgsConstructor @ToString
334    public static class Coordinates extends MavenTaglet {
335        @CompileTimeCheck
336        private static final Pattern PATTERN =
337            Pattern.compile("META-INF/maven/(?<g>[^/]+)/(?<a>[^/]+)/pom[.]properties");
338
339        @Override
340        public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
341            var properties = new POMProperties();
342            Class<?> type = null;
343
344            if (context instanceof PackageElement) {
345                type = asPackageInfoClass((PackageElement) context);
346            } else {
347                type = asClass(getEnclosingTypeElement(context));
348            }
349
350            var url = getResourceURLOf(type);
351            var protocol = Protocol.of(url);
352
353            switch (protocol) {
354            case FILE:
355                var document =
356                    DocumentBuilderFactory.newInstance()
357                    .newDocumentBuilder()
358                    .parse(getPomFileFor(tag, context));
359
360                Stream.of(GROUP_ID, ARTIFACT_ID, VERSION)
361                    .forEach(t -> properties.load(t, document, "/project/"));
362                Stream.of(VERSION)
363                    .forEach(t -> properties.load(t, document, "/project/parent/"));
364                break;
365
366            case JAR:
367                try (var jar = protocol.getJarFile(url)) {
368                    var entry =
369                        jar.stream()
370                        .filter(t -> PATTERN.matcher(t.getName()).matches())
371                        .findFirst().orElse(null);
372
373                    if (entry != null) {
374                        try (var in = jar.getInputStream(entry)) {
375                            properties.load(in);
376                        }
377                    }
378                }
379                break;
380            }
381
382            return pre("xml",
383                       render(element(DEPENDENCY,
384                                      Stream.of(GROUP_ID, ARTIFACT_ID, VERSION)
385                                      .map(t -> element(t).content(properties.getProperty(t, "unknown")))),
386                              2));
387        }
388    }
389
390    private static enum Protocol {
391        FILE, JAR;
392
393        public static Protocol of(URL url) {
394            return valueOf(url.getProtocol().toUpperCase());
395        }
396
397        public JarFile getJarFile(URL url) throws IOException {
398            JarFile jar = null;
399
400            switch (this) {
401            case JAR:
402                jar = ((JarURLConnection) url.openConnection()).getJarFile();
403                break;
404
405            default:
406                throw new IllegalStateException();
407                /* break; */
408            }
409
410            return jar;
411        }
412    }
413
414    private static class POMProperties extends PropertiesImpl {
415        private static final long serialVersionUID = 1354153174028868013L;
416
417        public String load(String key, Document document, String prefix) {
418            if (document != null) {
419                computeIfAbsent(key, k -> evaluate(prefix + k, document));
420            }
421
422            return super.getProperty(key);
423        }
424
425        private String evaluate(String expression, Document document) {
426            String value = null;
427
428            try {
429                value = XPATH.evaluate(expression, document);
430            } catch (Exception exception) {
431            }
432
433            return isNotEmpty(value) ? value : null;
434        }
435    }
