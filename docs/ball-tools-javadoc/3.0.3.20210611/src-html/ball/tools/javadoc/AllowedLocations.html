023import ball.annotation.ServiceProviderFor;
024import ball.annotation.processing.AnnotatedProcessor;
025import ball.annotation.processing.For;
026import ball.annotation.processing.TargetMustExtend;
027import java.lang.annotation.Documented;
028import java.lang.annotation.Retention;
029import java.lang.annotation.Target;
030import javax.annotation.processing.Processor;
031import javax.annotation.processing.RoundEnvironment;
032import javax.lang.model.element.Element;
033import javax.lang.model.element.TypeElement;
034import jdk.javadoc.doclet.Taglet;
035import lombok.NoArgsConstructor;
036import lombok.ToString;
037
038import static java.lang.annotation.ElementType.TYPE;
039import static java.lang.annotation.RetentionPolicy.RUNTIME;
040import static javax.tools.Diagnostic.Kind.ERROR;
041
042/**
043 * {@link java.lang.annotation.Annotation} to mark {@link Taglet}s with
044 * their {@link Taglet#getAllowedLocations() allowed locations}.
045 *
046 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
047 * @version $Revision: 7215 $
048 */
049@Documented
050@Retention(RUNTIME)
051@Target({ TYPE })
052@TargetMustExtend(AnnotatedTaglet.class)
053public @interface AllowedLocations {
054    Taglet.Location[] value();
055
056    /**
057     * {@link Processor} implementation.
058     */
059    @ServiceProviderFor({ Processor.class })
060    @For({ AllowedLocations.class })
061    @NoArgsConstructor @ToString
062    public class ProcessorImpl extends AnnotatedProcessor {
063        @Override
064        protected void process(RoundEnvironment roundEnv,
065                               TypeElement annotation, Element element) {
066            super.process(roundEnv, annotation, element);
067
068            var mirror = getAnnotationMirror(element, annotation);
069            var value = getAnnotationValue(mirror, "value");
070
071            if (isEmptyArray(value)) {
072                print(ERROR, element, mirror, value, "value() is empty");
073            }
074        }
075    }
076}
