023import com.sun.source.doctree.DocTree;
024import com.sun.source.doctree.UnknownInlineTagTree;
025import java.util.List;
026import javax.lang.model.element.Element;
027import lombok.NoArgsConstructor;
028import org.w3c.dom.Node;
029
030import static lombok.AccessLevel.PROTECTED;
031
032/**
033 * Abstract base class for inline {@link jdk.javadoc.doclet.Taglet}
034 * implementations.
035 *
036 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
037 * @version $Revision: 7215 $
038 */
039@NoArgsConstructor(access = PROTECTED)
040public abstract class AbstractInlineTaglet extends AbstractTaglet {
041    @Override
042    public boolean isInlineTag() { return true; }
043
044    @Override
045    protected Node toNode(List<? extends DocTree> tags, Element element) throws Throwable {
046        return toNode((UnknownInlineTagTree) tags.get(0), element);
047    }
048
049    /**
050     * Abstract method to be overridden by subclass implementations.
051     *
052     * @param   tag             The instance of this tag.
053     * @param   element         The element to which the enclosing comment
054     *                          belongs.
055     *
056     * @return  The {@link Node} representing the output.
057     *
058     * @throws  Throwable       As required by the subclass.
059     */
060    protected abstract Node toNode(UnknownInlineTagTree tag, Element element) throws Throwable;
061}
