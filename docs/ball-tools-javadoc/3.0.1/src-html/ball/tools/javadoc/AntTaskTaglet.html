023import ball.annotation.ServiceProviderFor;
024import ball.util.ant.taskdefs.AntTask;
025import ball.xml.FluentNode;
026import com.sun.source.doctree.UnknownInlineTagTree;
027import java.util.AbstractMap.SimpleEntry;
028import java.util.HashSet;
029import java.util.Map;
030import java.util.Set;
031import java.util.regex.Pattern;
032import java.util.stream.Stream;
033import javax.lang.model.element.Element;
034import javax.lang.model.element.TypeElement;
035import jdk.javadoc.doclet.Taglet;
036import lombok.NoArgsConstructor;
037import lombok.ToString;
038import org.apache.tools.ant.ComponentHelper;
039import org.apache.tools.ant.IntrospectionHelper;
040import org.apache.tools.ant.Project;
041import org.apache.tools.ant.Task;
042import org.apache.tools.ant.taskdefs.Antlib;
043import org.w3c.dom.Node;
044
045import static java.util.stream.Collectors.joining;
046import static org.apache.commons.lang3.StringUtils.EMPTY;
047import static org.apache.commons.lang3.StringUtils.isNotEmpty;
048import static org.apache.tools.ant.MagicNames.ANTLIB_PREFIX;
049
050/**
051 * Inline {@link Taglet} to document {@link.uri http://ant.apache.org/ Ant}
052 * {@link Task}s.
053 *
054 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
055 * @version $Revision: 7215 $
056 */
057@TagletName("ant.task")
058@ServiceProviderFor({ Taglet.class })
059@NoArgsConstructor @ToString
060public class AntTaskTaglet extends AbstractInlineTaglet {
061    private static final String NO = "no";
062    private static final String YES = "yes";
063
064    private static final String INDENTATION = "  ";
065
066    private static final String DOCUMENTED = "DOCUMENTED";
067
068    @Override
069    public FluentNode toNode(UnknownInlineTagTree tag, Element element) throws Throwable {
070        FluentNode node = null;
071        TypeElement type = null;
072        var name = getText(tag).trim();
073
074        if (isNotEmpty(name)) {
075            type = getTypeElementFor(element, name);
076        } else {
077            type = getEnclosingTypeElement(element);
078        }
079
080        if (! isAssignableTo(Task.class).test(type)) {
081            throw new IllegalArgumentException(type.getQualifiedName()
082                                               + " is not a subclass of "
083                                               + Task.class.getCanonicalName());
084        }
085
086        var template =
087            render(template(tag, asClass(type)), INDENTATION.length())
088            .replaceAll(Pattern.quote(DOCUMENTED + "=\"\""), "...");
089
090        return div(attr("class", "block"), pre("xml", template));
091    }
092
093    private FluentNode template(UnknownInlineTagTree tag, Class<?> type) {
094        String name = null;
095
096        if (name == null) {
097            var project = new Project();
098            var pkg = type.getPackage().getName();
099
100            while (pkg != null) {
101                var url =
102                    type.getResource("/"
103                                     + String.join("/", pkg.split(Pattern.quote(".")))
104                                     + "/antlib.xml");
105
106                if (url != null) {
107                    try {
108                        Antlib.createAntlib(project, url, ANTLIB_PREFIX + pkg)
109                            .execute();
110                        break;
111                    } catch (Exception exception) {
112                    }
113                }
114
115                int index = pkg.lastIndexOf(".");
116
117                if (! (index < 0)) {
118                    pkg = pkg.substring(0, index);
119                } else {
120                    pkg = null;
121                }
122            }
123
124            var helper = ComponentHelper.getComponentHelper(project);
125
126            name =
127                helper.getTaskDefinitions().entrySet()
128                .stream()
129                .filter(t -> t.getValue().equals(type))
130                .map(t -> t.getKey())
131                .findFirst().orElse(null);
132        }
133
134        if (name == null) {
135            var annotation = type.getAnnotation(AntTask.class);
136
137            name = (annotation != null) ? annotation.value() : null;
138        }
139
140        if (name == null) {
141            name = type.getSimpleName();
142        }
143
144        return type(0, new HashSet<>(), tag, new SimpleEntry<>(name, type));
145    }
146
147    private FluentNode type(int depth, Set<Map.Entry<?,?>> set,
148                            UnknownInlineTagTree tag,
149                            Map.Entry<String,Class<?>> entry) {
150        var helper = IntrospectionHelper.getHelper(entry.getValue());
151        var node = element(entry.getKey());
152
153        if (set.add(entry)
154            && (! entry.getValue().getName()
155                  .startsWith(Task.class.getPackage().getName()))) {
156            node
157                .add(attributes(tag, helper))
158                .add(content(depth + 1, set, tag, helper));
159
160            if (helper.supportsCharacters()) {
161                var content = "... text ...";
162
163                if (node.hasChildNodes()) {
164                    content =
165                        Stream.generate(() -> INDENTATION)
166                        .limit(depth + 1)
167                        .collect(joining(EMPTY, "\n", content + "\n"));
168                }
169
170                node.add(text(content));
171            }
172        } else {
173            node.add(attr(DOCUMENTED));
174        }
175
176        return node;
177    }
178
179    private Node[] attributes(UnknownInlineTagTree tag,
180                              IntrospectionHelper helper) {
181        var array =
182            helper.getAttributeMap().entrySet()
183            .stream()
184            .map(t -> attr(t.getKey(), t.getValue().getSimpleName()))
185            .toArray(Node[]::new);
186
187        return array;
188    }
189
190    private FluentNode content(int depth, Set<Map.Entry<?,?>> set,
191                               UnknownInlineTagTree tag,
192                               IntrospectionHelper helper) {
193        return fragment(helper.getNestedElementMap().entrySet()
194                        .stream()
195                        .map(t -> type(depth, set, tag, t)));
196    }
197}
