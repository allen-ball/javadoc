060/**
061 * Abstract base class for inline {@link Taglet}s that load
062 * {@link.uri https://maven.apache.org/index.html Maven} artifacts.
063 *
064 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
065 * @version $Revision: 7215 $
066 */
067@NoArgsConstructor(access = PROTECTED)
068public abstract class MavenTaglet extends AbstractInlineTaglet {
069    private static final XPath XPATH = XPathFactory.newInstance().newXPath();
070
071    private static final String POM_XML = "pom.xml";
072    private static final String DEPENDENCY = "dependency";
073    private static final String GROUP_ID = "groupId";
074    private static final String ARTIFACT_ID = "artifactId";
075    private static final String VERSION = "version";
076
077    protected XPathExpression compile(String format, Object... argv) {
078        XPathExpression expression = null;
079
080        try {
081            expression = XPATH.compile(String.format(format, argv));
082        } catch (Exception exception) {
083            throw new IllegalStateException(exception);
084        }
085
086        return expression;
087    }

089    /**
090     * Method to locate the POM from an {@link Element}.
091     *
092     * @param   tag             The {@link UnknownInlineTagTree}.
093     * @param   element         The {@link Element}.
094     *
095     * @return  The POM {@link File}.
096     *
097     * @throws  Exception       If the POM {@link File} cannot be found.
098     */
099    protected File getPomFileFor(UnknownInlineTagTree tag,
100                                 Element element) throws Exception {
101        var name = defaultIfBlank(getText(tag).trim(), POM_XML);
102        var parent =
103            new File(trees.getPath(element).getCompilationUnit()
104                     .getSourceFile().toUri())
105            .getParentFile();
106        var file = new File(parent, name);
107
108        while (parent != null) {
109            file = new File(parent, name);
110
111            if (file.isFile()) {
112                break;
113            } else {
114                file = null;
115            }
116
117            parent = parent.getParentFile();
118        }
119
120        if (! (file != null && file.isFile())) {
121            throw new FileNotFoundException(name);
122        }
123
124        return file;
125    }

127    /**
128     * Inline {@link Taglet} to provide a report of fields whose values are
129     * configured by the
130     * {@link.uri https://maven.apache.org/index.html Maven}
131     * {@link.uri https://maven.apache.org/plugin-developers/index.html Plugin}
132     * {@code plugin.xml}.
133     */
134    @TagletName("maven.plugin.fields")
135    @ServiceProviderFor({ Taglet.class })
136    @NoArgsConstructor @ToString
137    public static class PluginFields extends MavenTaglet {
138        private static final String PLUGIN_XML = "META-INF/maven/plugin.xml";
139
140        @Override
141        public FluentNode toNode(UnknownInlineTagTree tag, Element element) throws Throwable {
142            TypeElement type = null;
143            var argv = getText(tag).trim().split("[\\p{Space}]+", 2);
144
145            if (isNotEmpty(argv[0])) {
146                type = getTypeElementFor(element, argv[0]);
147            } else {
148                type = getEnclosingTypeElement(element);
149            }
150
151            var url = getResourceURLOf(asClass(type));
152            var protocol = Protocol.of(url);
153            Document document = null;
154
155            switch (protocol) {
156            case FILE:
157                var root =
158                    url.getPath()
159                    .replaceAll(Pattern.quote(getResourcePathOf(asClass(type))), EMPTY);
160
161                document =
162                    DocumentBuilderFactory.newInstance()
163                    .newDocumentBuilder()
164                    .parse(root + PLUGIN_XML);
165                break;
166
167            case JAR:
168                var jar = protocol.getJarFile(url);
169                var entry = jar.getEntry(PLUGIN_XML);
170
171                try (var in = jar.getInputStream(entry)) {
172                    document =
173                        DocumentBuilderFactory.newInstance()
174                        .newDocumentBuilder()
175                        .parse(in);
176                }
177                break;
178            }
179
180            if (document == null) {
181                throw new IllegalStateException("Cannot find " + PLUGIN_XML);
182            }
183
184            var mojo =
185                (Node)
186                compile("/plugin/mojos/mojo[implementation='%s']",
187                        type.getQualifiedName())
188                .evaluate(document, NODE);
189
190            return div(attr("class", "summary"),
191                       h3("Maven Plugin Parameter Summary"),
192                       table(tag, element, asClass(type), mojo,
193                             asStream((NodeList)
194                                      compile("parameters/parameter")
195                                      .evaluate(mojo, NODESET))));
196        }
197
198        private FluentNode table(UnknownInlineTagTree tag, Element element,
199                                 Class<?> type,
200                                 Node mojo, Stream<Node> parameters) {
201            return table(thead(tr(th(EMPTY), th("Field"),
202                                  th("Default"), th("Property"),
203                                  th("Required"), th("Editable"),
204                                  th("Description"))),
205                         tbody(parameters.map(t -> tr(tag, element, type, mojo, t))));
206        }
207
208        private FluentNode tr(UnknownInlineTagTree tag, Element element,
209                              Class<?> type, Node mojo, Node parameter) {
210            var tr = fragment();
211
212            try {
213                var name = compile("name").evaluate(parameter);
214                var field = FieldUtils.getField(type, name, true);
215
216                if (field != null) {
217                    tr =
218                        tr(td((! type.equals(field.getDeclaringClass()))
219                                  ? type(tag, element, field.getDeclaringClass())
220                                  : text(EMPTY)),
221                           td(declaration(tag, element, field)),
222                           td(code(compile("configuration/%s/@default-value", name)
223                                   .evaluate(mojo))),
224                           td(code(compile("configuration/%s", name)
225                                   .evaluate(mojo))),
226                           td(code(compile("required").evaluate(parameter))),
227                           td(code(compile("editable").evaluate(parameter))),
228                           td(p(compile("description").evaluate(parameter))));
229                }
230            } catch (RuntimeException exception) {
231                throw exception;
232            } catch (Exception exception) {
233                throw new IllegalStateException(exception);
234            }
235
236            return tr;
237        }
238    }

240    /**
241     * Inline {@jdk.javadoc.doclet.link Taglet} to include generated
242     * {@link.uri https://maven.apache.org/index.html Maven}
243     * {@link.uri https://maven.apache.org/plugin-developers/index.html Plugin}
244     * help documentation.
245     */
246    @TagletName("maven.plugin.help")
247    @ServiceProviderFor({ Taglet.class })
248    @NoArgsConstructor @ToString
249    public static class PluginHelp extends MavenTaglet {
250        private static final String NAME = "plugin-help.xml";
251        @CompileTimeCheck
252        private static final Pattern PATTERN =
253            Pattern.compile("META-INF/maven/(?<g>[^/]+)/(?<a>[^/]+)/"
254                            + Pattern.quote(NAME));
255
256        @Override
257        public FluentNode toNode(UnknownInlineTagTree tag, Element element) throws Throwable {
258            Class<?> type = null;
259
260            if (element instanceof PackageElement) {
261                type = asPackageInfoClass((PackageElement) element);
262            } else {
263                type = asClass(getEnclosingTypeElement(element));
264            }
265
266            var url = getResourceURLOf(type);
267            var protocol = Protocol.of(url);
268            Document document = null;
269
270            switch (protocol) {
271            case FILE:
272                var root =
273                    Paths.get(url.getPath()
274                              .replaceAll(Pattern.quote(getResourcePathOf(type)), EMPTY));
275                var path =
276                    Files.walk(root, Integer.MAX_VALUE)
277                    .filter(Files::isRegularFile)
278                    .filter(t -> PATTERN.matcher(root.relativize(t).toString()).matches())
279                    .findFirst().orElse(null);
280
281                document =
282                    DocumentBuilderFactory.newInstance()
283                    .newDocumentBuilder()
284                    .parse(path.toFile());
285                break;
286
287            case JAR:
288                try (var jar = protocol.getJarFile(url)) {
289                    var entry =
290                        jar.stream()
291                        .filter(t -> PATTERN.matcher(t.getName()).matches())
292                        .findFirst().orElse(null);
293
294                    try (var in = jar.getInputStream(entry)) {
295                        document =
296                            DocumentBuilderFactory.newInstance()
297                            .newDocumentBuilder()
298                            .parse(in);
299                    }
300                }
301                break;
302            }
303
304            if (document == null) {
305                throw new IllegalStateException("Cannot find " + NAME);
306            }
307
308            return div(attr("class", "summary"),
309                       h3(compile("/plugin/name").evaluate(document)),
310                       p(compile("/plugin/description").evaluate(document)),
311                       table(tag, element,
312                             asStream((NodeList)
313                                      compile("/plugin/mojos/mojo")
314                                      .evaluate(document, NODESET))));
315        }
316
317        private FluentNode table(UnknownInlineTagTree tag, Element element,
318                                 Stream<Node> mojos) {
319            return table(thead(tr(th("Goal"), th("Phase"), th("Description"))),
320                         tbody(mojos.map(t -> tr(tag, element, t))));
321        }
322
323        private FluentNode tr(UnknownInlineTagTree tag, Element element,
324                              Node mojo) {
325            var tr = fragment();
326
327            try {
328                tr =
329                    tr(td(a(tag, element,
330                            compile("implementation").evaluate(mojo),
331                            code(compile("goal").evaluate(mojo)))),
332                       td(code(compile("phase").evaluate(mojo))),
333                       td(p(code(compile("description").evaluate(mojo)))));
334            } catch (RuntimeException exception) {
335                throw exception;
336            } catch (Exception exception) {
337                throw new IllegalStateException(exception);
338            }
339
340            return tr;
341        }
342    }

344    /**
345     * {@link jdk.javadoc.doclet.Taglet} to provide
346     * {@link.uri https://maven.apache.org/pom.html POM} coordinates as
347     * a {@code <dependency/>} element to include this documented
348     * {@link Class} or {@link Package}.
349     *
350     * <p>For example:</p>
351     *
352     * {@pom.coordinates}
353     */
354    @TagletName("pom.coordinates")
355    @ServiceProviderFor({ Taglet.class })
356    @NoArgsConstructor @ToString
357    public static class Coordinates extends MavenTaglet {
358        @CompileTimeCheck
359        private static final Pattern PATTERN =
360            Pattern.compile("META-INF/maven/(?<g>[^/]+)/(?<a>[^/]+)/pom[.]properties");
361
362        @Override
363        public FluentNode toNode(UnknownInlineTagTree tag, Element element) throws Throwable {
364            var properties = new POMProperties();
365            Class<?> type = null;
366
367            if (element instanceof PackageElement) {
368                type = asPackageInfoClass((PackageElement) element);
369            } else {
370                type = asClass(getEnclosingTypeElement(element));
371            }
372
373            var url = getResourceURLOf(type);
374            var protocol = Protocol.of(url);
375
376            switch (protocol) {
377            case FILE:
378                var document =
379                    DocumentBuilderFactory.newInstance()
380                    .newDocumentBuilder()
381                    .parse(getPomFileFor(tag, element));
382
383                Stream.of(GROUP_ID, ARTIFACT_ID, VERSION)
384                    .forEach(t -> properties.load(t, document, "/project/"));
385                Stream.of(VERSION)
386                    .forEach(t -> properties.load(t, document, "/project/parent/"));
387                break;
388
389            case JAR:
390                try (var jar = protocol.getJarFile(url)) {
391                    var entry =
392                        jar.stream()
393                        .filter(t -> PATTERN.matcher(t.getName()).matches())
394                        .findFirst().orElse(null);
395
396                    if (entry != null) {
397                        try (var in = jar.getInputStream(entry)) {
398                            properties.load(in);
399                        }
400                    }
401                }
402                break;
403            }
404
405            return pre("xml",
406                       render(element(DEPENDENCY,
407                                      Stream.of(GROUP_ID, ARTIFACT_ID, VERSION)
408                                      .map(t -> element(t).content(properties.getProperty(t, "unknown")))),
409                              2));
410        }
411    }
412
413    private static enum Protocol {
414        FILE, JAR;
415
416        public static Protocol of(URL url) {
417            return valueOf(url.getProtocol().toUpperCase());
418        }
419
420        public JarFile getJarFile(URL url) throws IOException {
421            JarFile jar = null;
422
423            switch (this) {
424            case JAR:
425                jar = ((JarURLConnection) url.openConnection()).getJarFile();
426                break;
427
428            default:
429                throw new IllegalStateException();
430                /* break; */
431            }
432
433            return jar;
434        }
435    }
436
437    private static class POMProperties extends PropertiesImpl {
438        private static final long serialVersionUID = 1354153174028868013L;
439
440        public String load(String key, Document document, String prefix) {
441            if (document != null) {
442                computeIfAbsent(key, k -> evaluate(prefix + k, document));
443            }
444
445            return super.getProperty(key);
446        }
447
448        private String evaluate(String expression, Document document) {
449            String value = null;
450
451            try {
452                value = XPATH.evaluate(expression, document);
453            } catch (Exception exception) {
454            }
455
456            return isNotEmpty(value) ? value : null;
457        }
458    }
