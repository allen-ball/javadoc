023import ball.annotation.MatcherGroup;
024import ball.annotation.PatternRegex;
025import ball.annotation.ServiceProviderFor;
026import ball.util.PatternMatcherBean;
027import ball.xml.FluentNode;
028import com.sun.source.doctree.UnknownInlineTagTree;
029import java.io.File;
030import javax.lang.model.element.Element;
031import jdk.javadoc.doclet.Taglet;
032import lombok.NoArgsConstructor;
033import lombok.ToString;
034
035/**
036 * Inline {@link Taglet} providing links to {@link.man man(1)} pages.
037 *
038 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
039 * @version $Revision: 7215 $
040 */
041@TagletName("link.man")
042@ServiceProviderFor({ Taglet.class })
043@PatternRegex("(?is)(?<name>.+)[(](?<section>[\\p{Alnum}]+)[)]")
044@NoArgsConstructor @ToString
045public class LinkManTaglet extends AbstractInlineTaglet implements PatternMatcherBean {
046    private String name = null;
047    private String section = null;
048
049    @MatcherGroup(1)
050    protected void setName(String string) { name = string; }
051
052    @MatcherGroup(2)
053    protected void setSection(String string) { section = string; }
054
055    @Override
056    public FluentNode toNode(UnknownInlineTagTree tag, Element element) throws Throwable {
057        PatternMatcherBean.super.initialize(getText(tag).trim());
058
059        var path = new File(File.separator);
060
061        path = new File(path, "usr");
062        path = new File(path, "share");
063        path = new File(path, "man");
064        path = new File(path, "htmlman" + section);
065        path = new File(path, name + "." + section + ".html");
066
067        return a(path.toURI(), code(name + "(" + section + ")"));
068    }
069}
