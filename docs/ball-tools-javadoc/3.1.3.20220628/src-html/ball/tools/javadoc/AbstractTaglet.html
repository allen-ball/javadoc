130    @Override
131    public String toString(List<? extends DocTree> tags, Element element) {
132        Node node = null;
133
134        try {
135            node = toNode(tags, element);
136        } catch (Throwable throwable) {
137            print(WARNING, tags.get(0), element, "%s", throwable.toString());
138            node = toNode(tags.get(0), element, throwable);
139        }
140
141        return render(node);
142    }

157    private FluentNode toNode(DocTree tag, Element context, Throwable throwable) {
158        var string = "@" + getName();
159
160        if (isNotEmpty(getText(tag))) {
161            string += SPACE + getText(tag);
162        }
163
164        if (isInlineTag()) {
165            string = "{" + string + "}";
166        }
167
168        return fragment(p(b(u(string))),
169                        comment(ExceptionUtils.getStackTrace(throwable)));
170    }

185    /**
186     * Method to render a {@link Node} to a {@link String} with or without
187     * formatting or indentation.
188     *
189     * @param   node            The {@link Node}.
190     * @param   indent          The amount to indent; {@code <= 0} for no
191     *                          indentation.
192     *
193     * @return  The {@link String} representation.
194     *
195     * @throws  RuntimeException
196     *                          Instead of checked {@link Exception}.
197     */
198    protected String render(Node node, int indent) {
199        var writer = new StringWriter();
200
201        try {
202            transformer.setOutputProperty(INDENT, (indent > 0) ? YES : NO);
203            transformer.setOutputProperty(XALAN_INDENT_AMOUNT.toString(), String.valueOf(indent > 0 ? indent : 0));
204            transformer.transform(new DOMSource(node), new StreamResult(writer));
205        } catch (RuntimeException exception) {
206            throw exception;
207        } catch (Error error) {
208            throw error;
209        } catch (Exception exception) {
210            throw new RuntimeException(exception);
211        }
212
213        return writer.toString();
214    }

327    @Override
328    public URI href(DocTree tag, Element context, Object target) {
329        URI uri = null;
330
331        if (href == null) {
332            href = new Object() { }.getClass().getEnclosingMethod();
333        }
334
335        if (target != null) {
336            var parameters =
337                Stream.of(DocTree.class, Element.class, target.getClass())
338                .toArray(Class<?>[]::new);
339
340            try {
341                var method = AbstractTaglet.class.getDeclaredMethod(href.getName(), parameters);
342
343                if (Objects.equals(href, method) || (! href.getReturnType().isAssignableFrom(method.getReturnType()))) {
344                    throw new NoSuchMethodException();
345                }
346
347                var arguments =
348                    Stream.of(tag, context, target)
349                    .toArray(Object[]::new);
350
351                uri = (URI) method.invoke(this, arguments);
352            } catch (Exception exception) {
353                print(WARNING, tag, context,
354                      "No method to get href for %s",
355                      parameters[parameters.length - 1].getName());
356            }
357        }
358
359        return uri;
360    }

386        URI href = null;

415    private URI href(DocTree tag, Element context, Class<?> target, String fragment) {
416        var href = href(tag, context, asTypeElement(target), fragment);
417
418        if (href == null) {
419            href = extern(tag, context).get(target);
420
421            if (href != null) {
422                String path = target.getCanonicalName() + ".html";
423
424                path =
425                    Stream.concat(Stream.of(getComponentsOf(target.getPackage())), Stream.of(path))
426                    .collect(joining("/"));
427
428                href = href(href, path, fragment);
429            }
430        }
431
432        return href;
433    }
434
435    private URI href(DocTree tag, Element context, TypeElement target, String fragment) {
436        URI href = null;
437
438        if (target != null) {
439            var path = getCanonicalNameOf(target) + ".html";
440
441            path =
442                Stream.concat(Stream.of(getComponentsOf(elements.getPackageOf(target))), Stream.of(path))
443                .collect(joining("/"));
444
445            if (env.isIncluded(target)) {
446                int depth = getComponentsOf(elements.getPackageOf(context)).length;
447                path =
448                    Stream.concat(Stream.generate(() -> "..").limit(depth), Stream.of(path))
449                    .collect(joining("/"));
450
451                href = href(null, path, fragment);
452            } else {
453                href = extern(tag, context).get(elements.getPackageOf(target));
454
455                if (href != null) {
456                    href = href(href, path, fragment);
457                }
458            }
459        }
460
461        return href;
462    }
463
464    private URI href(URI base, String path, String fragment) {
465        URI href = null;
466
467        if (base != null) {
468            href = base.resolve(path + "?is-external=true");
469        } else {
470            href = URI.create("./" + path);
471        }
472
473        href = href.normalize();
474
475        if (fragment != null) {
476            href = href.resolve("#" + fragment);
477        }
478
479        return href;
480    }

482    @SuppressWarnings({ "unchecked" })
483    private Map<String,URI> extern(DocTree tag, Element context) {
484        if (extern == null) {
485            try {
486                /*
487                 * The Javadoc tool creates each Doclet and Taglet class in
488                 * a different ClassLoader necessitating access by
489                 * reflection look-up.
490                 */
491                extern = (Map<String,URI>) doclet.getClass().getField("extern").get(doclet);
492            } catch (Exception exception) {
493                extern = new Extern();
494                print(WARNING, tag, context,
495                      "Configure '-doclet %s' for external links", StandardDoclet.class.getCanonicalName());
496            }
497        }
498
499        return extern;
500    }

514    private String[] getComponentsOf(CharSequence sequence) {
515        var strings = new String[] { };
516
517        if (sequence != null && sequence.length() > 0) {
518            strings = sequence.toString().split(Pattern.quote("."));
519        }
520
521        return strings;

524    private String getCanonicalNameOf(TypeElement element) {
525        var string =
526            Stream.of(getComponentsOf(element))
527            .skip(getComponentsOf(elements.getPackageOf(element)).length)
528            .collect(joining("."));
529
530        return string;
531    }

533    @Override
534    public FluentNode a(DocTree tag, Element element, TypeElement target, Node node) {
535        URI href = href(tag, element, target);
536
537        if (node == null) {
538            var name = (href != null) ? target.getSimpleName() : target.getQualifiedName();
539
540            node = code(name.toString());
541        }
542
543        return a(href, node);
544    }

546    @Override
547    public FluentNode a(DocTree tag, Element element, Class<?> target, Node node) {
548        var brackets = EMPTY;
549
550        while (target.isArray()) {
551            brackets = "[]" + brackets;
552            target = target.getComponentType();
553        }
554
555        var href = href(tag, element, target);
556
557        if (node == null) {
558            var name = target.getCanonicalName();
559
560            if (href != null) {
561                name = target.getSimpleName();
562
563                var enclosing = target.getEnclosingClass();
564
565                while (enclosing != null) {
566                    name = enclosing.getSimpleName() + "." + name;
567                    enclosing = enclosing.getEnclosingClass();
568                }
569            }
570
571            node = code(name + brackets);
572        }
573
574        return a(href, node);
575    }

577    @Override
578    public FluentNode a(DocTree tag, Element element, Member member, Node node) {
579        if (node == null) {
580            node = code(member.getName());
581        }
582
583        return a(href(tag, element, member), node);

586    @Override
587    public FluentNode a(DocTree tag, Element element, String name, Node node) {
588        TypeElement target = null;
589
590        if (node == null) {
591            if (target != null) {
592                node = code(target.getSimpleName().toString());
593            } else {
594                node = code(name);
595            }
596        }
597
598        return a(tag, element, target, node);
599    }

613    @NoArgsConstructor @ToString
614    private class GetTextVisitor extends SimpleDocTreeVisitor<String,Void> {
615        @Override
616        public String visitUnknownInlineTag(UnknownInlineTagTree node, Void p) {
617            var text =
618                node.getContent().stream()
619                .map(t -> t.accept(this, p))
620                .filter(Objects::nonNull)
621                .collect(joining(SPACE, EMPTY, EMPTY));
622
623            return text.trim();
624        }
625
626        @Override
627        public String visitText(TextTree node, Void p) {
628            return node.getBody();
629        }

632        protected String defaultAction(DocTree node, Void p) { return EMPTY; }
