021import ball.annotation.ServiceProviderFor;
022import ball.swing.table.ListTableModel;
023import ball.swing.table.MapTableModel;
024import ball.xml.FluentNode;
025import com.sun.source.doctree.UnknownInlineTagTree;
026import java.lang.reflect.Field;
027import java.util.Collection;
028import java.util.Map;
029import java.util.regex.Pattern;
030import javax.lang.model.element.Element;
031import javax.lang.model.element.PackageElement;
032import javax.lang.model.element.TypeElement;
033import javax.lang.model.element.VariableElement;
034import jdk.javadoc.doclet.Taglet;
035import lombok.NoArgsConstructor;
036import lombok.ToString;
037import org.apache.commons.io.IOUtils;
038
039import static java.util.stream.Collectors.toList;
040import static javax.lang.model.util.ElementFilter.fieldsIn;
041import static org.apache.commons.lang3.StringUtils.isNotEmpty;
042
043/**
044 * Inline {@link Taglet} to include a static {@link Class}
045 * {@link java.lang.reflect.Field} or resource in the Javadoc output.
046 *
047 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
048 */
049@TagletName("include")
050@ServiceProviderFor({ Taglet.class })
051@NoArgsConstructor @ToString
052public class IncludeTaglet extends AbstractInlineTaglet {
053    @Override
054    public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
055        FluentNode node = null;
056        var argv = getText(tag).trim().split("[\\p{Space}]+", 2);
057
058        if (isNotEmpty(argv[0])) {
059            var target = argv[0].trim().split(Pattern.quote("#"), 2);
060
061            if (target.length > 1) {
062                var type =
063                    isNotEmpty(target[0])
064                        ? getTypeElementFor(context, target[0])
065                        : getEnclosingTypeElement(context);
066                var field =
067                    fieldsIn(type.getEnclosedElements()).stream()
068                    .filter(t -> t.getSimpleName().contentEquals(target[1]))
069                    .findFirst().get();
070
071                node = field(tag, context, field);
072            } else {
073                var type =
074                    (context instanceof PackageElement)
075                        ? asPackageInfoClass((PackageElement) context)
076                        : asClass(getEnclosingTypeElement(context));
077
078                node = resource(tag, context, type, target[0]);
079            }
080        } else {
081            node = field(tag, context, (VariableElement) context);
082        }
083
084        return node;
085    }
086
087    private FluentNode field(UnknownInlineTagTree tag, Element context, VariableElement element) throws Exception {
088        FluentNode node = null;
089        var type = asClass((TypeElement) element.getEnclosingElement());
090        var field = type.getDeclaredField(element.getSimpleName().toString());
091
092        field.setAccessible(true);
093
094        var value = field.get(null);
095
096        if (value instanceof Collection<?>) {
097            node =
098                table(tag, context,
099                      new ListTableModel(((Collection<?>) value).stream().collect(toList()), "Element"));
100        } else if (value instanceof Map<?,?>) {
101            node = table(tag, context, new MapTableModel((Map<?,?>) value, "Key", "Value"));
102        } else {
103            node = pre(String.valueOf(value));
104        }
105
106        return div(attr("class", "block"), node);
107    }
108
109    private FluentNode resource(UnknownInlineTagTree tag, Element context, Class<?> type, String name) throws Exception {
110        String string = null;
111
112        if (type == null) {
113            type = getClass();
114        }
115
116        try (var in = type.getResourceAsStream(name)) {
117            string = IOUtils.toString(in, "UTF-8");
118        }
119
120        return pre(string);
121    }
122}
