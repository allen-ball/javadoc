023import ball.annotation.ServiceProviderFor;
024import ball.xml.FluentNode;
025import com.sun.source.doctree.UnknownInlineTagTree;
026import java.net.URI;
027import java.util.LinkedHashMap;
028import javax.lang.model.element.Element;
029import jdk.javadoc.doclet.Taglet;
030import lombok.NoArgsConstructor;
031import lombok.ToString;
032
033/**
034 * Inline {@link jdk.javadoc.doclet.Taglet} to provide external links.
035 *
036 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
037 * @version $Revision: 7354 $
038 */
039@TagletName("link.uri")
040@ServiceProviderFor({ Taglet.class })
041@NoArgsConstructor @ToString
042public class LinkURITaglet extends AbstractInlineTaglet {
043    private static final String SPACES = "[\\p{Space}]+";
044
045    @Override
046    public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
047        var text = getText(tag).trim();
048        var argv = text.split(SPACES, 2);
049        var href = new URI(argv[0]);
050
051        text = (argv.length > 1) ? argv[1] : null;
052
053        var map = new LinkedHashMap<String,String>();
054
055        if (text != null) {
056            for (;;) {
057                argv = text.split(SPACES, 2);
058
059                var nvp = argv[0].split("=", 2);
060
061                if (argv.length > 1 && nvp.length > 1) {
062                    map.put(nvp[0], nvp[1]);
063                    text = argv[1];
064                } else {
065                    break;
066                }
067            }
068        }
069
070        return a(href, text)
071                   .add(map.entrySet()
072                        .stream()
073                        .map(t -> attr(t.getKey(), t.getValue())));
074    }
075}
