021import ball.lang.reflect.InterceptingInvocationHandler;
022import java.io.File;
023import java.net.URI;
024import java.util.List;
025import java.util.Locale;
026import java.util.Map;
027import java.util.Set;
028import java.util.TreeMap;
029import javax.tools.Diagnostic;
030import jdk.javadoc.doclet.DocletEnvironment;
031import jdk.javadoc.doclet.Reporter;
032import lombok.NoArgsConstructor;
033import lombok.ToString;
034
035import static java.util.stream.Collectors.toSet;
036import static javax.tools.Diagnostic.Kind.WARNING;
037
038/**
039 * {@link jdk.javadoc.doclet.StandardDoclet} subclass which provides
040 * additional services to {@link AbstractTaglet} implementations.
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 */
044@NoArgsConstructor @ToString
045public class StandardDoclet extends jdk.javadoc.doclet.StandardDoclet {
046    private final Map<URI,URI> links = new TreeMap<>();
047    private Locale locale = null;
048    private Reporter reporter = null;
049
050    /**
051     * {@link Extern} {@link Map} for {@link AbstractTaglet} instances.
052     */
053    public final Map<String,URI> extern = new Extern();
054
055    @Override
056    public void init(Locale locale, Reporter reporter) {
057        this.locale = locale;
058        this.reporter = reporter;
059
060        super.init(locale, reporter);
061    }
062
063    @Override
064    public Set<Option> getSupportedOptions() {
065        var set =
066            super.getSupportedOptions().stream()
067            .map(OptionWrapper::new)
068            .map(t -> (Option) t.newProxyInstance(Option.class))
069            .collect(toSet());
070
071        return set;
072    }
073
074    @Override
075    public boolean run(DocletEnvironment env) {
076        for (var key : links.keySet()) {
077            var value = links.get(key);
078
079            try {
080                ((Extern) extern).link(key, (value != null) ? value : key);
081            } catch (Exception exception) {
082                print(WARNING, "%s", exception.getMessage());
083            }
084        }
085
086        return super.run(env);
087    }
088
089    /**
090     * Method to print a diagnostic message.
091     *
092     * @param   kind            The {@link javax.tools.Diagnostic.Kind}.
093     * @param   format          The message format {@link String}.
094     * @param   argv            Optional arguments to the message format
095     *                          {@link String}.
096     *
097     * @see Reporter#print(Diagnostic.Kind,String)
098     */
099    protected void print(Diagnostic.Kind kind, String format, Object... argv) {
100        reporter.print(kind, String.format(format, argv));
101    }
102
103    private class OptionWrapper extends InterceptingInvocationHandler<Option> {
104        public OptionWrapper(Option option) { super(option); }
105
106        public void process(String option, List<String> argv) {
107            if (getTarget().getNames().contains("-link")) {
108                links.put(asURI(argv.get(0)), null);
109            } else if (getTarget().getNames().contains("-linkoffline")) {
110                links.put(asURI(argv.get(0)), new File(argv.get(1)).toURI());
111            }
112        }
113
114        private URI asURI(String string) {
115            return URI.create(string + "/").normalize();
116        }
117    }
118}
