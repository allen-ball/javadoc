021import ball.annotation.ServiceProviderFor;
022import ball.beans.PropertyDescriptorsTableModel;
023import ball.xml.FluentNode;
024import com.sun.source.doctree.UnknownInlineTagTree;
025import javax.lang.model.element.Element;
026import javax.lang.model.element.TypeElement;
027import jdk.javadoc.doclet.Taglet;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030
031import static org.apache.commons.lang3.StringUtils.isNotEmpty;
032
033/**
034 * Inline {@link jdk.javadoc.doclet.Taglet} to provide a table of bean
035 * properties.
036 *
037 * For example:
038 *
039 * {@bean.info}
040 *
041 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
042 */
043@TagletName("bean.info")
044@ServiceProviderFor({ Taglet.class })
045@NoArgsConstructor @ToString
046public class BeanInfoTaglet extends AbstractInlineTaglet {
047    @Override
048    public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
049        TypeElement type = null;
050        var name = getText(tag);
051
052        if (isNotEmpty(name)) {
053            type = getTypeElementFor(context, name);
054        } else {
055            type = getEnclosingTypeElement(context);
056        }
057
058        var model = new PropertyDescriptorsTableModel(getBeanInfo(asClass(type)).getPropertyDescriptors());
059
060        return div(attr("class", "summary"),
061                   h3("Bean Property Summary"),
062                   table(tag, context, model));
063    }
064}
