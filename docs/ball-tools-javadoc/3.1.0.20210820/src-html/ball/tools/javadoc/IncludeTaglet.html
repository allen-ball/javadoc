023import ball.annotation.ServiceProviderFor;
024import ball.swing.table.ListTableModel;
025import ball.swing.table.MapTableModel;
026import ball.xml.FluentNode;
027import com.sun.source.doctree.UnknownInlineTagTree;
028import java.lang.reflect.Field;
029import java.util.Collection;
030import java.util.Map;
031import java.util.regex.Pattern;
032import javax.lang.model.element.Element;
033import javax.lang.model.element.PackageElement;
034import javax.lang.model.element.TypeElement;
035import javax.lang.model.element.VariableElement;
036import jdk.javadoc.doclet.Taglet;
037import lombok.NoArgsConstructor;
038import lombok.ToString;
039import org.apache.commons.io.IOUtils;
040
041import static java.util.stream.Collectors.toList;
042import static javax.lang.model.util.ElementFilter.fieldsIn;
043import static org.apache.commons.lang3.StringUtils.isNotEmpty;
044
045/**
046 * Inline {@link Taglet} to include a static {@link Class}
047 * {@link java.lang.reflect.Field} or resource in the Javadoc output.
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 * @version $Revision: 7393 $
051 */
052@TagletName("include")
053@ServiceProviderFor({ Taglet.class })
054@NoArgsConstructor @ToString
055public class IncludeTaglet extends AbstractInlineTaglet {
056    @Override
057    public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
058        FluentNode node = null;
059        var argv = getText(tag).trim().split("[\\p{Space}]+", 2);
060
061        if (isNotEmpty(argv[0])) {
062            var target = argv[0].trim().split(Pattern.quote("#"), 2);
063
064            if (target.length > 1) {
065                var type =
066                    isNotEmpty(target[0])
067                        ? getTypeElementFor(context, target[0])
068                        : getEnclosingTypeElement(context);
069                var field =
070                    fieldsIn(type.getEnclosedElements())
071                    .stream()
072                    .filter(t -> t.getSimpleName().contentEquals(target[1]))
073                    .findFirst().get();
074
075                node = field(tag, context, field);
076            } else {
077                var type =
078                    (context instanceof PackageElement)
079                        ? asPackageInfoClass((PackageElement) context)
080                        : asClass(getEnclosingTypeElement(context));
081
082                node = resource(tag, context, type, target[0]);
083            }
084        } else {
085            node = field(tag, context, (VariableElement) context);
086        }
087
088        return node;
089    }
090
091    private FluentNode field(UnknownInlineTagTree tag, Element context,
092                             VariableElement element) throws Exception {
093        FluentNode node = null;
094        var type = asClass((TypeElement) element.getEnclosingElement());
095        var field = type.getDeclaredField(element.getSimpleName().toString());
096
097        field.setAccessible(true);
098
099        var value = field.get(null);
100
101        if (value instanceof Collection<?>) {
102            node =
103                table(tag, context,
104                      new ListTableModel(((Collection<?>) value)
105                                         .stream()
106                                         .collect(toList()),
107                                         "Element"));
108        } else if (value instanceof Map<?,?>) {
109            node =
110                table(tag, context,
111                      new MapTableModel((Map<?,?>) value, "Key", "Value"));
112        } else {
113            node = pre(String.valueOf(value));
114        }
115
116        return div(attr("class", "block"), node);
117    }
118
119    private FluentNode resource(UnknownInlineTagTree tag, Element context,
120                                Class<?> type, String name) throws Exception {
121        String string = null;
122
123        if (type == null) {
124            type = getClass();
125        }
126
127        try (var in = type.getResourceAsStream(name)) {
128            string = IOUtils.toString(in, "UTF-8");
129        }
130
131        return pre(string);
132    }
133}
