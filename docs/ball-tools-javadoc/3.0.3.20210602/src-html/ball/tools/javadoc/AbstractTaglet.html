136    @Override
137    public String toString(List<? extends DocTree> tags, Element element) {
138        Node node = null;
139
140        try {
141            node = toNode(tags, element);
142        } catch (Throwable throwable) {
143            print(WARNING, tags.get(0), element, "%s", throwable.toString());
144            node = toNode(tags.get(0), element, throwable);
145        }
146
147        return render(node);
148    }

163    private FluentNode toNode(DocTree tag, Element context, Throwable throwable) {
164        var string = "@" + getName();
165
166        if (isNotEmpty(getText(tag))) {
167            string += SPACE + getText(tag);
168        }
169
170        if (isInlineTag()) {
171            string = "{" + string + "}";
172        }
173
174        return fragment(p(b(u(string))),
175                        comment(ExceptionUtils.getStackTrace(throwable)));
176    }

191    /**
192     * Method to render a {@link Node} to a {@link String} with or without
193     * formatting or indentation.
194     *
195     * @param   node            The {@link Node}.
196     * @param   indent          The amount to indent; {@code <= 0} for no
197     *                          indentation.
198     *
199     * @return  The {@link String} representation.
200     *
201     * @throws  RuntimeException
202     *                          Instead of checked {@link Exception}.
203     */
204    protected String render(Node node, int indent) {
205        var writer = new StringWriter();
206
207        try {
208            transformer
209                .setOutputProperty(INDENT, (indent > 0) ? YES : NO);
210            transformer
211                .setOutputProperty(XALAN_INDENT_AMOUNT.toString(),
212                                   String.valueOf(indent > 0 ? indent : 0));
213            transformer
214                .transform(new DOMSource(node),
215                           new StreamResult(writer));
216        } catch (RuntimeException exception) {
217            throw exception;
218        } catch (Error error) {
219            throw error;
220        } catch (Exception exception) {
221            throw new RuntimeException(exception);
222        }
223
224        return writer.toString();
225    }

344    @Override
345    public URI href(DocTree tag, Element context, Object target) {
346        URI uri = null;
347
348        if (href == null) {
349            href = new Object() { }.getClass().getEnclosingMethod();
350        }
351
352        if (target != null) {
353            var parameters =
354                Stream.of(DocTree.class, Element.class, target.getClass())
355                .toArray(Class<?>[]::new);
356
357            try {
358                var method =
359                    AbstractTaglet.class
360                    .getDeclaredMethod(href.getName(), parameters);
361
362                if (Objects.equals(href, method)
363                    || (! href.getReturnType().isAssignableFrom(method.getReturnType()))) {
364                    throw new NoSuchMethodException();
365                }
366
367                var arguments =
368                    Stream.of(tag, context, target)
369                    .toArray(Object[]::new);
370
371                uri = (URI) method.invoke(this, arguments);
372            } catch (Exception exception) {
373                print(WARNING, tag, context,
374                      "No method to get href for %s",
375                      parameters[parameters.length - 1].getName());
376            }
377        }
378
379        return uri;
380    }

438    private URI href(DocTree tag, Element context, Class<?> target, String fragment) {
439        var href = href(tag, context, asTypeElement(target), fragment);
440
441        if (href == null) {
442            href = extern(tag, context).get(target);
443
444            if (href != null) {
445                String path = target.getCanonicalName() + ".html";
446
447                path =
448                    Stream.concat(Stream.of(getComponentsOf(target.getPackage())),
449                                  Stream.of(path))
450                    .collect(joining("/"));
451
452                href = href(href, path, fragment);
453            }
454        }
455
456        return href;
457    }

459    private URI href(DocTree tag, Element context, TypeElement target, String fragment) {
460        URI href = null;
461
462        if (target != null) {
463            var path = getCanonicalNameOf(target) + ".html";
464
465            path =
466                Stream.concat(Stream.of(getComponentsOf(elements.getPackageOf(target))),
467                              Stream.of(path))
468                .collect(joining("/"));
469
470            if (env.isIncluded(target)) {
471                int depth = getComponentsOf(elements.getPackageOf(context)).length;
472                path =
473                    Stream.concat(Stream.generate(() -> "..").limit(depth),
474                                  Stream.of(path))
475                    .collect(joining("/"));
476
477                href = href(null, path, fragment);
478            } else {
479                href = extern(tag, context).get(elements.getPackageOf(target));
480
481                if (href != null) {
482                    href = href(href, path, fragment);
483                }
484            }
485        }
486
487        return href;
488    }

490    private URI href(URI base, String path, String fragment) {
491        URI href = null;
492
493        if (base != null) {
494            href = base.resolve(path + "?is-external=true");
495        } else {
496            href = URI.create("./" + path);
497        }
498
499        href = href.normalize();
500
501        if (fragment != null) {
502            href = href.resolve("#" + fragment);
503        }
504
505        return href;
506    }

508    @SuppressWarnings({ "unchecked" })
509    private Map<String,URI> extern(DocTree tag, Element context) {
510        if (extern == null) {
511            try {
512                /*
513                 * The Javadoc tool creates each Doclet and Taglet class in
514                 * a different ClassLoader necessitating access by
515                 * reflection look-up.
516                 */
517                extern =
518                    (Map<String,URI>)
519                    doclet.getClass().getField("extern").get(doclet);
520            } catch (Exception exception) {
521                extern = new Extern();
522                print(WARNING, tag, context,
523                      "Configure '-doclet %s' for external links",
524                      StandardDoclet.class.getCanonicalName());
525            }
526        }
527
528        return extern;
529    }

553    private String getCanonicalNameOf(TypeElement element) {
554        var string =
555            Stream.of(getComponentsOf(element))
556            .skip(getComponentsOf(elements.getPackageOf(element)).length)
557            .collect(joining("."));
558
559        return string;

562    @Override
563    public FluentNode a(DocTree tag, Element element, TypeElement target, Node node) {
564        URI href = href(tag, element, target);
565
566        if (node == null) {
567            var name =
568                (href != null) ? target.getSimpleName() : target.getQualifiedName();
569
570            node = code(name.toString());
571        }
572
573        return a(href, node);
574    }

576    @Override
577    public FluentNode a(DocTree tag, Element element, Class<?> target, Node node) {
578        var brackets = EMPTY;
579
580        while (target.isArray()) {
581            brackets = "[]" + brackets;
582            target = target.getComponentType();
583        }
584
585        var href = href(tag, element, target);
586
587        if (node == null) {
588            var name = target.getCanonicalName();
589
590            if (href != null) {
591                name = target.getSimpleName();
592
593                var enclosing = target.getEnclosingClass();
594
595                while (enclosing != null) {
596                    name = enclosing.getSimpleName() + "." + name;
597                    enclosing = enclosing.getEnclosingClass();
598                }
599            }
600
601            node = code(name + brackets);
602        }
603
604        return a(href, node);
605    }

607    @Override
608    public FluentNode a(DocTree tag, Element element, Member member, Node node) {
609        if (node == null) {
610            node = code(member.getName());
611        }
612
613        return a(href(tag, element, member), node);

616    @Override
617    public FluentNode a(DocTree tag, Element element, String name, Node node) {
618        TypeElement target = null;
619
620        if (node == null) {
621            if (target != null) {
622                node = code(target.getSimpleName().toString());
623            } else {
624                node = code(name);
625            }
626        }
627
628        return a(tag, element, target, node);

643    @NoArgsConstructor @ToString
644    private class GetTextVisitor extends SimpleDocTreeVisitor<String,Void> {
645        @Override
646        public String visitUnknownInlineTag(UnknownInlineTagTree node, Void p) {
647            var text =
648                node.getContent()
649                .stream()
650                .map(t -> t.accept(this, p))
651                .filter(Objects::nonNull)
652                .collect(joining(SPACE, EMPTY, EMPTY));
653
654            return text.trim();
655        }

657        @Override
658        public String visitText(TextTree node, Void p) {
659            return node.getBody();
660        }
