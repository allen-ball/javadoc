023import java.io.IOException;
024import java.net.URI;
025import java.net.http.HttpClient;
026import java.net.http.HttpRequest;
027import java.net.http.HttpResponse;
028import java.nio.file.Files;
029import java.nio.file.Paths;
030import java.util.List;
031import java.util.TreeMap;
032import javax.lang.model.element.PackageElement;
033import lombok.NoArgsConstructor;
034
035import static java.net.http.HttpResponse.BodyHandlers;
036import static java.nio.charset.StandardCharsets.UTF_8;
037import static java.util.stream.Collectors.toList;
038
039/**
040 * package::URI {@link java.util.Map} for externally-linked Javadoc.
041 *
042 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
043 * @version $Revision: 7215 $
044 */
045@NoArgsConstructor
046public class Extern extends TreeMap<String,URI> {
047    private static final long serialVersionUID = 8076330291756669682L;
048
049    private static final String ELEMENT_LIST = "element-list";
050    private static final String PACKAGE_LIST = "package-list";
051
052    private static final String MODULE_PREFIX = "module:";
053
054    /**
055     * Method to configure an external Javadoc document for linking.
056     * ("{code -link}" option).
057     *
058     * @param   javadoc         The {@link URI} of the external Javadoc.
059     */
060    public void link(URI javadoc) throws InterruptedException,
061                                         IOException {
062        link(javadoc, javadoc);
063    }
064
065    /**
066     * Method to configure an external Javadoc document for linking
067     * ("{code -linkoffline}" option).
068     *
069     * @param   javadoc         The {@link URI} of the external Javadoc.
070     * @param   packageList     The {@link URI} of the folder containing the
071     *                          package list.
072     */
073    public void link(URI javadoc, URI packageList) throws InterruptedException,
074                                                          IOException {
075        List<String> list = null;
076
077        for (String name : List.of(ELEMENT_LIST, PACKAGE_LIST)) {
078            var uri = packageList.resolve(name);
079
080            try {
081                if (uri.getScheme().equals("file")) {
082                    list = Files.lines(Paths.get(uri), UTF_8).collect(toList());
083                    break;
084                } else {
085                    var client = HttpClient.newHttpClient();
086                    var request = HttpRequest.newBuilder().uri(uri).build();
087                    var response = client.send(request, BodyHandlers.ofLines());
088
089                    if (response.statusCode() == 200) {
090                        list = response.body().collect(toList());
091
092                        if (! list.isEmpty()) {
093                            break;
094                        } else {
095                            throw new IOException("No packages found in " + uri);
096                        }
097                    } else {
098                        throw new IOException(String.valueOf(uri));
099                    }
100                }
101            } catch (IOException exception) {
102                continue;
103            }
104        }
105
106        if (! (list == null || list.isEmpty())) {
107            var value = javadoc;
108
109            for (var key : list) {
110                if (key.startsWith(MODULE_PREFIX)) {
111                    value =
112                        javadoc
113                        .resolve(key.substring(MODULE_PREFIX.length()) + "/")
114                        .normalize();
115                } else {
116                    putIfAbsent(key, value);
117                }
118            }
119        } else {
120            throw new IOException("Cannot get package list from " + packageList);
121        }
122    }
123
124    @Override
125    public URI get(Object key) {
126        URI value = null;
127
128        if (key instanceof Class<?>) {
129            value = get(((Class<?>) key).getPackage());
130        } else if (key instanceof Package) {
131            value = get(((Package) key).getName());
132        } else if (key instanceof PackageElement) {
133            var element = (PackageElement) key;
134
135            if (! element.isUnnamed()) {
136                value = get(element.getQualifiedName().toString());
137            }
138        } else {
139            value = (key != null) ? super.get(key) : null;
140        }
141
142        return value;
143    }
144}
