021import ball.annotation.ServiceProviderFor;
022import ball.util.ant.taskdefs.AntTask;
023import ball.xml.FluentNode;
024import com.sun.source.doctree.UnknownInlineTagTree;
025import java.util.AbstractMap.SimpleEntry;
026import java.util.HashSet;
027import java.util.Map;
028import java.util.Set;
029import java.util.regex.Pattern;
030import java.util.stream.Stream;
031import javax.lang.model.element.Element;
032import javax.lang.model.element.TypeElement;
033import jdk.javadoc.doclet.Taglet;
034import lombok.NoArgsConstructor;
035import lombok.ToString;
036import org.apache.tools.ant.ComponentHelper;
037import org.apache.tools.ant.IntrospectionHelper;
038import org.apache.tools.ant.Project;
039import org.apache.tools.ant.Task;
040import org.apache.tools.ant.taskdefs.Antlib;
041import org.w3c.dom.Node;
042
043import static java.util.stream.Collectors.joining;
044import static org.apache.commons.lang3.StringUtils.EMPTY;
045import static org.apache.commons.lang3.StringUtils.isNotEmpty;
046import static org.apache.tools.ant.MagicNames.ANTLIB_PREFIX;
047
048/**
049 * Inline {@link Taglet} to document {@link.uri http://ant.apache.org/ Ant}
050 * {@link Task}s.
051 *
052 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
053 */
054@TagletName("ant.task")
055@ServiceProviderFor({ Taglet.class })
056@NoArgsConstructor @ToString
057public class AntTaskTaglet extends AbstractInlineTaglet {
058    private static final String NO = "no";
059    private static final String YES = "yes";
060
061    private static final String INDENTATION = "  ";
062
063    private static final String DOCUMENTED = "DOCUMENTED";
064
065    @Override
066    public FluentNode toNode(UnknownInlineTagTree tag, Element context) throws Throwable {
067        FluentNode node = null;
068        TypeElement type = null;
069        var name = getText(tag).trim();
070
071        if (isNotEmpty(name)) {
072            type = getTypeElementFor(context, name);
073        } else {
074            type = getEnclosingTypeElement(context);
075        }
076
077        if (! isAssignableTo(Task.class).test(type)) {
078            throw new IllegalArgumentException(type.getQualifiedName()
079                                               + " is not a subclass of " + Task.class.getCanonicalName());
080        }
081
082        var template =
083            render(template(tag, context, asClass(type)), INDENTATION.length())
084            .replaceAll(Pattern.quote(DOCUMENTED + "=\"\""), "...");
085
086        return div(attr("class", "block"), pre("xml", template));
087    }
088
089    private FluentNode template(UnknownInlineTagTree tag, Element context, Class<?> type) {
090        String name = null;
091
092        if (name == null) {
093            var project = new Project();
094            var pkg = type.getPackage().getName();
095
096            while (pkg != null) {
097                var url = type.getResource("/" + String.join("/", pkg.split(Pattern.quote("."))) + "/antlib.xml");
098
099                if (url != null) {
100                    try {
101                        Antlib.createAntlib(project, url, ANTLIB_PREFIX + pkg)
102                            .execute();
103                        break;
104                    } catch (Exception exception) {
105                    }
106                }
107
108                int index = pkg.lastIndexOf(".");
109
110                if (! (index < 0)) {
111                    pkg = pkg.substring(0, index);
112                } else {
113                    pkg = null;
114                }
115            }
116
117            var helper = ComponentHelper.getComponentHelper(project);
118
119            name =
120                helper.getTaskDefinitions().entrySet().stream()
121                .filter(t -> t.getValue().equals(type))
122                .map(t -> t.getKey())
123                .findFirst().orElse(null);
124        }
125
126        if (name == null) {
127            var annotation = type.getAnnotation(AntTask.class);
128
129            name = (annotation != null) ? annotation.value() : null;
130        }
131
132        if (name == null) {
133            name = type.getSimpleName();
134        }
135
136        return type(0, new HashSet<>(), tag, context, new SimpleEntry<>(name, type));
137    }
138
139    private FluentNode type(int depth, Set<Map.Entry<?,?>> set, UnknownInlineTagTree tag, Element context, Map.Entry<String,Class<?>> entry) {
140        var helper = IntrospectionHelper.getHelper(entry.getValue());
141        var node = element(entry.getKey());
142
143        if (set.add(entry) && (! entry.getValue().getName().startsWith(Task.class.getPackage().getName()))) {
144            node
145                .add(attributes(tag, context, helper))
146                .add(content(depth + 1, set, tag, context, helper));
147
148            if (helper.supportsCharacters()) {
149                var content = "... text ...";
150
151                if (node.hasChildNodes()) {
152                    content =
153                        Stream.generate(() -> INDENTATION)
154                        .limit(depth + 1)
155                        .collect(joining(EMPTY, "\n", content + "\n"));
156                }
157
158                node.add(text(content));
159            }
160        } else {
161            node.add(attr(DOCUMENTED));
162        }
163
164        return node;
165    }
166
167    private Node[] attributes(UnknownInlineTagTree tag, Element context, IntrospectionHelper helper) {
168        var array =
169            helper.getAttributeMap().entrySet().stream()
170            .map(t -> attr(t.getKey(), t.getValue().getSimpleName()))
171            .toArray(Node[]::new);
172
173        return array;
174    }
175
176    private FluentNode content(int depth, Set<Map.Entry<?,?>> set, UnknownInlineTagTree tag, Element context, IntrospectionHelper helper) {
177        return fragment(helper.getNestedElementMap().entrySet().stream()
178                        .map(t -> type(depth, set, tag, context, t)));
179    }
180}
