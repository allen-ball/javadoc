090    /**
091     * {@code <a href="}{@link VariableElement type}{@code ">}{@link Node node}{@code </a>}
092     *
093     * @param   tag             The context {@link DocTree}.
094     * @param   context         The context {@link Element}.
095     * @param   target          The target {@link VariableElement}.
096     * @param   node            The child {@link Node} (may be
097     *                          {@code null}).
098     *
099     * @return  {@link org.w3c.dom.Element}
100     */
101    public default FluentNode a(DocTree tag, Element context, VariableElement target, Node node) {
102        var href = href(tag, context, (TypeElement) target.getEnclosingElement());
103        var name = target.getSimpleName().toString();
104
105        if (href != null) {
106            href = href.resolve("#" + name);
107        }
108
109        if (node == null) {
110            node = code(name);
111        }
112
113        return a(href, node);
114    }

280    /**
281     * Method to generate a {@link Method} declaration with javadoc
282     * hyperlinks.
283     *
284     * @param   tag             The context {@link DocTree}.
285     * @param   context         The context {@link Element}.
286     * @param   method          The target {@link Method}.
287     *
288     * @return  {@link org.w3c.dom.DocumentFragment}
289     */
290    public default FluentNode declaration(DocTree tag, Element context, Method method) {
291        var node =
292            fragment(modifiers(method.getModifiers()),
293                     type(tag, context, method.getGenericReturnType()),
294                     code(SPACE),
295                     a(tag, context, method, (String) null));
296        var parameters = method.getParameters();
297
298        node.add(code("("));
299
300        for (int i = 0; i < parameters.length; i += 1) {
301            if (i > 0) {
302                node.add(code(", "));
303            }
304
305            node.add(declaration(tag, context, parameters[i]));
306        }
307
308        node.add(code(")"));
309
310        return node;
311    }

348    public default FluentNode annotation(DocTree tag, Element context, Annotation annotation) {
349        var type = annotation.annotationType();
350        var string =
351            String.valueOf(annotation)
352            .replace(type.getCanonicalName(), type.getSimpleName())
353            .replaceAll("[(][)]$", "");
354
355        return fragment().add(a(href(tag, context, type), code(string)));
356    }

409     * Method to generate types for {@code declaration()} methods.
410     *
411     * @param   tag             The context {@link DocTree}.
412     * @param   context         The context {@link Element}.
413     * @param   type            The target {@link DeclaredType}.
414     *
415     * @return  {@link org.w3c.dom.DocumentFragment}
416     */
417    public default FluentNode type(DocTree tag, Element context, DeclaredType type) {
418        var node = fragment(a(tag, context, (TypeElement) type.asElement(), null));
419        var types = type.getTypeArguments();
420
421        if (! types.isEmpty()) {
422            node = node.add(code("<"));
423
424            for (int i = 0, n = types.size(); i < n; i += 1) {
425                if (i > 0) {
426                    node.add(code(","));
427                }
428
429                node.add(type(tag, context, types.get(i)));
430            }
431
432            node.add(code(">"));
433        }
434
435        return node;
436    }

452     * Method to generate types for {@code declaration()} methods.
453     *
454     * @param   tag             The context {@link DocTree}.
455     * @param   context         The context {@link Element}.
456     * @param   type            The target {@link Type}.
457     *
458     * @return  {@link org.w3c.dom.DocumentFragment}
459     */
460    public default FluentNode type(DocTree tag, Element context, Type type) {
461        FluentNode node = null;
462
463        if (type instanceof ParameterizedType) {
464            node = fragment(type(tag, context, ((ParameterizedType) type).getRawType()));
465
466            var types = ((ParameterizedType) type).getActualTypeArguments();
467
468            node = node.add(code("<"));
469
470            for (int i = 0; i < types.length; i += 1) {
471                if (i > 0) {
472                    node.add(code(","));
473                }
474
475                node.add(type(tag, context, types[i]));
476            }
477
478            node.add(code(">"));
479        } else if (type instanceof Class<?>) {
480            node = a(tag, context, (Class<?>) type);
481        } else {
482            node = code(type.getTypeName());
483        }
484
485        return node;
486    }

505    /**
506     * {@code <table>}{@link TableModel model}{@code </table>}
507     *
508     * @param   tag             The context {@link DocTree}.
509     * @param   context         The context {@link Element}.
510     * @param   model           The {@link TableModel} to use to create the
511     *                          new table {@link org.w3c.dom.Element}.
512     * @param   nodes           The {@link Node}s to append to the newly
513     *                          created
514     *                          {@link org.w3c.dom.Element}.
515     *
516     * @return  {@link org.w3c.dom.Element}
517     */
518    public default FluentNode table(DocTree tag, Element context, TableModel model, Node... nodes) {
519        var table = table();
520        var names =
521            IntStream.range(0, model.getColumnCount())
522            .boxed()
523            .map(model::getColumnName)
524            .toArray(String[]::new);
525
526        if (! isAllBlank(names)) {
527            table.add(thead(tr(Stream.of(names).map(this::th))));
528        }
529
530        table
531            .add(tbody(IntStream.range(0, model.getRowCount())
532                       .boxed()
533                       .map(y -> tr(IntStream.range(0, names.length)
534                                    .boxed()
535                                    .map(x -> td(toHTML(tag, context, model.getValueAt(y, x))))))));
536
537        return table.add(nodes);
538    }

540    /**
541     * Method to get a Javadoc HTML representation of an {@link Object}.
542     *
543     * @param   tag             The context {@link DocTree}.
544     * @param   context         The context {@link Element}.
545     * @param   object          The target {@link Object}.
546     *
547     * @return  {@link org.w3c.dom.Node}
548     */
549    public default FluentNode toHTML(DocTree tag, Element context, Object object) {
550        FluentNode node = null;
551
552        if (object instanceof byte[]) {
553            node =
554                text(Stream.of(ArrayUtils.toObject((byte[]) object))
555                     .map (t -> String.format("0x%02X", t))
556                     .collect(joining(", ", "[", "]")));
557        } else if (object instanceof boolean[]) {
558            node = text(Arrays.toString((boolean[]) object));
559        } else if (object instanceof double[]) {
560            node = text(Arrays.toString((double[]) object));
561        } else if (object instanceof float[]) {
562            node = text(Arrays.toString((float[]) object));
563        } else if (object instanceof int[]) {
564            node = text(Arrays.toString((int[]) object));
565        } else if (object instanceof long[]) {
566            node = text(Arrays.toString((long[]) object));
567        } else if (object instanceof Object[]) {
568            node = toHTML(tag, context, Arrays.asList((Object[]) object));
569        } else if (object instanceof Type) {
570            node = type(tag, context, (Type) object);
571        } else if (object instanceof Enum<?>) {
572            node = a(tag, context, (Enum<?>) object);
573        } else if (object instanceof Field) {
574            node = a(tag, context, (Field) object);
575        } else if (object instanceof Constructor) {
576            node = a(tag, context, (Constructor) object);
577        } else if (object instanceof Method) {
578            node = a(tag, context, (Method) object);
579        } else if (object instanceof Collection<?>) {
580            List<Node> nodes =
581                ((Collection<?>) object).stream()
582                .map(t -> toHTML(tag, context, t))
583                .collect(toList());
584
585            for (int i = nodes.size() - 1; i > 0; i -= 1) {
586                nodes.add(i, text(", "));
587            }
588
589            node =
590                fragment()
591                .add(text("["))
592                .add(nodes.stream())
593                .add(text("]"));
594        } else {
595            node = text(String.valueOf(object));
596        }
597
598        return node;
599    }
