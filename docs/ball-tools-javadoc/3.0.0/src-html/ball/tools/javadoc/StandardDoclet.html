023import ball.lang.reflect.InterceptingInvocationHandler;
024import java.io.File;
025import java.net.URI;
026import java.util.List;
027import java.util.Locale;
028import java.util.Map;
029import java.util.Set;
030import java.util.TreeMap;
031import javax.tools.Diagnostic;
032import jdk.javadoc.doclet.DocletEnvironment;
033import jdk.javadoc.doclet.Reporter;
034import lombok.NoArgsConstructor;
035import lombok.ToString;
036
037import static java.util.stream.Collectors.toSet;
038import static javax.tools.Diagnostic.Kind.WARNING;
039
040/**
041 * {@link jdk.javadoc.doclet.StandardDoclet} subclass which provides
042 * additional services to {@link AbstractTaglet} implementations.
043 *
044 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
045 * @version $Revision: 7075 $
046 */
047@NoArgsConstructor @ToString
048public class StandardDoclet extends jdk.javadoc.doclet.StandardDoclet {
049    private final Map<URI,URI> links = new TreeMap<>();
050    private Locale locale = null;
051    private Reporter reporter = null;
052
053    /**
054     * {@link Extern} {@link Map} for {@link AbstractTaglet} instances.
055     */
056    public final Map<String,URI> extern = new Extern();
057
058    @Override
059    public void init(Locale locale, Reporter reporter) {
060        this.locale = locale;
061        this.reporter = reporter;
062
063        super.init(locale, reporter);
064    }
065
066    @Override
067    public Set<Option> getSupportedOptions() {
068        var set =
069            super.getSupportedOptions()
070            .stream()
071            .map(OptionWrapper::new)
072            .map(t -> (Option) t.newProxyInstance(Option.class))
073            .collect(toSet());
074
075        return set;
076    }
077
078    @Override
079    public boolean run(DocletEnvironment env) {
080        for (var key : links.keySet()) {
081            var value = links.get(key);
082
083            try {
084                ((Extern) extern).link(key, (value != null) ? value : key);
085            } catch (Exception exception) {
086                print(WARNING, "%s", exception.getMessage());
087            }
088        }
089
090        return super.run(env);
091    }
092
093    /**
094     * Method to print a diagnostic message.
095     *
096     * @param   kind            The {@link javax.tools.Diagnostic.Kind}.
097     * @param   format          The message format {@link String}.
098     * @param   argv            Optional arguments to the message format
099     *                          {@link String}.
100     *
101     * @see Reporter#print(Diagnostic.Kind,String)
102     */
103    protected void print(Diagnostic.Kind kind, String format, Object... argv) {
104        reporter.print(kind, String.format(format, argv));
105    }
106
107    private class OptionWrapper extends InterceptingInvocationHandler<Option> {
108        public OptionWrapper(Option option) { super(option); }
109
110        public void process(String option, List<String> argv) {
111            if (getTarget().getNames().contains("-link")) {
112                links.put(asURI(argv.get(0)), null);
113            } else if (getTarget().getNames().contains("-linkoffline")) {
114                links.put(asURI(argv.get(0)), new File(argv.get(1)).toURI());
115            }
116        }
117
118        private URI asURI(String string) {
119            return URI.create(string + "/").normalize();
120        }
121    }
122}
