192    /**
193     * Dispatches call to {@link #declaration(DocTree,Element,Field)} or
194     * {@link #declaration(DocTree,Element,Method)} as appropriate.
195     *
196     * @param   tag             The context {@link DocTree}.
197     * @param   element         The context {@link Element}.
198     * @param   member          The target {@link Member}.
199     *
200     * @return  {@link org.w3c.dom.DocumentFragment}
201     */
202    public default FluentNode declaration(DocTree tag, Element element, Member member) {
203        FluentNode node = null;
204
205        if (member instanceof Field) {
206            node = declaration(tag, element, (Field) member);
207        } else if (member instanceof Method) {
208            node = declaration(tag, element, (Method) member);
209        } else {
210            throw new IllegalArgumentException(String.valueOf(member));
211        }
212
213        return node;
214    }

216    /**
217     * Method to generate a {@link Field} declaration with javadoc
218     * hyperlinks.
219     *
220     * @param   tag             The context {@link DocTree}.
221     * @param   element         The context {@link Element}.
222     * @param   field           The target {@link Field}.
223     *
224     * @return  {@link org.w3c.dom.DocumentFragment}
225     */
226    public default FluentNode declaration(DocTree tag, Element element, Field field) {
227        return fragment(modifiers(field.getModifiers()),
228                        type(tag, element, field.getGenericType()),
229                        code(SPACE),
230                        a(tag, element, field, (String) null));
231    }

233    /**
234     * Method to generate a {@link Method} declaration with javadoc
235     * hyperlinks.
236     *
237     * @param   tag             The context {@link DocTree}.
238     * @param   element         The context {@link Element}.
239     * @param   method          The target {@link Method}.
240     *
241     * @return  {@link org.w3c.dom.DocumentFragment}
242     */
243    public default FluentNode declaration(DocTree tag, Element element, Method method) {
244        var node =
245            fragment(modifiers(method.getModifiers()),
246                     type(tag, element, method.getGenericReturnType()),
247                     code(SPACE),
248                     a(tag, element, method, (String) null));
249        var parameters = method.getParameters();
250
251        node.add(code("("));
252
253        for (int i = 0; i < parameters.length; i += 1) {
254            if (i > 0) {
255                node.add(code(", "));
256            }
257
258            node.add(declaration(tag, element, parameters[i]));
259        }
260
261        node.add(code(")"));
262
263        return node;
264    }

293     * {@code <a href="}{@link Annotation}{@code ">}{@link #code(String) code(String.valueOf(annotation))}{@code </a>}
294     *
295     * @param   tag             The context {@link DocTree}.
296     * @param   element         The context {@link Element}.
297     * @param   annotation      The target {@link Annotation}.
298     *
299     * @return  {@link org.w3c.dom.Element}
300     */
301    public default FluentNode annotation(DocTree tag, Element element, Annotation annotation) {
302        var type = annotation.annotationType();
303        var string =
304            String.valueOf(annotation)
305            .replace(type.getCanonicalName(), type.getSimpleName());
306
307        return fragment().add(a(href(tag, element, type), code(string)));
308    }

311     * Method to generate modifiers for {@code declaration()} methods.
312     *
313     * @param   modifiers       See {@link Modifier}.
314     *
315     * @return  {@link org.w3c.dom.DocumentFragment}
316     */
317    public default FluentNode modifiers(int modifiers) {
318        var node = fragment();
319        var string = Modifier.toString(modifiers);
320
321        if (isNotEmpty(string)) {
322            node.add(code(string + SPACE));
323        }
324
325        return node;
326    }

328    /**
329     * Method to generate types for {@code declaration()} methods.
330     *
331     * @param   tag             The context {@link DocTree}.
332     * @param   element         The context {@link Element}.
333     * @param   type            The target {@link Type}.
334     *
335     * @return  {@link org.w3c.dom.DocumentFragment}
336     */
337    public default FluentNode type(DocTree tag, Element element, Type type) {
338        FluentNode node = null;
339
340        if (type instanceof ParameterizedType) {
341            node =
342                fragment(type(tag, element, ((ParameterizedType) type).getRawType()));
343
344            var types = ((ParameterizedType) type).getActualTypeArguments();
345
346            node = node.add(code("<"));
347
348            for (int i = 0; i < types.length; i += 1) {
349                if (i > 0) {
350                    node.add(code(","));
351                }
352
353                node.add(type(tag, element, types[i]));
354            }
355
356            node.add(code(">"));
357        } else if (type instanceof Class<?>) {
358            node = a(tag, element, (Class<?>) type);
359        } else {
360            node = code(type.getTypeName());
361        }
362
363        return node;
364    }

384    /**
385     * {@code <table>}{@link TableModel model}{@code </table>}
386     *
387     * @param   tag             The context {@link DocTree}.
388     * @param   element         The context {@link Element}.
389     * @param   model           The {@link TableModel} to use to create the
390     *                          new table {@link org.w3c.dom.Element}.
391     * @param   nodes           The {@link Node}s to append to the newly
392     *                          created
393     *                          {@link org.w3c.dom.Element}.
394     *
395     * @return  {@link org.w3c.dom.Element}
396     */
397    public default FluentNode table(DocTree tag, Element element,
398                                    TableModel model, Node... nodes) {
399        var table = table();
400        var names =
401            IntStream.range(0, model.getColumnCount())
402            .boxed()
403            .map(model::getColumnName)
404            .toArray(String[]::new);
405
406        if (! isAllBlank(names)) {
407            table.add(thead(tr(Stream.of(names).map(this::th))));
408        }
409
410        table
411            .add(tbody(IntStream.range(0, model.getRowCount())
412                       .boxed()
413                       .map(y -> tr(IntStream.range(0, names.length)
414                                    .boxed()
415                                    .map(x -> td(toHTML(tag, element,
416                                                        model.getValueAt(y, x))))))));
417
418        return table.add(nodes);
419    }

421    /**
422     * Method to get a Javadoc HTML representation of an {@link Object}.
423     *
424     * @param   tag             The context {@link DocTree}.
425     * @param   element         The context {@link Element}.
426     * @param   object          The target {@link Object}.
427     *
428     * @return  {@link org.w3c.dom.Node}
429     */
430    public default FluentNode toHTML(DocTree tag, Element element, Object object) {
431        FluentNode node = null;
432
433        if (object instanceof byte[]) {
434            node =
435                text(Stream.of(ArrayUtils.toObject((byte[]) object))
436                     .map (t -> String.format("0x%02X", t))
437                     .collect(joining(", ", "[", "]")));
438        } else if (object instanceof boolean[]) {
439            node = text(Arrays.toString((boolean[]) object));
440        } else if (object instanceof double[]) {
441            node = text(Arrays.toString((double[]) object));
442        } else if (object instanceof float[]) {
443            node = text(Arrays.toString((float[]) object));
444        } else if (object instanceof int[]) {
445            node = text(Arrays.toString((int[]) object));
446        } else if (object instanceof long[]) {
447            node = text(Arrays.toString((long[]) object));
448        } else if (object instanceof Object[]) {
449            node = toHTML(tag, element, Arrays.asList((Object[]) object));
450        } else if (object instanceof Type) {
451            node = type(tag, element, (Type) object);
452        } else if (object instanceof Enum<?>) {
453            node = a(tag, element, (Enum<?>) object);
454        } else if (object instanceof Field) {
455            node = a(tag, element, (Field) object);
456        } else if (object instanceof Constructor) {
457            node = a(tag, element, (Constructor) object);
458        } else if (object instanceof Method) {
459            node = a(tag, element, (Method) object);
460        } else if (object instanceof Collection<?>) {
461            List<Node> nodes =
462                ((Collection<?>) object)
463                .stream()
464                .map(t -> toHTML(tag, element, t))
465                .collect(toList());
466
467            for (int i = nodes.size() - 1; i > 0; i -= 1) {
468                nodes.add(i, text(", "));
469            }
470
471            node =
472                fragment()
473                .add(text("["))
474                .add(nodes.stream())
475                .add(text("]"));
476        } else {
477            node = text(String.valueOf(object));
478        }
479
480        return node;
481    }
